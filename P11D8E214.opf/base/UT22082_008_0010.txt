༄༅། །དཔལ་ཀྱཻ་རྡོ་རྗེ་ལྷ་དགུའི་སྒྲུབ་ཐབས་དཀྱིལ་ཆོག་རྣལ་འབྱོར་རྫོགས་པའི་ཕྲེང་བ་

ཞེས་བྱ་བ་བཞུགས་སོ།། བདེ་བའི་རང་བཞིན་ཨ་ཡིག་ལས་འོངས་ཐབས་ཀྱི་མཆོག་གྱུར་ཕ་རྒྱུད་དང༌། །སྟོང་གསལ་འགག་མེད་ཧཾ་ཡིག་ལས་འོངས་ཤེས་རབ་ངོ་བོ་མ་རྒྱུད་དེ། བདེ་སྟོང་ཨ་ཧཾ་ཐབས་ཤེས་ཟུང་འཇུག་བདེ་ཆེན་འཁོར་ལོའི་ཡེ་ཤེས་ནི། །ཕ་རྒྱུད་མ་རྒྱུད་སྡོམ་པའི་སྙིང་རྗེ་གཉིས་མེད་དགྱེས་རྡོར་ལ་ཕྱག་འཚལ། །གང་ཞིག་འོག་མིན་ཞིང་དུ་སངས་རྒྱས་ཀྱང༌། །སླར་ཡང་གདུལ་བྱའི་སྣང་ངོར་ཞིང་ཁམས་འདིར། །འགྲོ་དོན་མཛད་ལ་ཆོག་ཤེས་མི་མངའ་བ། ་༌།བརྒྱུད་པའི་བླ་མ་རྣམས་ལ་ཕྱག་འཚལ་ལོ། ་༌།མཁས་དང་གྲུབ་པའི་དཔལ་མངའ་ཞིང༌། །སངས་རྒྱས་བསྟན་པའི་ཁུར་ལྕི་བ། །འདེགས་བྱེད་སྤོས་ཀྱི་གླང་པོ་ཆེ། །བྱམས་པའི་མཚན་ཅན་རྒྱལ་གྱུར་ཅིག །གསང་སྔགས་རྒྱུད་སྡེ་རྒྱ་མཚོ་ཆེན་པོ་ཡི༑ ༑སྙིང་པོ་ནོར་བུ་དགྱེས་པ་རྡོ་རྗེ་འདི། །ལན་ཅིག་ཉམས་སུ་བླང་བར་འདོད་རྣམས་ཀྱི། །ཆོག་གཞུང་གྲུ་གཟིངས་བཟང་པོ་འདིར་ཞུགས་ཏེ། །ངལ་བ་མེད་པར་འདོང་ཕྱིར་འདི་བྲིའོ། ། འདིར་དཀྱིལ་འཁོར་དུ་དབང་བསྐུར་བ་ཐོབ་ཅིང༌། དམ་ཚིག་དང་ལྡན་པའི་སྒྲུབ་པ་པོའི་ཡིད་དང་མཐུན་པའི་གནས་སུ་དཀྱིལ་འཁོར་དང༌། བུམ་པ་སོགས་འོག་ནས་འབྱུང་བ་ལྟར་ཉེར་བསྡོགས་བྱས་ལ། ཐོག་མར་སྐྱབས་སུ་འགྲོ་བ་དང་སེམས་བསྐྱེད་ལ། རང་ཉིད་སྐད་ཅིག་གིས་དཔལ་དགྱེས་པ་རྡོ་རྗེ་ལྷན་ཅིག་སྐྱེས་པའི་སྐུར་གསལ་བར་གྱུར་ པའི་ངང་ནས། བདུད་རྩི། ཨོཾ་བཛྲ་ཨ་མྲྀ་ཏ ས་བསངས། ཨོཾ་སྭ་བྷཱ་ཝ ས་སྦྱངས། སྟོང་པའི་ངང་ལས་ཡཾ་ལས་རླུང་གི་དཀྱིལ་འཁོར་སྔོན་པོ་གཞུ་ལྟ་བུ་བ་དན་གྱིས་མཚན་པ། དེའི་སྟེང་དུ་རཾ་ལས་མེའི་དཀྱིལ་འཁོར་དམར་པོ་གྲུ་གསུམ་པ་མེ་རབ་ཏུ་འབར་བས་མཚན་པ། དེའི་སྟེང་དུ་ཀཾ་ལས་མི་མགོའི་སྒྱེད་པུ་གསུམ་གྱི་སྟེང་དུ་ཨ་དཀར་པོ་ལས་བྱུང་བའི་ཐོད་པ་དུམ་བུ་གཅིག་པ། ཕྱི་དཀར་ལ་ནང་དམར་བ་ཡངས་ཤིང་རྒྱ་ཆེ་བའི་ནང་དུ་དབུས་སུ་ན་ལས་མིའི་ཤ༑ ཤར་དུ་གོ་ལས་བ་ལང་གི་ཤ།

ལྷོར་ཀུ་ལས་ཁྱིའི་ཤ། ནུབ་ཏུ་ད་ལས་རྟའི་ཤ། བྱང་དུ་ཧ་ལས་གླང་པོ་ཆེའི་ཤ། དེ་རྣམས་ཀྱི་སྟེང་གི་དབུས་སུ་ཧཱུྂ་ལས་དྲི་ཆུ་དང་ཆོས་ཀྱི་དབྱིངས་ཀྱི་ཡེ་ཤེས།་་་ཤར་དུ་བྷྲཱུཾ་ལས་དྲི་ཆེན་དང་མེ་ལོང་ལྟ་བུའི་ཡེ་ཤེས། ལྷོར་ཨཾ་ལས་རཀྟ་དང་མཉམ་པ་ཉིད་ཀྱི་ཡེ་ཤེས། ནུབ་ཏུ་ཛྲཱིཾ་ལས་བྱང་སེམས་དང་སོ་སོར་རྟོག་པའི་ཡེ་ཤེས། བྱང་དུ་ཁཾ་ལས་ཤ་ཆེན་དང་བྱ་བ་གྲུབ་པའི་ཡེ་ཤེས། དེའི་སྟེང་དུ་ཁ་ཆོད་ཀྱི་ཚུལ་དུ་ཟླ་བའི་དཀྱིལ་འཁོར་གྱི་གཡོན་སྐོར་དུ། ཨ་ཨཱ། ཨི་ཨཱི། ཨུ་ཨཱུ། ནས། ཨཾ་ཨཱ ཿ དང༌། གཡས་སྐོར་དུ། ཀ་ཁ་ག་གྷ་ང༌། ནས༑ ཤ་ཥ་ས་ཧ་ཀྵཿ དེ་རྣམས་ཀྱི་སྟེང་གི་དབུས་སུ་རྡོ་རྗེ་སྔོན་པོ་རྩེ་ལྔ་པའི་ལྟེ་བར་ཨོཾ་ཨཱཿཧཱུྂ་ལས་འོད་ཟེར་འཕྲོས། འགྲོ་བའི་དོན་བྱས། དེ་བཞིན་གཤེགས་པ་ཐམས་ཅད་ཀྱི་བྱིན་རླབས་འོད་ཟེར་གྱི་རྣམ་པར་བསྡུས་ཧཱུྂ་ལ་ཐིམ་པས་རྡོ་རྗེའི་རྩེ་གསུམ་དང༌། རྡོ་རྗེ་དབྱངས་གསལ་ཟླ་བའི་དཀྱིལ་འཁོར་དང་བཅས་པ་འོད་དུ་ཞུ། བདུད་རྩི་ལ་ཐིམ་པས་ཡེ་ཤེས་ཀྱི་བདུད་རྩིའི་རྒྱ་མཚོ་ཆེན་པོར་གྱུར། ཨོཾ་ཨཱཿཧཱུྂ ་ཞེས་བརྗོད། དེ་ནས་སྔོན་འགྲོའི་གཏོར་མ་གོང་ལྟར་བྱིན་རླབས་བྱས་ལ། རྡོ་རྗེ་ཐལ་མོ་ཁ་ཕྱེ་བའི་ཕྱག་རྒྱས། རང་གི་ཐུགས་ཀའི་ཧཱུྂ་གི་འོད་ཀྱིས་ཕྱོགས་སྐྱོང་འཁོར་བཅས་སྤྱན་དྲངས། ཨོཾ་ཁ་ཁ་ཁཱ་ཧི་ཁཱ་ཧི། སརྦ་ཡཀྵ། རཀྵ་ས། བྷཱུ་ཏ། པྲེ་ཏ། པི་ཤཱ་ཙཱ། ཨུནྨཱ་ད། ཨ་པ་སྨ་ར་ཌཱ་ཀ་ཌཱ་ཀི་ནྱ་དཱ་ཡ་ཨི་མཾ་བ་ལིཾ་གྲྀཧྞནྟུ། ས་མ་ཡ་རཀྵནྟུ༑ མ་མ་སརྦ་སིདྡྷི་མྨེ་པྲ་ཡཙྪནྟུ། ཡ་ཐེ་བཾ། ཡ་ཐཻཥྚཾ། བྷུཉྫ་ཐ། པི་བ་ཐ། ཛི་གྷྲ་ཐ། མཱ་ཏི་ཀྲ་མ་ཐ། མ་མ་སརྦ་ཀཱརྟ་ཡ་སཏྶུ་ཁཾ་བི་ཤུདྡྷ་ཡ་ས་ཧཱ་ཡི་ཀཾ་བྷ་ཝནྟུ་ཧཱུྂ་ཧཱུྂ་ཕཊ་སྭཱ་ཧཱ། ཞེས་ལན་གསུམ་གྱིས་གཏོར་མ་ཕུལ་ནས། ལྷ་ཡི་ཚོགས་རྣམས་མ་ལུས་དང༌། །ཀླུ་ཡི་ཚོགས་རྣམས་མ་ལུས་དང༌། །གནོད་སྦྱིན་ཚོགས་རྣམས་མ་ལུས་དང༌། །སྲིན་པོའི་ཚོགས་རྣམས་མ་ལུས་དང༌༑ ༑འབྱུང་པོའི་ཚོགས་རྣམས་མ་ལུས་དང༌། །ཡི་དྭགས་ཚོགས་རྣམས་མ་ལུས་དང༌། །ཤ་ཟའི་ཚོགས་རྣམས་མ་ལུས་དང༌། །སྨྱོ་བྱེད་ཚོགས་རྣམས་མ་ལུས་དང༌། །རྗེད་བྱེད་ཚོགས་རྣམས་མ་ལུས་དང༌། །མཁའ་འགྲོའི་ཚོགས་རྣམས་མ་ལུས་དང༌། །མ་མོའི་ཚོགས་རྣམས་མ་ལུས་དང༌། །མ་ལུས་པ་ནི་ཐམས་ཅད་དག

།འདིར་གཤེགས་བདག་ལ་དགོངས་སུ་གསོལ། །མཆོད་སྦྱིན་གཏོར་མ་འདི་བཞེས་ལ། །རྣལ་འབྱོར་བདག་ཅག་འཁོར་བཅས་ལ། །ནད་མེད་ཚེ་དང་དབང་ཕྱུག་དང༌། །དཔལ་དང་གྲགས་དང་སྐལ་པ་བཟང༌༑ ༑ལོངས་སྤྱོད་རྒྱ་ཆེན་ཀུན་ཐོབ་ཅིང༌། །ཞི་དང་རྒྱས་ལ་སོགས་པ་ཡི། །ལས་ཀྱི་དངོས་གྲུབ་བདག་ལ་སྩོལ། །དམ་ཚིག་ཅན་གྱིས་བདག་ལ་སྲུངས། །དངོས་གྲུབ་ཀུན་གྱི་བཀའ་དྲིན་སྩོལ༑ ༑དུས་མིན་འཆི་དང་ནད་རྣམས་དང༌། །གདོན་དང་བགེགས་རྣམས་ཞི་བ་དང༌། །རྨི་ལམ་ངན་དང་མཚན་མ་ངན། །བྱད་སྟེམས་ངན་པ་མེད་པར་མཛོད། །འཇིག་རྟེན་བདེ་ཞིང་ལོ་ལེགས་དང༌། །འབྲུ་རྣམས་འཕེལ་ཞིང་ཆོས་འཕེལ་བ། །བདེ་ལེགས་ཐམས་ཅད་འགྲུབ་པ་ཡི། །ཡིད་ལ་འདོད་པ་ཀུན་འགྲུབ་མཛོད། ། ཅེས་གཏོར་མ་ཕུལ། མཆོད་པ་བསངས་སྦྱངས། སྟོང་པའི་ངང་ལས་ཨ་ལས་ཐོད་པའི་སྣོད་དུ་ཧཱུྂ་ཡོངས་སུ་གྱུར་པ་ལས། མཆོད་ཡོན། ཞབས་བསིལ། བསང་གཏོར། ཞལ་བསིལ། མེ་ཏོག །བདུག་སྤོས། སྣང་གསལ། དྲི་ཆབ། ཞལ་ཟས། རོལ་མོ་རྣམས་དྭངས་ཤིང་ཐོགས་པ་མེད་པ་རང་བཞིན་སྟོང་པ་ཟག་མེད་ཀྱི་བདེ་བ་བསྐྱེད་ནུས་པ་བསམ་གྱིས་མི་ཁྱབ་པར་གྱུར། ཨོཾ་བཛྲ་ཨརྒྷཾ་ཨཱཿཧཱུྂ། དེ་བཞིན་དུ། པཱ་དྱཾ༴ ཨཱ་ཙ་མཱ་ནཾ༴ པྲོཀྵ་ཎཾ༴ པུཥྤཾ༴ དྷཱུ་པཾ༴ ཨཱ་ལོ་ཀཾ༴ གནྡྷཾ༴ ནཻ་ཝི་དྱཾ༴ ཤཔད ་ཞེས་སྦྱར་བས་བྱིན་གྱིས་བརླབས། དེ་ནས། སྭ་བྷཱ་ཝ ས་སྦྱངས། སྟོང་པའི་ངང་ལས་པདྨ་དང༌། རཾ་ལས་ཉི་མའི་དཀྱིལ་འཁོར་གྱི་སྟེང་དུ། ཧཱུྂ་ལས་སྣ་ཚོགས་རྡོ་རྗེ། ཤར་དཀར། ལྷོ་སེར། ནུབ་དམར། བྱང་ལྗང༌། ལྟེ་བ་མཐིང་ནག་ཧཱུྂ་གིས་མཚན་པ་ལས་འོད་འཕྲོས། སྟེང་འོག་ཐམས་ཅད་རྡོ་རྗེའི་ས་གཞི། ར་བ། དྲ་བ། གུར༑ བླ་བྲེ། མདའ་ཡི་དྲ་བ། ཕྱི་རོལ་དུ་ཡེ་ཤེས་ཀྱི་མེ་དཔུང་འབར་བའི་དབུས་སུ་བྷྲཱུཾ་ལས་འཁོར་ལོ་སྔོན་པོ་རྩིབས་བཅུ་པ། རྩིབས་སོ་སོའི་ནང་དུ་པད་ཉིའི་གདན་ལ་ཧཱུྂ་རེ་རེ་ཡོངས་སུ་གྱུར་པ་ལས། དབུས་སུ་རང་ཉིད་བྷྲཱུཾ་ལས་གཙུག་ཏོར་འཁོར་ལོས་སྒྱུར་བའི་རྒྱལ་པོ་སྐུ་མདོག་སེར་པོ་འཁོར་ལོ་འཛིན་པ།

ཤར་དུ་གཤིན་རྗེ་མཐར་བྱེད་སྔོན་པོ་ཐོ་བ་འཛིན་པ། ལྷོར་ཤེས་རབ་མཐར་བྱེད་དཀར་པོ་བེ་ཅོན་འཛིན་པ། ནུབ་ཏུ་པདྨ་མཐར་བྱེད་དམར་པོ་པདྨ་འཛིན་པ། བྱང་དུ་བགེགས་མཐར་བྱེད་སྔོན་པོ་རྡོ་རྗེ་འཛིན་པ། བྱང་ཤར་དུ་མི་གཡོ་བ་སྔོན་པོ་རལ་གྲི་འཛིན་པ། ཤར་ལྷོར་འདོད་རྒྱལ་སྔོན་པོ་ལྕགས་ཀྱུ་འཛིན་པ། ལྷོ་ནུབ་ཏུ་དབྱུག་སྔོན་ཅན་སྔོན་པོ་དབྱུག་པ་འཛིན་པ། ནུབ་བྱང་དུ་སྟོབས་པོ་ཆེ་སྔོན་པོ་རྩེ་གསུམ་འཛིན་པ། སྟེང་དུ་ཧཱུྂ་དཀར་སྔོན་པོ་རྡོར་དྲིལ་འཛིན་པ། འོག་ཏུ་གནོད་མཛེས་སྔོན་པོ་གཏུན་ཤིང་འཛིན་པ། དེ་ཐམས་ཅད་ཀྱང་ཁྲོ་ཞིང་གཏུམ་པ། སྐུ་སྦོམ་ཞིང་གསུས་ཁྱིམ་ཆེ་བ། སྟག་གི་པགས་པའི་ཤམ་ཐབས་དང་སྦྲུལ་གྱིས་བརྒྱན་པ། ཞབས་དོར་སྟབས་སུ་བཞུགས་པ། དེ་རྣམས་ཀྱི་མིག་ཏུ་གཏི་མུག་རྡོ་རྗེ། རྣ་བར་ཞེ་སྡང་རྡོ་རྗེ། སྣ་ལ་སེར་སྣ་རྡོ་རྗེ། ཁ་ལ་འདོད་ཆགས་རྡོ་རྗེ། ལུས་ལ་ཕྲག་དོག་རྡོ་རྗེ། ཡིད་ལ་བདག་མེད་རྣལ་འབྱོར་མ་རྣམས་དང༌། སྤྱི་བོར་ཨོཾ་ལས་སྐུ་རྡོ་རྗེ་དཀར་མོ། མགྲིན་པར་ཨཱཿལས་གསུང་རྡོ་རྗེ་དམར་མོ། ་་ཐུགས་ཀར་ཧཱུྂ་ལས་ཐུགས་རྡོ་རྗེ་མ་སྔོན་མོ་རྣམས་སུ་གསལ་བ། ཡེ་ཤེས་ཀྱི་མེ་དཔུང་འབར་བའི་དབུས་སུ་བཞུགས་པའི་ཐུགས་ཀར་ཧཱུྂ་ལས་རང་འདྲ་བའི་ཁྲོ་བོའི་ཚོགས་དཔག་ཏུ་མེད་པ་འཕྲོས་པས། བགེགས་ཐམས་ཅད་དཔག་ཚད་དང་རྒྱང་གྲགས་སུ་བསྐྲད་ཅིང་མ་སོང་བ་རྣམས་ཀྱང་ཐལ་བའི་རྡུལ་ཕྲན་བཞིན་དུ་བརླག་པར་འགྱུར། ཨོཾ་དེ་ཝ་པི་ཙུ་བཛྲ་ཧཱུྂ་ཧཱུྂ་ཧཱུྂ་ཕཊ། ཨོཾ་བྷྲཱུཾ་ཧཱུྂ་ཧཱུྂ་ཧཱུྂ་ཧཱུྂ་ཧཱུྂ་ཧཱུྂ་ཧཱུྂ་ཧཱུྂ་ཧཱུྂ་ཧཱུྂ། ཨོཾ་སུཾ་བྷ་ནི་སུཾ་བྷ་ནི་ཧཱུྂ་ཧཱུྂ་ཕཊ། ནས། བིདྱཱ་རཱ་ཛ་ཧཱུྂ་ཧཱུྂ་ཕཊ། ཅེས་སྲུང་འཁོར་བསྒོམ། རང་ཉིད་སྐད་ཅིག་གིས་ཀྱེ་རྡོ་རྗེ་གསལ་བའི་ཐུགས་ཀར་པཾ་ལས་པདྨ་དང༌། རཾ་ལས་ཉི་མའི་དཀྱིལ་འཁོར་གྱི་སྟེང་དུ་ཧཱུྂ་སྔོན་པོ་ལས་འོད་ཟེར་འཕྲོས་པས། ཕྱོགས་བཅུའི་སངས་རྒྱས་དང་བྱང་ཆུབ་སེམས་དཔའ་ཐམས་ཅད་དཔལ་ཧེ་རུ་ཀ་དགྱེས་པ་རྡོ་རྗེ་ལྷ་དགུའི་དཀྱིལ་འཁོར་གྱི་རྣམ་པར་མདུན་གྱི་ནམ་མཁར་སྤྱན་དྲངས།

བཛྲ་ས་མཱ་ཛཿ ཞེས་སྤྱན་དྲངས་པར་མོས་ཏེ། མཆོད་པ་ནི། དཀར་མོ་རི་དྭགས་མཚན་མ་འཛིན། །ཆོམ་རྐུན་བདུད་ལས་རྒྱལ་བའི་སྣོད། །རོ་ལངས་མ་ནི་ཆུ་ལག་མ། །གྷསྨ་རཱི་ནི་སྨན་འཛིན་ཅིང༌། །པུཀྐ་སཱི་ནི་རྡོ་རྗེའི་ལག །དེ་བཞིན་རི་ཁྲོད་མི་རོ་འཛིན། །གདོལ་པ་མོ་ནི་ཅང་ཏེའུ་བརྡུང༌། །འདི་རྣམས་ཀྱིས་ནི་གཙོ་བོར་མཆོད། །འདོད་ཆགས་ཆེན་པོའི་རྗེས་ཆགས་པས༑ ༑གཡུང་མོས་སྐུ་ལ་འཁྱུད་པ་ཉིད། ། ཅེས་དང༌། ཨརྒྷཾ་ སོགས་ཀྱིས་མཆོད་ལ། དཀོན་མཆོག་གསུམ་ལ་བདག་སྐྱབས་མཆི། །སྡིག་པ་ཐམས་ཅད་སོ་སོར་བཤགས། །འགྲོ་བའི་དགེ་ལ་རྗེས་ཡི་རངས། །སངས་རྒྱས་བྱང་ཆུབ་ཡིད་ཀྱིས་བཟུང༌། །སངས་རྒྱས་ཆོས་དང་ཚོགས་མཆོག་ལ། །བྱང་ཆུབ་བར་དུ་སྐྱབས་སུ་མཆི། །རང་གཞན་དོན་གཉིས་རབ་བསྒྲུབ་ཕྱིར། །བྱང་ཆུབ་སེམས་ནི་བསྐྱེད་པར་བགྱི། །བྱང་ཆུབ་མཆོག་གི་སེམས་ནི་བསྐྱེད་བགྱིས་ནས། །སེམས་ཅན་ཐམས་ཅད་བདག་གི་མགྲོན་དུ་གཉེར། །བྱང་ཆུབ་སྤྱོད་མཆོག་ཡིད་འོང་བསྐྱེད་པར་བགྱི། །འགྲོ་ལ་ཕན་ཕྱིར་སངས་རྒྱས་ཐོབ་པར་ཤོག ། ཅེས་གསུམ་བརྗོད། ཚོགས་ཞིང་རང་ལ་ཐིམ་པར་གྱུར། སེམས་ཅན་ཐམས་ཅད་བདེ་བ་དང་བདེ་བའི་རྒྱུ་དང་ལྡན་པར་གྱུར་ཅིག ༑སྡུག་བསྔལ་དང་སྡུག་བསྔལ་གྱི་རྒྱུ་དང་བྲལ་བར་གྱུར་ཅིག །སྡུག་བསྔལ་མེད་པའི་བདེ་བ་དང་མི་འབྲལ་བར་གྱུར་ཅིག །ཉེ་རིང་ཆགས་སྡང་གཉིས་དང་བྲལ་བའི་བཏང་སྙོམས་ཆེན་པོ་ལ་ཡུན་རིང་དུ་གནས་པར་གྱུར་ཅིག ། ཅེས་གསུམ་བརྗོད། ཨོཾ་ཤཱུ་ནྱ་ཏཱ་ཛྙཱ་ན་བཛྲ་སྭ་བྷཱ་ཝ་ཨཱཏྨ་ཀོ྅ཧཾ། སྟོང་པའི་ངང་ལས་རང་ཉིད་གཙུག་ཏོར་འཁོར་ལོས་སྒྱུར་བའི་རྒྱལ་པོ་ཡོངས་སུ་གྱུར་པ་ལས༑ ཨེ་ལས་ཆོས་ཀྱི་འབྱུང་གནས་སྔོན་པོ་གྲུ་གསུམ་པ་རྩེ་མོ་ཕྲ་བ་ཐུར་དུ་ཟུག་པ། གདེངས་ཀ་ཆེ་བ་གྱེན་དུ་བསྟན་པ། དེའི་སྟེང་དུ་ཡཾ་ལས་རླུང་གི་དཀྱིལ་འཁོར་སྔོན་པོ་གཞུ་ལྟ་བུའི་བ་དན་ལ་ཡཾ་གྱིས་མཚན་པ། དེའི་སྟེང་དུ་རཾ་ལས་མེའི་དཀྱིལ་འཁོར་དམར་པོ་གྲུ་གསུམ་པ་རཾ་གྱིས་མཚན་པ།

དེའི་སྟེང་དུ་ཆུའི་དཀྱིལ་འཁོར་དཀར་པོ་ཟླུམ་པོ་བུམ་པ་ལ་བཾ་གྱིས་མཚན་པ། དེའི་སྟེང་དུ་ལཾ་ལས་སའི་དཀྱིལ་འཁོར་སེར་པོ་གྲུ་བཞི་པ་ཟུར་བཞིར་རྡོ་རྗེ་ཕྱེད་པ་དང་ལཾ་གྱིས་མཚན་པ། དེའི་སྟེང་དུ་བྷྲཱུཾ་ལས་འཁོར་ལོ་སེར་པོ་རྩིབས་བཅུ་པ་བྷྲཱུཾ་གྱིས་མཚན་པ། དེའི་སྟེང་དུ་རིན་པོ་ཆེ་སྣ་ཚོགས་ལས་གྲུབ་པའི་གཞལ་ཡས་ཁང་གྲུ་བཞི་སྒོ་བཞི་པ་ནང་ནས་རིམ་བཞིན་དུ། ནག །ལྗང༌། དམར། སེར། སྔོ་བའི་རྩིག་པ་རིམ་པ་ལྔ་དང་ལྡན་པའི་སྟེང་དུ་ཕ་གུ་སེར་པོ་དང༌། དྲ་བ་དང༌། དྲ་ཕྱེད་འཕྱང་བ། དེའི་སྟེང་དུ་བ་ཀུ་ལི་དང༌། རིན་པོ་ཆེའི་མདའ་ཡབ་པུ་ཤུས་བརྒྱན་པ། རྩིག་པའི་ཕྱི་རོལ་དུ་འདོད་ཡོན་གྱི་སྣམ་བུ་དམར་པོ་ལ་གནས་པའི་ལྷ་མོ་བཅུ་དྲུག་གིས་མཆོད་རྫས་ཐོགས་ནས་བཅོམ་ལྡན་འདས་ལ་མཆོད་པ། ནང་ཀ་བ་བརྒྱད་དང་ལྡན་པའི་སྟེང་དུ་རྡོ་རྗེའི་གདུང་བཞི་མིག་མང་རིས་སུ་བྲིས་ཤིང་མཛེས་པར་ཕུབ་པ། དབུས་ཕྱུར་བུར་གྱུར་པའི་སྟེང་དུ་རྡོ་རྗེའི་མེ་ཏོག་གིས་སྤྲས་པ། སྒོ་བཞིའི་སྟེང་དུ་ཀ་བ་བཞི་བཞིས་བཏེག་པའི་རྟ་བབས་སྣམ་བུ་བཅུ་གཅིག་དང་ལྡན་པའི་སྟེང་དུ་པདྨ་ལ་གནས་པའི་ཆོས་ཀྱི་འཁོར་ལོ་དང༌། གདུགས་དང༌། རི་དྭགས་དང་བཅས་པ། གཞན་ཡང་གོས་དང༌། འཕན་དང༌། མེ་ཏོག་གི་ཕྲེང་བ་དང༌། ཟླ་བ་རླུང་གིས་བསྐྱོད་པ་དང༌། རྔ་ཡབ་ནོར་བུའི་ཡུ་བ་ཅན་ལ་སོགས་པ་རྒྱན་དུ་མས་བརྒྱན་པའི་གཞལ་ཡས་ཁང་ཆེན་པོ་དེའི་ཕྱི་རོལ་དུ་སྣ་ཚོགས་པདྨ་དང་རྡོ་རྗེའི་ར་བ། དེའི་ཕྱི་རོལ་དུ་ཡེ་ཤེས་ཀྱི་མེ་ཁ་དོག་ལྔ་དང་ལྡན་པ་རབ་ཏུ་འབར་བས་བསྐོར་བ། དེའི་ཕྱི་རོལ་དུ་ཆུ་བརྒྱད་ཀྱིས་ཕྱེ་བའི་དུར་ཁྲོད་བརྒྱད་དེ། གཏུམ་དྲག །ཚང་ཚིང་འཁྲིགས་པ། རྡོ་རྗེ་འབར་བ། ཀེང་རུས་ཅན། མི་བཟད་བཞེད་པ། བཀྲ་ཤིས་ཚལ། མུན་པ་དྲག་པོ། ཀཱི་ལི་ཀཱི་ལ་སྒྲོག་པ་དེ་རྣམས་སུ་ཤིང་ཆེན་པོ་བརྒྱད་དེ། ཤིརྵ། ཙཱུཏྟ། ཀངྐ་ལ། ཨ་ཤྭཏྠ། དྷུ་པ་ཊ། ཀ་རཉྫ། འཁྲི་ཤིང་ཅན། ཨརྫུ་ན།

དེ་རྣམས་ཀྱི་རྩ་བར་ཕྱོགས་སྐྱོང་བརྒྱད་ནི། དབང་པོ་བརྒྱ་བྱིན་སེར་པོ། ཆུ་ལྷ་དཀར་པོ། གཤིན་རྗེ་སྔོན་པོ། དབང་ཕྱུག་དཀར་པོ། མེ་ལྷ་དམར་པོ། སྲིན་པོ་སྔོན་པོ། རླུང་ལྷ་དུད་ཁ་རྣམས་དང༌། ཤིང་གི་རྩེ་མོར་ཞིང་སྐྱོང་བརྒྱད་དེ། གླང་པོ་ཆེ་དང༌། མི་དང༌། ཆུ་སྲིན་དང༌། མ་ཧེ་དང༌། གླང་དང༌། ར་དང༌། རོ་ལངས་དང༌། ཤ་བའི་གདོང་ཅན་དང༌། ཆུ་ལ་གནས་པའི་ཀླུ་བརྒྱད་ནི། ནོར་རྒྱས། འཇོག་པོ། སྟོབས་རྒྱུ། དུང་སྐྱོང༌། པདྨ། པདྨ་ཆེན་པོ། མཐའ་ཡས། རིགས་ལྡན་དང༌། དེ་དག་གི་སྟེང་དུ། སྤྲིན་ཆེན་པོ་བརྒྱད་ནི། སྒྲོག་པ། ཆེར་སྒྲོག་པ། དྲག་པོ། གཡོ་བ། བརྟན་པ། གང་བ། འབེབས་པ། གཏུམ་པོ་རྣམས་དང༌། རི་བརྒྱད་ནི། ལྷུན་པོ། མནྡྷ་ར། ཀཻ་ལ་ཤ། མཱ་ལ་ཡ། དབང་ཆེན། སྤོས་ངད་ལྡན། གངས་ཅན། དཔལ་གྱི་རི་བོ་རྣམས་སོ། །མཆོད་རྟེན་བརྒྱད་ནི། རྡོ་རྗེ། རྡོ་རྗེ་འདུས་བྱས། རྡོ་རྗེ་ཕྱག་པ། རྡོ་རྗེ་ནག་པོ། ཐུགས་རྡོ་རྗེ། སྐུ་རྡོ་རྗེ། རིན་ཆེན་རྡོ་རྗེ་རྣམས་དང༌། གཞན་ཡང་ཀེཾ་རུས་དང༌། ཅེ་སྤྱང་དང༌། བྱ་ཁྱི་གཅན་གཟན་ལ་སོགས་པས་ཡོངས་སུ་བསྐོར་བར་གྱུར། དེ་ལྟ་བུའི་གཞལ་ཡས་ཁང་གི་དབུས་སུ་སྣ་ཚོགས་པདྨ་འདབ་མ་བརྒྱད་པའི་ལྟེ་བར་ཉི་མའི་དཀྱིལ་འཁོར་གྱི་སྟེང་དུ་བདུད་བཞིའི་ངོ་བོ་ཚངས་པ་སེར་པོ། དབང་པོ་དཀར་པོ། ཉེ་དབང་སྔོན་པོ། དྲག་པོ་ནག་པོ་རྣམས་སྟེང་ནས་སྟེང་དུ་བརྩེགས་པ་འདབ་མ་བརྒྱད་པའི་ཟླ་བའི་དཀྱིལ་འཁོར་རེ་རེའི་དབུས་སུ། ཤར་དུ་ཚངས་པ་སེར་པོ། ལྷོར་དབང་པོ་དཀར་པོ། ནུབ་ཏུ་ཉེ་དབང་སྔོན་པོ། བྱང་དུ་དྲག་པོ་ནག་པོ། བྱང་ཤར་དུ་གཤིན་རྗེ་སྔོན་པོ། ཤར་ལྷོར་ནོར་བདག་སེར་པོ། ལྷོ་ནུབ་ཏུ་སྲིན་པོ་ནག་པོ། ནུབ་བྱང་དུ་ཐགས་བཟང་རིས་དུད་ཁ་རྣམས་གན་རྐྱལ་དུ་ཕྱོགས་པའི་གདན་དང་བཅས་པར་གྱུར། ཨོཾ་སརྦ་ཏ་ཐཱ་ག་ཏ་བཛྲ་སྭ་བྷཱ་ཝ་ཨཱཏྨ་ཀོ྅ཧཾ། དབུས་ཀྱི་གདན་གྱི་སྟེང་དུ་མཚན་བཟང་པོ་སུམ་ཅུའི་ངོ་བོ། ཨ་ཨཱ། ཨི་ཨཱི། ནས། ཨཾ་ཨཱཿ སོགས། ཕྲག་གཉིས་ཞུ་བ་ལས་བྱུང་བའི་རི་བོང་འཛིན་པའི་དཀྱིལ་འཁོར་དཀར་པོ་མེ་ལོང་ལྟ་བུའི་ཡེ་ཤེས།

དེའི་སྟེང་དུ་དཔེ་བྱད་བརྒྱད་ཅུའི་ངོ་བོ། ཀ་ཁ་ག་གྷ་ང༌། ནས། ཤ་ཥ་ས་ཧ་ཀྵཿ ཡ་ར་ལ་ཝ་ཌ་ཌྷ། སོགས། ཕྲག་གཉིས་ཞུ་བ་ལས་བྱུང་བའི་ཉི་མའི་དཀྱིལ་འཁོར་དམར་པོ་མཉམ་པ་ཉིད་ཀྱི་ཡེ་ཤེས་དང༌། དེ་གཉིས་ཀྱི་བར་དུ་ཧཱུྂ་ལས་རྡོ་རྗེ་སྔོན་པོ་རྩེ་དགུ་པའི་ལྟེ་བར་ཧཱུྂ་གིས་མཚན་པ་དང༌། དེའི་གཡོན་དུ་ཨཾ་ཨཿལས་གྲི་གུག་སྔོ་ནག་ཡུ་བ་ལ་ཨཾ་གིས་མཚན་པ་འོད་ཟེར་འཕྲོ་བ་སོ་སོར་རྟོག་པའི་ཡེ་ཤེས། ཉི་ཟླ་ཕྱག་མཚན་ས་བོན་དང་བཅས་པ་ཡོངས་སུ་གྱུར་པ་ལས་བྱ་བ་གྲུབ་པའི་ཡེ་ཤེས། དེ་ལས་དཔལ་ཧེ་རུ་ཀ་དགྱེས་པ་རྡོ་རྗེ་ལྷ་དགུའི་རྟེན་དང་བརྟེན་པར་བཅས་པ་དུས་གཅིག་གྲུབ་པ་ནི་ཆོས་ཀྱི་དབྱིངས་ཀྱི་ཡེ་ཤེས་སོ། །དབུས་ཀྱི་གདན་ལ་ཧཱུྂ་དང་རྡོ་རྗེ་ལས་འོད་འཕྲོས། སེམས་ཅན་ཐམས་ཅད་ཀྱི་རྣམ་པར་ཤེས་པའི་ཕུང་པོ་སྦྱངས༑ ཆོས་ཀྱི་དབྱིངས་ཀྱི་ཡེ་ཤེས་ལ་བཀོད། ཚུར་འདུས་ཡོངས་སུ་གྱུར་པ་ལས་སྐད་ཅིག་གིས་བཅོམ་ལྡན་འདས་དཔལ་ཧེ་རུ་ཀ་དགྱེས་པ་རྡོ་རྗེ་སྐུ་སྔོ་སངས་ཞལ་བརྒྱད་ཕྱག་བཅུ་དྲུག་པ། ཞབས་བཞི་པ། རྩ་བའི་ཞལ་སྔོན་པོ། གཡས་དཀར། གཡོན་དམར། སྟེང་ཞལ་དུད་ཁ། ཞལ་ལྷག་མ་གཉིས་གཉིས་ནག་པོ། སྤྱན་གསུམ་གསུམ་དང་ལྡན་པ། མཆེ་བ་བཞི་བཞི་གཙིགས་པ། དབུ་སྐྲ་སེར་པོ་གྱེན་དུ་བཅིངས་པ་ལ་སྣ་ཚོགས་རྡོ་རྗེས་མཚན་པ། ཕྱག་བཅུ་དྲུག་གིས་ཐོད་པ་བཅུ་དྲུག་བཟུང་བའི་གཡས་ཀྱི་དང་པོ་ན་གླང་པོ་དཀར་པོ། གཡོན་གྱི་དང་པོ་ན་སའི་ལྷ་སེར་པོ་བཟུང་བའི་ཕྱག་གཉིས་ཀྱིས་འཁྱུད་པ། གཡས་ཀྱི་གཉིས་པ་ན་རྟ་སྔོན་པོ། གསུམ་པ་ན་བོང་བུ་གྭ་པ། བཞི་པ་ན་གླང་སེར་པོ། ལྔ་པ་ན་རྔ་མོང་ཐལ་ཀ །དྲུག་པ་ན་མི་དམར་པོ། བདུན་པ་ན་ཤ་ར་བྷ་སྔོན་པོ། བརྒྱད་པ་ན་བྱི་ལ་གྭ་པ། གཡོན་གྱི་གཉིས་པ་ན་ཆུ་ལྷ་དཀར་པོ། གསུམ་པ་ན་མེའི་ལྷ་དམར་པོ། བཞི་པ་ན་རླུང་གི་ལྷ་ལྗང་གུ། ལྔ་པ་ན་ཟླ་བའི་ལྷ་དཀར་པོ། དྲུག་པ་ན་ཉི་མའི་ལྷ་དམར་པོ། བདུན་པ་ན་གཤིན་རྗེ་སྔོན་པོ།

བརྒྱད་པ་ན་ནོར་འཛིན་སེར་པོ་རྣམས། མི་མགོ་སྐམ་པོ་ལྔའི་དབུ་རྒྱན་ལ་རིན་པོ་ཆེའི་རྩེ་བྲན་གྱིས་སྤྲས་པ། རློན་པ་ལྔ་བཅུས་དོ་ཤལ་འཕྱང་བ། རུས་པའི་རྒྱན་དྲུག་དང་ལྡན་པ། ཞབས་གཡས་གཉིས་བརྐྱང་བའི་བརླ་ལ་གཡོན་བསྐུམ་པའི་སོར་མོ་རྣམས་སྐྱིལ་ཀྲུང་ཕྱེད་པའི་གར་སྟབས་ཀྱིས་བཞེངས་པ། སྒེག་ཅིང་དཔའ་བ་འཇིགས་སུ་རུང༌། །རྒོད་ཅིང་དྲག་ཤུལ་རྔམ་པ་དང༌། །སྙིང་རྗེ་ཞི་བའི་ངོ་མཚར་ཆེ། །གར་གྱི་ཉམས་དགུ་ལྡན་པ་ཉིད། །དེའི་པང་ན་ཡུམ་རྡོ་རྗེ་བདག་མེད་མ་སྐུ་མདོག་སྔོན་མོ་སྤྱན་གསུམ་མ། སྐྲ་སེར་པོ་ཕྱེད་བཅིངས་ཀྱིས་རྒྱབ་ཏུ་གྲོལ་བ། ཕྱག་གཉིས་ཀྱི་གཡས་གྲི་གུག་དང༌། གཡོན་ཐོད་པ་འཛིན་པས་ཡབ་ལ་འཁྱུད་པ། ཐོད་པ་སྐམ་པོ་ལྔས་དབུ་བརྒྱན་པ་དང༌། སྐམ་པོ་ལྔ་བཅུའི་དོ་ཤལ་འཕྱང་བ། རུས་པ་དང་རིན་པོ་ཆེའི་རྒྱན་གྱིས་བརྒྱན་པ། ཞབས་གཡོན་ཡབ་ཀྱི་གཡས་དང་ལྷན་གཅིག་ཏུ་བརྐྱང་ཞིང་གཡས་ཡབ་ཀྱི་སྐེད་ལ་འཁྲིལ་བ། ཆགས་པ་ཆེན་པོའི་ཉམས་དང་ལྡན་པ་གཉིས་ཀའང་ཡེ་ཤེས་ཀྱི་མེ་རབ་ཏུ་འབར་བའི་དབུས་ན་བཞུགས་པ། ཡུམ་གྱི་མཁའ་གསང་མི་དམིགས་ཏེ་སྟོང་པའི་ངང་ལས་པདྨ་དམར་པོ་འདབ་མ་བརྒྱད་པ། ལྟེ་བར་ཨཱཿཡིག་གིས་མཚན་པ། ཡབ་ཀྱི་གསང་བའི་ངོ་བོ་མི་དམིགས་ཏེ་སྟོང་པའི་ངང་ལས་ཧཱུྂ་ལས་རྡོ་རྗེ་རྩེ་དགུ་པའི་ལྟེ་བར་ཧཱུྂ་གིས་མཚན་པར་གྱུར། རྡོ་རྗེ་ཞེ་སྡང་ཆེན་པོ་ནི། །དགའ་བ་བཞི་ནི་སྟེར་བར་བྱེད། །མཁའ་འགྲོའི་གདོང་པ་རོ་གཅིག་མགོན། །བྱ་བར་མཛོད་ཅིག་བདག་ལའོ། །པདྨ་བདེ་བར་འཛིན་པ་ནི། །འདོད་ཆགས་ཆེན་པོས་བདེ་བ་སྟེར། །དགའ་བཞིའི་སྐལ་པ་སྣ་ཚོགས་པ། །བྱ་བར་མཛོད་ཅིག་བདག་ལའོ། །ཞེས་པའི་སྒྲས་བསྐུལ་ཏེ། ཡབ་ཡུམ་སྙོམས་པར་ཞུགས་པའི་བདེ་བ་ཆེན་པོའི་སྒྲ་གསང་བའི་གནས་སུ་བསྒྲགས་པས་ཕྱོགས་བཅུའི་དེ་བཞིན་གཤེགས་པ་ཐམས་ཅད་སྤྱན་དྲངས། ཡབ་ཀྱི་ཞལ་དུ་ཞུགས་ཐུགས་ཀར་རྗེས་ཆགས་ཀྱི་མེས་འོད་དུ་ཞུ། སྐུའི་དབྱིབས་བརྒྱུད་རྡོ་རྗེའི་ལམ་ནས་ཡུམ་གྱི་མཁར་བབ་པས་ཐིག་ལེ་བརྒྱད་དུ་གྱུར་ཅིང༌།

དེ་རྣམས་གཾ་ཙཾ་བཾ་གྷཾ་པཾ་ཤཾ་ལཾ་ཌཾ་ཞེས་པའི་ས་བོན་རྣམས་ལྷ་མོ་བརྒྱད་དུ་གྱུར་ཏེ། མཁའ་ནང་གཽ་རཱི་དཔག་ཏུ་མེད་པ་འཕྲོས༑ སེམས་ཅན་ཐམས་ཅད་ཀྱི་གཏི་མུག་དང་གཟུགས་ཀྱི་ཕུང་པོ་སྦྱངས། མེ་ལོང་ལྟ་བུའི་ཡེ་ཤེས་ལ་བཀོད། ཚུར་འདུས་ཤར་དུ་ཚངས་པའི་གདན་ལ་འཁོད་པ་ནི། གཽ་རཱི་ནག་མོ་གཡས་གྲི་གུག་དང༌། གཡོན་ཉ་རོ་ཧི་ཏ་འཛིན་པ། མཁའ་ནང་གི་ཙཽ་རཱི་དཔག་ཏུ་མེད་པ་འཕྲོས། སེམས་ཅན་ཐམས་ཅད་ཀྱི་ང་རྒྱལ་དང་ཚོར་བའི་ཕུང་པོ་སྦྱངས། མཉམ་པ་ཉིད་ཀྱི་ཡེ་ཤེས་ལ་བཀོད། ཚུར་འདུས་ལྷོར་དབང་པོའི་གདན་ལ་འཁོད་པ་ནི། ཙཽ་རཱི་མ་དམར་པོ་གཡས་ཅང་ཏེའུ་དང༌། གཡོན་ཕག་པ་འཛིན་པ། མཁའ་ནང་གི་བེ་ཏཱ་ལཱི་དཔག་ཏུ་མེད་པ་འཕྲོས། སེམས་ཅན་གྱི་འདོད་ཆགས་དང་འདུ་ཤེས་ཀྱི་ཕུང་པོ་སྦྱངས། སོ་སོར་རྟོག་པའི་ཡེ་ཤེས་ལ་བཀོད། ཚུར་འདུས་ནུབ་ཏུ་ཉེ་དབང་གི་གདན་ལ་འཁོད་པ་ནི། བེ་ཏཱ་ལཱི་སེར་མོ་གཡས་རུས་སྦལ་དང༌། གཡོན་ཐོད་པ་འཛིན་པ། མཁའ་ནང་གི་གྷསྨ་རཱི་དཔག་ཏུ་མེད་པ་འཕྲོས། སེམས་ཅན་ཐམས་ཅད་ཀྱི་ཕྲག་དོག་དང་འདུ་བྱེད་ཀྱི་ཕུང་པོ་སྦྱངས། བྱ་བ་གྲུབ་པའི་ཡེ་ཤེས་ལ་བཀོད། ཚུར་འདུས་བྱང་དུ་དྲག་པོའི་གདན་ལ་འཁོད་པ་ནི། གྷསྨ་རཱི་ལྗང་གུ་གཡས་སྦྲུལ་དང༌། གཡོན་ཐོད་པ་འཛིན་པ། མཁའ་ནང་གི་པུཀྐ་སིའ་དཔག་ཏུ་མེད་པ་འཕྲོས༑ སེམས་ཅན་ཐམས་ཅད་ཀྱི་སའི་ཁམས་སྦྱངས། བྱམས་པ་ཚད་མེད་པ་ལ་བཀོད། ཚུར་འདུས་བྱང་ཤར་དུ་གཤིན་རྗེའི་གདན་ལ་འཁོད་པ་ནི། པུཀྐ་སཱི་སྔོན་མོ་གཡས་སེང་གེ་དང༌། གཡོན་རྡོ་རྗེའི་དགྲ་སྟ་འཛིན་པ། མཁའ་ནང་གི་རི་ཁྲོད་མ་དཔག་ཏུ་མེད་པ་འཕྲོས། སེམས་ཅན་ཐམས་ཅད་ཀྱི་ཆུའི་ཁམས་སྦྱངས། སྙིང་རྗེ་ཚད་མེད་པ་ལ་བཀོད། ཚུར་འདུས་ཤར་ལྷོར་ནོར་བདག་གི་གདན་ལ་འཁོད་པ་ནི། རི་ཁྲོད་མ་དཀར་མོ་གཡས་དགེ་སློང་དང༌། གཡོན་གསིལ་བྱེད་འཛིན་པ། མཁའ་ནང་གི་གདོལ་པ་མོ་དཔག་ཏུ་མེད་པ་འཕྲོས། སེམས་ཅན་ཐམས་ཅད་ཀྱི་མེའི་ཁམས་སྦྱངས། དགའ་བ་ཚད་མེད་པ་ལ་བཀོད།

ཚུར་འདུས་ལྷོ་ནུབ་ཏུ་སྲིན་པོའི་གདན་ལ་འཁོད་པ་ནི། གདོལ་པ་མོ་སྔོན་མོ་གཡས་འཁོར་ལོ་དང༌། གཡོན་ཐོང་གཤོལ་འཛིན་པ། མཁའ་ནང་གི་གཡུང་མོ་དཔག་ཏུ་མེད་པ་འཕྲོས། སེམས་ཅན་ཐམས་ཅད་ཀྱི་རླུང་གི་ཁམས་སྦྱངས། བཏང་སྙོམས་ཚད་མེད་པ་ལ་བཀོད། ཚུར་འདུས་ནུབ་བྱང་དུ་ཐགས་བཟང་རིས་ཀྱི་གདན་ལ་འཁོད་པ་ནི། གཡུང་མོ་ཁ་དོག་སྣ་ཚོགས་པ་གཡས་རྡོ་རྗེ་གཡོན་སྡིགས་མཛུབ་མཛད་པ། ཐམས་ཅད་ཀྱང་ཞལ་གཅིག་ཕྱག་གཉིས་པ་སྐྲ་སེར་པོ་གྱེན་དུ་བརྫེས་པ་གཅེར་བུ་རུས་པ་དང་རིན་པོ་ཆེས་བརྒྱན་པ། མི་མགོ་སྐམ་པོ་ལྔའི་དབུ་རྒྱན་དང༌། སྐམ་པོ་ལྔ་བཅུའི་དོ་ཤལ་བྱས་པ། ཞབས་གཡོན་བརྐྱང་བའི་སྟབས་ཀྱིས་ཡེ་ཤེས་ཀྱི་མེ་དཔུང་འབར་བའི་ཁྲོད་ན་བཞུགས་པ། ལྷ་ཐམས་ཅད་ཀྱི་ཐུགས་ཀར་པདྨ་དང་ཉི་མའི་སྟེང་དུ་རང་རང་གི་ས་བོན་གྱིས་མཚན་པ། མིག་ཏུ་བཾ་ལས་གཏི་མུག་རྡོ་རྗེ་མ་དཀར་མོ། །རྣ་བར་དཾ་ལས་ཞེ་སྡང་རྡོ་རྗེ་མ་སྔོན་པོ། སྣར་པཾ་ལས་སེར་སྣ་རྡོ་རྗེ་མ་སེར་མོ། ཁར་རཾ་ལས་འདོད་ཆགས་རྡོ་རྗེ་མ་དམར་མོ། དཔྲལ་བར་ཨཾ་ལས་ཕྲག་དོག་རྡོ་རྗེ་མ་ལྗང་གུ། སྙིང་ཁར་ནཾ་ལས་བདག་མེད་རྣལ་འབྱོར་མ་ནག་མོ། ། སྤྱི་བོར་ཨོཾ་ལས་སྐུ་རྡོ་རྗེ་མ་སྐུ་མདོག་དཀར་མོ། མགྲིན་པར་ཨཱཿལས་གསུང་རྡོ་རྗེ་མ་དམར་མོ། ཐུགས་ཀར་ཧཱུྂ་ལས་ཐུགས་རྡོ་རྗེ་མ་ནག་མོ། ཐམས་ཅད་ཀྱང་ཞལ་གཅིག་ཕྱག་གཉིས་གྲི་གུག་དང་ཐོད་པ་འཛིན་པ། རིན་པོ་ཆེ་དང་རུས་པས་བརྒྱན་ཅིང་མི་མགོ་སྐམ་པོའི་ཕྲེང་བས་བརྒྱན་པ། རོའི་གདན་ལ་སྐྱིལ་ཀྲུང་ཕྱེད་པའི་སྟབས་ཀྱིས་བཞུགས་པ། ལྷ་ཚོགས་ཐམས་ཅད་ཀྱི་ཐུགས་ཀ་ནས་འོད་ཟེར་འཕྲོས་པས་ཕྱོགས་བཅུའི་འཇིག་རྟེན་གྱི་ཁམས་ན་བཞུགས་པའི་དཔལ་ཧེ་རུ་ཀ་དགྱེས་པ་རྡོ་རྗེ་ལྷ་དགུའི་དཀྱིལ་འཁོར་དང་དབང་གི་ལྷ་རྣམས་མདུན་གྱི་ནམ་མཁར་སྤྱན་དྲངས། ཨོཾ་བཛྲ་ཨརྒྷཾ་པྲ་ཏཱིཙྪ་སྭཱ་ཧཱ། 

དེ་བཞིན་དུ། ཤཔད འི་བར་གྱིས་མཆོད་ལ། ཡེ་ཤེས་པ་དང་དམ་ཚིག་པ་དབྱེར་མེད་དུ་གྱུར། ཛཿཧཱུྂ་བཾ་ཧོཿ སླར་ཡང་དབང་གི་ལྷ་རྣམས་མདུན་གྱི་ནམ་མཁར་བཞུགས་པ་ལ། བཅོམ་ལྡན་དེ་བཞིན་གཤེགས་པ་ཀུན། །བཅོམ་ལྡན་བདག་ཉིད་ཆེན་པོ་རྣམས། །ཐུགས་རྗེའི་རང་བཞིན་རྣམས་ཀྱི་སྐུ། །ཁྱོད་ཀྱིས་བདག་ལ་དབང་བསྐུར་མཛོད། །ཞེས་གསོལ་བ་བཏབ་པས། བུམ་པའི་ཆུ་རྒྱུན་གྱིས་སྤྱི་བོ་ནས་དབང་བསྐུར། སྐུ་གང་དྲི་མ་དག །ཆུ་ལྷག་མ་ཡར་ལུད་པ་ལས། གཙོ་བོ་ཡབ་ཡུམ་གཉིས་གཽ་རཱི་དང༌༑ པུཀྐ་སཱི་ལ་མི་བསྐྱོད་པ། ཙཽ་རཱི་དང་ཤ་ཝ་རཱི་ལ་རྣམ་པར་སྣང་མཛད། བེ་ཏཱ་ལཱི་དང་ཙཎྜ་ལཱི་ལ་རིན་ཆེན་འབྱུང་ལྡན། གྷསྨ་རཱི་དང་ཌོཾ་བྷི་ནཱི་ལ་འོད་དཔག་མེད་ཀྱིས་དབུར་བརྒྱན། དབང་གི་ལྷ་རྣམས་ཀྱང་དེ་རྣམས་ལ་ཐིམ་པར་གྱུར། ཅེས་བརྗོད། ཨརྒྷཾ་ སོགས་ཆུ་བཞི་རོལ་མོ་དང་བཅས་པས་མཆོད། ཨོཾ་བཛྲ་ཝཱི་ཎི་ཧུཾ་ཧཱུྂ་ཕཊ། ཝཾ་ཤེ། མྲྀ་ཏཾ་གེ། མུ་རཉྫེ། ཧཱ་སྱ། ལཱ་སྱ། གཱི་ཏི། ནྲྀ་ཏྱཻ། པུཥྤེ། དྷཱུ་པེ། དཱི་པེ། གནྡྷེ། རྣམས་དང༌། ཨོཾ་རཱུ་པ་བཛྲ་ཧུཾ་ཧཱུྂ་ཕཊ། རཱུ་པའི་གནས་སུ་ ར་ས། སྤརྴེ། དྷརྨ་དྷཱ་ཏུ། ཞེས་སོགས་རིག་མ་བཅུ་དྲུག་གིས་ཀྱང་མཆོད། ནང་མཆོད་བདུད་རྩི་མཆོད་པ་ནི། ཨོཾ་ཨཱཿཧཱུྂ། ཁྱབ་བདག་རྡོ་རྗེ་འཆང་དབང་བདག་མེད་མ། །ནག་པོ་པ་དང་ཌཱ་མ་རུ་པ་དང༌། །ཨ་ཝ་དྷཱུ་ཏི་ག་ཡ་དྷ་ར་དང༌། །འབྲོག་མི་ལོ་ཙཱ་སེ་སྟོན་ཀུན་རིག་དང༌། །ཞང་སྟོན་ཆོས་འབར་ཀུན་དགའ་སྙིང་པོའི་མཚན། །བསོད་ནམས་རྩེ་མོ་གྲགས་པ་རྒྱལ་མཚན་དང༌། །ཀུན་དགའ་རྒྱལ་མཚན་ཆོས་རྒྱལ་འཕགས་པ་དང༌། །དཀོན་མཆོག་དཔལ་དང་ཆོས་རྗེ་བསོད་ནམས་དཔལ༑ ༑བསོད་ནམས་རྒྱལ་མཚན་དཔལ་ལྡན་ཚུལ་ཁྲིམས་དང༌། །བུདྡྷ་ཤྲཱི་ཀུན་དགའ་བཟང་པོ་དང༌། །ཤཱཀྱ་མཆོག་ལྡན་གཞོན་ནུ་མཆོག་གྲུབ་དང༌། །ཀུན་དགའ་སྒྲོལ་མཆོག་ཀུན་དགའ་རྒྱལ་མཚན་དང༌། །བྱམས་པ་ཆོས་ཀྱི་རྒྱ་མཚོ་ལ་སོགས་པ། །དགྱེས་པ་རྡོ་རྗེའི་སྤྲུལ་པའི་ལྷ་དགུའི་ཚོགས། །ཆོས་སྐྱོང་སྲུང་མའི་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྂ། །

ཞེས་མཆོད་ལ༑ བསྟོད་པ་ནི། ཞེ་སྡང་རྡོ་རྗེ་ཕྱག་འཚལ་ལོ། །གཏི་མུག་རྡོ་རྗེ་ཕྱག་འཚལ་བསྟོད། །སེར་སྣ་རྡོ་རྗེ་བདག་ལ་སྩོལ། །འདོད་ཆགས་རྡོ་རྗེ་བདག་ལ་སྐྱོབས། །ཡིད་དང་ཕྲག་དོག་ཆེན་པོ་ཀུན༑ ༑རྡོ་རྗེ་ཅན་གྱི་དམ་ཚིག་མཆོག །ལས་རྣམས་ཐམས་ཅད་སྒྲུབ་བྱེད་པ། །རྡོ་རྗེ་ཅན་ཀུན་བདག་དགྱེས་མཛོད། ། ཅེས་བསྟོད་ལ། རང་གི་ཐུགས་ཀ་ནས་སྔགས་ཀྱི་ཕྲེང་བ་བྱུང་ཞལ་ནས་ཐོན། ཡུམ་གྱི་ཞལ་དུ་ཞུགས། སྦྱོར་མཚམས་ནས་ཐོན་ལྷ་མོ་བརྒྱད་ཀྱི་ཞལ་དུ་ཞུགས། པདྨ་ནས་ཐོན་ཁམས་གསུམ་གྱི་སེམས་ཅན་ཐམས་ཅད་ཀྱཻའི་རྡོ་རྗེའི་གོ་འཕང་ལ་བཀོད། ཚུར་འདུས་རང་གི་ཐུགས་ཀའི་སྔགས་ཀྱི་ཕྲེང་བ་ས་བོན་དང་བཅས་པ་ལ་ཐིམ་པར་གྱུར། ཅེས་དམིགས་ལ། ཨོཾ་དེ་ཝ་པི་ཙུ་བཛྲ་ཧུཾ་ཧཱུྂ་ཕཊ། ཅེས་ཡབ་ཀྱི་སྙིང་པོ་དང༌། ཨོཾ་ལཾ་ཧཱུྂ་ཕཊ། ཅེས་ཡུམ་གྱི་སྙིང་པོ་དང༌། ཨོཾ་བཛྲ་གཾ་ཙཾ་བཾ་གྷཾ་པཾ་ཤཾ་ལཾ་ཌཾ་ཧཱུྂ་ཕཊ། འཁོར་གྱི་སྙིང་པོའོ། །དེ་ལྟར་གཙོ་འཁོར་གྱི་བཟླས་པ་ཅི་རིགས་པར་བྱས་ལ། ཐུན་བསྡུ་བ་ནི། རང་གི་ཐུགས་ཀའི་ཧཱུྂ་ལས་འོད་ཟེར་འཕྲོས། སྣོད་བཅུད་ཐམས་ཅད་འོད་དུ་ཞུ་བསྲུང་བའི་འཁོར་ལོ་ལ་ཐིམ། དེ་དུར་ཁྲོད་བརྒྱད་ལ་ཐིམ། གཞལ་ཡས་ཁང་ལྷ་མོ་བརྒྱད་ལ་ཐིམ། ལྷ་མོ་བརྒྱད་ཡུམ་ལ་ཐིམ། ཡུམ་ཡབ་ལ་ཐིམ། ཡབ་ཐུགས་ཀའི་ཧཱུྂ་ལ་ཐིམ། ཧཱུྂ་ཡང་མི་དམིགས་པ་སྟོང་པ་ཉིད་དུ་གྱུར། ཨོཾ་ཡོ་ག་ཤུདྡྷཿསརྦ་དྷརྨཱཿཡོ་ག་ཤུདྡྷོ྅ཧཾ། སླར་ཡང་རང་ཉིད་དགྱེས་པ་རྡོ་རྗེ་ཞལ་གཅིག་ཕྱག་གཉིས་པའི་རྣམ་པར་གྱུར། རིགས་སུ་སྐྱེས་དང་དམ་ཚིག་ཅན༑ ༑གཉིས་མེད་དགྱེས་པ་རྡོ་རྗེ་སྟོན། །བླ་མ་ལ་གུས་སྙིང་རྗེར་ལྡན། །སྐྱེ་ཞིང་སྐྱེ་བར་འགྱུར་བར་ཤོག །ལག་པས་རྡོ་རྗེ་དྲིལ་བུ་འཁྲོལ། །ཟབ་མོའི་ཆོས་ནི་ཀློག་པ་དང༌། །བཙུན་མོའི་ཁུ་བ་མཉམ་ཟས་ཅན། །གང་གི་ལམ་འདིར་གྲོལ་འགྱུར་བ། །སྐྱེ་ཞིང་སྐྱེ་བར་འགྱུར་བར་ཤོག ། ཅེས་སྨོན་ལམ་གདབ་བོ། །དེ་དག་ནི་བདག་གི་རྣལ་འབྱོར་རྫོགས་པར་བསྒོམ་པའི་རིམ་པའོ། །དེ་ནས་བུམ་པ་གཉིས་ཁ་རྒྱན་མགུལ་ཆིངས་དང་བཅས།

བུམ་རྫས་དང་ལྡན་པ་ཆུས་སུམ་གཉིས་བཀང་བ། གཙོ་བོ་རྣམ་རྒྱལ་བུམ་པ་ལ་བུམ་མཚན་ཐོད་པ་རྡོ་རྗེའི་མཚན་མ་བྲིས་པའི་སྟེང་དུ། དུང་ཆོས་དྲི་ཆུས་བཀང་བ་དང་གཟུང་ཐག་བཏགས་པའི་རྡོ་རྗེའང་བཞག །ལས་བུམ་ལ་སྣ་ཚོགས་རྡོ་རྗེ་བྲིས་པའི་བུམ་མཚན་དང་ལྡན་པ། གཞན་ཡང་མཆོད་པ་སོགས་བརྒྱུད་པའི་ཕྱག་ལེན་ལྟར་བཤམས་ལ། བུམ་པ་གཉིས་བསངས་སྦྱངས། སྟོང་པའི་ངང་ལས་པདྨ་དང་ཟླ་བའི་དཀྱིལ་འཁོར་གྱི་སྟེང་དུ་བྷྲཱུཾ་ལས་རིན་པོ་ཆེའི་བུམ་པ་མཚན་ཉིད་ཡངས་ཤིང་རྒྱ་ཆེ་བ། ཞབས་ཞུམ་པ། ལྟོ་བ་ལྡིར་བ། མགྲིན་པ་ཕྲ་བ། ཁ་གཡེལ་བ། ཆུ་འཕྱངས་པ། ལྷའི་གོས་ཀྱིས་མགུལ་བཅིངས་ཤིང༌། དཔག་བསམ་ཤིང་གིས་ཁ་བརྒྱན་པ། བཅུད་བཟང་པོ་རྣམས་ཀྱིས་ཡོངས་སུ་གང་བའི་རྣམ་རྒྱལ་བུམ་པའི་ནང་དུ། སྣ་ཚོགས་པདྨ་འདབ་མ་བརྒྱད་པའི་ལྟེ་བར་ཉི་མའི་དཀྱིལ་འཁོར་གྱི་སྟེང་དུ། བདུད་བཞིའི་ངོ་བོ་ ཞེས་པ་ནས། དབང་གི་ལྷ་རྣམས་ཀྱང་དེ་རྣམས་ལ་ཐིམ་པར་གྱུར། ཅེས་པའི་བར་གོང་ལྟར་རོ། ། ལས་བུམ་གྱི་དབུས་པད་ཉིའི་གདན་ལ། སླར་ཡང་བདུད་རྩི་འཁྱིལ་བ་སྐུ་མདོག་ལྗང་གུ་ཞལ་གཅིག་ཕྱག་གཉིས་པ་གཡས་སྣ་ཚོགས་རྡོ་རྗེ་དང༌། གཡོན་དེས་ཡུ་བ་བྱས་པའི་དྲིལ་བུ་དཀུར་བརྟེན་པ། རིན་པོ་ཆེ་དང་སྦྲུལ་གྱིས་བརྒྱན་ཅིང༌། སྟག་གི་པགས་པའི་ཤམ་ཐབས་ཅན༑ ཞབས་དོར་སྟབས་སུ་ཡེ་ཤེས་ཀྱི་མེ་འབར་བའི་དབུས་ན་བཞུགས་པའི་ཐུགས་ཀའི་ཧཱུྂ་གིས་ཡེ་ཤེས་སེམས་དཔའ་སྤྱན་དྲངས། དབང་སྐུར་ཞིང་དོན་ཡོད་གྲུབ་པས་དབུ་བརྒྱན་པར་གྱུར། ཡེ་ཤེས་པ་དང་དམ་ཚིག་པ་གཉིས་སུ་མེད་པར་གྱུར། ཅེས་བརྗོད་ལ། སྔར་ལྟར་མཆོད་པ་བྱིན་བརླབས་ལ། ཆུ་བཞི་རོལ་མོ་དང་བཅས་པས་མཆོད་ལ། དྲི་མེད་ཡིད་དུ་འོང་བ་ཡི། །ལྷ་རྫས་དམ་པའི་མཆོད་ཡོན་འདི༑ ༑བཅོམ་ལྡན་འཁོར་དང་བཅས་ལ་འབུལ། །བཞེས་ནས་བདག་ལ་དགྱེས་པར་མཛོད། །ཨོཾ་སརྦ་ཏ་ཐཱ་ག་ཏ་ཨརྒྷཾ་པྲ་ཏཱིཙྪ་ཡེ་སྭཱ་ཧཱ། ཞེས་དང༌། མཆོད་ཡོན་གྱི་གནས་སུ ། ཞབས་བསིལ།

འཐོར་འཐུང༌། ཞལ་བསིལ། མེ་ཏོག །བདུག་སྤོས། སྣང་གསལ། དྲི་ཆབ། ཞལ་ཟས། སྒྲ་སྙན་ ཞེས་བསྒྱུར་ཞིང༌། ཨརྒྷཾ ་ གྱི་མལ་དུ། པཱ་དྱཾ། ཨཱ་ཙ་མཱ་ནཾ། པྲོཀྵ་ཎཾ། པུཥྤེ། དྷཱུ་པེ། ཨཱ་ལོ་ཀེ། གནྡྷེ། ནཻ་ཝི་དྱེ། ཤཔད། ཞེས་མཆོད་ཅིང༌། རང་གི་ཐུགས་ཀའི་ཧཱུྂ་ལས་འཕྲོས་པ་ཡི། །པི་ཝཾ་གླིང་བུ་རྫ་རྔ་རྔ་ཆེན་མ། །སྔོན་མོ་སེར་མོ་དམར་མོ་ལྗང་གུ་རྣམས། །དང་པོའི་ཕྱག་གཉིས་རང་རྟགས་ཕྱག་མཚན་བསྣམས། །དེ་ལྷག་མཐའ་གཉིས་རྡོ་རྗེ་དྲིལ་བུ་འཛིན། །བཞད་མ་སྒེག་མ་གླུ་མ་གར་མ་རྣམས། །དམར་སྐྱ་སྔོན་མོ་སེར་མོ་ལྗང་གུ་སྟེ། །དང་པོའི་ཕྱག་གཉིས་རང་རང་རྟགས་མཚན་ལྡན། །གཡོན་མཐའ་ཐོད་པ་ཁ་ཊྭཱཾ་ཅང་ཏེའུ། །མེ་ཏོག་བདུག་པ་སྣང་གསལ་དྲི་ཆབ་དང༌། །དཀར་མོ་མཐིང་དམར་དམར་སེར་དམར་མོ་རྣམས། །གཡོན་གཉིས་ཐོད་པ་ཁ་ཊྭཱཾ་གཡས་དང་པོ། །རང་མིན་རྗེས་མཐུན་གཉིས་པ་ཅང་ཏེའུ། །གཟུགས་མ་རོ་མ་རེག་མ་ཆོས་མ་རྣམས། །དཀར་མོ་དམར་མོ་ལྗང་གུ་དཀར་མོ་སྟེ། །གཡོན་གཉིས་ཐོད་པ་ཁ་ཊྭཱཾ་གཡས་དང་པོ། །རང་མིན་རྗེས་མཐུན་གཉིས་པ་ཅང་ཏེའུ། །ཀུན་བཟང་ཞལ་གཅིག་ཕྱག་བཞི་མ་ལ་སྒེག །ལང་ཚོ་གཞོན་པས་རོལ་ཞིང་འགྱིང་བག་ཅན། །རིན་ཆེན་རྒྱན་ལྡན་རབ་ཀྱིས་རབ་ཏུ་མཛེས། །དཔག་མེད་རབ་བསྔགས་བརྗོད་ཅིང་མགོན་སོགས་མཆོད། །ཨོཾ་བཛྲ་ཝཱི་ཎི་ཧུཾ་ཧཱུྂ་ཕཊ། དེ་བཞིན་དུ་ཝཱི་ཎིའི་གནས་སུ། ཝཾ་ཤེ། མྲྀ་ཏཾ་གེ། མུ་རཉྫེ། ཧཱ་སྱ། ལཱ་སྱ། གཱི་ཏི། ནྲྀ་ཏྱཻ། པུཥྤེ། དྷཱུ་པེ། དཱི་པེ། གནྡྷེ། རྣམས་དང༌། ཨོཾ་རཱུ་པ་བཛྲ་ཧུཾ་ཧཱུྂ་ཕཊ། རཱུ་པའི་གནས་སུ་ ར་ས། སྤརྴེ། ཨོཾ་དྷརྨ་དྷཱ་ཏུ་བཛྲ་ཧུཾ་ཧཱུྂ་ཕཊ། ཅེས་མཆོད་ལ། བསྟོད་པ་ནི། ཧཱུྂ དྲག་བྱེད་ཆེན་པོ་ཞལ་བརྒྱད་མཆེ་གཙིགས་ཕྱག་བཅུ་དྲུག །རལ་པ་གྱེན་བརྫེས་མུན་པ་ལྟ་བུ་ཐོད་པས་དབུ་བརྒྱན་པ། །དབང་ཕྱུག་རུས་པ་སྦྲུལ་གྱིས་བརྒྱན་ཅིང་ཧཱུྂ་ཕཊ་འབྲུག་སྒྲ་སྒྲོག །དོར་སྟབས་བདུད་བཞིའི་གདན་ལ་ཞབས་བཞི་བརྐྱང་བསྐུམ་གར་གྱིས་བཞུགས། །འཇིགས་བྱེད་བཻ་ཌཱུར་མདོག་ཅན་ཁྱོད་ལ་རབ་ཏུ་ཕྱག་འཚལ་ལོ། །རང་འོད་ལས་སྤྲུལ་སྒྱུ་མའི་ལྷ་མོ་བདག་མེད་སྔོ་སངས་ཆགས་ཆེན་འདུལ་མཁས་མ། །དབུ་སྐྲ་ཕྱེད་བཅིངས་སིལ་བུ་གྲོལ་ཞིང་ཡབ་ལ་དང་བས་ལྷན་ཅིག་འཁྱུད་མཁས་མ། །གཡས་བསྐུམ་གཡོན་བརྐྱང་དོར་སྟབས་འགྱིང་ཚུལ་ཐོད་སྐམ་དབུ་བརྒྱན་དོ་ཤལ་འཆང་མཁས་མ། །དགའ་བཞིའི་བདག་མོ་ཁྱོད་ལ་གུས་བསྟོད་བདེ་བའི་རང་བཞིན་གསང་གསུམ་བདག་མེད་མ། །ཡེ་ཤེས་བཞི་ཡི་ངོ་བོ་ཉིད་འཛིན་མ། །གཽ་རཱི་ལ་སོགས་ཕྱོགས་ཀྱི་ལྷ་མོ་བཞི། །འབྱུང་བཞི་དག་ཅིང་ཚད་མེད་བཞི་མངའ་མ། །པུཀྐ་སཱི་སོགས་མཚམས་ཀྱི་ལྷ་མོ་སྟེ། །སྤྲུལ་པའི་ལྷ་མོ་བརྒྱད་ལ་ཕྱག་འཚལ་ལོ། །ཧཱུྂ

འཇིགས་པའི་ཧཱུྂ་སྒྲ་ངེས་སྒྲོག་ཅིང༌། །བགེགས་རྣམ་མ་ལུས་འཇོམས་མཛད་པ། །དངོས་གྲུབ་མ་ལུས་སྩོལ་བའི་ལྷ། །བགེགས་ཀྱི་དགྲ་ལ་ཕྱག་འཚལ་བསྟོད། ། སོགས་བསྟོད། རང་གི་ཐུགས་ཀ་ནས་སྔགས་ཕྲེང་བྱུང༌༑ ཞལ་ནས་ཐོན་གཟུང་ཐག་ལ་བརྒྱུད་དེ་བུམ་པའི་ལྷ་གཙོ་འཁོར་གྱི་ཐུགས་བརྒྱུད་བསྐུལ། མཁའ་གསང་ནས་བདུད་རྩི་བབ་པས་བུམ་པ་གང་ཞིང་རྒྱས་པར་གྱུར། པར་དམིགས་ལ། ཨོཾ་དེ་ཝ་པི་ཙུ་བཛྲ་ཧུཾ་ཧཱུྂ་ཕཊ་སྭཱ་ཧཱ། ཞེས་ཡབ་ཀྱི་སྙིང་པོ་དང༌། ཨོཾ་ཨཱཿནཾ་བཛྲ་ནཻ་རཱཏྨཱ་ཡ་ཧུཾ་ཧཱུྂ་ཕཊ་སྭཱ་ཧཱ། ཞེས་ཡུམ་གྱི་སྙིང་པོ་དང༌། ཨོཾ་བཛྲ་གཽ་རཱི་ཧུཾ་ཧཱུྂ་ཕཊ། ཨོཾ་བཛྲ་ཙཽ་རཱི་ཧུཾ་ཧཱུྂ་ཕཊ། ཨོཾ་བཛྲ་བེ་ཏཱ་ལཱི་ཧུཾ་ཧཱུྂ་ཕཊ། ཨོཾ་བཛྲ་གྷསྨ་རཱི་ཧུཾ་ཧཱུྂ་ཕཊ། ཨོཾ་བཛྲ་པུཀྐ་སཱི་ཧུཾ་ཧཱུྂ་ཕཊ། ཨོཾ་བཛྲ་ཤ་ཝ་རཱི་ཧུཾ་ཧཱུྂ་ཕཊ། ཨོཾ་བཛྲ་ཙཎྜ་ལིའ་ཧུཾ་ཧཱུྂ་ཕཊ། ཨོཾ་བཛྲ་ཌོཾ་བྷི་ནཱི་ཧུཾ་ཧཱུྂ་ཕཊ། ཅེས་འཁོར་གྱི་སྙིང་པོ་རྣམས་ཅི་རིགས་པར་བཟླས། སླར་ཡང་ལས་བུམ་ལ་དམིགས་ནས། ཨོཾ་བཛྲ་ཨ་མྲྀ་ཏ་ཀུཎྜ་ལི་ཧ་ན་ཧ་ན་ཧཱུྂ་ཕཊ། ཅེས་ཅི་རིགས་པར་བཟླས། སླར་ཡང་མཆོད་ཅིང་བསྟོད་ལ། རྣམ་རྒྱལ་བུམ་པའི་ལྷ་རྣམས་རང་བཞིན་བདེ་བ་ཆེན་པོའི་ཆགས་པའི་སྒྲས་བསྐུལ་ལ། འོད་དུ་ཞུ་བ་བུམ་པའི་ཆུ་དང་རོ་གཅིག་ཏུ་གྱུར་པ་ཡེ་ཤེས་ལྔའི་བདག་ཉིད་དུ་གྱུར། ཅེས་དང༌། ལས་བུམ་གྱི་ལྷས་ཇི་སྲིད་སྒྲུབ་པའི་འབྲས་བུ་མཆིས་པར་མ་གྱུར་གྱི་བར་དུ་བགེགས་རྣམས་བཟློག་པར་མཛོད་ཅིག ། ཅེས་ཞུའོ། །དེ་ནས་མདུན་བསྐྱེད་སྒྲུབ་ཅིང་དབང་བསྐུར་བའི་རིམ་པ་ནི། རྡོ་རྗེ་སློབ་དཔོན་དུ་རུང་བར་སོ་སོར་ཐར་པའི་སྡོམ་པ་ལ་ཚུལ་བཞིན་དུ་གནས་པ། བྱང་ཆུབ་སེམས་དཔའི་སེམས་བསྐྱེད་དང་ལྡན་པ། དཀྱིལ་འཁོར་འདི་ཉིད་ཀྱི་དབང་མནོས་ཤིང་དམ་ཚིག་དང་ལྡན་པས། ས་ཆོག་དང༌། ཐིག་གདབ་པ་དང༌། ཚོན་བཀྱེ་བ་སོགས་དབང་གི་ཆུ་བོ་མཛེས་རྒྱན་ལས་འབྱུང་བ་རྣམས་དང༌། གཞན་ཡང་གཞུང་ལས།

མིག་རས་མེ་ཏོག་གུ་གུལ་ཏིལ། །རིགས་དྲུག་མཆོད་པ་དབུ་རྒྱན་དང༌། །རྡོར་དྲིལ་མིག་ཐུར་མེ་ལོང་དང༌། །འབུད་དུང་འཁོར་ལོ་པོ་ཏི་དང༌། །མདའ་གཞུ་སྣམ་སྦྱར་བདུད་རྩི་སོགས། །ཉེར་མཁོ་ཐམས་ཅད་ཚོགས་བྱའོ། །ཞེས་འབྱུང་བ་ལྟར་དཀྱིལ་འཁོར་ལ་མཁོ་བའི་ཡོ་བྱད་འདུ་བྱ་བ་དང༌། གཞན་ཡང་མཆོད་པ་སོགས་མཐོང་བ་བརྒྱུད་པ་ལས་འབྱུང་བ་ལྟར་བཤམས་ལ། མདུན་གྱི་དཀྱིལ་འཁོར་བསངས་སྦྱངས། སྟོང་པའི་ངང་ལས་མདུན་དུ་གཙུག་ཏོར་འཁོར་ལོས་སྒྱུར་རྒྱལ་ཡོངས་སུ་གྱུར་པ་ལས། ཨེ་ལས་ཆོས་ཀྱི་འབྱུང་གནས་སྔོན་པོ་གྲུ་གསུམ་ ཞེས་པ་ནས། དབང་གི་ལྷ་རྣམས་ཀྱང་དེ་རྣམས་ལ་ཐིམ་པར་གྱུར། ཅེས་པའི་བར་གོང་ལྟར་བརྗོད་ལ། མཆོད་པ་སྔར་ལྟར་བྱིན་བརླབས་ལ། ཆུ་བཞི་ཉེར་སྤྱོད་རོལ་མོ་དང་བཅས་པས་མཆོད། སླར་ཡང་རང་གི་སྙིང་ཁ་ནས་མཆོད་པའི་ལྷ་མོ་དཔག་ཏུ་མེད་པ་སྤྲོས་ཏེ། དཀྱིལ་འཁོར་ལ་མཆོད་པར་ དམིགས་ལ། མཐའ་ཡས་ཞིང་ཁམས་རྒྱ་མཚོ་དཔག་མེད་ན། །དངོས་སུ་འབྱོར་དང་ཡིད་ཀྱིས་རྣམ་སྤྲུལ་ཏེ། །བདག་པོས་ཡོངས་སུ་བཟུང་དང་མ་བཟུང་བའི། །མཆོད་ཡོན་རྣམ་པ་ཇི་སྙེད་ཡོད་པ་ཀུན། །གཉིས་མེད་དགྱེས་པ་རྡོ་རྗེའི་ལྷ་ཚོགས་ལ། །ཚོགས་གཉིས་བསག་ཕྱིར་གུས་པས་འབུལ་ལགས་ན། །བཞེས་ནས་དངོས་གྲུབ་དམ་པའི་བཀའ་དྲིན་སྩོལ། །ཨོཾ་ཤྲཱི་ཧེ་བཛྲ་སརྦ་ཏ་ཐཱ་ག་ཏ་ཨརྒྷཾ་པྲ་ཏཱིཙྪ་པཱུ་ཛ་མེ་གྷ་ས་མུ་དྲ་སྥ་ར་ཎ་ས་མ་ཡ་ཤྲཱི་ཡེ་ཧཱུྂ། ཞེས་དང ་ མཆོད་ཡོན་གྱི་གནས་སུ། ཞབས་བསིལ། འཐོར་འཐུང༌། ཞལ་བསིལ། མེ་ཏོག །བདུག་སྤོས། སྣང་གསལ། དྲི་ཆབ། ཞལ་ཟས། སྒྲ་སྙན། ཞེས་དང༌། ཨརྒྷཾ་གྱི་གནས་སུ་ པཱ་དྱཾ། ཨཱ་ཙ་མཱ་ནཾ། པྲོཀྵ་ཎཾ། པུཥྤེ། དྷཱུ་པེ། ཨཱ་ལོ་ཀེ། གནྡྷེ། ནཻ་ཝི་དྱེ། ཤཔད། ཞེས་སྦྱར་ལ་མཆོད། ཁ་དོག་དབྱིབས་ལེགས་ཡིད་འོང་རྒྱན་ཕྲེང་ལྡན། །ཕྱོགས་དུས་ཀུན་ན་གཟུགས་མཆོག་ཅི་མཆིས་པ། །མོས་བློས་དཀྱིལ་འཁོར་ལྷ་ལ་བདག་འབུལ་ན། །བཞེས་ཤིང་བླ་མེད་དངོས་གྲུབ་བསྩལ་དུ་གསོལ། །ཨོཾ་ཤྲཱི་ཧེ་བཛྲ་སརྦ་ཏ་ཐཱ་ག་ཏ་རཱུ་པ་ཀཱ་མ་གུ་ཎ་བཛྲ་པཱུ་ཛི་ཏེ་ཨཱཿཧཱུྂ།

ཕྱོགས་དུས་ཀུན་ན་ཟིན་དང་མ་ཟིན་པའི། །ཡོན་ཏན་ཚིག་བརྗོད་སྒྲ་སྙན་ཅི་མཆིས་པ། །མོས་བློས་དཀྱིལ་འཁོར་ལྷ་ལ་བདག་འབུལ་ན༑ ༑བཞེས་ཤིང་བླ་མེད་དངོས་གྲུབ་བསྩལ་དུ་གསོལ། །ཤཔད༴ ཕྱོགས་དུས་ཀུན་ན་ཙནྡན་སྦྲུལ་སྙིང་དང༌། །ཨ་གར་ལ་སོགས་དྲི་ཆབ་ཅི་མཆིས་པ། །མོས་བློས་དཀྱིལ་འཁོར་ལྷ་ལ་བདག་འབུལ་ན། །བཞེས་ཤིང་བླ་མེད་དངོས་གྲུབ་བསྩལ་དུ་གསོལ། །གནྡྷེ༴ ཕྱོགས་དུས་ཀུན་ན་རོ་དང་བཅུད་ལྡན་པའི། །ལུས་སེམས་ཚིམ་བྱེད་རོ་མཆོག་ཅི་མཆིས་པ། །མོས་བློས་དཀྱིལ་འཁོར་ལྷ་ལ་བདག་འབུལ་ན། །བཞེས་ཤིང་བླ་མེད་དངོས་གྲུབ་བསྩལ་དུ་གསོལ། །ར་ས༴ ཕྱོགས་དུས་ཀུན་ན་འཇམ་མཉེན་འཁྲིལ་ཆགས་པའི། །རེག་བྱའི་ཁྱད་པར་འཕགས་པ་གང་མཆིས་པ། །མོས་བློས་དཀྱིལ་འཁོར་ལྷ་ལ་བདག་འབུལ་ན། །བཞེས་ཤིང་བླ་མེད་དངོས་གྲུབ་བསྩལ་དུ་གསོལ། །སྤརྴེ༴ དེ་ནས་རིག་མ་བཅུ་དྲུག་གི་མཆོད་པ་ནི། དཀྱིལ་འཁོར་ཆེན་པོའི་འདོད་ཡོན་ལ་གནས་པའི། །ཡིད་འཕྲོག་གཟུགས་མཛེས་ལང་ཚོས་རབ་བརྒྱན་མ། །འཇོ་སྒེག་རྣམ་ལྡན་རྨད་བྱུང་བདེ་སྟེར་མ། །སྤྲུལ་པའི་ལྷ་མོ་རིན་ཆེན་རྒྱན་སྤྲས་མ། །སྔོན་མོ་པི་ཝང་སྙན་པར་སྒྲོག་མཛད་མ། །སེར་མོ་ཡིད་འོང་གླིང་བུ་སྒྱུར་མཛད་མ། །དམར་མོ་རྔ་ཟླུམ་རབ་ཏུ་བསྐུལ་མཛད་མ།་་༌།ལྗང་གུ་མཛེས་པའི་རྫ་རྔ་བརྡུང་མཛད་མ། །དམར་སྐྱ་འཛུམ་ཞིང་བཞད་པའི་ཚུལ་མཛད་མ། །སྔོན་མོ་སྒེག་ཅིང་འགྱིང་བའི་ཉམས་མཛད་མ། །སེར་མོ་སྙན་པའི་གླུ་དབྱངས་ལེན་མཛད་མ། །ལྗང་གུ་གར་གྱི་རོལ་རྩེད་སྒྱུར་མཛད་མ། །དཀར་མོ་སྣ་ཚོགས་མེ་ཏོག་ཕྲེང་འཛིན་མ། །སྔོན་མོ་དྲི་ཞིམ་བདུག་པའི་སྤོས་འཛིན་མ། །དམར་མོ་འོད་འབར་སྣང་གསལ་མཆོག་འཛིན་མ། །ལྗང་གུ་དྲི་བཟང་ཆུ་ཡི་སྣོད་འཛིན་མ། །དཀར་མོ་དྲི་མེད་མེ་ལོང་རབ་འཛིན་མ༑ ༑དམར་མོ་རོ་དྲུག་བདུད་རྩིའི་སྣོད་འཛིན་མ། །ལྗང་གུ་སྲབ་འཇམ་རྣམ་བཀྲའི་གོས་འཛིན་མ། །དཀར་མོ་ཆོས་ཀྱི་འབྱུང་གནས་མཆོག་འཛིན་མ། །དཔག་མེད་སྤྲུལ་ནས་དཀྱིལ་འཁོར་མཆོད་པར་བགྱི། །ཨོཾ་བཛྲ་ཝཱི་ཎི་ཧུཾ་ཧཱུྂ་ཕཊ།

དེ་བཞིན་དུ་ཝཱི་ཎིའི་གནས་སུ། ཝཾ་ཤེ། མྲྀ་ཏཾ་གེ། མུ་རཉྫེ། ཧཱ་སྱ། ལཱ་སྱེ། གཱི་ཏི། ནྲྀ་ཏྱཻ། པུཥྤེ། དྷཱུ་པེ། དཱི་པེ། གནྡྷེ། རྣམས་དང༌། ཨོཾ་རཱུ་པ་བཛྲ་ཧུཾ་ཧཱུྂ་ཕཊ། རཱུ་པའི་གནས་སུ། ར་ས། སྤརྴེ། ཨོཾ་དྷརྨ་དྷཱ་ཏུ་བཛྲ་ཧུཾ་ཧཱུྂ་ཕཊ། 

ཅེས་བཅུ་དྲུག་གིས་མཆོད། ནང་གི་མཆོད་པ་ནི། བདུད་རྩི་ལག་ཏུ་ཐོགས་ལ། རང་བཞིན་དག་པའི་དྲི་ཆེན་ཀླད་པ་དང༌། །བྱང་སེམས་རཀྟ་དྲི་ཆུ་བདུད་རྩི་ལྔ། །མི་ཁྱི་བ་གླང་རྟ་ཡི་ཤ་ལ་སོགས། །ཡོངས་སུ་སྦྱངས་ཤིང་སྦྱང་སྤེལ་བསྒྱུར་བ་ཡི། །རྨད་བྱུང་བདུད་རྩིའི་རྒྱ་མཚོ་ཆེན་པོ་འདིས། །རྩ་བརྒྱུད་བླ་མ་ཡི་དམ་མཁའ་འགྲོ་མཆོད། །ཁྱབ་བདག་རྡོ་རྗེ་འཆང་ཆེན་པོའི་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྂ། ཌཱ་ཀི་བདག་མེད་མའི་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྂ། རྣལ་འབྱོར་ནག་པོ་པའི་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྂ། གྲུབ་ཆེན་ཌཱ་མ་རུ་པའི་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྂ། ཨ་ཝ་དྷཱུ་ཏི་པའི་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྂ། ག་ཡ་དྷ་རའི་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྂ། འབྲོག་མིའི་ལོ་ཙཱ་བ་ཤཱཀྱ་ཡེ་ཤེས་ཀྱི་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྂ། སེ་སྟོན་ཀུན་དགའ་རིག་གི་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྂ། ཞང་སྟོན་ཆོས་འབར་གྱི་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྂ། ས་ཆེན་ཀུན་དགའ་སྙིང་པོའི་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྂ། རྗེ་བཙུན་བསོད་ནམས་རྩེ་མོའི་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྂ། རྗེ་བཙུན་གྲགས་པ་རྒྱལ་མཚན་གྱི་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྂ། ཆོས་རྗེ་པཎྜི་ཏ་ཀུན་དགའ་རྒྱལ་མཚན་དཔལ་བཟང་པོའི་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྂ། འགྲོ་བའི་མགོན་པོ་ཆོས་རྒྱལ་ཨཱརྻའི་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྂ། ཞང་དཀོན་མཆོག་དཔལ་གྱི་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྂ། ཆོས་རྗེ་བསོད་ནམས་ཤྲཱིའི་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྂ། ཆོས་རྗེ་བསོད་ནམས་རྒྱལ་མཚན་གྱི་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྂ། དཔལ་ལྡན་ཚུལ་ཁྲིམས་ཡེ་ཤེས་ཀྱི་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྂ། གྲུབ་ཆེན་བུདྡྷ་ཤྲཱིའི་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྂ། ངོར་པ་ཀུན་དགའ་རཏྣའི་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྂ། པཎྜི་ཏ་ཆེན་པོ་ཤཱཀྱ་མཆོག་ལྡན་གྱི་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྂ། ལྕགས་ལུང་པ་ཆེན་པོ་གཞོན་ནུ་ཆོས་གྲུབ་ཀྱི་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྂ། རྗེ་བཙུན་ཀུན་དགའ་སྒྲོལ་མཆོག་གི་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྂ། རྗེ་རྡོ་རིང་པ་ཀུན་དགའ་རྒྱལ་མཚན་གྱི་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྂ། མཁན་ཆེན་བྱམས་པ་ཆོས་ཀྱི་རྒྱ་མཚོའི་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྂ།

ཨོཾ་བཛྲ་ཀརྟཱ་རི་ཧེ་བཛྲ་ཨོཾ་ཨཱཿ་ཧཱུྂ། ཨོཾ་ཨཱཿནཾ་བཛྲ་ནཻ་རཱཏྨཱ་ཡ་ཨོཾ་ཨཱཿཧཱུྂ། ཨོཾ་བཛྲ་གཽ་རཱི་གཾ་ཨོཾ་ཨཱཿཧཱུྂ། ཨོཾ་ཙཽ་རཱི་ཙཾ་ཨོཾ་ཨཱཿཧཱུྂ། ཨོཾ་བེ་ཏཱ་ལཱི་བཾ་ཨོཾ་ཨཱཿཧཱུྂ། ཨོཾ་གྷསྨ་རཱི་གྷཾ་ཨོཾ་ཨཱཿཧཱུྂ། ཨོཾ་པུཀྐ་སཱི་པཾ་ཨོཾ་ཨཱཿཧཱུྂ། ཨོཾ་ཤ་ཝ་རཱི་ཤཾ་ཨོཾ་ཨཱཿཧཱུྂ། ཨོཾ་ཙཎྜ་ལཱི་ལཾ་ཨོཾ་ཨཱཿཧཱུྂ། ཨོཾ་ཌོཾ་བྷི་ནཱི་ཌཾ་ཨོཾ་ཨཱཿཧཱུྂ། ཨོཾ་མ་ཧཱ་ཀཱ་ལཱ་བི་ཀཱ་ལ་རཱ་ཏྲི་ཏ་ཨོཾ་ཨཱཿཧཱུྂ། ཌོཾ་བི་ནཱི་ཨོཾ་ཨཱཿཧཱུྂ། ཙཎྜ་ལཱི་ཨོཾ་ཨཱཿཧཱུྂ། རཀྵ་སཱི་ཨོཾ་ཨཱཿཧཱུྂ། སིངྒ་ལཱི་ཨོཾ་ཨཱཿཧཱུྂ། དེ་ཝཱི་ཨོཾ་ཨཱཿཧཱུྂ། ཁྱད་པར་དུ་དཔལ་ལྡན་ལྷ་མོ་དུད་པའི་སོལ་བ་མོ་འཁོར་དང་བཅས་པའི་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྂ། གཞན་ཡང་ཕྱོགས་སྐྱོང་ལྷ་ཀླུ་གནོད་སྦྱིན་འཁོར་དང་བཅས་པའི་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྂ། གཞན་ཡང་གཞི་བདག་འགྲོ་དྲུག་གི་སེམས་ཅན་དཔལ་དགྱེས་པ་རྡོ་རྗེ་དང་ཙ་ཏུརྨུ་ཁའི་ལྷར་གྱུར་པའི་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྂ། དེ་ནས་གསང་བའི་མཆོད་པ་ནི། ཡབ་ཀྱི་གསང་བའི་ངོ་བོ་རྡོ་རྗེ་སྔོན་པོ་རྩེ་ལྔ་པ་བུ་ག་ཕཊ་སེར་པོས་བཀག་པ། ཡུམ་གྱི་གསང་བ་པདྨ་དམར་པོ་འདབ་མ་བཞི་པ་ཨཱཿདམར་པོས་མཚན་པ། ཡབ་ཡུམ་བསྲུབས་པ་དང་བསྲུབ་པའི་སྦྱོར་བས། བདེ་བ་སྤྱི་བོ་ནས་མགྲིན་པར་སླེབ་པས་དགའ་བ། མགྲིན་པ་ནས་སྙིང་ཁར་སླེབ་པས་མཆོག་ཏུ་དགའ་བ། སྙིང་ཁ་ནས་ལྟེ་བར་སླེབ་པས་ཁྱད་པར་གྱི་དགའ་བ། ལྟེ་བ་ནས་རྡོ་རྗེ་ནོར་བུའི་རྩེར་སླེབ་པས་ལྷན་ཅིག་ཉམས་སུ་མྱོང་ཞིང༌། ལྷ་ཐམས་ཅད་སྙོམས་པར་ཞུགས་པས་ཆགས་པ་མེད་པའི་བདེ་བ་ཆེན་པོ་ཉམས་སུ་མྱོང་བར་གྱུར། མཆོད་བྱ་མཆོད་བྱེད་མཆོད་པ་རྣམས། །ཐ་དད་མི་དམིགས་གྲུབ་པ་མེད། །དེ་ཕྱིར་མཆོད་པ་ཆེན་པོ་སྟེ། །སྐྱེ་གནས་འགྲོ་འོང་བྲལ་བའི་ཕྱིར། །བླ་མེད་མཆོད་པ་མཆོག་གིས་མཆོད། ། ཅེས་མཆོད་དོ། །དེ་ནས་བསྟོད་པ་ནི། རྡོ་རྗེ་དྲིལ་བུ་འཛིན་ལ་འདི་སྐད་དོ། ། ཧཱུྂ ཆོས་དབྱིངས་མི་འགྱུར་བདེ་ཆེན་ཧེ་རུ་ཀཿ

། ཅེས་བསྟོད། རང་གི་ཐུགས་ཀ་ནས་སྔགས་ཕྲེང་བྱུང་ཞལ་ནས་ཐོན། མདུན་བསྐྱེད་ཀྱི་ཞལ་དུ་ཞུགས། ཐུགས་ཀའི་སྔགས་ཕྲེང་ལ་ཐིམ་ཞིང་འཁོར་བར་གྱུར། ཅེས་དམིགས་ཏེ། ཨཥྚ་ནི། ཨོཾ་ཨཥྚཱ་ན་ནཱ་ཡ། པིངྒ་ལོརྡྷཱ་ཀེ་ཤ་ཝརྡ་མཱ་ནེ། ཙ་ཏུརྦིཾ་ཤ་ཏི་ནཻ་ཏྲཱ་ཡ། ཥོ་ཌ་ཤ་བྷུ་ཛཱ་ཡ། ཀྲྀཥྞ་ཛཱི་མཱུ་ཏ་ཝ་པུ་ཥེ། ཀ་པཱ་ལ་མཱ་ལཱ་ནེ་ཀཱ་དྷ་རི་ཎེ། ཨདྷྨཱ་ཏ་ཀྲཱུར་ཙིཏྟཱ་ཡ། ཨརྡྷེནྡུ་དཾ་ཥྜཱི་ཎེ། མཱ་ར་ཡ་མཱ་ར་ཡ། ཀཱ་ར་ཡ་ཀཱ་ར་ཡ། གརྫ་ཡ་གརྫ་ཡ། ཏརྫ་ཡ་ཏརྫ་ཡ། ཤོ་ཥ་ཡ་ཤོ་ཥ་ཡ། སཔྟ་སཱ་ག་རཱ་ཎ། བྷནྡྷ་བྷནྡྷ། ནཱ་གཱཥྚ་ཀཱན། གྲྀཧྞ་གྲྀཧྞ། ཤ་ཏྲཱུཾ། ཧ་ཧཱ། ཧི་ཧཱི། ཧུ་ཧཱུ། ཧེ་ཧཻ། ཧོ་ཧཽ། ཧཾ་ཧཱཿཕཊ་སྭཱ་ཧཱ། ཞེས་པ་རྩ་སྔགས་སོ། ། ཨོཾ་དེ་ཝ་པི་ཙུ་བཛྲ་ཧཱུྂ་ཧཱུྂ་ཧཱུྂ་ཕཊ་སྭཱ་ཧཱ། ཡབ་ཀྱི་སྙིང་པོའོ། ། ཨོཾ་བཛྲ་ཀརྟཱ་རི་ཧེ་བཛྲ་ཡ་ཧཱུྂ་ཧཱུྂ་ཧཱུྂ་ཕཊ་སྭཱ་ཧཱ། ཡབ་ཀྱི་ཉེ་སྙིང་ངོ༌། ། ཨོཾ་ཨཱཿཨཾ་ཧཱུྂ་ཕཊ་སྭཱ་ཧཱ། ཡུམ་གྱི་སྙིང་པོའོ། ། ཨོཾ་ཨཱཿནཾ་བཛྲ་ནཻརཱཏྨཱ་ཡ་ཧཱུྂ་ཕཊ་སྭཱ་ཧཱ། ཡུམ་ཀྱི་ཉེ་སྙིང་ངོ༌། ། ཨོཾ་ཨཱཿགཾ་བཛྲ་གཽ་རཱི་ཧཱུྂ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་ཨཱཿཙཾ་བཛྲ་ཙཽ་རཱི་ཧཱུྂ་ཕཊ། ཨོཾ་ཨཱཿབཾ་བཛྲ་བེ་ཏཱ་ལཱི་ཧཱུྂ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་ཨཱཿགྷཾ་བཛྲ་གྷསྨ་རཱི་ཧཱུྂ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་ཨཱཿཔཾ་བཛྲ་པུཀྐ་སཱི་ཧཱུྂ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་ཨཱཿཤཾ་བཛྲ་ཤ་ཝ་རཱི་ཧཱུྂ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་ཨཱཿལཾ་བཛྲ་ཙཎྜ་ལཱི་ཧཱུྂ་ཕཊ་སྭཱ་ཧཱ༑ ཨོཾ་ཨཱཿཌཾ་བཛྲ་ཌོཾ་བྷི་ནཱི་ཧཱུྂ་ཕཊ་སྭཱ་ཧཱ། ཞེས་པ་ལྷ་མོ་བརྒྱད་ཀྱི་སྙིང་པོའོ། ། ཨོཾ་ཨཱཿགཾ་ཧཱུྂ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་ཨཱཿཙཾ་ཧཱུྂ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་ཨཱཿབཾ་ཧཱུྂ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་ཨཱཿགྷཾ་ཧཱུྂ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་ཨཱཿ་པཾ་ཧཱུྂ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་ཨཱཿཤཾ་ཧཱུྂ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་ཨཱཿལཾ་ཧཱུྂ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་ཨཱཿཌཾ་ཧཱུྂ་ཕཊ་སྭཱ་ཧཱ། ཞེས་ལྷ་མོ་བརྒྱད་ཀྱི་ཉེ་སྙིང་པོའོ། །དེ་ལྟར་མདུན་བསྐྱེད་ལ་དམིགས་པའི་བཟླས་པ་ཅི་རིགས་བྱའོ། །དེ་ནས་གཏོར་མ་བསངས་སྦྱངས།

བྱིན་རླབས་སྔར་ལྟར་བྱས་ལ། ཨོཾ་བཛྲ་ཨཱ་རལླི་ཧོཿ ཛཿཧཱུྂ་བཾ་ཧོཿ ཨོཾ་ཨཥྚ་ན་ནཱ་ཡཿ ཞེས་པའི་བར་སྔགས་ལན་གསུམ་བརྗོད་ལ། ལྷ་ཐམས་ཅད་ཀྱི་ལྗགས་རྡོ་རྗེ་འོད་ཀྱི་སྦུ་གུ་ཅན་གྱིས་བདུད་རྩིའི་གཏོར་མ་གསོལ་བས་མཉེས་ཤིང་ཚིམ་པར་གྱུར། ཅེས་དང༌། ཆུ་བཞི་ཉེར་སྤྱོད་རོལ་མོ་གར་དང་བཅས་པས་མཆོད་ལ། བསྟོད་པ་ནི། ཁྱོད་ནི་རྣལ་འབྱོར་མ་ཚོགས་ཡོངས་སུ་བསྐོར༑ ༑ཁྱོད་ནི་དེ་ཉིད་ཐུགས་ཀྱི་མཆོག་བརྙེས་པ། །ཁྱོད་ཀྱིས་རྒྱུ་དང་མི་རྒྱུའི་འགྲོ་བ་ཀུན་བརྒྱན་ཏེ། །སྙིང་རྗེའི་ཐུགས་ཀྱིས་སེམས་ཅན་སད་མཛད་པ། །སྒྱུ་མའི་སྤྲོས་པ་རྡོ་རྗེ་བསྒྲུབ་པའི་ཡིད༑ ༑ཧེ་རུ་ཀ་དཔལ་རྡོ་རྗེ་ཡིད་ཀྱིས་དགྱེས། ། ཞེས་བརྗོད། དེ་ནས་ཕྱི་རོལ་ཕྱོགས་སྐྱོང་ལ་གཏོར་མ་འབུལ་བ་ནི། སྒྲུབ་གྲོགས་ལ་དཀྱིལ་འཁོར་བསྲུང་དུ་བཅུག་ལ། གཞན་རྣམས་དཀྱིལ་འཁོར་གྱི་ཕྱི་རོལ་ཕྱིན་ལ། གཏོར་མ་སྔར་ལྟར་བྱིན་གྱིས་བརླབས། མདུན་དུ་པད་ཟླའི་གདན་ལ་སོ་སོར་གསལ་བ། རང་གི་ཐུགས་ཀའི་འོད་ཀྱིས་དུར་ཁྲོད་བརྒྱད་ན་གནས་པའི་ཕྱོགས་སྐྱོང་རྣམས་དང༌། གཞན་ཡང་ཕྱོགས་བཅུའི་འཇིག་རྟེན་ཁམས་ན་གནས་པའི་ཕྱོགས་སྐྱོང་འཁོར་བཅས་སྤྱན་དྲངས། མདུན་དུ་རང་རང་གི་གདན་སོ་སོ་ལ་འཁོད་པར་གྱུར་ པར་དམིགས་ལ། ཨོཾ་ཨིནྡྲ་ཡེ། ཡ་མཱ་ཡེ། ཀུ་བེ་ར། མ་ཧཱ་བྷཱུ་ཏ། ཨཱགྣ་ཡེ། རཀྵ་ས། བཱ་ཡུ་དེ་ཝ་ཏ། ཙནྡྲ། སཱུརྻ། བྲཧྨ། ཨ་དེ་ཝ། བེ་མ་ཙེ་ཏྲི། པྲྀ་ཐི་ཝཱི་དེ་ཝ་ཏ་ས་པ་རི་ཝཱ་ར་ཨཱ་ཀརྵ་ཡ་ཛཿཧཱུྂ་བཾ་ཧོཿ པཎྜི་ཏ་སརྦ་དེ་ཝ་ཏ། སཱ་དྷ་ཀ་སྱ་ཤཱནྟིཾ་ཀུ་རུ། པུཥྚིཾ་ཀུ་རུ༑ སརྦ་ཀརཡནྟུ། སིདྡྷིབྷ་ཝནྟུ། ཡ་ཐཱ་སུ་ཁཾ་སུ་ཁེ། ཞེས་སྤྱན་དྲངས་ལ། ཆུ་བཞི་ཉེར་སྤྱོད་རོལ་མོས་ཀྱང་མཆོད་ལ། མགྲོན་རྣམས་ཀྱི་ལྗགས་ཧཱུྂ་དཀར་པོ་ལས་གྱུར་པའི་རྡོ་རྗེ་རྩེ་གཅིག་པའི་རྣམ་པས་འོད་ཟེར་གྱི་ལྗགས་སྦུ་གུས་གཏོར་མ་གསོལ་བར་གྱུར ་པར་དམིགས་ལ། ཨོཾ་ཁ་ཁ་ཁཱ་ཧི་ཁཱ་ཧི། སརྦ་ཡཀྵ། རཀྵ་ས། བྷཱུ་ཏ།

པྲེ་ཏ། པི་ཤཱ་ཙཱ། ཨུནྨཱ་ད། ཨ་པ་སྨ་ར་ཌཱ་ཀ་ཌཱ་ཀི་ནྱ་དཱ་ཡ་ཨི་མཾ་བ་ལིཾ་གྲྀཧྞནྟུ། ས་མ་ཡ་རཀྵནྟུ། མ་མ་སརྦ་སིདྡྷི་མྨེ་པྲ་ཡཙྪནྟུ། ཡ་ཐེ་བཾ། ཡ་ཐཻཥྚཾ། བྷུཉྫ་ཐ། པི་བ་ཐ། ཛི་གྷྲ་ཐ། མཱ་ཏི་ཀྲ་མ་ཐ། མ་མ་སརྦ་ཀཱརྟ་ཡ་སཏྶུ་ཁཾ་བི་ཤུདྡྷ་ཡ་ས་ཧཱ་ཡི་ཀཾ་བྷ་ཝནྟུ་ཧཱུྂ་ཧཱུྂ་ཕཊ་སྭཱ་ཧཱ། ཞེས་ལན་གསུམ་དང༌། ལྷ་ཡི་ཚོགས་རྣམས་མ་ལུས་དང༌། ། ཞེས་པ་ནས། ཡིད་ལ་འདོད་པ་ཀུན་འགྲུབ་མཛོད། ། ཅེས་པའི་བར་གྱིས་གཏོར་མ་འབུལ་ཞིང་ཕྲིན་ལས་བཅོལ། ཧེ་རུ་ཀའི་ཡི་གེ་བརྒྱ་པས་ལྷག་ཆད་ཁ་བསྐངས་ནང་དུ་ལོག །དེ་ནས་སློབ་མ་རྗེས་སུ་བཟུང་བའི་ཕྱིར། བདག་ཉིད་འཇུག་པའི་དོན་དུ་བླང་བ་ནི། དབང་རྫས་རྣམས་གཞུང་ལྟར་འདུ་བྱས་ལ། གཏོར་ཆུང་ཏིང་ལོ་ཆངས་བུ་རྣམས་ཉེར་བསྡོགས་བྱ། སློབ་མ་སྟ་གོན་དགོས་ན་སྐབས་འདིར་བྱ། དེ་ནས་ལས་ཐམས་ཅད་པའི་བུམ་པའི་ཆུས་ཁྲུས་བྱ། ཧཱུྂ་བཞིའི་སྔགས་ཀྱིས་བགེགས་བསྐྲད། དཔལ་ལྡན་བླ་མ་དམ་པ་སྤྲུལ་པའི་སྐུར་བཞུགས་པ་དང༌། དཀྱིལ་འཁོར་གྱི་གཙོ་བོ་ཐ་མི་དད་པ་ལ་དཀྱིལ་འཁོར་དུ་འཇུག་པའི་ཡོན་དུ་མཎྜལ་ཕུལ། གསོལ་བ་གདབ་པ་ནི། བདག་མེད་མ་ཡིས་མགྲིན་འཁྱུད་པའི། །དགའ་ཆེན་ཁྱོད་བདག་སྟོན་པ་སྟེ། །སློབ་དཔོན་གྱིས་ནི་དགོངས་སུ་གསོལ། །བྱང་ཆུབ་ཆེན་པོའི་ཚུལ་བསྟན་པ། །མགོན་པོ་ཆེན་པོ་བདག་འཚལ་ལོ། །དམ་ཚིག་དེ་ཉིད་བདག་ལ་སྩོལ། །བྱང་ཆུབ་སེམས་ཀྱང་བདག་ལ་སྩོལ། །སངས་རྒྱས་ཆོས་དང་དགེ་འདུན་ཏེ། །སྐྱབས་གསུམ་དག་ཀྱང་དག་ལ་སྩོལ། །ཐར་པ་ཆེན་པོའི་གྲོང་མཆོག་ཏུ། །མགོན་པོས་བདག་ནི་གཞུག་ཏུ་གསོལ། ། ཞེས་ལན་གསུམ་གྱིས་གསོལ་བ་གདབ། དཀོན་མཆོག་གསུམ་ལ་བདག་སྐྱབས་མཆི། ། ཞེས་པ་ནས། འགྲོ་ལ་ཕན་ཕྱིར་སངས་རྒྱས་འགྲུབ་པར་ཤོག ། ཅེས་པའི་བར་གྱིས་ཡན་ལག་བདུན་པ་བྱ། དེ་ནས་རིགས་ལྔའི་སྡོམ་པ་གཟུང་བ་ནི། སངས་རྒྱས་སྲས་བཅས་ཐམས་ཅད་དང༌། །དཔའ་བོ་རྣལ་འབྱོར་མ་རྣམས་ཀུན། །བདག་ལ་དགོངས་པར་མཛད་དུ་གསོལ། །བདག་མིང་འདི་ཞེས་བགྱི་བས་ནི། །དུས་འདི་ནས་ནི་བཟུང་ནས་སུ། །བྱང་ཆུབ་སྙིང་པོར་མཆིས་ཀྱི་བར། །ཇི་ལྟར་དུས་གསུམ་མགོན་པོ་རྣམས། །བྱང་ཆུབ་ཏུ་ནི་ངེས་མཛད་པའི། །བྱང་ཆུབ་སེམས་ནི་བླ་ན་མེད། །དམ་པ་བདག་གིས་བསྐྱེད་པར་བགྱི། །སངས་རྒྱས་རྣལ་འབྱོར་སྡོམ་པ་ལ། །ཚུལ་ཁྲིམས་ཀྱི་ནི་བསླབ་པ་དང༌། །དགེ་བའི་ཆོས་ནི་སྡུད་པ་དང༌། །སེམས་ཅན་དོན་བྱེད་ཚུལ་ཁྲིམས་གསུམ། །བདག་གིས་བརྟན་པོར་གཟུང་བར་བགྱི། །སངས་རྒྱས་ཆོས་དང་དགེ་འདུན་ཏེ། །བླ་ན་མེད་པའི་དཀོན་མཆོག་གསུམ། །དེང་ནས་བརྩམས་ཏེ་གཟུང་བར་བགྱི། །ནམ་དུ་ཡང་ནི་མི་སྤང་ངོ༌། །རྡོ་རྗེའི་རིགས་མཆོག་ཆེན་པོ་ལ། །རྡོ་རྗེ་དྲིལ་བུའི་ཕྱག་རྒྱ་ཡང༌༑ ༑ཡང་དག་ཉིད་དུ་གཟུང་བར་བགྱི། །སློབ་དཔོན་དག་ཀྱང་གཟུང་བར་བགྱི། །རིན་ཆེན་རིགས་མཆོག་ཆེན་པོ་ཡི། །དམ་ཚིག་ཡིད་དུ་འོང་བ་ལ། །ཉིན་རེ་བཞིན་དུ་ལན་དྲུག་ཏུ། །སྦྱིན་པ་རྣམ་བཞི་རྟག་ཏུ་སྦྱིན། །བྱང་ཆུབ་ཆེན་པོ་ལས་བྱུང་བའི། །པདྨའི་རིགས་ཆེན་དག་པ་ལ། །ཕྱི་ནང་གསང་བ་ཐེག་པ་གསུམ། །དམ་པའི་ཆོས་ནི་གཟུང་བར་བགྱི། །ལས་ཀྱི་རིགས་མཆོག་ཆེན་པོ་ལ། །སྡོམ་པ་ཐམས་ཅད་ལྡན་པ་ནི། །ཡང་དག་ཉིད་དུ་གཟུང་བར་བགྱི། །མཆོད་པའི་ལས་ཀྱང་ཅི་ནུས་བགྱི། །བླ་ན་མེད་པའི་བྱང་ཆུབ་སེམས། །དམ་པ་བདག་གིས་བསྐྱེད་བགྱིས་ནས། །སེམས་ཅན་ཀུན་གྱི་དོན་གྱི་ཕྱིར། །བདག་གིས་སྡོམ་པ་མ་ལུས་གཟུང༌། །མ་བསྒྲལ་བ་རྣམས་བསྒྲལ་བ་དང༌། །མ་གྲོལ་བ་རྣམས་བདག་གིས་དགྲོལ༑ ༑དབུགས་མ་ཕྱིན་པ་དབུགས་དབྱུང་ཞིང༌། །སེམས་ཅན་མྱ་ངན་འདས་ལ་འགོད། །

ཅེས་ལན་གསུམ་གྱིས་སྡོམ་པ་བཟུང༌། ཨོཾ་ཙཀྵུརྦཧནྡྷ་ཝ་ར་མཱ་ན་ཡ་ཧཱུྂ། ཞེས་གདོང་བཅིངས། ཨ་ཁཾ་བཱི་ར་ཧཱུྂ། ཞེས་མེ་ཏོག་གཏད་ལ། བུ་ཁྱོད་སུ་ཡིན་ཅི་ལ་དགའ། ། ཞེས་དང༌། སྐལ་བཟང་བདག་ནི་བདེ་ཆེན་ལའོ། ། ཞེས་དང༌། དེ་ཡིས་བུ་ལ་ཅི་ཞིག་བྱ། ཞེས་འདྲི། སངས་རྒྱས་མཆོག་གི་དམ་ཚིག་གོ། ། ཞེས་ལན་བཏབ་པར་མོས་ལ། སྙིང་ཁར་ཟླ་བའི་དཀྱིལ་འཁོར་གྱི་རྡོ་རྗེ་དཀར་པོ་རྩེ་ལྔ་པ་འོད་ཟེར་འཕྲོ་བ་དང་བཅས་པར་གྱུར། ཨོཾ་སརྦ་ཡོ་ག་ཙིཏྟ་ཨུཏྤཱ་ད་ཡཱ་མི། སུ་ར་ཏེ་ས་མ་ཡ་སྟྭཾ་ཧོཿ སིདྡྷི་བཛྲ་ཡ་ཐཱ་སུ་ཁཾ། ཞེས་བརྗོད། གསང་བར་གདམ་པ་ནི། སྤྱི་བོར་རྡོ་རྗེ་བཞག་ལ། དེང་ཁྱོད་དེ་བཞིན་གཤེགས་པ་རྣམས་དང་དཔའ་བོ་དང་རྣལ་འབྱོར་མ་རྣམས་ཀྱི་རིགས་སུ་ཞུགས་ཀྱི་དེ་བཞིན་གཤེགས་པ་ཐམས་ཅད་ཀྱི་གསང་བ་དམ་པ་འདི་ཁྱོད་ཀྱིས་དཀྱིལ་འཁོར་ཆེན་པོ་མ་མཐོང་བ་རྣམས་ཀྱི་མདུན་དུ་སྨྲ་བར་མི་བྱ་སྟེ། མ་དད་པ་རྣམས་ལ་ཡང་མ་ཡིན་ནོ། །ཧཱུྂ་ཛཿཧཱུྂ། ཞེས་བརྗོད། ཕན་ཡོན་གྱི་སྒོ་ནས་དམ་ལ་བཞག་པ་ནི། རྡོ་རྗེ་མགོར་བཞག ། ཡེ་ཤེས་གང་གིས་ཁྱོད་དེ་བཞིན་གཤེགས་པ་ཐམས་ཅད་ཀྱི་དངོས་གྲུབ་ཀྱང་ཐོབ་པར་འགྱུར་ན་དངོས་གྲུབ་གཞན་ལྟ་སྨོས་ཀྱང་ཅི་དགོས། དེ་ལྟ་བུའི་ཡེ་ཤེས་བསྐྱེད་པར་བྱ་ཡི། ཁྱོད་ཀྱིས་དཀྱིལ་འཁོར་ཆེན་པོ་འདི་མ་མཐོང་བ་རྣམས་ཀྱི་མདུན་དུ་སྨྲ་བར་མི་བྱ་སྟེ། བུ་ཁྱོད་དམ་ཚིག་ཉམས་པར་འགྱུར་ཏ་རེ། ཞེས་དང༌། ཉེས་དམིགས་ཀྱི་སྒོ་ནས་དམ་ལ་བཞག་པ་ནི། རྡོ་རྗེ་སྙིང་ག་དང་སྤྱི་བོར་བཞག་ལ། ཧེ་རུ་ཀ་དཔལ་དེང་ཁྱོད་ཀྱི། །སྙིང་ལ་ཡང་དག་ཞུགས་པར་འགྱུར། །གལ་ཏེ་ཚུལ་འདི་སྨྲས་ན་ནི། །དེ་མ་ཐག་ཏུ་དྲལ་ཏེ་གཤེགས།་་༌།

ཞེས་དང༌། ། འདི་ནི་དམ་ཚིག་རྡོ་རྗེ་སྟེ། གལ་ཏེ་ཚུལ་འདི་ཁྱོད་ཀྱིས་སུ་ལ་ཡང་སྨྲས་ན། འདིས་ཁྱོད་ཀྱིས་མགོ་བོ་འགེམས་པར་འགྱུར་ཏ་རེ། ཞེས་བརྗོད། གཉིས་ཀའི་སྒོ་ནས་དམ་ལ་བཞག་པ་ནི། བདུད་རྩི་ལྕེ་ཐོག་ཏུ་བཞག་ལ། འདི་ནི་ཁྱོད་ཀྱི་དམྱལ་བའི་ཆུ། །དམ་ཚིག་འདས་ན་སྲེག་པར་བྱེད། །དམ་ཚིག་བསྲུང་ན་དངོས་གྲུབ་ནི། །རྡོ་རྗེ་བདུད་རྩིའི་ཆུ་འདིས་འགྲུབ། །ཨོཾ་པཉྩ་ཨ་མྲྀ་ཏ་ཨུ་ད་ཀ་ཋ་ཋ་ཋཿ ཞེས་སྦྱིན། སློབ་དཔོན་བཀའ་གཉན་པའི་སྒོ་ནས་དམ་ལ་བཞག་པ་ནི། རྡོ་རྗེ་སྤྱི་བོར་བཞག་ལ། དེང་ནས་བཟུང་སྟེ་བུ་ཁྱོད་ཀྱི་རྡོ་རྗེ་སློབ་དཔོན་དཔལ་ཧེ་རུ་ཀ་ཀྱཻ་རྡོ་རྗེ་ང་ཡིན་གྱི། ངས་འདི་གྱིས་ཤིག་ཅེས་བསྒོ་བ་གང་ཡིན་པ་དེ་ཁྱོད་ཀྱིས་བྱ་དགོས་སོ། །ང་ལ་བརྙས་པར་མ་བྱེད་ཅིག །བྱས་ན་ཡ་ང་བ་མ་སྤང་བར་འཆི་བའི་དུས་བྱས་ཏེ་སེམས་ཅན་དམྱལ་བ་ཆེན་པོར་སྐྱེ་བར་འགྱུར་ཏ་རེ། ཞེས་དམ་ལ་བཞག །དེ་ནས་ཡེ་ཤེས་དབབ་པའི་དོན་དུ་མཎྜལ་ཕུལ་ལ། གསོལ་བ་གདབ་པ་ནི། དེ་བཞིན་གཤེགས་པ་ཐམས་ཅད་ཀྱི་བྱིན་རླབས་ལས་དཔལ་ཧེ་རུ་ཀ་བདག་ལ་དབབ་ཏུ་གསོལ། ཞེས་གསུམ་བརྗོད། རང་ཉིད་ཧེ་རུ་ཀར་གསལ་བའི་སྐུའི་ནང་དུ་རྩ་གསུམ་འཁོར་ལོ་བཞི་གསལ་བའི་ལྟེ་བར་ཧཱུྂ་གཏུམ་མོའི་མེ་དང༌། སྤྱི་བོར་ཧཾ་དཀར་པོ་མགོ་ཐུར་དུ་བལྟ་བ། རོ་རྐྱང་རླུང་གིས་གཏུམ་མོའི་མེ་ཆེར་སྦར་བས། སྤྱི་བོའི་ཧཾ་ལས་དཀར་ཆའི་རྒྱུན་བབ། ལུས་ཐམས་ཅད་བརྗོད་མེད་ཀྱི་བདེ་བས་ཁྱབ་པ་དང༌། སླར་ཡང་གཙོ་བོའི་ཐུགས་ཀ་ནས། འོད་ཟེར་ལྕགས་ཀྱུ་ལྟ་བུར་འཕྲོས། ཕྱོགས་བཅུའི་འཇིག་རྟེན་གྱི་ཁམས་སུ་བཞུགས་པའི་ཡེ་ཤེས་པ་དཔལ་ཀྱཻའི་རྡོ་རྗེའི་ལྷ་ཚོགས་དཔག་ཏུ་མེད་པ་སྤྱན་དྲངས། རང་གི་བ་སྤུའི་བུ་ག་ཐམས་ཅད་དུ་མཚོ་ལ་ཁ་བ་བབ་པ་ལྟར་སིབ་སིབ་ཐིམ་པར་གྱུར། ཅེས་དམིགས་ལ། གུ་གུལ་གྱིས་བདུག །རོལ་མོ་དཀྲོལ། ཨོཾ་བཛྲ་ཧེ་རུ་ཀ་སརྦ་མཎྜ་ལ་ཨཱ་ཝེ་ཤ་ཡ་ཨ་ཨ་ཨཱཿ ཞེས་མང་དུ་བརྗོད།

ཏིཥྛ་བཛྲ་ཧཱུྂ། ཞེས་རྡོ་རྗེ་སྤྱི་བོར་བཞག་ལ་བརྟན་པར་བྱའོ༑ ༑དེའི་ཚེ་བར་སྣང་གི་ཁ་དོག་བརྟག་པ་ནི། དཀྱིལ་འཁོར་དབུས་ཀྱི་ནམ་མཁའ་ལ།་་༌།ཁྱོད་ཀྱི་མིག་ལ་ཇི་ལྟར་སྣང༌། ཞེས་དྲི། དཀར་སེར་དམར་ལྗང་མཐིང་ཁྲར་མཐོང༌། ། ཞེས་དང༌། དེང་ནས་བཟུང་སྟེ་ཁྱོད་རྣམས་ཀྱིས། །ཞི་དང་རྒྱས་པའི་ལས་རྣམས་དང༌། །དབང་དུ་བྱ་དང་དགུག་པ་དང༌། །ལས་རྣམས་ཐམས་ཅད་འགྲུབ་པར་འགྱུར། ། ཞེས་བརྗོད། རིགས་ཀྱི་ལྷ་བརྟག་པའི་མེ་ཏོག་དོར་བ་ནི། ཧེ་རུ་ཀ་ཡི་དཀྱིལ་འཁོར་འདིར། །རིག་པའི་མེ་ཏོག་འདི་དོར་བས། །སྔོན་ཚེ་གང་ལ་སྦྱངས་པའི་ལྷ། །གང་ཡིན་དེ་ལ་ཕོག་པར་ཤོག །པྲ་ཏཱིཙྪ་བཛྲ་ཧོཿ ཞེས་དོར། མེ་ཏོག་དབུས་སུ་བབ་པར་མཚོན་ཏེ། དགའ་བའི་རྡོ་རྗེ་ སོགས་གསང་མིང་བཏགས། དེ་ནས་མིག་ཐུར་ཐོགས་ཏེ། མིག་དབྱེ་བ་ནི། ཀྱཻ་རྡོ་རྗེ་དེང་ཁྱོད་ཀྱི། །མིག་འབྱེད་པར་ནི་བརྩོན་པར་བྱ། །ཕྱེ་བས་ཐམས་ཅད་མཐོང་འགྱུར་ཏེ༑ ༑རྡོ་རྗེའི་མིག་ནི་བླ་ན་མེད། །ཧེ་བཛྲ་པ་ཤྱ་ཧོཿ ། ཞེས་གདོང་གཡོགས་དཀྲོལ་ལ། ཐར་པ་ཆེན་པོའི་གྲོང་མཆོག་ཏུ། །ལྟོས་ཤིག་ལྟོས་ཤིག་སེམས་དཔའ་ཆེ། །མེ་ཏོག་གང་དུ་ལྷུང་གྱུར་པ། །དེ་ནི་ཁྱོད་ཀྱི་ཧེ་རུ་ཀཿ །དཀྱིལ་འཁོར་བཟང་པོ་འདི་ལ་ལྟོས། །ད་ནི་དད་པ་རབ་ཏུ་སྐྱེད། །ཁྱོད་ནི་སངས་རྒྱས་རིགས་སུ་སྐྱེས། །སྔགས་དང་ཕྱག་རྒྱས་བྱིན་གྱིས་བརླབས། །ཁྱོད་ནི་རྣལ་འབྱོར་མ་རིགས་སྐྱེས། །དམ་ཚིག་ཐམས་ཅད་སྐྱོང་བ་དང༌། །གསང་སྔགས་འདི་ལ་བརྩོན་པར་གྱིས། །འདི་ལྟ་བུའི་དཀྱིལ་འཁོར་གྲུ་བཞི་སྒོ་བཞི་པ། རྟ་བབས་དུར་ཁྲོད་མེ་རི་དང་བཅས་པའི་ནང༌། དཔལ་ཀྱཻ་རྡོ་རྗེ་ལྷ་དགུ་པ་ནི། ཁྱོད་ཀྱི་ཡི་དམ་གྱི་ལྷ་ཡིན་པས་ངོ་ཤེས་པར་གྱིས་ཤིག ། ཅེས་བསྟན་ཏོ། །དེ་དག་ནི་དཀྱིལ་འཁོར་དུ་འཇུག་པའི་རིམ་པའོ། །དངོས་གཞི་ཐོག་མར་སྦྱང་བ་ནི། གཏོར་ཆུང་ཏིང་ལོ་ཆབ་ཀྱིས་བྲན་ཏེ། ཨོཾ་ཨ་ཀཱ་རོ་ གསུམ་གྱིས་བྱིན་གྱིས་བརླབས་ནས།

གཏོར་ཆུང་རིམ་པ་ལྟར་བླངས་ལ་གཡས་སུ་བསྐོར་ཞིང༌། ན་མཿསརྦ་ཏ་ཐཱ་ག་ཏེ་བྷྱོ་བི་ཤྭ་མུ་ཁེ་བྷྱཿ སོགས་བརྗོད་ཅིང༌། སླར་ཡང་ཏིང་ལོ་གསུམ་རིམ་གྱིས་གཡོན་དུ་བསྐོར་ཞིང༌། ཨོཾ་སུཾ་བྷ་ནི་ སོགས་བརྗོད་ཅིང་དོར། དེ་ནས་རང་གི་མདུན་དུ་གཡས་སུ་མེ། གཡོན་དུ་ཆུ། དབུས་སུ་ཏིང་ལོ་བཀོད་ལ། བསངས་སྦྱངས། སྟོང་པའི་ངང་ལས་ཐབ་ཁུང་ཟླུམ་པོ་མཚན་ཉིད་ཐམས་ཅད་ཡོངས་སུ་རྫོགས་པའི་ནང་དུ་རྡོ་རྗེ་མཁའ་འགྲོ་ བསམས་ལ། རང་གི་རྐང་ཞབས་ཀྱི་རླུང་གིས་ལྟེ་བའི་མེ་སྦར། སྙིང་ཁར་སྡིག་པའི་ས་བོན་པཾ་ནག་པོ་གནས་པ་དེ་མེ་ལྕེས་དེད་སྡིག་པ་རྭ་ཅན་གྱི་རྣམ་པར་གྱུར། སྣ་ནས་ཐོན། མདུན་གྱི་ཏིལ་ལ་ཐིམ། རྡོ་རྗེ་མཁའ་འགྲོ་ཞལ་གདངས་པའི་ལྗགས་ལ་རཾ་ཡིག་གིས་མཚན་པར་གྱུར། ཨོཾ་བཛྲ་ཌཱ་ཀ་ཁ་ཁ་ཁཱ་ཧི་ཁཱ་ཧི། སརྦ་པཱ་པཾ་ད་ཧ་ན་བྷསྨིཾ་ཀུ་རུ་སྭཱ་ཧཱ། ཞེས་རྡོར་དྲིལ་དང་ལྡན་པས་མཐེབ་སྲིན་གྱིས་བླངས་ལ། གཡས་ཀྱིས་གཡས་དང༌། གཡོན་གྱིས་གཡོན་དུ་བསྐོར་ཞིང༌། མེ་དང་ཆུར་དོར་བ་རྡོ་རྗེ་མཁའ་འགྲོ་མའི་ཞལ་དང་རྒྱ་མཚོར་དོར་བར་བསམ། ཉེར་སྤྱོད་ལྔས་མཆོད་ལ། མི་བསྐྱོད་རྡོ་རྗེ་ཡེ་ཤེས་ཆེ། །རྡོ་རྗེ་དབྱིངས་ནི་མཁའ་ཆེན་པོ། །རྡོ་རྗེ་གསུམ་མཆོག་དཀྱིལ་འཁོར་གསུམ། །རྡོ་རྗེ་མཁའ་འགྲོར་ཕྱག་འཚལ་བསྟོད། ། ཅེས་བསྟོད། ཡི་གེ་བརྒྱ་པས་ལྷག་ཆད་ཁ་བསྐངས། བཛྲ་མུཿ ཞེས་གཤེགས་སུ་གསོལ། དེ་དག་སྦྱང་བའི་རིམ་པའོ། །དེ་ནས་རྡོ་རྗེའི་སློབ་མས་དབང་གི་དོན་དུ་མཎྜལ་ཕུལ་ལ། གསོལ་བ་གདབ་པ་ནི། བྱང་ཆུབ་རྡོ་རྗེས་སངས་རྒྱས་ལ། །མཆོད་ཆེན་ཇི་ལྟར་བསྩལ་བ་ལྟར། །བདག་ཀྱང་བསྐྱབ་པའི་དོན་གྱི་ཕྱིར། །ནམ་མཁའི་རྡོ་རྗེས་དེང་བདག་སྩོལ། ། ཅེས་གསུམ་གྱིས་གསོལ་བ་བཏབ། བསངས་སྦྱངས། སྟོང་པའི་ངང་ལས་པད་ཟླའི་གདན་ལ་བདག་དང་དབང་རྫས་བུམ་ཆུ་ཧཱུྂ་དང་རྡོ་རྗེ་ཡོངས་སུ་གྱུར་པ་ལས། མི་བསྐྱོད་པ་ཀྱཻ་རྡོ་རྗེ་སྐུ་མདོག་སྔོན་པོ་ཞལ་བརྒྱད་ཕྱག་བཅུ་དྲུག་ཞབས་བཞི་དང་ལྡན་པ།

གཡས་བརྒྱད་ཀྱིས་རྡོ་རྗེ་གཡོན་བརྒྱད་ཀྱིས་དྲིལ་བུ་འཛིན་པ། ཡུམ་བདག་མེད་མ་དང་མཉམ་པར་སྦྱོར་བ། རིན་པོ་ཆེ་དང་རུས་པས་བརྒྱན་པའི་ཐུགས་ཀའི་འོད་ཀྱིས་དབང་གི་ལྷ་རྣམས་སྤྱན་དྲངས། དེ་བཞིན་གཤེགས་པ་ཐམས་ཅད་ཀྱིས་བདག་ལ་མངོན་པར་དབང་བསྐུར་དུ་གསོལ། ཞེས་གསོལ་བ་བཏབ་པས། བྱང་ཆུབ་སེམས་དཔའ་རྣམས་ཀྱིས་བཀྲ་ཤིས་བརྗོད། རིག་མ་རྣམས་ཀྱིས་རྡོ་རྗེའི་གླུ་ལེན། ཁྲོ་བོ་རྣམས་ཀྱིས་བགེགས་བསྐྲད། དབང་གི་ལྷ་རྣམས་ཀྱིས་བུམ་པ་བྱང་ཆུབ་སེམས་ཀྱི་བདུད་རྩིས་གང་བ་ཅུང་ཟད་ཡོ་བར་བཟུང་སྟེ་དབང་བསྐུར་བར་གྱུར། སངས་རྒྱས་རྣམས་ནི་ཐམས་ཅད་ཀྱི། །གསང་གསུམ་གནས་ལས་བྱུང་བ་ཡི། །ཁམས་གསུམ་ཀུན་གྱིས་ཕྱག་བྱས་པ། །བུམ་པའི་དབང་བསྐུར་ཆེན་པོ་སྦྱིན། །ཨོཾ་སརྦ་ཏ་ཐཱ་ག་ཏ་ཨ་བྷི་ཥེ་ཀ་ཏ་ས་མ་ཡ་ཤྲཱི་ཡེ་ཧཱུྂ། ཞེས་བརྗོད་ཅིང་དབང་བསྐུར། ཆུ་ལྷག་མ་ཡར་ལུད་པ་ལས་མི་བསྐྱོད་པས་དབུ་བརྒྱན། དབང་གི་ལྷ་རྣམས་རང་ཉིད་ལ་ཐིམ་པར་གྱུར་ པ་ལ། པུཥྤེ་ སོགས་ཉེར་སྤྱོད་ལྔས་མཆོད། དེས་བུམ་པའི་དབང་ཐོབ། ཉོན་མོངས་པ་ཞེ་སྡང་གི་དྲི་མ་དག །རྣམ་པར་ཤེས་པའི་ཕུང་པོ་གནས་འགྱུར། ཆོས་ཀྱི་དབྱིངས་ཀྱི་ཡེ་ཤེས་མངོན་དུ་བྱས། མི་བསྐྱོད་པ་ཀྱཻ་རྡོ་རྗེའི་ས་བོན་བག་ཆགས་མ་ལུས་པ་རྒྱུད་ལ་བཞག ། ཅེས་བརྗོད། དབུ་རྒྱན་བསངས་སྦྱངས། སྟོང་པའི་ངང་ལས་པད་ཟླའི་གདན་ལ་བདག་དང་དབང་རྫས་རིགས་ལྔ་རིན་པོ་ཆེ་ཨཾ་གིས་མཚན་པ་ཡོངས་སུ་གྱུར་པ་ལས། རིན་ཆེན་འབྱུང་ལྡན་ཀྱཻ་རྡོ་རྗེ་སེར་པོ་ཞལ་བརྒྱད་ཕྱག་བཅུ་དྲུག་པ། ཞབས་བཞི་དང་ལྡན་པ། གཡས་བརྒྱད་ཀྱིས་རིན་པོ་ཆེ་དང༌། གཡོན་བརྒྱད་ཀྱིས་དྲིལ་བུ་འཛིན་པ། ཡུམ་སངས་རྒྱས་སྤྱན་མ་བདག་མེད་མ་དང་སྙོམས་པར་སྦྱོར་བ། རིན་པོ་ཆེ་དང་རུས་པའི་རྒྱན་གྱིས་བརྒྱན་པའི་ཐུགས་ཀའི་ཧཱུྂ་ལས་འོད་ཟེར་འཕྲོས་པས་དབང་གི་ལྷ་རྣམས་སྤྱན་དྲངས། དེ་བཞིན་གཤེགས་པ་རྣམས་ཀྱིས་བདག་ལ་མངོན་པར་དབང་བསྐུར་དུ་གསོལ།

ཞེས་གསོལ་བ་བཏབ་པས། བྱང་ཆུབ་སེམས་དཔའ་རྣམས་ཀྱིས་བཀྲ་ཤིས་བརྗོད། རིག་མ་རྣམས་ཀྱིས་རྡོ་རྗེའི་གླུ་ལེན། ཁྲོ་བོ་རྣམས་ཀྱིས་བགེགས་བསྐྲད། དབང་གི་ལྷ་རྣམས་ཀྱིས་དབང་རྫས་རིགས་ལྔ་དང་བུམ་པ་རྣམས་ཀྱིས་དབང་བསྐུར་བར་གྱུར། སངས་རྒྱས་ཀུན་གྱི་རྒྱལ་སྲིད་ཆེ། །དབུ་རྒྱན་རིན་ཆེན་དམ་པ་སྟེ། །རིན་ཆེན་ཀུན་གྱི་མཆོག་གྱུར་པ། །རྫོགས་པའི་སངས་རྒྱས་དབང་བསྐུར་རོ། །ཨོཾ་བཛྲ་མུ་ཀུ་ཊ་ཨ་བྷི་ཥིཉྩ་ཧཱུྂ། ཞེས་དབང་བསྐུར། བཛྲཱི་བྷ་ཝ་ཨ་བྷི་ཥིཉྩ་ཧཱུྂ། བུམ་ཆུ་སྦྱིན། ཆུ་ལྷག་མ་ཡར་ལུད་པ་ལས་རིན་ཆེན་འབྱུང་ལྡན་གྱིས་དབུ་བརྒྱན། དབང་གི་ལྷ་རྣམས་རང་ཉིད་ལ་ཐིམ་པར་གྱུར་ པ་ལ། པུཥྤེ་ སོགས་ཉེར་སྤྱོད་ལྔས་མཆོད། དེ་ལྟར་རིན་ཆེན་འབྱུང་ལྡན་ཅོད་པན་གྱི་དབང་བསྐུར་བས། ང་རྒྱལ་དང་སེར་སྣའི་དྲི་མ་དག །ཚོར་བའི་ཕུང་པོ་གནས་འགྱུར། མཉམ་ཉིད་ཀྱི་ཡེ་ཤེས་མངོན་དུ་བྱས། རིན་ཆེན་འབྱུང་ལྡན་གྱི་བག་ཆགས་དང་ས་བོན་རྒྱུད་ལ་བཞག ། ཅེས་བརྗོད། རྡོ་རྗེ་བསངས་སྦྱངས། སྟོང་པའི་ངང་ལས་པད་ཟླའི་གདན་ལ་བདག་དང་དབང་རྫས་པདྨ་ཛྲཱིཾ་གིས་མཚན་པ་ཡོངས་སུ་གྱུར་པ་ལས། སྣང་བ་མཐའ་ཡས་ཀྱཻ་རྡོ་རྗེ་དམར་པོ་ཞལ་བརྒྱད་ཕྱག་བཅུ་དྲུག་པ། ཞབས་བཞི་དང་ལྡན་པ། གཡས་བརྒྱད་ཀྱིས་པདྨ་དང༌། གཡོན་བརྒྱད་ཀྱིས་དྲིལ་བུ་འཛིན་པ། ཡུམ་གོས་དཀར་མོ་དང་མཉམ་པར་སྦྱོར་བ། རིན་པོ་ཆེ་དང་རུས་པའི་རྒྱན་གྱིས་བརྒྱན་པའི་ཐུགས་ཀའི་ཧཱུྂ་ལས་འོད་ཟེར་འཕྲོས་པས་དབང་གི་ལྷ་རྣམས་སྤྱན་དྲངས། དེ་བཞིན་གཤེགས་པ་རྣམས་ཀྱིས་བདག་ལ་མངོན་པར་དབང་བསྐུར་དུ་གསོལ། ཞེས་གསོལ་བ་བཏབ་པས། བྱང་ཆུབ་སེམས་དཔའ་རྣམས་ཀྱིས་བཀྲ་ཤིས་བརྗོད། རིག་མ་རྣམས་ཀྱིས་རྡོ་རྗེའི་གླུ་ལེན། ཁྲོ་བོ་རྣམས་ཀྱིས་བགེགས་བསྐྲད། དབང་གི་ལྷ་རྣམས་ཀྱིས་དབང་རྫས་རྡོ་རྗེ་དང་བུམ་པ་བཟུང་ནས་དབང་བསྐུར་བར་གྱུར། སངས་རྒྱས་ཀུན་གྱི་རྡོ་རྗེའི་དབང༌། །དེ་རིང་ཁྱོད་ལ་བསྐུར་བར་བྱ། །སངས་རྒྱས་ཉིད་དུ་རབ་བསྒྲུབ་ཕྱིར། །རྡོ་རྗེ་འདི་ནི་ཁྱོད་ཀྱིས་ཟུང༌༑ ༑ཨོཾ་མ་ཧཱ་བཛྲ་ཨ་བྷི་ཥིཉྩ་ཧཱུྂ།

ཞེས་དབང་བསྐུར། བཛྲཱི་བྷ་ཝ་ཨ་བྷི་ཥིཉྩ་ཧཱུྂ། ཞེས་བུམ་ཆུ་སྦྱིན། ཆུ་ལྷག་མ་ཡར་ལུད་པ་ལས་སྣང་བ་མཐའ་ཡས་ཀྱིས་དབུ་བརྒྱན། དབང་གི་ལྷ་རྣམས་རང་ཉིད་ལ་ཐིམ་པར་གྱུར་ པ་ལ། པུཥྤེ་ སོགས་ཉེར་སྤྱོད་ལྔས་མཆོད། དེ་ལྟར་སྣང་བ་མཐའ་ཡས་རྡོ་རྗེའི་དབང་བསྐུར་བས། འདོད་ཆགས་ཀྱི་དྲི་མ་དག །འདུ་བྱེད་ཀྱི་ཕུང་པོ་གནས་འགྱུར། སོ་སོར་རྟོག་པའི་ཡེ་ཤེས་མངོན་དུ་བྱས། སྣང་བ་མཐའ་ཡས་ཀྱི་ས་བོན་དང་བག་ཆགས་རྒྱུད་ལ་བཞག ། ཅེས་བརྗོད། དྲིལ་བུ་བསངས་སྦྱངས། སྟོང་པའི་ངང་ལས་པད་ཟླའི་གདན་ལ་བདག་དང་དབང་རྫས་རལ་གྲི་ཁཾ་གིས་མཚན་པ་ཡོངས་སུ་གྱུར་པ་ལས། དོན་ཡོད་གྲུབ་པ་ཀྱཻ་རྡོ་རྗེ་ལྗང་ཁུ་ཞལ་བརྒྱད་ཕྱག་བཅུ་དྲུག་པ། ཞབས་བཞི་དང་ལྡན་པ། གཡས་བརྒྱད་ཀྱིས་རལ་གྲི་དང༌། གཡོན་བརྒྱད་ཀྱིས་དྲིལ་བུ་འཛིན་པ། ཡུམ་དམ་ཚིག་སྒྲོལ་མ་དང་སྙོམས་པར་སྦྱོར་བ་རིན་པོ་ཆེ་དང་རུས་པའི་རྒྱན་གྱིས་བརྒྱན་པའི་ཐུགས་ཀའི་ཧཱུྂ་ལས་འོད་ཟེར་འཕྲོས་པས་དབང་གི་ལྷ་རྣམས་སྤྱན་དྲངས༑ དེ་བཞིན་གཤེགས་པ་རྣམས་ཀྱིས་བདག་ལ་མངོན་པར་དབང་བསྐུར་དུ་གསོལ། ཞེས་གསོལ་བ་བཏབ་པས་བྱང་ཆུབ་སེམས་དཔའ་རྣམས་ཀྱིས་བཀྲ་ཤིས་བརྗོད། རིག་མ་རྣམས་ཀྱིས་རྡོ་རྗེའི་གླུ་ལེན། ཁྲོ་བོ་རྣམས་ཀྱིས་བགེགས་བསྐྲད། དབང་གི་ལྷས་དྲིལ་བུ་དང་བུམ་པ་མཉམ་པར་བཟུང་སྟེ་དབང་བསྐུར་བར་གྱུར། དྲིལ་བུ་ཤེས་རབ་ཉིད་ལྡན་པའི༑ ༑འདི་ལྟར་སངས་རྒྱས་ཐམས་ཅད་ཀྱི། །དམ་ཚིག་མཆོག་ཏུ་བསྔགས་པའི་ཕྱིར། །རབ་ཏུ་བསྒྲུབ་ཕྱིར་དྲིལ་བུ་སྦྱིན། །ཨོཾ་བཛྲ་གྷཎྚེ་ཨ་བྷི་ཥིཉྩ་ཧཱུྂ། ཞེས་དབང་བསྐུར། བཛྲཱི་བྷ་ཝ་ཨ་བྷི་ཥིཉྩ་ཧཱུྂ། ཞེས་བུམ་ཆུ་སྦྱིན། ཆུ་ལྷག་མ་ཡར་ལུད་པ་ལས་དོན་ཡོད་གྲུབ་པས་དབུ་བརྒྱན། དབང་གི་ལྷ་རྣམས་རང་ཉིད་ལ་ཐིམ་པར་གྱུར་ པ་ལ། པུཥྤེ་ སོགས་ཉེར་སྤྱོད་ལྔས་མཆོད། དེ་ལྟར་དོན་ཡོད་གྲུབ་པ་དྲིལ་བུའི་དབང་བསྐུར་བས། ཕྲག་དོག་གི་དྲི་མ་དག །འདུ་བྱེད་ཀྱི་ཕུང་པོ་གནས་འགྱུར།

བྱ་བ་གྲུབ་པའི་ཡེ་ཤེས་མངོན་དུ་བྱས། དོན་ཡོད་གྲུབ་པའི་བག་ཆགས་དང་ས་བོན་རྒྱུད་ལ་བཞག ༑ ཅེས་བརྗོད། རང་དང་རྡོར་དྲིལ་བསངས་སྦྱངས། སྟོང་པའི་ངང་ལས་པད་ཟླའི་གདན་ལ་འཁོར་ལོ་བྷྲཱུཾ་གྱིས་མཚན་པ་ཡོངས་སུ་གྱུར་པ་ལས། རང་དང་རྡོ་རྗེ་དྲིལ་བུ་རྣམས་ཀྱཻ་རྡོ་རྗེ་དཀར་པོ་ཞལ་བརྒྱད་ཕྱག་བཅུ་དྲུག་པ། ཞབས་བཞི་དང་ལྡན་པ། གཡས་བརྒྱད་ཀྱིས་འཁོར་ལོ་དང༌། གཡོན་བརྒྱད་ཀྱིས་དྲིལ་བུ་འཛིན་པ། ཡུམ་སངས་རྒྱས་སྤྱན་མ་དང་སྙོམས་པར་སྦྱོར་བ་རིན་པོ་ཆེ་དང་རུས་པའི་རྒྱན་གྱིས་བརྒྱན་པའི་ཐུགས་ཀའི་ཧཱུྂ་ལས་འོད་ཟེར་འཕྲོས་པས་དབང་གི་ལྷ་རྣམས་སྤྱན་དྲངས། དེ་བཞིན་གཤེགས་པ་རྣམས་ཀྱིས་བདག་ལ་མངོན་པར་དབང་བསྐུར་དུ་གསོལ། ཞེས་གསོལ་བ་བཏབ་པས། བྱང་ཆུབ་སེམས་དཔའ་རྣམས་ཀྱིས་བཀྲ་ཤིས་བརྗོད། རིག་མ་རྣམས་ཀྱིས་རྡོ་རྗེའི་གླུ་ལེན། ཁྲོ་བོ་རྣམས་ཀྱིས་བགེགས་བསྐྲད། དབང་གི་ལྷས་བུམ་པ་དང་རྡོར་དྲིལ་མཉམ་དུ་བཟུང་ནས་དབང་བསྐུར་བར་གྱུར། རྡོ་རྗེ་མིང་གིས་དབང་བསྐུར་བས། །རྟག་པའི་རྡོ་རྗེ་རབ་བཞུགས་ཏེ། །མི་ཤིགས་རྡོ་རྗེ་མིང་ལྡན་པས། །བུ་ཁྱོད་དེ་རིང་དབང་བསྐུར་རོ༑ ༑ཨོཾ་བཛྲ་སཏྭ་ཏྭཾ་བཛྲ་ནཱ་མ་ཨ་བྷི་ཥིཉྩ་ཧཱུྂ། ཞེས་དང༌། ཀྱཻ་དཔལ་ཞེས་འདྲི། ཆེ་གེ་མོ་ ཞེས་བྱ་བའི་དེ་བཞིན་གཤེགས་པའོ། །བཛྲཱི་བྷ་ཝ་ཨ་བྷི་ཥིཉྩ་ཧཱུྂ། ཞེས་བུམ་ཆུ་སྦྱིན། ཆུ་ལྷག་མ་གྱེན་དུ་འཁྱིལ་བ་ལས་རྣམ་པར་སྣང་མཛད་ཀྱིས་དབུ་བརྒྱན། དབང་གི་ལྷ་རྣམས་རང་ཉིད་ལ་ཐིམ་པར་གྱུར་ པ་ལ། པུཥྤེ་ སོགས་ཉེར་སྤྱོད་ལྔས་མཆོད། དེ་ལྟར་མིང་གི་དབང་བསྐུར་བས། གཏི་མུག་གི་དྲི་མ་དག །གཟུགས་ཀྱི་ཕུང་པོ་གནས་འགྱུར། མེ་ལོང་ལྟ་བུ་ཡེ་ཤེས་མངོན་དུ་བྱས། རྣམ་པར་སྣང་མཛད་ཀྱི་བག་ཆགས་དང་ས་བོན་རྒྱུད་ལ་བཞག །དེ་ལྟར་བུམ་ཆུའི་བྱ་བ་རྗེས་སུ་འགྲོ་བས་རིག་པའི་དབང་ལྔ་ཐོབ། ཕུང་པོ་ལྔ་སྦྱངས། ཡེ་ཤེས་ལྔ་མངོན་དུ་བྱས། རིགས་ལྔའི་ས་བོན་དང་བག་ཆགས་རྒྱུད་ལ་བཞག

།འཐོབ་བྱའི་འབྲས་བུ་བྱ་སྤྱོད་ཀྱི་རྒྱུད་མཐའ་དག་ཉན་པ་དང༌། འཆད་པ། སྔགས་བསྒྲུབ་པ་ལ་དབང་བར་བྱས་སོ། ། ཞེས་དོན་བཤད། གཏང་རག་གི་མཎྜལ་ཕུལ། དེ་ནས་རྡོ་རྗེ་སློབ་དཔོན་གྱི་དབང་གི་དོན་དུ་མཎྜལ་ཕུལ་ལ། གསོལ་བ་གདབ་པ་ནི། མེ་ཏོག་འཁོར་ལོའི་དབང་བསྐུར་ཆེ། །མགོན་པོས་བདག་ལ་བསྩལ་ནས་ཀྱང༌། །འཁོར་ལོ་ལྷ་ཡི་དེ་ཉིད་དང༌། །སློབ་དཔོན་ཕྲིན་ལས་ཡོངས་སུ་བཟུང༌། །སངས་རྒྱས་ཀུན་གྱི་དམ་ཚིག་དང༌། །སྡོམ་པ་བླ་ན་མེད་པ་སྩོལ། །རྡོ་རྗེ་སློབ་དཔོན་རྒྱལ་སྲིད་ཆེ། །བདག་པོས་དབང་བསྐུར་བདག་ལ་སྩོལ། ། ཅེས་གསུམ་བརྗོད། རང་ཉིད་དབང་སྟེགས་པདྨ་འདབ་མ་བརྒྱད་པའི་སྟེང་དུ། སྐད་ཅིག་གིས་རྡོ་རྗེ་སེམས་དཔའ་སྐུ་མདོག་དཀར་པོ། རྡོ་རྗེ་དང་དྲིལ་བུ་འཛིན་པས་ཡུམ་རྡོ་རྗེ་སྙེམས་མ་སྔོན་མོ་གྲི་ཐོད་འཛིན་པ་ལ་འཁྱུད་པ། རུས་པ་དང་རིན་པོ་ཆེས་བརྒྱན་པའི་སྟེང་དུ་གདུགས་དཀར་པོ་གསེར་གྱི་ཡུ་བ་ཅན་དང༌། ་་གཡས་སུ་རྒྱལ་མཚན། གཡོན་དུ་བ་དན་རྣམས་གསལ་བར་གྱུར། ཅེས་དམིགས་ཏེ། རྡོ་རྗེ་ཧཱུྂ་ལས་བྱུང་བའི་རྡོ་རྗེར་གྱུར། ཐོག་མཐའ་མེད་པའི་སེམས་དཔའ་ནི༑ ༑རྡོ་རྗེ་སེམས་དཔའ་དགའ་བ་ཆེ། །ཀུན་ཏུ་བཟང་པོ་ཀུན་བདག་ཉིད། །རྡོ་རྗེ་སྙེམས་ཚུལ་རྡོ་རྗེའི་བདག །བཅོམ་ལྡན་འདས་དཔལ་མཆོག་གི་དང་པོའི་སྐྱེས་བུའོ། །ཞེས་རྡོ་རྗེ་གཡས་སུ་སྦྱིན། ་དྲིལ་བུ་ཨཿལས་བྱུང་བའི་དྲིལ་བུར་གྱུར། འདི་དེ་སངས་རྒྱས་ཐམས་ཅད་ཀྱི། །ཤེས་རབ་དབྱངས་ཀྱི་རྗེས་འགྲོར་བཤད། །ཁྱོད་ཀྱིས་ཀྱང་ནི་རྟག་ཏུ་བཅང༌། །བྱང་ཆུབ་མཆོག་ཏུ་རྒྱལ་བས་བཤད། །

ཅེས་གཡོན་དུ་སྦྱིན། ཡིད་ཀྱི་ལུས་ནི་བརྟན་པའི་ཕྱིར། །ཕྱག་རྒྱའི་དམ་ཚིག་རབ་ཏུ་བརྗོད། །གང་གི་ལུས་ནི་ཐམས་ཅད་བརྟན། །དེ་ནི་ཕྱག་བྱར་རབ་ཏུ་གྲགས། །མི་ཤིགས་ཕྱག་རྒྱར་བྱིན་བརླབས་ནས། །འདོད་པ་ཐམས་ཅད་ལོངས་སྤྱོད་དོ། །དངོས་གྲུབ་ཐམས་ཅད་བསྒྲུབ་པར་བྱ། །རང་ཉིད་རྡོ་རྗེ་སེམས་དཔའ་དང་ཡུམ་རྡོ་རྗེ་སྙེམས་མ་གཉིས་ཙུམྦ་ནའི་སྦྱོར་བས། བདེ་བ་དང་བཅས་པའི་ཏིང་ངེ་འཛིན་རྒྱུད་ལ་སྐྱེ་བར་གྱུར། ཤེས་རབ་བཅུ་དྲུག་ལོན་པ་ལ། །ལག་པ་དག་གིས་ཡང་དག་འཁྱུད། །རྡོ་རྗེ་དྲིལ་བུ་མཉམ་སྦྱོར་བ། །སློབ་དཔོན་གྱི་ནི་དབང་དུ་འདོད། །སླར་ཡང་རང་གི་ཐུགས་ཀ་ནས་འོད་ཟེར་འཕྲོས་དབང་གི་ལྷ་སྤྱན་དྲངས། དེ་བཞིན་གཤེགས་པ་ཐམས་ཅད་ཀྱིས་བདག་ལ་མངོན་པར་དབང་བསྐུར་དུ་གསོལ། ཞེས་གསོལ་བ་བཏབ་པས། བྱང་ཆུབ་སེམས་དཔའ་རྣམས་ཀྱིས་བཀྲ་ཤིས་བརྗོད། རིག་མ་རྣམས་ཀྱིས་རྡོ་རྗེའི་གླུ་ལེན། ཁྲོ་བོ་རྣམས་ཀྱིས་བགེགས་བསྐྲད། དབང་གི་ལྷས་རྡོར་དྲིལ་དང་བུམ་ཆུ་བསྣམས་ནས་དབང་བསྐུར་བར་གྱུར། རྡོ་རྗེ་ཆེ་དང་དྲིལ་བུ་ཆེ། །ཟུངས་ཤིག་རྡོ་རྗེ་བྱིན་གྱིས་བརླབས། །དེ་རིང་རྡོ་རྗེ་སློབ་དཔོན་གྱུར། །སློབ་མའི་ཚོགས་ནི་བསྡུ་བར་གྱིས། །ཨོཾ་བཛྲཱི་བྷ་ཝ་ཨ་བྷི་ཥིཉྩ་ཧཱུྂ། ཞེས་བུམ་ཆུ་སྦྱིན། འདི་དེ་ཕྱི་རོལ་ཡུལ་སྣང་ཀུན། །དག་ཕྱིར་གཞལ་ཡས་ཁང་དུ་ཤར། །མཚན་ཉིད་ཐམས་ཅད་ཡོངས་སུ་རྫོགས། །སྣ་ཚོགས་པདྨ་འདབ་བརྒྱད་སྟེང༌། །སྤྲུལ་པའི་ལྷ་དགུ་ཡབ་དང་བཅས། །སོ་སོར་བཞུགས་ཀྱང་དོན་གྱིས་ནི། །རྡོ་རྗེ་སློབ་དཔོན་ལུས་ཀྱི་ནང༌། །རྒྱལ་བའི་སྐུ་ནི་རིམ་བཞིན་གནས། །ཀུན་གྱི་ངོ་བོ་ཧེ་རུ་ཀཿ ༑དེ་ལྟར་ཆུ་ལྷག་མ་གྱེན་དུ་འཁྱིལ་བ་ལས། རྡོ་རྗེ་སེམས་དཔས་དབུ་བརྒྱན། དབང་གི་ལྷ་རྣམས་རང་ཉིད་ལ་ཐིམ་པར་གྱུར་ པ་ལ། པུཥྤེ་ སོགས་ཉེར་སྤྱོད་ལྔས་མཆོད། ཆོས་རྣམས་གཟུགས་བརྙན་ལྟ་བུ་སྟེ། ་༌།གསལ་ཞིང་དག་ལ་རྙོག་པ་མེད། །གཟུང་དུ་མེད་ཅིང་བརྗོད་དུ་མེད། །རྒྱུ་དང་ལས་ལ་ཀུན་ཏུ་འབྱུང༌། །ངོ་བོ་ཉིད་མེད་གནས་མེད་པ། །དེ་ལྟར་ཆོས་རྣམས་ཤེས་གྱིས་ལ༑

༑སེམས་ཅན་དོན་ནི་མཉམ་མེད་བྱོས། །འོན་ཀྱང་སྐྱོབ་པའི་སྲས་སུ་སྐྱེ། །དེ་ལྟར་ཆོས་རྣམས་ཐམས་ཅད་མེ་ལོང་གི་གཟུགས་བརྙན་དང་འདྲ་བར་གྱིས་ཤིག །སེམས་ཅན་གྱི་དོན་གྱིས་ཤིག ། ཅེས་བརྗོད་དོ། ། དེ་ནས་རང་ཉིད་ཀྱཻ་རྡོ་རྗེ་སྐུ་མདོག་སྔོན་པོ་ཕྱག་གཉིས་རྡོར་དྲིལ་འཛིན་པ། ཡུམ་བདག་མེད་མ་ལ་འཁྱུད་པ། རུས་པ་དང་རིན་པོ་ཆེས་བརྒྱན་པའི་སྐུར་གསལ་བར་གྱུར་པར་ དམིགས་ལ། ཌཱ་རུ། རྡོར་དྲིལ། ཚངས་སྐུད། ཐོད་པ། ཁ་ཊྭཱཾ་གཏད་ལ། ཁ་ཊྭཱཾ་ཅང་ཏེའུ་ལྷུང་བཟེད་འདི། །ཀྱཻ་ཡི་རྡོ་རྗེ་ཁ་ཊྭཱཾ་ག །བདག་མེད་མ་ནི་ཌཱ་མ་རུ། །རྡོ་རྗེ་དྲིལ་བུ་མཁའ་འགྲོ་བརྒྱད། །ཚངས་པའི་སྐུད་པ་ཀ་པཱ་ལ། །དཔའ་བོ་ཆེན་པོས་བླང་བར་བགྱི། ། ཞེས་སྦྱིན། དེ་ནས། གཙོ་བོའི་ཐུགས་ཀ་ནས་སྔགས་ཕྲེང་བྱུང་ཞལ་ནས་ཐོན་རང་གི་ཞལ་དུ་ཞུགས། ཐུགས་ཀའི་སྔགས་ཕྲེང་ལ་ཐིམ་པར་གྱུར། ཅེས་དང༌། སྔགས་རྣམས་ལན་གསུམ་བརྗོད་ལ། བཅོམ་ལྡན་བདག་གིས་བཟུང་ལགས་ཀྱི། །བདག་ལ་ཉེ་བར་མཛད་དུ་གསོལ། ། ཞེས་བརྗོད། དེ་ནས་ལག་པ་གཡོན་པས་ཆོས་གོས་གྲྭ་གཉིས་དང་བཅས་པའི་དྲིལ་བུ་སྙིང་ཁར་བཟུང་ལ། གཡས་མི་འཇིགས་པའི་ཕྱག་རྒྱ་བཅས་ཏེ། འདི་སྐད་དུ། རྡོ་རྗེ་སློབ་དཔོན་དང་དཀྱིལ་འཁོར་གྱི་ལྷ་ཚོགས་ཀྱིས་མགྲིན་གཅིག་ཏུ་ལུང་བསྟན་པར་གྱུར། ཨོཾ། འདིར་ནི་ཁྱོད་ལ་ངས་ལུང་བསྟན། །རྡོ་རྗེ་སེམས་དཔའ་དེ་བཞིན་གཤེགས། །སྲིད་པ་ཤིན་ཏུ་དག་པས་ན། །སྲིད་པའི་ངན་འགྲོ་རྣམས་ལས་བཏོན། ། ཆེ་གེ་མོ་ ཞེས་བྱ་བའི་དེ་བཞིན་གཤེགས་པའོ། །སིདྡྷི་ས་མ་ཡ་སྟྭཾ་བྷུརྦཧུ་ཝཿསྭཿ གང་ལ་ངས་ལུང་བསྟན་པ་འདི་ལ་སངས་རྒྱས་དང་བྱང་ཆུབ་སེམས་དཔའ་རྣམས་དང་དཀྱིལ་འཁོར་གྱི་ལྷ་ཚོགས་ཐམས་ཅད་ཀྱིས་མགྲིན་གཅིག་ཏུ་བྱང་ཆུབ་ཆེན་པོར་ལུང་བསྟན་པར་མཛད་དོ། །སྔགས་འདི་དང་ཕྱག་རྒྱ་འདིའི་མཐུ་ཡིན་ནོ། ། ཞེས་བརྗོད། པུཥྤེ་ སོགས་ཉེར་སྤྱོད་ལྔས་མཆོད། དེ་ནས་མཐའ་རྟེན་གྱི་རྗེས་གནང་སྦྱིན་པ་ནི།

མེ་ལོང་གཏད་ལ། ཨཱཿལས་བྱུང་བའི་མེ་ལོང་འདི། །རྡོ་རྗེ་སེམས་དཔའི་མེ་ལོང་སྟེ། །གསལ་ཞིང་དག་ལ་རྙོག་པ་མེད། །བུ་ཁྱོད་ཀྱི་ནི་སྙིང་ལ་ཞུགས། །སྐྱོབ་པ་རྣམས་ཀྱི་སྲས་སུ་སྐྱེ། ། ཞེས་དང༌། མདའ་གཞུ་གཏད་ལ། ཧོཿལས་བྱུང་བའི་མདའ་གཞུ་འདི། །ཆོས་ཉིད་གཞུ་ལ་ཡེ་ཤེས་མདའ། །ཕྱོགས་མཚམས་སྟེང་འོག་ཀུན་ལ་འཕེན། །རྒྱལ་བ་སྲས་དང་བཅས་པ་རྣམས། །ཐམས་ཅད་རྗེས་སུ་མཉེས་པར་གྱིས། ། ཞེས་དང༌། འཁོར་ལོ་གཏད་ལ། བྷྲཱུཾ་ལས་བྱུང་བའི་འཁོར་ལོ་འདི། །ཨོཾ་བཛྲ་ཧེ་ཏུ་མཱཾ། །དེང་ནས་བརྩམས་ཏེ་སེམས་བསྐྱེད་པ། །བྱ་བ་སྤྱོད་ཚུལ་གསང་སྔགས་ཀྱི། །ཆོས་ཀྱི་འཁོར་ལོ་བསྐོར་བར་གྱིས། ། ཞེས་དང༌། དུང་གཏད་ལ། ཨཱཿལས་བྱུང་བའི་དུང་དཀར་འདི། །ཨོཾ་བཛྲ་བྷཱ་ཥ་རཾ། །ཁྱོད་ཀྱིས་ཡིད་གཉིས་སྤངས་བྱས་ཏེ། །ཆོས་ཀྱི་དུང་ནི་བླ་ན་མེད། །དྲི་མེད་ཀུན་ཏུ་གང་བར་གྱིས།་་༌། ཞེས་དང༌། པོ་ཏི་གཏད་ལ། ཨཱཿལས་བྱུང་བའི་པོ་ཏི་ནི། །གསང་སྔགས་ཀུན་ལ་རྟག་ཏུ་བཤད། །དེ་ལྟར་བྱས་ན་སངས་རྒྱས་ལ། །ཕན་འདོགས་དྲིན་དུ་གཟོ་ཞེས་བྱ། །རྡོ་རྗེ་འཛིན་ཀུན་ཁྱོད་ལ་སྲུང༌། ། ཞེས་དང༌། དྲིལ་བུ་གཏད་ལ། ཨཿལས་བྱུང་བའི་དྲིལ་བུ་འདི།་་༌།ཐམས་ཅད་ནམ་མཁའི་མཚན་ཉིད་དེ། །ནམ་མཁའ་ལ་ཡང་མཚན་ཉིད་མེད། །ནམ་མཁའ་སྟོང་པ་ཉིད་སྦྱོར་བ། །ཀུན་མཆོག་མཉམ་པ་ཉིད་དུ་གསལ། ། ཞེས་བརྗོད། དེ་དག་གིས་རྡོ་རྗེ་སློབ་དཔོན་གྱི་དབང་འདོགས་བཅས་གྲུབ་པས་གཏང་རག་གི་མཎྜལ་ཕུལ། ། དེ་ནས། གསང་དབང་གི་མཎྜལ་ཕུལ་ནས༑ གསོལ་བ་གདབ་པ་ནི། ཁྱོད་ནི་རྣལ་འབྱོར་མ་ཚོགས་ཡོངས་སུ་བསྐོར། །ཁྱོད་ནི་དེ་ཉིད་ཐུགས་ཀྱི་མཆོག་བརྙེས་པ། །ཁྱོད་ཀྱིས་རྒྱུ་དང་མི་རྒྱུའི་འགྲོ་བ་ཀུན་བརྒྱན་ཏེ། །སྙིང་རྗེའི་ཐུགས་ཀྱིས་སེམས་ཅན་སད་མཛད་པ། །སྒྱུ་མའི་སྤྲོས་པ་རྡོ་རྗེ་བསྒྲུབ་པའི་ཡིད། །ཧེ་རུ་ཀ་དཔལ་རྡོ་རྗེ་ཡིད་ཀྱིས་དགྱེས། ། ཞེས་གསུམ་བརྗོད། མིག་དར་བཅིངས་ལ།

སློབ་དཔོན་དང་དཀྱིལ་འཁོར་གྱི་གཙོ་བོ་ཐ་མི་དད་པའི་བསྲུབས་པ་དང་བསྲུབ་པའི་སྦྱོར་བ་མཛད་པའི་སྒྲས། ཧ་ཧ་སི་སི་ཀུ་རུ་བཛྲ་དྡྷྲྀཀ། ཅེས་པའི་སྒྲས་བསྐུལ་ཏེ། ཕྱོགས་བཅུའི་སངས་རྒྱས་ཐམས་ཅད་སྤྱན་དྲངས་ཡུམ་གྱི་ཞལ་དུ་ཞུགས། སྙིང་ཁར་རྗེས་ཆགས་ཀྱི་མེས་འོད་དུ་ཞུ། བྱང་ཆུབ་སེམས་ཀྱི་ངོ་བོར་གྱུར་པ་མཁའ་གསང་ནས་ཐོན་ཏེ། ཐོད་པའི་བདུད་རྩི་དང་གཉིས་སུ་མེད་པར་གྱུར་ པར་དམིགས་ལ། བདུད་རྩི་བྱིན་ལ། ཇི་ལྟར་འདས་པའི་སངས་རྒྱས་ཀྱིས། །བྱང་ཆུབ་སྲས་རྣམས་དབང་བསྐུར་ལྟར། །བདག་གིས་གསང་བའི་དབང་བསྐུར་གྱི། །བུ་ཁྱོད་དེ་ལྟར་དབང་བསྐུར་རོ༑ ༑ ཞེས་བརྗོད་ཅིང༌། ཨ་ཧོཿམ་ཧཱ་སུ་ཁ། ཞེས་ལྕེ་ཐོག་ཏུ་མྱང༌། ཨོཾ་ཨཱཿཧཱུྂ་ཕཊ། ཅེས་མིག་དབྱེ། པུཥྤེ་ སོགས་ཉེར་སྤྱོད་ལྔས་མཆོད། དེ་དག་གིས་བླ་མ་ཡབ་ཡུམ་གྱི་མཁའ་གསང་ལ་བརྟེན་ནས་གསང་བའི་དབང་ཐོབ། ངག་གི་དྲི་མ་དག །ལམ་ཙཎྜ་ལཱི་བསྒོམ་པ་ལ་དབང༌། འབྲས་བུ་ལོངས་སྤྱོད་རྫོགས་པའི་སྐུ་འགྲུབ་པར་བྱས་སོ། ། ཞེས་དོན་བཤད། གཏང་རག་གི་མཎྜལ་ཕུལ། དེ་ནས་ཤེས་རབ་ཡེ་ཤེས་ཀྱི་དབང་གི་དོན་དུ་མཎྜལ་ཕུལ་ཏེ། གསོལ་བ་གདབ་པ་ནི། བཅོམ་ལྡན་ཞི་བ་ཆེན་པོ་ཀྱེ། །རྡོ་རྗེ་རྣལ་འབྱོར་གཅིག་བཞེད་པ། །མི་ཕྱེད་ཕྱག་རྒྱ་སྒྲུབ་པ་པོ། །རྡོ་རྗེ་རྣལ་འབྱོར་ལས་བྱུང་བ། །ཇི་ལྟར་ཁྱོད་ནི་བདག་ཉིད་ཅན། །མངའ་བདག་བདག་ལའང་དེ་ལྟར་མཛོད། །འཁོར་བ་འདམ་གྱི་ཚོགས་ནང་དུ། །བྱིང་བ་སྐྱབས་མེད་བདག་ལ་སྐྱོབས། ། ཞེས་གསུམ་བརྗོད། བླ་མ་དང་དཀྱིལ་འཁོར་གྱི་གཙོ་བོ་ཐ་མི་དད་པས་བདག་མེད་མ་བཅུ་དྲུག་ལོན་པའི་ན་ཚོད་ཅན་རང་ལ་གཏད་ཅིང༌། དགའ་སྟེར་གནས་སྟེར་ལྷ་མོ་ནི། །སྣ་ཚོགས་གཟུགས་ཅན་ཡིད་འོང་མ། །ཁྱེར་ཅིག་ཁྱེར་ཅིག་སེམས་དཔའ་ཆེ། །ཁྱེར་ལ་མཆོད་པ་རྟག་ཏུ་གྱིས། ། ཞེས་གནང་བ་ལ། ལྷ་མོས། བཤང་གཅི་དག་ནི་ཟ་བ་དང༌། །ཁྲག་དང་ཁུ་བ་དེ་བཞིན་ཏེ། །ཤ་ཆེན་བསྟེན་པར་བུ་སྤྲོའམ། །བུད་མེད་རྣམས་ལ་རྟག་གུས་དང༌། །བྷ་གའི་པདྨར་འོ་བྱེད་པ། །བུ་ཁྱོད་ཇི་ལྟར་དགའ་བ་སྨྲོས། །

ཞེས་དྲིས་བ་ལ། རང་གིས། ཁུ་ཁྲག་ལ་སོགས་ཟ་བ་དང༌། །བུད་མེད་རྣམས་ལ་རྟག་གུས་དང༌། །བྷ་གའི་པདྨར་འོ་བྱེད་པ༑ ༑ལྷ་མོ་ཅི་ཕྱིར་བདག་མི་སྤྲོ། ། ཞེས་བརྗོད་ལ། ལྷ་མོས་པདྨ་རྒྱས་པ་ཁ་ཕྱེས་ནས་བསྟན་ཏེ། ཨེ་མ་བདག་གི་པདྨ་འདི། །བདེ་བ་ཐམས་ཅད་ཡང་དག་ལྡན། །གང་ཞིག་ཆོ་ག་བཞིན་བསྟེན་ན། །དེ་ཡི་མདུན་དུ་བདག་གནས་སོ། །གསང་གནས་བསྟན་པ་ལ་སོགས་པ། །པདྨར་སྦྱར་བ་ཇི་བཞིན་ཏེ། །བདེ་ཆེན་རྒྱལ་པོ་རང་ཉིད་ནི། །འདི་ཉིད་དུ་ནི་རྟག་ཏུ་བཞུགས། །བླུན་པོ་གང་ཞིག་འདི་སྤངས་པ། །དེ་ལ་མཆོག་གི་དངོས་གྲུབ་མེད། །བྷཉྫ་མོཀྵ་ཧོཿ ཞེས་བརྗོད། རང་གི་པང་དུ་བྱུང་བར་བསམས་ཏེ། རང་ཉིད་ཀྱཻ་རྡོ་རྗེར་གསལ་བའི་ཡུམ་དང་སྦྱོར་བ་མཛད་པས༑ བདེ་སྟོང་ལྷན་ཅིག་སྐྱེས་པའི་དགའ་བས་ལུས་ཐམས་ཅད་ཁྱབ་པར་གྱུར། མཚན་ལྡན་བུད་མེད་མཐོང་བ་དང༌། །གཡོན་ནས་ལན་གསུམ་བསྐོར་བྱས་ཏེ། །ཐལ་སྦྱར་ནས་ནི་གསོལ་བ་བཏབ༑ ༑འདོད་ཆགས་བྲལ་བ་ལྟ་བུའི་སྡིག །ཁམས་གསུམ་ལ་ཡང་གཞན་མེད་དོ། །དེ་བས་འདོད་ལ་ཆགས་བྲལ་བ། །ཁྱོད་ནི་ནམ་ཡང་མི་བྱའོ། །མ་ཧཱ་ས་མ་ཡ་ཧ་ན་ཧཱུྂ་ཕཊ། ཅེས་དམ་ཚིག་བསྒྲགས་ལ། ཨོཾ་ཨཱཿཧཱུྂ་ གིས་མིག་དབྱེ། པུཥྤེ་ སོགས་ཉེར་སྤྱོད་ལྔས་མཆོད། དེ་དག་གིས་བྷ་གའི་དཀྱིལ་འཁོར་མཐོང༌། ཤེས་རབ་ཡེ་ཤེས་ཀྱི་དབང་ཐོབ། ཡིད་ཀྱི་དྲི་མ་དག །ལམ་ཕྱག་རྒྱ་ཆེན་པོ་བསྒོམ་པ་ལ་དབང༌། འབྲས་བུ་ཆོས་སྐུ་འགྲུབ་པའི་སྐལ་པ་དང་ལྡན་པར་བྱས་སོ། ། ཞེས་བརྗོད། གཏང་རག་གི་མཎྜལ་ཕུལ། དེ་ནས་དབང་བཞི་པའི་དོན་དུ་མཎྜལ་ཕུལ་ཏེ། གསོལ་བ་གདབ་པ་ནི། བདག་ཉིད་ཆེ་ཁྱོད་བཀའ་དྲིན་གྱིས། །གཙོ་བོའི་དབང་བསྐུར་རྣམ་གསུམ་ཐོབ། །དེ་རིང་རིན་ཆེན་བཞི་པ་ཡང༌། །བཀའ་དྲིན་གྱིས་ནི་བསྩལ་དུ་གསོལ། ། ཞེས་གསུམ་བརྗོད་གདོང་གཡོག་བཅིངས་ལ། འདི་ནི་དབང་གསུམ་པའི་དུས་སུ་ཉམས་སུ་མྱོང་པའི་ལྷན་ཅིག་སྐྱེས་པའི་བདེ་བ་བརྗོད་དུ་མེད་ཅིང་སྣང་ལ་རང་བཞིན་མེད་ཅིང་གསལ་ལ་རྫོགས་པ།

དབང་པོའི་ཚོན་དང་ཆུ་ཟླ་ལྟ་བུ་སྟེ། །ཡེ་ཤེས་འདི་ནི་ཆེས་ཕྲ་ཞིང༌། །རྡོ་རྗེ་ནམ་མཁའི་དཀྱིལ་ལྟ་བུ། །རྡུལ་བྲལ་མཐར་ཕྱིན་ཞི་བ་ཉིད། །ཁྱོད་རང་ཡང་ནི་དེ་ཡི་ཕ། །ཞེས་པའི་དོན་གཟུང་འཛིན་བྲལ་བའི་སེམས་དེ་དབང་བཞི་པའི་ངོ་བོ་ཡིན་ནོ༑ ༑ ཞེས་ངོ་སྤྲད། འདི་ནི་རང་རིག་མཛེས་མ་སྟེ། །རིག་པའི་བརྟུལ་ཞུགས་ཆེད་གཅིག་པུ། །རིགས་འདི་བོར་ནས་དེ་ལ་ནི། །གཞན་པའི་ཐབས་ཀྱི་སངས་རྒྱས་མེད། །ཁྱོད་ཀྱིས་སྲོག་ཆགས་གསད་པར་བྱ། །མ་བྱིན་པ་ཡང་ཁྱོད་ཀྱིས་ལོངས། །ཐམས་ཅད་ལ་ཡང་རྫུན་ཚིག་སྨྲོས། །བརྟུལ་ཞུགས་རྣམས་ཀྱི་བླ་མེད་པ། །མཚན་མེད་བསམ་གྱིས་མི་ཁྱབ་པ། །འདི་ལ་རྟག་ཏུ་གནས་པར་གྱིས། ། ཞེས་བརྗོད། ཨོཾ་ཨཱཿཧཱུྂ་ གིས་མིག་དབྱེ། པུཥྤེ་ སོགས་ཉེར་སྤྱོད་ལྔས་མཆོད། དེ་དག་གིས་དོན་དམ་བྱང་ཆུབ་སེམས་ཀྱི་དཀྱིལ་འཁོར་དུ་དབང་བཞི་པ་ཐོབ། སྒོ་གསུམ་གྱི་དྲི་མ་དག །ལམ་བདེན་གཉིས་དབྱེར་མེད་ཟུང་འཇུག་གི་བསྒོམ་པའི་རིམ་པ་ལ་དབང༌། འབྲས་བུ་ངོ་བོ་ཉིད་ཀྱི་སྐུ་འགྲུབ་པའི་སྐལ་པ་ལྡན་པར་བྱས་སོ། ། ཞེས་དོན་བཤད། གཙོ་བོས་ཇི་ལྟར་བཀའ་བསྩལ་པ། ། ཞེས་ནས། ཆ་ཡིས་ཀྱང་ནི་སྤྱད་དུ་གསོལ། ། ཞེས་པའི་བར་བརྗོད། གཏང་རག་གི་མཎྜལ་ཕུལ། དེ་དག་གིས་དབང་གི་རིམ་པ་ལེགས་པར་གྲུབ་པའོ། །དེ་ནས་སྐབས་འདིར་ལྷག་ཆད་ཁ་བསྐང་བའི་ཕྱིར། ལས་བཞིའི་སྦྱིན་སྲེག་ཀྱང་བྱའོ། །དེ་ནས་ཚོགས་རྫས་ཤ་ཆང་ལ་སོགས་པའི་ཡོ་བྱད་རྣམས་བསངས་སྦྱངས། སྟོང་པའི་ངང་ལས། བདུད་རྩི་བྱིན་གྱིས་བརླབས་ལ། ཨོཾ་ཨཱཿཧཱུྂ་ གསུམ་བརྗོད། དེ་ནས་ལག་པ་གཡོན་གྱི་སྟེང་དུ་ཚོམ་བུ་དགུ་བཀོད་ལ། བསངས་སྦྱངས། སྟོང་པའི་ངང་ལས། ལག་མཐིལ་པདྨ་དམར་པོ་འདབ་མ་བརྒྱད་པའི་ལྟེ་བར་པད་ཉིའི་གདན་ལ། ཀྱཻ་རྡོ་རྗེ་ཞལ་གཅིག་ཕྱག་གཉིས་སྐུ་མདོག་སྔོན་པོ་ཕྱག་གཉིས་ཐོད་པ་འཛིན་པ། ཡུམ་བདག་མེད་མ་ལ་འཁྱུད་པ། འདབ་མ་བརྒྱད་ལ། མདུན་དུ་གཽ་རཱི་མ། གཡས་སུ་ཙཽ་རཱི་མ།

རྒྱབ་ཏུ་བཻ་ཏཱ་ལཱི། གཡོན་དུ་གྷསྨ་རཱི། བྱང་ཤར་པུཀྐ་སཱི། ཤར་ལྷོར་རི་ཁྲོད་མ། ལྷོ་ནུབ་ཏུ་གདོལ་པ་མོ། ནུབ་བྱང་དུ་གཡུང་མོ་རྣམས་གསལ་བའི་ཐུགས་ཀའི་ཧཱུྂ་ལས་འོད་ཟེར་འཕྲོས། ཡེ་ཤེས་པ་སྤྱན་དྲངས་གཉིས་སུ་མེད་པར་གྱུར། ཆུ་བཞི་ཉེར་སྤྱོད་རོལ་མོ་དང་བཅས་པས་མཆོད་ལ། ཧེ་རུ་ཀ་དཔལ་བདག་མེད་མ། །སྤྲུལ་པའི་ལྷ་མོ་བརྒྱད་ཀྱིས་བསྐོར། །ཀུན་ནས་དགྱེས་པ་བསྐྱེད་པའི་ཕྱིར། །ཤ་ལྔ་བདུད་རྩི་ཁུ་ཁྲག་དང༌། །བཤང་གཅི་གོ་རོ་ཙ་ན་སོགས༑ ༑ག་ཎ་ཙཀྲའི་མཆོད་པ་འབུལ། །བཞེས་ནས་བདག་ལ་དགྱེས་པར་མཛོད། ། ཅེས་དབྱངས་གསལ་བརྗོད། ཡིག་བརྒྱས་ལྷག་ཆད་ཁ་བསྐངས། ཨོཾ། ནམ་མཁའ་སྐྱེ་བ་མཚན་མ་མེད། །ཐོག་མ་ཐ་མ་མེད་པའི་ཕྱིར། །སེམས་དཔའ་རྡོ་རྗེ་ཆེ་རང་བཞིན། །ཀྱཻ་རྡོ་རྗེ་བདག་འགྲུབ་མཛོད། ། ཅེས་ལན་གསུམ་བརྗོད། ལག་པར་གནས་པའི་ལྷ་ཚོགས་རང་ཉིད་ལ་ཐིམ་པར་གྱུར། ཅེས་བརྗོད། ལས་ཀྱི་རྡོ་རྗེས་ཚོགས་དྲངས་ཤིང༌། གཟིགས་ཤིག་མཛེས་ལྡན་དམ་པའི་ཆོས། ། ནས། རང་བཞིན་གཅིག་ཏུ་དགོངས་ཏེ་རོལ། ། སོགས་བརྗོད། མཆེད་ལྕམ་རྣམས་ཀྱིས། འདོད་ཆགས་དྲི་མ་དང་བྲལ་བ། །རིན་ཐང་མེད་པ་ལེགས་འོང་ཆོས། །གཟུང་དང་འཛིན་པ་རྣམ་པར་སྤངས། །ཇི་བཞིན་བསྐྱེད་པར་བདག་འཚལ་ལོ། ། ཞེས་བརྗོད་ཅིང་བཟའ་བར་བྱའོ། །དེ་ནས་མཆེད་ལྕམ་རྣམས་ཀྱིས་རྡོ་རྗེའི་གླུ་བླང་བ་ནི། འདི་སྐད་དོ། ། ཀྱེ་ཧོཿཚོགས་སུ་ཅང་ཏེའུ་འཁྲོལ། །སྙིང་རྗེའི་ཀུ་ཅོ་མི་བྱའོ། །དེ་ནི་ཤ་རུ་བཟའ་བར་བྱ། །འབད་པས་ཆང་ནི་བཏུང་བར་བྱ། ཀྱེ་ཧོཿསྐལ་ལྡན་འཇུག་པར་བྱ། །སྐལ་མེད་རྣམས་ནི་སྤང་བ་བྱ༑ ༑ག་པུར་རྣམས་ནི་སྤང་བར་བྱ། །ཚོད་མ་དང་ནི་ཤ་ཆེན་པོ། །དེ་ནི་མང་དུ་བཟའ་བར་བྱ། །འོང་བ་དང་ནི་འགྲོ་བར་བྱ། །དག་དང་མ་དག་མི་དམིགས་བྱ། །རུས་པའི་རྒྱན་ནི་ལུས་ལ་གདགས། །དེར་ནི་རུམ་དུ་གཞུག་པར་བྱ། །འདུ་བར་འཁྲིག་ལ་འཇུག་པ་བྱ། །དེར་ནི་རིག་མིན་སྤང་བར་བྱ། །དགག་སྒྲུབ་བྲལ་སླད་མ་བཅོས་པ། །ལམ་འདི་འགྲུབ་ཕྱིར་ཨ་ལ་ལ། །

ཞེས་བརྗོད། ལྷག་མ་བསྡུས་ལ་རྡོ་རྗེ་སློབ་དཔོན་གྱི་ཁ་ཆུས་བྲན། ཨ་ཀཱ་རོ་ གསུམ་གྱིས་བྱིན་གྱིས་བརླབས། ཨོཾ་སརྦ་ཁ་ཁ་ཁཱ་ཧི་ཁཱ་ཧི་ཨུཙྪིཥྚ་བ་ལིངྟ་བྷཀྵ་བྷྱཿསྭཱ་ཧཱ། ཞེས་བརྗོད་ཅིང་རོལ་མོ་དང་བཅས་གཙང་སར་བསྐྱལ་ལོ། །དེ་ནས་རྗེས་ཀྱི་ཆོ་ག་ནི། མཆོད་པ་སྔར་ལྟར་བྱིན་གྱིས་བརླབས་ལ། ཆུ་བཞི་ཉེར་སྤྱོད་རོལ་མོ་དང་བཅས་པས་མཆོད་ལ། ཧེ་རུ་ཀ་དཔལ་དབང་ཕྱུག་བཙུན་མོར་བཅས། །དཔའ་བོ་རྣལ་འབྱོར་མ་རྣམས་མཉེས་པའི་ཕྱིར། །གཟུང་འཛིན་བྲལ་བའི་མཆོད་ཡོན་འབུལ་ལགས་ན། །བཞེས་ནས་བདག་ལ་བདེ་བའི་མཆོག་སྩོལ་མཛོད། །ཨོཾ་སརྦ་ཏ་ཐཱ་ག་ཏ་ཨརྒྷཾ་པྲ་ཏཱིཙྪ་སྭཱ་ཧཱ། ཞེས་དང༌། མཆོད་ཡོན་གྱི་གནས་སུ། ཞབས་བསིལ། བསང་གཏོར། ཞལ་བསིལ། མེ་ཏོག །བདུག་སྤོས། སྣང་གསལ། དྲི་ཆབ། ཞལ་ཟས། སྒྲ་སྙན། ཞེས་བསྒྱུར། ཨརྒྷཾ ་ སོགས་ཀྱི་མལ་དུ། པཱ་དྱཾ། ཨཱ་ཙ་མཱ་ནཾ། པྲོཀྵ་ཎཾ། པུཥྤེ། དྷཱུ་པེ། ཨཱ་ལོ་ཀེ། གནྡྷེ། ནཻ་ཝི་དྱེ། ཤཔད། ཞེས་སྦྱར་ལ་མཆོད། བསྟོད་པ་ནི། ཧཱུྂ བདེ་སྟོང་དབྱེར་མེད་ཁྲག་འཐུང་ཧེ་རུ་ཀཿ །མཆེ་གཙིགས་དྲག་བྱེད་བཻ་ཌཱུརྻ་ལྟ་བུ། །དགའ་བཞིའི་བདག་མེད་ཡུམ་དང་གཉིས་མེད་སྦྱོར། །ཉིད་སྤྲུལ་ལྷ་མོ་བརྒྱད་དང་གཞལ་ཡས་ཁང༌། །སྲུང་འཁོར་འོད་ཕྲེང་པད་རྡོར་མེ་རིར་བཅས། །རྟེན་དང་བརྟེན་པར་བཅས་ལ་ཕྱག་འཚལ་ལོ། །བདག་ཅག་སྒྲུབ་པའི་མཆེད་ལྕམ་འཁོར་བཅས་ཀྱི། །དུས་གསུམ་བསགས་དང་ཡོད་པའི་དགེ་བ་དང༌། །ལྷག་པའི་ལྷ་མཆོག་ཁྱོད་ཀྱི་ཐུགས་རྗེ་ཡིས། །གསང་སྔགས་ཟབ་མོའི་ཆོས་སྒོར་འཇུག་པ་དང༌། །ཉན་དང་འཆད་དང་འབྲི་སོགས་ལེགས་མནོས་ཤིང༌། །དགྱེས་པ་རྡོ་རྗེའི་དཀྱིལ་འཁོར་ལྷ་ཚོགས་མཆོག །བདག་སོགས་ཀུན་གྱིས་གེགས་མེད་འགྲུབ་པར་ཤོག །

ཅེས་བསྟོད། བཟོད་པར་གསོལ་བ་ནི། གང་ཡང་བདག་བློ་རྨོངས་པ་ཡིས། །ཉེས་བྱས་ཅུང་ཟད་གང་བགྱིས་པ། །མ་མཉེས་ཡོངས་སུ་མ་མཆིས་པ། །རྣལ་འབྱོར་མ་ཐོབ་དབང་གིས་ནི། །གསང་སྔགས་ཉམས་དང་བྱ་བ་ཉམས། །ཆོ་ག་ཉམས་པར་གྱུར་པ་ཀུན། །དེ་ཡང་ཁྱོད་ཀྱིས་བཟོད་པར་མཛོད། ། ཅེས་བརྗོད། ཡི་གེ་བརྒྱ་པ་ལན་གསུམ་བརྗོད་ལ་ལྷག་ཆད་ཁ་བསྐངས། ཨོཾ། ཧེ་རུ་ཀ་དཔལ་བདག་མེད་ཡུམ་དང་བཅས། །སྤྲུལ་པའི་མཁའ་འགྲོ་བརྒྱད་ལ་བརྟེན་བཅས་དང༌། །ལྷ་ཚོགས་སྲུང་འཁོར་ཆོས་ཀྱི་དབྱིངས་སུ་གཤེགས། །ཁྱོད་ཀྱིས་སེམས་ཅན་དོན་ཀུན་མཛོད། ། ནས། སླར་ཡང་འབྱོན་པར་མཛད་དུ་གསོལ། །བཛྲ་མུཿ དཀྱིལ་འཁོར་གྱི་ཡེ་ཤེས་པ་དང༌། བུམ་བསྐྱེད་ཀྱི་ཡེ་ཤེས་པ་རྣམས་ཆོས་ཀྱི་དབྱིངས་སུ་གཤེགས༑ དམ་ཚིག་པ་རྣམས་རང་གི་སྙིང་ཁར་ཐིམ་པར་གྱུར། ཨོཾ་ཨཱཿཧཱུྂ། ཞེས་གཤེགས་སུ་གསོལ། ཤིས་པ་བརྗོད་པ་ནི། བཀྲ་ཤིས་གང་ཞིག་ཀྱཻ་རྡོ་རྗེ་འཇིགས་བྱེད་འབར་བའི་སྐུ། །ཤེས་རབ་བདག་མེད་ཡུམ་གྱི་པདྨར་རྡོ་རྗེ་ཤིན་ཏུ་རོལ་མཛད་པ། །བདེ་ཆེན་གཉིས་མེད་འཁོར་ལོ་མཆོག་དེ་དགའ་བ་བཞི་ཡི་དབང་ཕྱུག་གཙོ། །བདེན་པ་དེས་ནི་བདེ་བ་མཆོག་ལ་གནས་པའི་བཀྲ་ཤིས་ཤོག ༑གང་ཞིག་ཡེ་ཤེས་བཞི་ཡི་རང་གཟུགས་རབ་འཆང་ཞིང༌། །ཚད་མེད་བཞི་ཡིས་འགྲོ་རྣམས་མ་ལུས་སྒྲོལ་མཛད་པ། །གཽ་རཱི་ལ་སོགས་ལྷ་མོ་བརྒྱད་དུ་ཤར་བ་འདི། །བདེན་པ་དེས་ནི་བདེ་བ་མཆོག་ལ་གནས་པའི་བཀྲ་ཤིས་ཤོག །ཕྱི་རོལ་ཡུལ་སྣང་རབ་ཏུ་དག་པའི་ཕྱིར། །གཞལ་ཡས་པད་རྡོར་རྡོ་ར་མེ་རི་དུར་ཁྲོད་རི། །རང་ལུས་རྩ་ཁམས་དཔའ་བོ་དཔའ་མོའི་ངང༌། །བདེན་པ་དེས་ནི་བདེ་བ་མཆོག་ལ་གནས་པའི་བཀྲ་ཤིས་ཤོག །སྣ་ཚོགས་སྣང་བའི་ཆོ་འཕྲུལ་མ་འགག་ཤར། །བུམ་ཐོད་རྡོར་དྲིལ་ཁ་ཊྭཱཾ་ག་ལ་སོགས། །ཡོ་བྱད་སོ་སོའི་རྣམ་པར་ཤིས་དེས་ཀྱང༌༑ ༑བདེན་པ་དེས་ནི་བདེ་བ་མཆོག་ལ་གནས་པའི་བཀྲ་ཤིས་ཤོག །ནམ་མཁའ་སྐྱེ་བ་མཚན་མེད་ཕྱིར། །ཐོག་མ་ཐ་མ་མེད་པའི་ཕྱིར། །སེམས་དཔའ་རྡོ་རྗེ་ཆེའི་རང་བཞིན། །ཀྱཻ་རྡོ་རྗེ་བདག་འགྲུབ་མཛོད། །ཀུན་མཆོག་དངོས་གྲུབ་ཆེན་པོ་སྟེ། །དབང་ཕྱུག་ཆེན་པོ་ལྷག་པའི་ལྷ། །རྡོ་རྗེ་འཛིན་པ་ཀུན་གྱི་རྒྱལ། །མི་འགྱུར་མཆོག་ཏུ་བདག་འགྲུབ་མཛོད། །ཉེས་པ་མི་མངའ་རྟག་པ་སྟེ། །འདོད་ཆགས་ཀུན་གྱི་རྗེས་སུ་ཆགས། །འདོད་ཆགས་ཆེན་པོ་དགའ་བ་ཆེ། །བཅོམ་ལྡན་དེ་ཉིད་ཀྱིས་གྲུབ་མཛོད། །ཤིན་ཏུ་དགའ་བ་ཀུན་གྱི་མཆོག
