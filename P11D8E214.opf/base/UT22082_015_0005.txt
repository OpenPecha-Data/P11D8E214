༄༅། །རིན་ཆེན་རིལ་བུ་གཏོང་ཚུལ་གྱི་ཟིན་བྲིས་བཞུགས་སོ།།

བཻ་ཌཱུརྻ་འོད་རྒྱལ་ལ་འདུད། །སྙིགས་དུས་སེམས་ཅན་ངན་སེམས་ཆེ། །རྫས་ངན་དུག་གི་སྦྱོར་བ་དར། །ཀུན་ལ་འཇིགས་པར་མི་འགྱུར་བའི། །སྲུང་བ་རྡོ་རྗེའི་ཕ་ལམ་བཤད། །སྲུང་དགོས་ཕན་ཡོན་དམ་ཚིག་བཞི། །དང་པོ་གཟའ་སྐར་ཚེས་བཟང་ལ། །ལྟོ་བཀུག་དགོང་མོ་གཡེར་མ་ཡི། །རྡོག་ཕྲལ་ཉེར་གཅིག་ཆུ་ལ་གདུས། །དྲོ་འཇམ་ཇ་སུན་ཙམ་ཞིག་བཏང༌། །སྲོད་ལ་རིན་ཆེན་རིལ་བུ་གཅིག །ནམ་མཁའ་བུད་མེད་བྱི་ལ་ཡིས། །མ་མཐོང་སུ་ར་ནར་སོན་དབུལ། །ནམ་ལངས་དུས་སུ་དོམ་གྱི་མཁྲིས། །ཐུན་གཅིག་འབྲི་མར་བཙག་བཅད་དབུལ། །ཟླ་བ་བཅུ་ཡི་བར་དུ་སྲུང༌། །གཉིས་པ་སྦྱར་དུག་གསོ་བ་ནི། །འོག་ཏུ་འཆད་པ་ཇི་བཞིན་ནོ། །གསུམ་པ་དུག་ཆུང་གསོ་བ་ཡང༌། །སླར་འབྱུང་ལྟར་རོ་བཞི་པ་ནི། །དུག་ཀུན་བསྡུས་ལ་བསད་པ་ཡི། །རྗེས་སུ་རིན་ཆེན་བཤལ་གྱིས་སྦྱང༌། །འདུས་ནད་སྨུག་པོ་གསོ་དཀའ་ལ། །རིན་ཆེན་མ་ནུ་བཞི་ཡིས་ཕུལ། །རྗེས་གཅོད་སྨུག་པོ་སྐྱུག་མདོག་འབྱུང༌། །རིན་ཆེན་སྦྱར་བས་ངེས་པར་གྲོལ། །རྙིང་ཚད་ག་པུར་ཉེར་ལྔ་ལ༑ ༑རིན་ཆེན་སྦྱར་བ་ཆུ་བསྐོལ་དབུལ། །ཡང་ན་རྩ་སྦྱོང་རིན་ཆེན་སྦྱར། །ཆང་གིས་ཕུལ་ན་ངེས་པར་ཕན། །བཙན་དུག་དྲག་པོས་བཏབ་གྱུར་ན༑ ༑རིན་ཆེན་ལོ་བརྒྱད་ཆུ་ཡིས་དབུལ། །ཚབས་ཆེ་བདུད་རྩི་དྭངས་མས་དབུལ། །སྐྱེ་འམ་ཐག་པས་གདུང་བསྐོར་བྱ། །དེ་ཡིས་འཁྲུ་སྐྱུག་མྱུར་དུ་འབྱུང༌༑ ༑ཁྱི་དུག་རྟ་གསེབ་ཆུ་ཡིས་དབུལ། །ཡང་ན་སྦུར་་་་་་སྲེ་སྦུར་ནག་གདུས་ཁུས་དབུལ། །ཡང་ན་སྟོང་རི་གདུས་ཁུས་དབུལ། །ཁྱི་དུག་འཇོམས་པའི་བདུད་རྩི་ཡིན། །མདའ་དུག་རིན་ཆེན་ཁར་བཟུང་འཇིབ། །ལོ་བརྒྱད་ཆུར་གདུས་ཁུ་བ་བཏུང༌། །ཚབས་ཆེན་བཞི་མཉམ་ཁོང་དུ་གཏང༌། །ན་ཚོད་སྦྱར་བའི་རྟ་ཕོར་བསྐྱོན། །ཤ་དུག་རིན་ཆེན་ཆང་གིས་དབུལ། །ཚབས་ཆེན་ལན་ཚྭ་གདུས་ཁུས་དབུལ། །ཡང་ན་ཉུང་སྐམ་གདུས་ཁུས་དབུལ༑ ༑འཆི་བདག་དགྲ་བགེགས་འཇོམས་པར་བྱེད། །སྒོང་དུག་རིན་ཆེན་སྲམ་ཤས་དབུལ། །ཡང་ན་མཐིང་ཤུན་ཁུ་བས་དབུལ། །སྒོང་དུག་གློ་སྐམ་གྱུར་པ་སེལ། །སྤུ་དུག་རིན་ཆེན་ཁྲག་དྲོན་དབུལ། །ཡང་ན་དོམ་གྱི་ཚིལ་བུས་དབུལ། །རྩར་བབ་རྨར་གྱུར་སེལ་བར་བྱེད། །གཡའ་དུག་ཚ་ལ་བཏུལ་མ་དང༌། །མུ་ཟི་གསེར་འདྲའི་ཁུ་བས་དབུལ། །ཡང་ན་ཟྭ་དཀར་ཁུ་བས་དབུལ། །སྡུད་དང་གསོད་པའི་གདམས་པ་ཡིན། །དུག་གིས་ཤི་བའི་མཆིན་དུག་ལ། །རིན་ཆེན་ཉ་བམ་ཟྭ་ཁུས་དབུལ། །དེ་རྗེས་ནད་ལྷག་བཤལ་གྱིས་འདེད། །གྱུར་པའི་དུག་ལ་རིན་ཆེན་ནི། །གང་གནོད་ཟས་ཀྱི་ཁུ་བས་དབུལ། །གཞན་ཡང་རྒྱུ་དང་མི་རྒྱུ་བའི། །དུག་རིགས་གང་དང་གང་ཡིན་ཡང༌། །རྒྱུད་ནས་བཤད་པའི་སྨན་རྣམས་དང༌། །རིན་ཆེན་བསྡེབས་པས་མ་ལུས་འཇོམས། །དེ་བཞིན་ཚ་བའི་ནད་ལ་བསིལ། །གྲང་བའི་ནད་ལ་དྲོད་དུ་འགྱུར། །བསིལ་སྨན་བསྡེབས་པས་ཚད་ནད་འཇོམས། །དྲོད་སྨན་དང་སྡེབ་གྲང་ནད་སེལ། །དྲོད་བཅུད་དང་སྡེབ་རླུང་ནད་གསོ། །སྦྱོར་བ་གང་དང་བསྡེབས་གྱུར་ཀྱང༌། །དེ་ཡི་ནུས་པ་བརྒྱ་འགྱུར་སྐྱེད། །གསུམ་པ་འདི་ཡི་ཕན་ཡོན་ཕྱི་རྒྱུད་ལས། །ཐང་ཕྱེ་རིལ་བུ་ལྡེ་གུ་སྨན་མར་སོགས། །ཡང་ཡང་བསྟེན་པས་སྨན་འདྲེད་ནད་མ་ཐོན། །ཞི་བྱེད་ཀུན་གྱི་ལོག་གནོན་རིན་པོ་ཆེ། །ཁྲག་མཁྲིས་བད་རླུང་བཞི་བརྒྱ་རྩ་བཞིའི་ནད། །མགོ་དང་བྱང་ཁོག་ཡན་ལག་རྨ་རྣམས་དང༌། །དམུ་འོར་མཛེ་འབྲས་གག་ལྷོག་གཉན་ནད་དང༌།
