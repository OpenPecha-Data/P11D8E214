༄༅། །ཡང་ཟློག་མེའི་སྤུ་གྲིའི་བསྙེན་སྒྲུབ་ཀྱི་ཞལ་གདམས་གུད་དུ་སྦས་པ་བཞུགས་སོ།།

བླ་མ་དང་འཇམ་དཔལ་གཤིན་རྗེའི་གཤེད་ལ་ཕྱག་འཚལ་ལོ། །རྣལ་འབྱོར་པ་གང་ཞིག་ཡང་ཟློག་མེའི་སྤུ་གྲི་ལ་ཡི་དམ་དུ་བྱ་བ་དག་གིས་ལས་སྦྱོར་སྲུང་ཟློག་བསད་པའི་སྔ་རོལ་དུ་བསྙེན་བསྒྲུབ་བྱ་བ་ཤིན་ཏུ་གལ་ཆེ་བའི་ཕྱིར། དེ་དག་གི་ཚུལ་ཟླ་གསང་འཇམ་དཔལ་ཐུགས་ཀྱི་རྩ་རྒྱུད་དང་ཟླ་གསང་དྲེགས་པ་ཚར་གཅོད་ཀྱི་རྒྱུད་སོགས་ནས་གསུངས་པ་རྣམས་ལེགས་པར་མཐོང་ཞིང་རྟོགས་པས། དེ་དག་གི་མན་ངག་ལག་ལེན་རྣམས་གོ་བདེ་བར་བཀོད་པ་ནི་འདི་ལྟ་སྟེ། འཇམ་དཔལ་གཤིན་རྗེའི་གཤེད་ཀྱི་དབང་ཐོབ་ཅིང་དམ་ཚིག་ལ་གནས་པའི་སྔགས་པས་དེ་དག་གི་བསྙེན་བསྒྲུབ་བྱ་བར་འདོད་ན། སྦྱོར་བའི་ཆོས་དང༌།་་་་་གཤིན་རྗེ་ནག་པོའི་སྔགས་རྒྱུད་ཀྱི་སྒྲུབ་པ་པོའི་མཚན་ཉིད་དང་ཉམས་སུ་བླང་བའི་གོ་རིམ་བསྟན་པའི་ལེའུ་བརྒྱད་པ་ལས། རིགས་ཡོད་སྡོམ་དང་ལྡན་ཞིང་དབང་ཐོབ་མན་ངག་མནོས། །རྒྱུད་ཐོས་དམ་ཚིག་བསྲུང་ཞིང་སྙིང་རྗེ་ཆེ་དང་བདུན། །དེ་བདུན་ལྡན་པའི་རྣལ་འབྱོར་པས། །འདི་ཉིད་ཉམས་སུ་བླང་བར་བྱ༑ ༑ཞེས་གསུངས་པས། མཚན་ཉིད་བདུན་པོ་ལྡན་པའི་སྔགས་པས་སྒྲུབ་ཅིང༌། གཞན་བྱུང་རྒྱལ་གང་རུང་གིས་སྒྲུབ་པར་མ་གསུངས་སོ། །དངོས་གཞི་གཉིས་སོ། །དང་པོ་སྦྱོར་བའི་ཆོས་ནི་ཡིད་མཐུན་པའི་གནས་སུ། འབུམ་འདོན། སཱཙྪ་གདབ། བསྐང་བཤགས་བྱ། ཚད་མེད་བཞི་ལ་བློ་སྦྱང༌། བྱང་ཆུབ་ཀྱི་སེམས་རྣམ་པ་གཉིས་བསྒོམ༑ རྣམ་པར་མི་རྟོག་པའི་དོན་དྲན་པར་བྱ། སྟོང་ཉིད་སྙིང་རྗེ་ཟུང་འཇུག་གི་ངང་ནས་འཇམ་དཔལ་ཡེ་ཤེས་སེམས་དཔའ་ངག་འདོན་གྱི་སྲུང་བའི་དུས་རྩིབས་ནང་དུ་བསྐྱེད་པ་ལྟར་རང་ཉིད་བསྒོམས་པའི་ཐུགས་ཀར་མུཾ་དམར་སེར་ལ་སྔགས་ཀྱིས་བསྐོར་བར་དམིགས་ལ། ཨོཾ་ཝཱ་གཱི་ཤྭ་རི་མུཾ། ཞེས་པ་དྲུག་འབུམ་བཟླ་ཞིང་འཇམ་དཔལ་མཚན་བརྗོད་རྟག་ཏུ་འདོན་པར་དམ་བཅའ་བ་སོགས་མདོར་ན་བསོད་ནམས་ཀྱི་ཚོགས་བསགས་པ་གྲུབ་ཚད་ལ་ནན་ཏན་དུ་བྱའོ། །གཉིས་པ་དངོས་་་་་་ཝཱ་གཱི་ཤྭ་ར་དྲུག་འབུམ་སྔོན་དུ་བཟླ་དགོས་པ་ནི།

ཟླ་གསང་རྩ་རྒྱུད་ཆེན་པོའི་ལེའུ་བཅུ་དགུ་པར། ཡེ་ཤེས་སེམས་དཔའ་བསྒོམས་བྱས་ནས། །ཝཱ་གཱི་ཤྭ་ར་དྲུག་འབུམ་འདོན། །ཞེས་གསུངས། ཡང་གསང་སྔགས་ལྔ་ཐམས་ཅད་འཁྲུལ་འཁོར་རྒྱ་བསྡེབས་པའི་རྒྱུད་ལས། ཝཱ་གཱི་ཤྭ་ར་མུཾ། ཐུགས་ཀར་མུཾ་ལས་བསྐྱེད་བྱས་ནས། །འཕྲོ་འདུ་དང་བཅས་འབུམ་ཚོ་གཅིག །སྔོན་དུ་འགྲོ་བའི་བསྙེན་པ་སྟེ། །ཞེས་གསུངས་པས། རབ་འབུམ་ཚོ་དྲུག །དེ་མ་བྱུང་ན་འབུམ་ཕྲག་གཅིག་གིས་ཀྱང་ཆོག་པར་གསུངས་སོ། །ཞི་བསྙེན་འདི་བྱ་ཚུལ་རྒྱས་པ་ལོགས་སུ་ཡོད་པ་དེ་བཞིན་དུ་རྫོགས་པར་གྲུབ་རྗེས། གཞི་བསྙེན་དངོས་ལ་འཇུག་དགོས་ཀྱི་འདིར་ཞི་བསྙེན་ས་བཅད་ཙམ་ལས་མ་བྲིས་སོ། །གཞི་ལ་བཞི།་་་་་འཕགས་པ་འཇམ་དཔལ་གཤིན་རྗེའི་གཤེད་ཀྱི་འཁྲུལ་འཁོར་སྔགས་རྒྱུད་ནག་པོའི་རྟོགས་པ་བཅུ་གཉིས་པར། བསྙེན་པ་དང་ནི་ཉེ་བསྙེན་དང༌། །སྒྲུབ་པ་དང་ནི་སྒྲུབ་ཆེན་པོ། །དགོས་པའི་དོན་ཆེན་གང་ལའང༌། །བསྙེན་པ་དངོས་གྲུབ་རྣམ་གསུམ་ཐོབ། །དེ་མ་ཡིན་པས་དངོས་མི་འགྲུབ། །ཅེས་གསུངས་པའི་ཕྱིར། ལྷ་བསྙེན་པ། ཉེ་བར་བསྙེན་པ། སྒྲུབ་པ་ཆེན་པོ། འཆི་བདག་བདུད་སྒྲུབ་རྣམས་སོ། །དང་པོ་ལྷ་བསྙེན་པ་ནི། ཟླ་བ་ཐ་ཆུང་བཞིའི་ཡར་ངོའི་གྲུབ་པའི་ཉི་མ་གང་ཡིན་ལ་བབ་པའི་ཚེ།་་་་་གཤིན་རྗེ་ནག་པོ་སྔགས་རྒྱུད་ཀྱི་ལེའུ་བཞི་པ་ལས། ཟླ་བ་ཡར་ངོའི་གཟའ་གང་བཟང༌། །མདའ་འཕེན་པ་ཡི་ཟླ་བའི་དུས། །ཞེས་དགུན་ཟླ་ཐ་ཆུང་གི་ཡར་ངོ་ལའང་སྒྲུབ་པར་གསུངས་ལ། མདའ་འཕེན་པའི་ཟླ་བ་ནི། དགུན་ཐ་མ་ཡིན་པ་རྡོ་རྗེ་གཙུག་ལག་གི་རྒྱུད་ནས་གསུངས་ཤིང༌། དགུན་ཟླ་ནི་ཧོར་ཟླ་བཅུ་གཅིག་པ་ལ་བྱེད་ན་རྒྱ་ནག་རྙིང་མ་བླ་མ་དགོངས་འདུས་ཀྱི་དགོངས་པ་རྣམ་པར་དག་པའོ། །གཤིན་རྗེ་མ་རུ་རྩེའི་སྔགས་རྒྱུད་ཀྱི་ལེའུ་ལྔ་པ་ལས། སྔགས་པས་མདའ་འཕེན་ཟླ་བ་ནི། །ཉི་མ་རླུང་བར་དབང་ནས་ནི། །སྨིན་དྲུག་ས་འོག་མ་སོང་བར་དུ་བྱ། །ཞེས་ཧོར་ཟླ་བཅུ་གཅིག་པ་ནས་བཟུང་ནས་སྨིན་དྲུག་རོལ་འོག་ལ་མ་སོང་གི་བར་དེ་རྣམས་གང་འཛོམས་ལ་བྱ་དགོས་པ་ཡིན་ནོ། །དུར་ཁྲོད་ལ་སོགས་པ་དབེན་པའི་གནས་སུ།་་་་་བསྙེན་པའི་གནས་ནི།

གཤིན་རྗེའི་གཤེད་ནག་པོ་ལས་ཐམས་ཅད་གྲུབ་པར་བྱེད་པ་འཁོར་ལོའི་རྒྱུད་ལས། ཆུ་འགྲམ་ཆེན་པོ་བས་མཐའ་དང༌། །དུར་ཁྲོད་ཤིང་དྲུང་དགོན་པ་སྟེ། །དབེན་ས་ཡུལ་དང་མ་མོའི་ཁྱིམ། །དེར་ནི་མགོན་པོ་དེ་ལ་འབད། །གཞན་དུ་བསྒྲུབས་ཀྱང་དེ་མི་འགྲུབ། །ཅེས་གསུངས་ཤིང༌། འཇིགས་བྱེད་རྟོག་བདུན་གྱི་རྟོག་པ་དང་པོར་ཡང༌། དེ་ལ་གནས་གསུངས་པ་ནི། དུར་ཁྲོད་དམ། ཆུ་ཀླུང་གི་འགྲམ་མམ། ཤིང་གཅིག་གི་མཚན་མ་གཅིག་པའམ། བཞི་མདོ་འམ། རིའམ། རྩེ་མོའམ། ཁང་སྟོང་ངམ། མ་མོའི་ཁྱིམ་མམ། ལྷའི་ཁང་པའམ། འཐབ་མོའི་གནས་སམ། གྲོང་ཁྱེར་རམ། ནགས་ཁྲོད་དམ། གྲོང་བར་ལ་སོགས་པའི་གནས་སུ་སྔགས་གནས་ཏེ། ཞེས་གསུངས་པས། གནས་འདི་རྣམས་གང་རུང་དུ་བྱའོ། །མཎྜལ་གཙང་མ་དྲིས་བྱུགས་པའི་སྟེང་དུ་སྨན་རཀ་གཏོར་མ།་་་་་རུས་སྦལ་ཁོག་པ་སོགས་སུ་མཆོད་གཏོར་བཤམས་ལ་མཚམས་གྲོལ་ཁར་སྒྲུབ་ཁང་སྟེང་དུ་འབུལ། རྟགས་གཏོར་གཅིག་ཟུར་དུ་བཤམས་ལ་རྟགས་ནམ་བྱུང་ཐོག་སྟེང་དུ་འབུལ་ཞིང་ཕུལ་ཕྲལ་སླར་གསར་པ་བཅོས་པ་བཞག་ནས་རྟགས་བྱུང་རིམ་གྱི་རྟགས་གཏོར་རེ་རེ་འབུལ་ལོ། །མཆོད་པ་དང་བཅས་པ་བཤམས། མདུན་དུ་འཇམ་དཔལ་་་་་་འཇམ་དཔལ་གཤིན་རྗེ་ཁྲོ་བོ་གསང་བའི་རྒྱུད་ཀྱི་སྒྲུབ་པ་རྩ་བའི་ལེའུ་ལས། དུར་ཁྲོད་རས་ལ་དཔའ་བོའི་གཟུགས། །མཚོན་རྩི་སྤྱིན་མེད་བཟང་པོ་ཡི། །ཁྲོས་པའི་གཟུགས་ལ་སྤྲིན་ནག་མདོག ༑མཁས་པས་ཡབ་ཡུམ་སྲས་བཅས་བྲི། །རབ་ཏུ་གནས་པའི་ཆོ་ག་བྱ། །ཞེས་པ་ལྟར། བྲིས་སྐུ་མཚན་ཉིད་དང་ལྡན་པ་གལ་ཆེའོ། །གཤིན་རྗེའི་བྲིས་སྐུ་བཀྲམ། ཕྱི་ནང་གི་མཚམས་དམ་པར་བཅད། །ཕྱོགས་སྐྱོང་དང་གཞི་བདག་བགེགས་ལ་སོ་སོར་གཏོར་མ་བྱིན་ལ་བཀའ་བསྒོ། ཀུ་ཤ་ལ་སོགས་པའི་སྟན་བདེ་བར་འདུག་ལ།་་་་་འཕགས་པ་འཇམ་དཔལ་གཤིན་རྗེའི་གཤེད་ཀྱི་སྔགས་རྒྱུད་འཁྲུལ་འཁོར་ནག་པོའི་ལེའུ་ལྔ་པ་བསྙེན་ཅིང་སྒྲུབ་པའི་སྐབས་ལས། དྲི་སྤོས་སྨན་དང་བདུད་རྩིའི་ཆུ། །འབར་བའི་སྔགས་ཀྱིས་ལུས་བཀྲུས་ཤིང༌། །བདུད་རྩི་འཁྱིལ་པས་རབ་ཏུ་བསྟོད། །ལྷོ་ཡི་ཕྱོགས་སུ་ཁ་བལྟས་ལ། །ཞེས་གསུངས་པས།

ལས་གཤིན་དང་འཁྱིལ་པ་བཟླས་པའི་དྲིའི་ཆུས་ཁྲུས་བྱས་ནས། ངེས་པར་ཁ་ལྷོ་ཕྱོགས་སུ་ཕྱོགས་པས་བསྙེན་སྒྲུབ་ལ་འཇུག་གོ། །སྐྱབས་སེམས་ཚད་མེད་བཞི་སྔོན་དུ་འགྲོ་བའི་་་་་་རྫོགས་པའི་སངས་རྒྱས་སོགས་སྐྱབས་སེམས་དང༌། མ་རིག་རྒྱུ་སོགས་སྙིང་རྗེ་རྣམ་གསུམ༑ དཀའ་ཐུབ་དྲག་པོས་སྡོམ་པ་དང་ལྡན་པས། སངས་རྒྱས་ཀྱི་བསྟན་པ་རྒྱས་པར་བྱ་བ་ནས། མཐུ་རྩལ་དང་ལྡན་པར་གྱུར་གྱི་བར་ངག་འདོན་དཀྱུས་ལྟར་ལ། སྦས་ལྷ་གཉིས་ཡོད་མེད་རང་གང་མོས་བྱས་པས་ཆོག་ཅིང་གང་ལྟར་ནའང་བསྐྱེད་རིམ་གསལ་བར་བྱ། མཆོད་པའི་སྐབས་བདག་ལྷར་གསལ་བའི་ཐུགས་ཀ་ནས་ཛཿཧཱུྂ་བཾ་ཧོཿའི་ཕྱག་རྒྱ་བཞི་ལས་མཆོད་པའི་ལྷ་མོ་བཞི་སྤྲོས་ནས། ཕྱི་ནང་གསང་བ་སྨན་རཀ་གཏོར་ཚོགས་ཀྱི་མཆོད་པར་བསམ་ཞིང་བསྟོད་པ་སོགས་བྱ། དེ་ནས་བསྙེན་པ་ལ་འཇུག་པ་ནི། མགོ་བོ་རྨ་བྱ་བཞིན་དུ་དགུག །རུ་རཀྴ་ཡི་ཕྲེང་བ་བགྲང༌།་་་་་བསྙེན་པའི་གོང་ནས་བགྲང་ཕྲེང་རབ་གནས་ཚུལ་བཞིན་དུ་བྱས་ནས་མ་ཉམས་པར་དགོས་ཤིང༌། ཐུན་སོ་སོའི་འགོར་ཡང་ཕྲེང་བ་བྱིན་རླབས་ཀྱི་སྔགས་ཉེར་གཅིག་ཙམ་རེ་བཟླ་བ་གལ་ཆེའོ། །འཇམ་དཔལ་རྩ་བའི་ཆོ་ག་ཞིབ་མོ་ཁྲོ་བོ་རྣམ་པར་རྒྱལ་བའི་རྒྱུད་ཀྱི་ལེའུ་བདུན་པར། དེ་ནས་གཞན་ཡང་བཟླས་པ་ཡི། །ཆོ་ག་ཇི་སྐད་གསུངས་པ་བཤད། །རི་མོ་བཀྲམ་ནས་མཆོད་བྱས་ཏེ། །དེ་བཞིན་གཤེགས་པ་ཀུན་མཆོད་ནས། །ཀུ་ཤའི་སྟན་ལ་འདུག་ནས་ནི། །མཉམ་པར་བཞག་སྟེ་མི་སྨྲ་བར། །རི་མོ་གཅིག་པོ་ལ་བལྟས་ཤིང༌། །མགོ་ནི་རྨ་བྱ་བཞིན་བཀུག་སྟེ། །རུ་རཀྴ་ཡི་ཕྲེང་བ་ནི། །ཇི་སྲིད་ཡིད་ནི་མི་སྐྱོའི་བར། །རྩེ་གཅིག་སེམས་ཀྱིས་བཟླས་བྱས་ཏེ། །སྐྱོ་ན་བཟླས་པ་གཏང་བར་བྱ། །ཞེས་སོགས་དོན་གཅིག་པོ་འདི་ཉིད་རྒྱུད་མང་པོར་ཡང་ཡང་བལྡབས་ནས་གསུངས་འདུག་པས་གལ་ཆེན་པོ་ཉིད་དུ་རིགས་པར་བྱའོ། །མགོན་པོ་ཉིད་ལ་གུས་བྱས་ཏེ། །དམིགས་པ་ནི།་་་་་ཞེས་པ་ཟླ་གསང་རྩ་རྒྱུད་ཀྱི་ལེའུ་བཅུ་བཞི་པ་དང༌།

ཟླ་གསང་དྲེགས་པ་ཚར་གཅོད་ཀྱི་སྐབས་བརྒྱད་པར་གསུངས་སོ༑ ༑འདི་འདྲའི་རིགས་ཀྱི་བསྙེན་སྒྲུབ་བཞི་གར་རྒྱུད་ཀྱི་ཚིག་རྣམས་བྲིས་ན་ཡིག་ཚོགས་ཆེ་བར་འཇིགས་ནས་དོན་རྣམས་བསྡུས་ཏེ་བྲིས་པ་ཡིན་ནོ། །ཕྱི་སྣོད་ཀྱི་འཇིག་རྟེན་གཞལ་ཡས་ཁང༌། ནང་བཅུད་ཀྱི་སེམས་ཅན་འགྲོ་བ་རིགས་དྲུག་ཐམས་ཅད་འཇམ་དཔལ་གཤིན་རྗེའི་གཤེད་ཀྱི་ལྷ་ཚོགས་སུ་མ་གྱུར་པ་གང་ཡང་མེད་པ་ཏིལ་གྱི་གང་བུ་ལྟར་གང་བ། བདག་དམ་ཚིག་སེམས་དཔར་གསལ་བའི་ཙིཏྟ་གཞལ་ཡས་ཁང་གུར་ཕུབ་པ་ལྟ་བུའི་དབུས་སུ་པདྨ་ཟླ་ཉིའི་སྟེང་དུ་རང་རིག་ཡེ་ཤེས་ཀྱི་ངོ་བོ་གདོད་ནས་ཡོངས་སུ་གྲུབ་པའི་འཇམ་དཔལ་ཡེ་ཤེས་སེམས་དཔའ་ཚོན་གང་པ། སྐུ་མདོག་དམར་སེར། ཞལ་གཅིག་ཕྱག་གཉིས། དབུ་སྐྲ་རལ་པའི་ཅོད་པན་ཅན། ཕྱག་གཡས་ཤེས་རབ་ཀྱི་རལ་གྲི་དཔུང་པའི་ཐད་དུ་གདེངས་ཤིང༌། གཡོན་པའི་མཐེབ་སྲིན་གྱིས་ཨུཏྤལའི་ཡུ་བ་ཐུགས་ཀར་འཛིན་པའི་འདབ་མ་སྙན་གྱི་ཐད་དུ་ཁ་བྱེ་བ་རྒྱས་པའི་སྟེང་དུ་སྟོང་ཕྲག་བརྒྱ་པའི་གླེགས་བམ་གྱིས་མཚན་པ། ཞབས་རྡོ་རྗེའི་སྐྱིལ་མོ་ཀྲུང་གིས་བཞུགས་པ། རིན་པོ་ཆེ་སྣ་ཚོགས་པའི་རྒྱན་ཅན། དར་སྣ་ཚོགས་ཀྱི་སྟོད་གཡོགས་དང་སྨད་གཡོགས་ཀྱིས་བཀླུབས་པ། སྐུ་སྣང་ལ་རང་བཞིན་མེད་པ་བཅུ་དྲུག་ལོན་པའི་ལང་ཚོ་ཅན་ཞི་ཞིང་འཛུམ་མདངས་ཅན༑ དེའི་ཐུགས་ཀར་ཉི་མའི་དཀྱིལ་འཁོར་ཀྱི་སྟེང་དུ་རྡོ་རྗེ་སྔོན་པོ་རྩེ་ལྔ་པའི་ལྟེ་བར་ཟླ་ཉིའི་སྟེང་དུ་ཏིང་འཛིན་སེམས་དཔའ་ཧཱུྂ་མཐིང་ནག་གི་མཐར་བཟླ་བྱའི་སྔགས་ཀྱིས་བསྐོར་བ། ལྷ་གཞན་རྣམས་ཀྱི་ཐུགས་ཀར་ཡང་དེ་འདྲ་བ་རེ་རེ་གསལ་ཞིང༌། བདག་གིས་སྔགས་བཟླས་པའི་ཚེ། ཐུགས་ཀ་ནས་སྔགས་ཕྲེང་མར་མེ་གཅིག་ལས་གཉིས་མཆེད་ཀྱི་ཚུལ་དུ་འཕྲོས། ལྗགས་སྟེང་བརྒྱུད་དེ་ཕྱིར་འདོམ་གང་ཙམ་ཕྱིན་པའི་ཚེ་སྔགས་འབྲུ་འོད་དང་བཅས་པ་རྣམས་ཀྱང་ལྷའི་སྐུ་ཕྲ་མོ་གྲངས་མེད་པར་གྱུར་ནས་དེ་རྣམས་ཀྱི་ཞལ་ནས་ཀྱང་སྔགས་བཟླ་ཞིང་འོད་ཟེར་རྣམས་མཆོད་པའི་ལྷ་དང་ལྷ་མོར་གྱུར།

ཕར་འཕྲོས་འཕགས་པ་རྣམས་མཆོད། སྐུ་གསུང་ཐུགས་མཉེས་པའི་མཆོད་པ་རྣམས་ཕུལ། འོད་ཟེར་དེ་དག་བདག་ལ་ཚུར་འདུས་པའི་ཚེ། ཕྱོགས་བཅུ་དུས་གསུམ་གྱི་རྒྱལ་བ་འཁོར་དང་བཅས་པའི་སྐུ་གསུང་ཐུགས་ཀྱི་བྱིན་རླབས་ཐམས་ཅད་བསྡུས་ཏེ་བདག་ཉིད་ལ་ཐིམ་པར་བསམ་པའི་ངང་ནས།་་་་་རྡོ་རྗེ་ཟླ་ཉི་ས་བོན་རྣམས་མི་འགུལ་བར་སྔགས་ཕྲེང་གཡས་སུ་དྲག་ལ་མྱུར་བར་དུས་རྟག་ཏུ་འཁོར་བར་དམིགས་པ་གལ་ཆེའོ། །ཨོཾ་ཨ་ཀྲོ་ཏེ་ཀ་ཡ་མཱནྟ་ཀ་ཧ་ན་མ་ཐ་བྷཉྫ་ཧཱུྂ་ཕཊ྄། ཅེས་པ་ནི་ཡི་གེ་བཅུ་བདུན་པའམ། ཡང་ན། ཨོཾ་ཨ་ཀྲོ་ཏེ་ཀ་ཡ་མཱནྟ་ཀ་ཧཱུྂ་ཕཊ྄༑ ཅེས་པའི་ཡི་གེ་བཅུ་གཅིག་པ་་་་་་རྙིང་རྒྱུད་པྲ་ཁོག་སླང་གི་རྟགས་ཞེས་དང༌། གསར་རྒྱུད་དུ་འཇམ་དཔལ་གྱི་རྩ་བའི་ཆོ་ག་ཞིབ་མོ་ཞེས་རྒྱུད་གཅིག་ལ་མཚན་གསར་རྙིང་སོ་སོར་བཏགས་པ་དེ་སྔགས་དེའི་མཐུ་ནུས་ཕན་ཡོན་གསུངས་པ་རྣམས་ཟུར་དུ་བྲིས་ཡོད་དོ། །དང༌། ཡ་མ་མེ་དཀྵེ་རཱ་ཙ་ས་དོ་རུ་ཎ་ཡོ་ནི་ར་ཡ་ཙ་ནི། ཞེས་པའི་ཡ་མ་བཅུ་བདུན་པ་སྟེ། སྔགས་གསུམ་པོ་གང་ཡང་རུང་བ། སྔགས་གསུམ་པོ་གང་རུང་གིས་ཆོག་པའི་རྒྱུ་མཚན་ནི། འཇམ་དཔལ་ཁྲོ་བོ་རིན་ཆེན་གསང་བའི་རྒྱུད་ཀྱི་སྒྲུབ་པ་རྩ་བའི་ལེའུ་ལས། ལྷ་སྲིན་སྲོག་གི་རྩ་བ་ཡིན། །ཡིད་བཞིན་ནོར་བུའི་ཕྲེང་བ་སྟེ། །རྣལ་མའི་ཡི་གེ་ངོ་བོ་ལྟ། །རང་བཞིན་སྟོང་པ་ནུས་པའི་མེ། །ཉི་ཟླ་རྒྱ་མཚོ་ལྷུན་པོར་བརྗོད། །ཡ་མཱནྟ་ཀའི་ངོ་བོ་ཉིད། །ཨོཾ་ཨ་ཀྲོ་ཏེ་ཀ་ཡ་མཱནྟ་ཀ་ཧ་ན་མ་ཐ་བྷཉྫ་ཧཱུྂ་ཕཊ྄།་་་ཅེས་དང༌། གཤིན་རྗེ་དམར་ནག་འབར་བའི་རྨེ་རྒྱུད་ཀྱི་ལེའུ་བདུན་པ་སྔགས་ཀྱི་ལེའུ་ལས། རྩ་བའི་སྔགས་དང་ལས་སྔགས་དང༌། །དམོད་པའི་སྔགས་དང་བསད་སྔགས་དང༌། །མ་ལུས་འདི་ཉིད་ཡིན་པ་རུ། །དམ་ཚིག་བདག་པོས་རབ་ཏུ་གསུངས། །ཨོཾ་ཨ་ཀྲོ་ཏེ་ཀ་ཡ་མཱནྟ་ཀ་ཧ་ན་མ་ཐ་བྷཉྫ་ཧཱུྂ་ཕཊ྄། བྱེ་བ་སྔོན་དུ་འགྲོ་བར་རིགས། །རི་རབ་རྡུལ་དུ་རློག་པ་དང༌། །རྒྱ་མཚོ་དག་ཀྱང་སྐེམ་པར་ནུས། །གནམ་ལྕགས་སེར་བ་འབེབས་པ་དང༌། །ཟློག་པ་དག་ཀྱང་ཐོགས་པ་མེད། །ཁྲོས་ན་བདུད་ཀྱང་བརླག་བྱེད་ལ། །བལྟས་པ་ཙམ་གྱིས་དགྲ་ཡང་ཏུངས། །དགའ་ན་སེམས་ཅན་གསོ་བ་དང༌། །འཁོར་བ་དོང་ནས་སྤྲུག་པར་ནུས། །སྔགས་ཀྱི་དོན་གཉིས་ལྷ་ཡིན་ཏེ། །ལྷ་ཡི་རང་བཞིན་སྔགས་ཉིད་དོ། །ཞེས་སོགས་གསུངས་པའི་ཕྱིར་དང༌།

འཇམ་དཔལ་ཁྲོ་བོ་རིན་པོ་ཆེ་གསང་བའི་རྒྱུད་ཀྱི་སྒྲུབ་པའི་རྩ་བའི་ལེའུ་ལས། འཇམ་དཔལ་ནང་ནས་འཇམ་དཔལ་ཉིད། །ཅེས་སོགས་ནས། དུས་ཀྱི་བསྙེན་པ་གཙོ་རུ་བླང༌། །ཞེས་པའི་བར་གསུངས་ཤིང༌། གསང་བ་དགུ་བཅུ་པའི་རྒྱུད་ཀྱི་སྔགས་ཀྱི་ཚད་མའི་ལེའུ་ལས། རྩ་བ་ཉིད་ལྟར་གྲང་བ་ནི། །ཨ་ཀྲོ་ཡིག་འབྲུ་བཅུ་བདུན་ཡིན། །ཞེས་དང་དེ་ཉིད་ལས། སྔོན་དུ་འགྲོ་བའི་བསྙེན་པ་ནི། །ཨ་ཀྲོ་བཅུ་བདུན་རྒྱུན་དུ་བགྲང༌༑ ༑ཞེས་སོགས་གསུངས་སོ། །གཞན་ཡང་འཇམ་དཔལ་གྱི་རྩ་བའི་ཆོ་ག་ཞིབ་མོ་ཁྲོ་བོ་རྣམ་པར་རྒྱལ་བའི་རྒྱུད་ཀྱི་ལེའུ་གསུམ་པར། ཨོཾ་ཨ་ཀྲོ་ཏེ་ཀ་ཡ་མཱནྟ་ཀ་ཧ་ན་མ་ཐ་བྷཉྫ་ཧཱུྂ་ཕཊ྄། འདི་ནི་ཁྲོ་བོའི་རྒྱལ་པོ་གང་གི་རྟོགས་པ་འདི་ཡིན་པ་དེའི་སྔགས་ཡིག་ཏུ་ཡིག་འབྲུ་བཅུ་བདུན་པའོ། །ཞེས་པ་དང༌། རྒྱུད་དེ་ཉིད་ཀྱི་ལེའུ་དྲུག་པར། ཁྲོ་བོ་ཡི་ནི་གཙོ་བོ་ཡི། །ཡི་གེ་བཅུ་བདུན་སྔགས་བཟླའོ། །ཨོཾ་ཨ་ཀྲོ་ཏེ་ཀ་ཡ་མཱནྟ་ཀ་ཧ་ན་མ་ཐ་བྷཉྫ་ཧཱུྂ་ཕཊ྄། འདི་ནི་རྩ་བའི་ཡིག་འབྲུ་བཅུ་བདུན་པའོ། །ཞེས་དང༌། ཡང་དེ་ཉིད་ཀྱི་ལེའུ་བདུན་པར། བཟླས་པའི་གྲངས་ནི་ངེས་པར་དག །ཆོ་ག་ཞིབ་མོ་འདིར་བཤད་བྱ། །ཐོག་མར་འབུམ་དྲུག་བཟླས་པ་ནི། །སྔོན་དུ་བསྙེན་པ་བྱས་པར་གྱུར། །དེ་ནས་བུ་དོན་ཆོ་ག་ཡི། །ཕྱིས་ཀྱང་དྲུག་འབུམ་བཟླ་བར་བྱ། །དེ་བཞིན་རི་རྩེར་འབུམ་བཟླས་ན། །འགྲུབ་པར་འགྱུར་ཏེ་གཞན་དུ་མིན། །ཞེས་རྒྱུད་འདིར་འབུམ་ཚོ་བཅུ་གསུམ་བཟླ་དགོས་པར་གསུངས་འདུག་པ་ནི་གོང་དུ་བཤད་པ་ལས་འབུམ་ཕྲག་གཉིས་ལྷག་འདུག་པའང་བྲིས་པ་ཡིན་ནོ། །ཡང་ན། གཤིན་རྗེ་དམར་ནག་འབར་བའི་རྨེ་རྒྱུད་ཀྱི་ཡི་གེ་འཁོར་ལོ་རྫོགས་ཆེན་གྱི་ལེའུ་ལས། ཨོཾ་ཨ་ཀྲོ་ཏེ་ཀ་ཡ་མཱནྟ་ཀ་ཧཱུྂ་ཕཊ྄། ཁྲོ་རྒྱལ་རྔམ་པའི་རྒྱལ་པོ་ཉིད། །བདག་དང་དབྱེར་མེད་རྫོགས་པའི་སྐུ། །ཞེས་པ་ནས། བརྟན་པ་དེར་གནས་དངོས་གྲུབ་བླང༌༑ ༑ཞེས་པའི་བར་རྒྱས་པར་གསུངས་སོ། །ཡང་ན་གསང་བ་དགུ་བཅུ་པའི་རྒྱུད་ཀྱི་སྲོག་ལ་དྲངས་པའི་ལེའུ་ལས།

སྲོག་སྙིང་རྣམ་པ་བཞིར་བསྟན་པ། །འདིར་ནི་འཁོར་ལོ་འཆི་བདག་སྲོག །ཡ་མ་མེ་དཀྵེ་རཱ་ཙ་ས་དོ་རུ་ཎ་ཡོ་ནི་ར་ཡ་ཙ་ནི། སྲོག་སྙིང་ཉི་མ་ཤར་ལྟས་འབྱུང༌། །ཞེས་པ་ནས། བྱ་རྒོད་གཟན་ལ་བཅུམ་བཞིན་འབྱུང༌། །ཞེས་པའི་བར་གསུངས་ཤིང༌། གསང་བ་དགུ་བཅུ་པའི་རྒྱུད་ཀྱི་སྔགས་ཀྱི་ཚད་མའི་ལེའུ་ལས། བྱ་རྒོད་གཟན་བཅུམ་དྲང་བ་ནི། །ཡ་མ་ཡི་གེ་བཅུ་བདུན་ཡིན། །ཞེས་གསུངས་པས། སྔགས་གསུམ་པོ་གང་རུང་རེ་རེ་ལ་གྲངས་ཚད་ཐེམས་ན་ངེས་པར་ཆོག་གོ། །མི་ཚིག་གིས་བར་མ་ཆོད་ཅིང་་་་་་མི་ཚིག་གིས་བར་མ་ཆོད་པར་དགོས་པ་ནི། ཟླ་གསང་རྩ་རྒྱུད་ཀྱི་ལེའུ་བཅུ་གསུམ་པ་ལས། རྣལ་འབྱོར་ནང་ལ་གཏམ་ཟེར་ན། །ལག་གི་བརྡ་ཡིས་བརྡ་བར་བྱ། །སྐོམ་དང་ཟ་བར་འདོད་པའི་ཚེ། །ལག་གི་བརྡ་ཡིས་བླང་བར་བྱ། །སྨྲས་ན་རྣལ་འབྱོར་རང་དོན་ཟད། །དེ་ཡི་ཕྱིར་ན་ལག་བརྡ་བྱ། །ཞེས་གསུངས་སོ། །ཟླ་གསང་མན་ངག་གི་རྒྱུད་ཀྱི་ལེའུ་བཅུ་གཅིག་པར། མི་ཚིག་བར་མ་ཆོད་པར་བྱ། །བར་དུ་ཆད་ན་དོན་མི་འགྲུབ། །ཅེས་མི་ཚིག་གིས་བར་ཆད་ན་མི་འགྲུབ་པར་གསུངས་སོ། །ལུས་ལྷ། ངག་སྔགས། སེམས་ཏིང་ངེ་འཛིན་དང་མ་བྲལ་བའི་ངང་ནས་འབུམ་ཕྲག་དྲུག་འབད་རྩོལ་གྱིས་བཟླ་དགོས་ཤིང༌།་་་་་འཕགས་པ་འཇམ་དཔལ་གཤིན་རྗེའི་གཤེད་ཀྱི་འཁྲུལ་འཁོར་ནག་པོའི་སྔགས་རྒྱུད་ཀྱི་རྟོགས་པ་བཅུ་གཉིས་པར། བསྙེན་པའི་སྔོན་དུ་ལྷ་བཀུག་ལ། །ལས་ཀྱི་ཡ་མ་དབང་དུ་བསྡུ། །དེ་ཡིས་ལྷ་དང་ལས་བྱེད་འགུགས། །རྟགས་ལ་རྣམ་པ་གསུམ་དུ་འདོད། །ཕྱི་ནང་གསང་བ་ཉེ་བར་བཟུང༌། །མགོ་ལ་སྐུ་གདུང་ཉི་ཟླ་དགབ། །བདུད་རྩི་ཆུ་ལ་ཁྲུས་བྱེད་དང༌། །མེ་ཆུ་འཚུབ་མ་འཁྱིལ་བ་དང༌། །ཀླུ་ལ་འདོད་ཕྱིར་གྲུ་རྫིངས་བསྐྱོད། །གླང་ཆེན་ཞོན་ནས་མཚོན་ཆ་འབར༑ ༑གཅན་གཟན་བདག་རྒོད་བསྲུན་པ་ལས། །ཕྱི་ཡི་རྟགས་ཏེ་སྤྱོད་ཡུལ་འགྱུར། །དམ་རྫས་གསར་ཏེ་མཆོད་པ་འབུལ། །གཏོར་མ་བྲན་ལ་སྨན་རཀ་བརླབས། །མཚན་མ་གསུམ་གྱི་དགུག་པར་བྱ། །ཞེས་གསུངས་སོ། །སྔགས་ཀྱི་བཟླ་ཚུལ་ནི།

འཇམ་དཔལ་ཁྲོ་བོ་གསང་བའི་རྒྱུད་ཀྱི་སྒྲུབ་པ་རྩ་བའི་ལེའུ་ལས། མྱུར་བ་མ་ཡིན་བུལ་མ་ཡིན། །དེ་བཞིན་ཡི་གེ་འདྲེས་པ་མིན། །ཤིན་ཏུ་གསལ་ཕྱིར་དག་པར་སྨྲ། །སྔགས་ཀྱི་བཟླས་གྲངས་བར་བར་དུ། །ཁྲོ་བོའི་བསྟོད་སྒྲ་གཏང་བར་བྱ། །རཀྟ་དམར་གྱི་ཨརྒྷཾ་འབུལ། །ཞེས་དང༌། གཤིན་རྗེ་ནག་པོའི་སྔགས་རྒྱུད་ནག་པོ་བསྙེན་པའི་ལེའུ་ལས། མི་སྨྲ་གླལ་སྦྲིད་གཉིད་རྨུགས་དང༌། །སྒྲ་ཆེན་སྒྲ་ཆུང་སྤངས་པ་དང༌། །དྲག་ལ་རིང་ཐུང་རྫོགས་པ་དང༌། །ཡིད་ལ་ཤི་ཕུར་བཟླས་པའི་ཐབས། །ཞེས་གསུངས་པ་ལྟར༑ སྐྱོན་རྣམས་དང་བྲལ་ཞིང་མཚན་ཉིད་ཚང་བར་བཟླ་དགོས་སོ། །དེའང་མཚམས་བཅད་པའི་ནུབ་དམིགས་པ་དང་བཅས་པའི་སྔགས་བརྒྱ་སྐོར་ཞིག་འབུལ། དེ་ནས་སྨན་མཆོད་བྱས་ནས་ངལ་གསོ་ཞིང་ཉལ༑ ཕྱི་ནངས་ནས་གོང་ལྟར་གསལ་བ་གསུམ་ལྡན་གྱི་ངང་ནས་བཟླས་པ་ལ་འབད། ཐུན་ཡང་ལྕོགས་ན་དྲུག་དང་མ་ལྕོགས་ན། ཉི་ཤར། ཉིན་གུང༌། ཉི་ནུབ། ནམ་སྲོད་དེ་ཐུན་བཞིར་བཅད་པའི་ཐུན་བར་དུ་གཏོར་མ་གསོས་ལ་སྨན་་་་་་ཟླ་གསང་འདུས་པ་རྩ་བའི་རྒྱུད་ཀྱི་ལེའུ་དྲུག་ཅུ་པ་ལས་ཀྱི་སྤྱིར་བསྟན་པའི་སྐབས་སུ། དུས་དྲུག་དག་ཏུ་པཱུ་ཛ་འབུལ། །ཞེས་གསུངས་པ་ཡིན་ནོ། །རཀ་བསྒྲོན་ནས་གཏོར་བསྔོས་རྒྱས་པར་བྱས་པའི་མཐར་མཆོད་སྦྱིན་ཐབས་མཆོག་བདེ་ཆེན་མས་བསྟོད་དོ། །་་་་་འཇམ་དཔལ་ཁྲོ་བོ་གསང་བའི་རྒྱུད་ཀྱི་ལེའུ་ལྔ་པ་ལས་སྒྲུབ་པ་རྩ་བའི་ལེའུ་ལས། སྔགས་ཀྱི་ཕྲེང་བ་སྟོང་རྫོགས་དང༌། །སྟོང་ཚོགས་མཆོད་དང་བསྟོད་པ་བྱ། །དྲག་ཏུ་གཤིན་རྗེ་དགུག་པ་ནི། །ཁུར་བས་བུས་ལ་གཤིན་མི་བཟློག །ཚིལ་ཆེན་མར་མེ་རྒྱུན་མི་འཆད། །རྫ་རྔ་རྡུང་ཞིང་བཤུག་པས་འབོད། །སྨུག་ཆེན་སྨུག་ཆུང་སྙིང་་་་་་སྨུག་ཆེན་སྨུག་ཆུང་ཞེས་པ་མི་རྟའི་སྙིང་ཡིན་པ་ཁ་ཐུན་ནས་གསུངས། བྱུང་ན་གྲི་སྙིང་དགོས། མ་བྱུང་ན་ཤི་སྙིང་དགོས་སྙིང་གཉིས་པོ་སྤྱན་གཟིགས་སུ་ངེས་པར་ཚང་དགོས་སོ། །ལ་འབོད། །དཔའ་བོ་གྲུབ་རྟགས་བྱུང་བ་ན། །དངོས་གྲུབ་དམ་ཞིང་བླང་བར་བྱ། །ཞེས་གསུངས་པ་ལྟར།

ཚིལ་ཆེན་གྱི་མར་མེ། ཤ་ཆེན་དང་གུ་གུལ་བསྲེས་པའི་བདུག་པ་དང༌། རྫ་རྔའི་རོལ་མོ་སོགས་བྱ་བ་གནད་ཆེའོ། གཤིན་རྗེ་དགྲ་ནག་གི་རྒྱུད་རྭ་སྔ་འགྱུར་ལས། གཞན་དུ་མི་འགྲོ་སེམས་ཀྱིས་ནི། །དེ་ལྟའི་ཚུལ་དུ་བཟླ་བར་བྱ། །ཡུལ་གཞན་བློ་ནི་འགྲོ་བ་ནི༑ ༑བསྐལ་པར་བཟླས་ཀྱང་འབྲས་མི་འབྱིན། །མི་འཕྲོ་སེམས་ཀྱིས་འདི་ཟློས་ཤིག །ཞེས་གསུངས་པས། སྔགས་བཟླ་བའི་ཚེ་འདི་གཞན་དུ་མི་ཡེང་བར་དམིགས་པ་ཁོ་ན་ལ་ཡིད་གཏོད་པར་གལ་ཆེའོ། །་་་་་དེ་ལྟར་མ་ཡེང་བར་བཟླ་བ་ནི་བསྙེན་པ་བྱ་བའི་རིམ་པའོ། །དེ་དག་ཟིན་ནས་གཉིས་པ་ཉེ་བའི་བསྙེན་པ་ནི།་་་་་གཤིན་རྗེའི་གཤེད་དམར་ནག་འབར་བའི་དམེ་རྒྱུད་ཀྱི་ཡི་གེ་འཁོར་ལོ་རྫོགས་ཆེན་གྱི་ལེའུ་ལས། སྒོ་ལྔ་རང་སོར་ཡུལ་མེད་བཞག །བེ་ཤིང་འཁར་བ་ཐོགས་ནས་ནི༑ ༑བྱིང་ནས་རྨུགས་པར་གྱུར་པས་ན། །འཇམ་དབྱངས་ཁྲོ་བོ་འཛབ་ཀྱིས་བསྐུལ། །གཡེལ་ཞིང་རྒོད་པར་གྱུར་པས་ན། །མཚན་མ་ལྷ་ཡི་སྐུ་ལ་བལྟ། །ཞེས་སོགས་གསུངས་པས། འདིའི་སྒྲུབ་མ་གྲོལ་བར་བྱིང་ན་འཛབ་དབྱངས་བྱ་ཞིང་དལ་གྱིས་བཅག །རྒོད་ན་བྲིས་སྐུ་སོགས་ལ་བལྟ་ཞིང་བྱིང་རྒོད་སོགས་ཀྱི་སྐྱོན་གང་ཡང་མེད་པར་དུས་ཀུན་ཏུ་བསྒྲིམས་ནས་བཟླ་བར་བཤད་པ་ལྟར་བྱའོ། །བཅའ་གཞི་་་་་་དཀྱིལ་འཁོར་གྱི་དབུས་སུ་འཇམ་དཔལ་ཞི་ཁྲོའི་སྐུ་གཉིས་བྲིས་འབུར་གང་རུང་ཞིག་ངེས་པར་བཤམ་དགོས་པར་གསུངས་འདུག་གོ། །གོང་མ་ལྟར་ལ་དམིགས་པ་སོགས་ཀྱང་སྔ་མ་ཁོ་ན་ལྟར་གྱི་སྔགས་འབུམ་ཕྲག་གསུམ་བཟླ་བ་ནི་ཉེ་བའི་བསྙེན་པའོ། །དེ་ལྟར་ཉེ་བའི་བསྙེན་པའི་བསྙེན་པ་བགྱིས་པའི་ཚེ་བྲིས་སྐུ་ཞལ་འཛུམ་པ་དང་དངོས་ཉམས་རྨི་ལམ་སོགས་སུ་ལྷ་དང་མཇལ་བ་རྟགས།་་་་་འཕགས་པ་འཇམ་དཔལ་གཤིན་རྗེའི་གཤེད་འཁྲུལ་འཁོར་ནག་པོའི་སྔགས་རྒྱུད་ཀྱི་རྟགས་པ་བཅུ་གཉིས་པར། འོད་སྒྲ་དུ་བ་བཞད་ཅིང་སྟོན༑ ༑དམག་འདྲེན་དྲག་འགུགས་དབང་བྱེད་དང༌། །ནགས་གཅོད་བཀའ་འབེབས་ལས་ཆུད་དང༌། །གཅོད་ཡུལ་རྣམ་གསུམ་ནང་དུ་བཟུང༌། །་་་་་བསྙེན་སྒྲུབ་ཀྱི་རྟགས་རྒྱུད་རྣམས་ནས་གསུངས་པ་རྣམས་ཟུར་དུ་ཕྱུང་ནས་བྲིས་ཡོད་པས་དེར་རྟོགས་པར་བྱོས་ཤིག

༑བཟང་པོ་མཐོང་ནས།་་་་་གཤིན་རྗེ་དམར་ནག་འབར་བའི་དམེ་རྒྱུད་ཀྱི་ལེའུ་བདུན་པ་ལས། ཡབ་ཡུམ་སྔགས་ཀྱི་གྲངས་ཐེམས་ཀྱང༌། །ལྷ་ཡི་མཚན་མ་མ་མཐོང་ན། །སྔགས་ཀྱི་དམ་ཚིག་ཉམས་སྲིད་པས། །ལྷུ་དུམ་བསིལ་ལ་བརྒྱ་བཤགས་བྱ། །དེ་ཚེ་མཚན་རྟགས་མངོན་དུ་སྟེར། །དེས་ཀྱང་རྟགས་རྣམས་མ་བྱུང་ན། །གསང་སྔགས་སིལ་བུར་ལྷོ་ནས་ཟློག །རུ་ཁུག་རུ་ཉལ་རུ་ཉམ་དང༌། །རུ་ཟློག་དྲག་པོ་གནད་ལ་དབབ། །བཞུག་ལ་དམ་ཚིག་བརྡར་ལ་བཞག །ལྷ་ཉིད་གནད་ལ་དབབ་པ་སྟེ། །ཀུན་གྱི་བྱ་བ་མ་ཡིན་ཏེ། །ལས་རྣམས་གདུགས་ན་བྱ་བ་ཡིན། །མི་འགྲུབ་འགྲུབ་པར་བྱ་བ་རུ། །དམ་ཚིག་བདག་པོས་རབ་ཏུ་གསུངས། །ཞེས་པ་ལྟར། བསྙེན་པ་ཐེམས་ཀྱང་རྟགས་མ་བྱུང་ན་བཤགས་པ་དང་སྔགས་བསྡེབས་སོགས་མན་ངག་ལྟར་བྱ་དགོས་སོ། །གསུམ་པ་སྒྲུབ་པ་ཆེན་པོ་ལ་འཇུག་པ་ནི། བཅའ་གཞི་སྔ་མ་ལྟར་ལ་དངོས་གྲུབ་ཀྱི་རྫས་ཐོད་པ་མཚན་ལྡན་དུ་་་་་་འཕགས་པ་འཇམ་དཔལ་གཤིན་རྗེའི་གཤེད་འཁྲུལ་འཁོར་ནག་པོའི་གསང་རྒྱུད་ཀྱི་ལེའུ་ལྔ་པ་བསྙེན་སྒྲུབ་ཀྱི་སྐབས་སུ། ཕྱི་ནང་གསང་བ་དྲོད་རྙེད་ནས། །དངོས་གྲུབ་རྫས་ལ་བརྟེན་ནས་བླང༌། །ཞེས་གསུངས་པའི་དོན་ནོ། །བ་དམར་སེར་གྱི་འོ་མ་དང༌། སྦྲང་རྩི། ཆོས་སྨན་སྦྱར་བའི་རྫས་རྣམས་བཀོད་ལ་་་་་མཆོད་གཏོར་སོགས་སྔ་མ་བས་ཆེས་ཆེར་བཟང་བ་བཤམས་དགོས་སོ། །བདག་གི་ཐུགས་ཀའི་ས་བོན་ལས་འོད་ཟེར་འཕྲོས། ་བདེ་བར་གཤེགས་པ་རྣམས་ཀྱི་ཐུགས་རྒྱུད་བསྐུལ། དེ་རྣམས་ཀྱི་ཐུགས་རྗེ་དང་ནུས་མཐུ་ཐམས་ཅད་བསྡུས་ནས་བྲིས་སྐུ་དང་དངོས་གྲུབ་ཀྱི་རྫས་ལ་ཐིམ་པར་དམིགས་ཤིང༌།་་་་་བརྟན་གཡོ་སྣོད་བཅུད་དང་ཚེ་སྲོག་ཕུན་ཚོགས་ཐམས་ཅད་འོད་ཀྱིས་བསྡུས་ནས་དངོས་གྲུབ་ཀྱི་རྫས་བསྟིམ་དམིགས་པ་བྱའོ། །གོང་གི་སྔགས་འབུམ་ཕྲག་གཅིག་གླལ་དང་སྦྲིད་པ་དང་བརྐྱང་བསྐུམ་དང་གཉིད་སོགས་ལེ་ལོ་སྤངས་ནས་ཤིན་ཏུ་བསྒྲིམས་ལ་བཟླ།་་་་་འཕགས་པ་འཇམ་དཔལ་གཤིན་རྗེའི་གཤེད་འཁྲུལ་འཁོར་ནག་པོའི་སྔགས་རྒྱུད་ཀྱི་རྟོགས་པ་བཅུ་གཉིས་པར།

ལུས་ཡང་སེམས་རྒོད་ལུང་པ་ཡངས། །ཟས་དང་སྐོམ་གྱི་འདུ་ཤེས་ཆུང༌། །ཉོན་མོངས་ཤེས་པ་ཡུལ་ལ་འབར། །སྔ་ན་མེད་པའི་རྣམ་པ་མྱོང༌། །ཐུན་ཟོར་ཤ་ཁྲག་རྔམས་ཤིང་རོལ། །གསང་བའི་རྟགས་དེ་དབང་སྡུད་སྦྱར། །སྒྲུབ་པའི་དངོས་གཞི་དྲན་པས་བཟུང༌། །ཐོད་པ་བདུད་རྩི་རིལ་བུ་བརྟེན། །དཀར་དམར་དར་དང་རུ་སྤོས་རྒྱས། །བཏབ་པ་དཀྱིལ་འཁོར་དབུས་སུ་བཞག །བརྒྱད་ནས་ཉའི་བར་དག་ཏུ། །དངོས་གྲུབ་དབང་གི་ཆོ་ག་བྱ། །རྣལ་འབྱོར་འདོད་པ་བླང་བར་བྱ། །ལྷ་སྒོམ་མཚམས་བཅད་སིདྡྷི་དབབ། །འོད་སྒྲ་དྲི་རླངས་དྲི་དང་སྐུར། །མཐོང་ན་ཕྱག་རྒྱ་གནས་ལྔར་བཏབ། །དངོས་གྲུབ་བླངས་ལ་སྦྱོར་བ་བརྩམ། །ཞེས་གསུངས་པས། དངོས་གྲུབ་ཀྱི་སྒྲུབ་པ་འདི་ཚེས་བརྒྱད་ནས་བཅོ་ལྔའི་བར་གསུངས་འདུག་གོ། །ཐུན་མཚམས་རྣམས་སུ་གཏོར་འབུལ་དང༌། མཆོད་བསྟོད། ཚོགས་མཆོད་སོགས་རྒྱས་པར་བྱ། སྔགས་འབུམ་ཕྲག་ཐེམས་མ་ཐག་དངོས་གྲུབ་ཀྱི་རྫས་གཞན་ལ་མི་སྟེར་བར་རང་ཉིད་ཀྱིས་ལོངས་སྤྱོད་དོ། །ཚུལ་དེ་ལྟ་བུ་བརྩོན་འགྲུས་དྲག་པོ་བགྱིས་ན་བསམ་པ་ཡིད་བཞིན་འགྲུབ་བོ། །བཞི་པ་འཆི་བདག་བདུད་དང་གཤིན་རྗེའི་བསྙེན་པ་ནི། སྔར་བཤད་པའི་གནས་སུ་རང་ཉིད་ཁ་ལྷོར་བལྟས། བཅའ་གཞི་སོགས་སྔར་ལྟར་ལྷ་སྒྲུབ་བདུད་སྒྲུབ་་་་་་འཕགས་པ་འཇམ་དཔལ་གཤིན་རྗེའི་གཤེད་འཁྲུལ་འཁོར་ནག་པོའི་གསང་རྒྱུད་ཀྱི་ལེའུ་ལྔ་པ་བསྙེན་སྒྲུབ་ཀྱི་སྐབས་སུ། ལས་ཀྱི་གཤིན་རྗེ་ལས་ལ་བསྙེན། །ཁྲོ་བོ་བཀའ་ཐུབ་ལྡན་པ་ཡིས། །སྣ་ཚོགས་མེ་རླུང་འཁོར་ལོ་ཅན། །འབུམ་ཕྲག་བཀུག་ལ་ཡ་སོགས་བཟླ། །ཞེས་གསུངས་པའི་དོན་ནོ། །བྱས་ལ། བདག་གི་ཐུགས་ཀའི་ས་བོན་ཧཱུྂ་མཐིང་ནག་གནས་འགྱུར་བ་ཡི་གེ་ཡ་ནག་པོ་མེ་འོད་འབར་བ་ལྟ་བུའི་མཐར་བཟླ་བྱའི་སྔགས་ཀྱིས་བསྐོར་བ། སྔགས་བཟླས་པའི་ཚེ་དེ་རྣམས་ལས་འོད་ཟེར་དྲག་པོ་ཕྱོགས་བཅུར་འཕྲོས། འཆི་བདག་ལས་ཀྱི་གཤིན་རྗེ་ལ་སོགས་པ་འཇིག་རྟེན་ཁམས་ཀྱི་དྲེགས་བྱེད་ཐམས་ཅད་ལ་ཕོག་པས།

དེ་ཐམས་ཅད་ཀྱི་ཐུགས་དམ་གྱི་རྒྱུད་བསྐུལ་ནས། རང་དབང་མེད་པར་བདག་གི་མདུན་དུ་སྐད་ཅིག་ལ་བཀུག་པས། བདག་ཉིད་ཀྱི་བཀའ་ཉན་གསུང་སྡོད་པར་བསམ་ལ་ཡིད་མ་ཡེང་བར་སྔགས་གོང་མ་འབུམ་ཕྲག་གཅིག་དྲག་ཏུ་བཟླའོ། །་་་་་ཐུན་མཚམས་རྣམས་སུ་དྲེགས་པ་རྣམས་ལ་གཏོར་མ་འབུལ༑ དེས་ནི་དྲེགས་པ་ལྷ་སྲིན་སྡེ་བརྒྱད་ཐམས་ཅད་དབང་དུ་འདུ་བར་འགྱུར་རོ། །་་་་་དེ་ལྟར་བསྙེན་པའི་གྲངས་ཚད་བསྡོམས་ཏེ་བསྟན་པ་དང༌། བསྙེན་སྒྲུབ་ཀྱི་རྟགས་དང་མཇུག་གི་ལས་བསྟན་པ་ནི། ཟླ་གསང་དྲེགས་པ་ཚར་གཅོད་ཀྱི་སྐབས་བརྒྱད་པ་ལས། བྱེ་བ་ཆིག་འབུམ་ཐོན་པ་དང༌། །མེ་ཏོག་ལ་སོགས་མཆོད་པར་བྱ། །ཕྱི་ནང་གི་ནི་མཆོད་པས་མཆོད། །བསྟོད་པ་ལ་སོགས་ཅི་རིགས་བྱ། །བསྙེན་པ་རྫོགས་ནས་རྟགས་རྣམས་འབྱུང༌། །འཇམ་དཔལ་སྐུ་ནི་ཞལ་འཛུམ་མམ། །རྟགས་ནི་བཟང་པོ་བྱུང་བ་དང༌། །དངོས་གྲུབ་ལ་སོགས་བླང་བར་བྱ། །དེ་ནས་དྲག་པོའི་སྦྱོར་ལ་འཇུག །ཅེས་གསུངས་པས། འདིའི་བསྙེན་པ་ལ་གྲངས་ཀྱི་བསྙེན་པ་བྱེ་བ་གཅིག་དང་འབུམ་ཕྲག་གཅིག །དུས་ཀྱི་བསྙེན་པ་ཟླ་བ་གཅིག །རྟགས་ཀྱི་བསྙེན་པ་འཇམ་དཔལ་ཞལ་འཛུམ་པ་སོགས་ཏེ། གཤིན་རྗེ་ནག་པོ་རྔམ་རྒྱུད་ཀྱི་ལེའུ་བཞི་པ་ལས། དུས་ཀྱི་བསྙེན་སྒྲུབ་ཟླ་བ་གཅིག་གསུངས་པ་ནི། བཤད་ཚོད་ཙམ་ཡིན་གྱི་ནང་རྟགས་ངེས་ཤེས་འདྲོངས་པ་ཞིག་མ་བྱུང་བར་དུ་འབད་དགོས་སོ། །མཐོང་དང་ཐོས་དང་རྨི་ལམ་གསུམ། །བསྙེན་པའི་རྟགས་སུ་ཤེས་པར་བྱ། །ཞེས་རབ་ཞལ་མཐོང་བ། འབྲིང་ལུང་བསྟན་ཐོབ་པ། ཐ་མ་རྨི་ལམ་དུ་རྟོགས་པ་དགོས། རྨིས་ལྟས་བཤད་ཚོད་དང་མཐུན་པ་བྱུང་ན་རབ། ཕྱེད་ཙམ་བྱུང་ན་འབྲིང༌། འགའ་ཟུང་ལ་ཆ་འགྲིག་ཙམ་བྱུང་ན་ཉན་ཙམ་ཡིན་ནོ། །དེ་ལྟར་གྲངས་དུས་རྟགས་གསུམ་གྱི་སྒོ་ནས་བསྙེན་པ་དང༌། ཉེ་བསྙེན། སྒྲུབ་ཆེན་འཆི་བདག་བདུད་སྒྲུབ་རྣམས་བག་ཡོད་བརྩོན་ལྡན་གྱིས་བསྒྲུབས་ཕྱིན་གཤིན་རྗེའི་གཤེད་ཀྱི་ཕྲིན་ལས་གང་སྒྲུབ་ཀྱང་མྱུར་དུ་འགྲུབ་པར་འགྱུར་རོ། །དེ་ལྟར་བསྙེན་སྒྲུབ་བཞི་པོ་ཐེམས་ནས་ཚོགས་འཁོར་དང༌།

བསྐང་བཤགས་དང༌། གཏོར་འབུལ་སོགས་གཏང་རག་གི་མཆོད་པ་རྒྱས་པ་བྱ།་་་་་གྲུབ་ན་བཟླས་པའི་བཅུ་ཆའི་ཞི་བའི་སྦྱིན་སྲེག་དང༌། རྒྱས་པ་དང༌། དབང་དང༌། དྲག་པོ་རྣམས་ཀྱང་ཚར་བཅུ་རེའམ། ལྔ་རེའམ། གསུམ་རེ་སོགས་གང་འགྲུབ་བྱ་དགོས་སོ། །དགུ་གསུམ་སོགས་ལ་ཚོགས་དང་གཏོར་མཆོད་ལ་འབད་ཅིང་རྒྱུན་གྱི་བསྙེན་པ་མ་ཆད་པར་བྱ་དགོས་པ་སྐབས་བསྟུན་གྱི་ལས་སྦྱོར་རྣམས་གང་དགོས་ཉམས་སུ་བླང་བས་འབད་མེད་དུ་འགྲུབ་པར་འགྱུར་རོ། །རྣལ་འབྱོར་རྒྱུན་གྱི་བསྙེན་པ་ཨ་ཀྲོ་ཏེ་ཀ་ལ་བྱེད་དགོས་པ་ནི། ཟླ་གསང་དྲེགས་པ་ཚར་གཅོད་ཀྱི་རྒྱུད་ཀྱི་ལེའུ་ཉེར་གསུམ་པ་ལས། ཨ་ཀྲོ་ཏེ་ཀ་རྒྱུན་གྱི་སྔགས། །དེ་དག་སྒྲུབ་པའི་དུས་སུ་ཡང༌། །མི་ལུས་ཕྲ་མེན་མགོ་ཅན་རབ། །སྤྲུལ་དང་བྲན་གཡོག་ཕ་མ་དང༌། །མཆོད་རྟེན་ཆེན་པོ་སྟོན་པ་དང༌། །མེ་ཆེན་རྟག་ཏུ་འབར་བ་དང༌། །རྒྱ་མཚོ་དག་ཏུ་མི་ནུབ་དང༌། །དེ་དག་མ་བྱུང་བར་དུ་བསྒྲང༌། །ཞེས་རྒྱུན་བསྙེན་ཨ་ཀྲོ་ཏེ་ཀ་བྱས་པས་འཕྲལ་དུ་རྟགས་དེ་རྣམས་འབྱུང་བར་གསུངས་སོ། །རྟགས་དེ་རྣམས་ནི་རྒྱུན་བསྙེན་གྱི་རྟགས་ཡིན་ནོ། །་་་་་ཟླ་གསང་འཇམ་དཔལ་ཐུགས་ཀྱི་རྩ་རྒྱུད་ལེའུ་སུམ་ཅུ་པར་ཡང་ཚིག་འདི་ཁོ་ན་གསུངས་འདུག་གོ། །རྒྱུན་སྔགས་ཨ་ཀྲོ་བཅུ་བདུན་པའམ་བཅུ་གཅིག་པ་ལ་བྱེད་པ་ནི་ཟླ་གསང་ཐུན་མོང་མ་ཡིན་པའི་རྒྱུད་ཀྱི་ལེའུ་བརྒྱད་པར། བཅོ་བརྒྱད་པ་དང་བཅུ་གཉིས་པ། །རྒྱུན་གྱི་བཟླས་པའི་བསྙེན་པའོ། །ཞེས་གསུངས་པའང་ཨ་ཀྲོ་གཉིས་ཀྱི་ཡ་མ་ཨནྟ་ཀ་ཞེས་མཚམས་མ་སྦྱར་བས་ཚེག་བར་རེ་རེ་བཅུ་བདུན་པ་ལ་བཅོ་བརྒྱད་པ་དང༌། བཅུ་གཅིག་པ་ལ་བཅུ་གཉིས་པ་ཞེས་སོ། །ཚུལ་འདི་ནི་དེང་སང་ཁ་ཅིག་བསྙེན་སྒྲུབ་མ་ཐེམས་པར་ལས་སྦྱོར་གྱི་སྒྲུབ་པ་ལ་ཞུགས་པས་གཞན་ཕུང་བདག་བརླག་བྱེད་པ་མང་དུ་མཐོང་བས་དད་བརྩོན་དང་ལྡན་པ་འགའ་ཞིག་ལ་ཕན་པའི་ཕྱིར། གཤིན་རྗེའི་གཤེད་ཀྱི་རྣལ་འབྱོར་པ་རིག་པ་འཛིན་པ་ཆོས་ཀྱི་གྲགས་པས༑

ཟླ་གསང་རྩ་བའི་རྒྱུད་ཆེན་པོའི་ལེའུ་བཅུ་དགུ་པ་དང༌། ཉེར་བཞི་པ་དང༌། ཉེར་ལྔ་པ་དང༌། ཉེར་དྲུག་པ་དང༌། ཉེར་བདུན་པ་དང༌། ཉེར་བརྒྱད་པ་རྣམས་དང༌། ཟླ་གསང་འཇམ་དཔལ་ཐུགས་ཀྱི་རྩ་རྒྱུད་ལེའུ་བཅུ་གཅིག་པ་དང༌། བཅུ་གསུམ་པ་དང༌། ཉེར་བདུན་པ་དང༌། ཟླ་གསང་དྲེགས་པ་ཚར་གཅོད་ཀྱི་རྒྱུད་ཀྱི་སྐབས་བརྒྱད་པ་དང༌། སྐབས་བཅུ་གསུམ་པ་དང༌། བཅུ་བཞི་པ་དང༌། བཅོ་ལྔ་པ་དང༌། བཅུ་དྲུག་པ་དང༌། བཅུ་བདུན་པ་རྣམས་ནས་ནོར་འཁྲུལ་མེད་ཅིང་འཇུག་པ་བདེ་བར། ལྕགས་རྟ་ཟླ་བ་དགུ་པའི་ཚེས་དྲུག་ལ་ཕོ་བྲང་བཀྲ་ཤིས་ཚུགས་ཀྱི་འདུན་སའི་རེའུ་མིག་པདྨ་འོད་དུ་བཀོད་པ་དགེ་བར་གྱུར་ཅིག །རབ་འབྱམས་རྒྱུད་སྡེ་གདེངས་ཅན་འབུམ་ཕྲག་དབུས། །མཐུ་སྟོབས་དབང་ཕྱུག་ཟླ་གསང་ཀླུ་དབང་གི། །སྙིང་བཅུད་རིན་ཆེན་དབང་གི་རྒྱལ་པོ་འདི། །བློ་ཅན་དེད་དཔོན་ང་མིན་གཞན་སུས་རྙེད།། །།སརྦ་མངྒ་ལཾ།།
