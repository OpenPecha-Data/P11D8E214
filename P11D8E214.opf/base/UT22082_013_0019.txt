ཁྱབ་བདག་རྡོ་རྗེ་འཆང་དང་ཤེས་རབ་བཟང༌། །ཡེ་ཤེས་དངོས་གྲུབ་བདེ་བའི་རྡོ་རྗེ་དང༌། །བཞད་པ་རྡོ་རྗེ་ཟླ་འོད་གཞོན་ནུའི་དཔལ། །གསོལ་བ་འདེབས་སོ་བདག་རྒྱུད་བྱིན་གྱིས་རློབས། །རྡོ་རྗེ་རྒྱལ་པོ་མཉམ་མེད་རཏྣ་ཤྲཱི། །བསོད་ནམས་གྲགས་དང་རྡོ་རྗེ་གྲགས་པ་དང༌༑ ༑རིན་ཆེན་སེང་གེ་གྲགས་པ་བསོད་ནམས་ཞབས། །རྡོ་རྗེ་རིན་ཆེན་རྡོ་རྗེ་རྒྱལ་པོ་དང༌། །ཆོས་ཀྱི་རྒྱལ་པོ་དོན་གྲུབ་རྒྱལ་པོའི་མཚན། །ནམ་མཁའ་རྒྱལ་མཚན་ངག་དབང་རྒྱལ་མཚན་ཞབས། །ངག་གི་དབང་པོ་རིན་ཆེན་ཆོས་རྒྱལ་དང༌། །ཨཱ་ནནྡ་ཞབས་རཏྣ་དྲག་པོ་རྩལ། །དཔལ་གྱི་རྒྱ་མཚོ་སྔགས་འཆང་མཆོག་གྲུབ་རྒྱལ། །རྡོ་རྗེ་འཆང་དབང་མཁན་ཆེན་རྣམ་འཇོམས་ཞབས། །བཀའ་དྲིན་སུམ་ལྡན་རྩ་བའི་བླ་མ་དང༌། །ཡི་དམ་དཔའ་བོ་མཁའ་འགྲོ་ཆོས་སྐྱོང་ལ། །གསོལ་བ་འདེབས་སོ་བྱིན་གྱིས་བརླབ་ཏུ་གསོལ། །མ་དག་ལས་ཀྱི་ཕུང་ཁམས་སྐྱེ་མཆེད་རྣམས། །རྣམ་དག་དབང་བཞིས་སྐུ་བཞི་ཡེ་ཤེས་ལྔར། །རབ་སྦྱངས་མངོན་བྱས་ས་ལམ་ཀུན་རྫོགས་པར། །བདག་རྒྱུད་ཡོངས་སུ་སྨིན་པར་བྱིན་གྱིས་རློབས། །མེད་བཞིན་ལས་འཁྲུལ་བག་ཆགས་ཀྱིས་བསླད་པ། །ངར་འཛིན་ལུས་ཀྱི་གྲོང་ཁྱེར་འདུས་བྱས་ཆོས། །གསལ་སྟོང་ཡི་དམ་ལྷ་ཡི་ཏིང་འཛིན་གྱིས། །ཟུང་འཇུག་བསྐྱེད་རིམ་རྟོགས་པ་མཐར་ཕྱིན་ཤོག །སྤྲུལ་པའི་འཁོར་ལོར་ཨ་ཐུང་སྦར་བའི་མེས། །ར་ལ་སྲོག་རྩོལ་ཉི་ཟླའི་རྒྱུ་བ་ཀུན༑ ༑དྷཱུ་ཏི་སྒྲ་གཅན་ཁྱིམ་དུ་རབ་བསད་ནས། །འབར་འཛག་བདེ་དྲོད་མི་རྟོག་མཐར་ཕྱིན་ཤོག །ཆོས་ཀྱི་འཁོར་ལོར་དྭངས་མ་འོད་ཀློང་དུ། །རླུང་སེམས་འབྱུང་བཞི་སྣང་མཆེད་ཐོབ་པའི་ཚོགས། །སྐད་ཅིག་ཐིམ་ནས་ཚོར་དྲུག་གིས་སྟོང་པའི། །དོན་གྱི་འོད་གསལ་མཆོག་ལ་གནས་པར་ཤོག །ལོངས་སྤྱོད་འཁོར་ལོར་རླུང་སེམས་རབ་འདུས་ཤིང༌། །ས་བོན་བསམ་གཏན་སྦྱོར་བས་ཉིང་འཁྲུལ་ཀུན། །རང་ངོ་ཤེས་ཤིང་སྤྲུལ་བསྒྱུར་སྒྱུ་མར་རྟོགས༑ ༑རྩལ་འབྱོང་བོགས་འདོན་འཁྲུལ་བ་དྲུང་འབྱིན་ཤོག

།བདེ་ཆེན་འཁོར་ལོར་ཕྱི་ནང་རྟེན་འབྱུང་གི། །དག་དང་མ་དག་བརྟན་གཡོ་མ་ལུས་པ། །བཅོས་མིན་སྒྱུ་མར་རྟོགས་ཤིང་འཇུག་ལྡང་གིས། །དོན་གྱི་སྒྱུ་ལུས་རྟོགས་པ་ཐོབ་གྱུར་ཅིག །འཕོ་བྱ་འཕོ་བྱེད་བྲལ་བ་རང་སེམས་ཉིད། །རིག་མདངས་ལོངས་སྐུ་རྣམ་པར་གོང་འཕེལ་ཞིང༌། །གནས་སྐབས་རླུང་སེམས་ཡིག་འབྲུའི་སྦྱོར་བ་ཡི། །འཕོ་དང་གྲོང་འཇུག་གེགས་མེད་འགྲུབ་པར་ཤོག །བར་དོ་རང་ངོ་ཤེས་ཤིང་དབང་འབྱོར་ཏེ། །འོད་གསལ་སྒྱུ་མའི་འཇུག་ལྡང་གནས་ཟབ་མོ། །འཁོར་བའི་རྒྱ་མཚོར་འཇུག་པ་རྩད་བཅད་ནས། །མཁའ་སྤྱོད་རིག་འཛིན་ཞིང་སོགས་བགྲོད་གྱུར་ཅིག །མ་དག་ཉོན་མོངས་རྩ་རླུང་ཐིག་ལེའི་འགྲོས། །ལྷ་སྔགས་ཏིང་འཛིན་སྦྱོར་བས་གནས་བསྒྱུར་ཞིང༌༑ ༑ཞུགས་གནས་ཐིམ་དང་འགྲོས་བཞིན་མཐར་ཕྱིན་ནས། །ཉོན་མོངས་མ་སྤང་ལམ་དུ་ལྷོང་བར་ཤོག །དངོས་གྲུབ་རྩ་བ་དམ་ཚིག་སྡོམ་པའི་ཚོགས། །རྩ་བ་ཡན་ལག་སྲུང་མཚམས་ཇི་སྙེད་པ། །ཉེས་པས་མ་གོས་ཚུལ་བཞིན་ལེགས་བསྲུངས་ནས། །མཆོག་དམན་དངོས་གྲུབ་བདེ་བླག་འགྲུབ་གྱུར་ཅིག །རང་དོན་ཡན་ལག་བདུན་ལྡན་ཆོས་ཀྱི་སྐུ། །སྐད་ཅིག་མངོན་ཐོབ་དེ་ཡི་རང་མདངས་རྩལ། །མཆོག་དམན་བར་མའི་སྣང་ངོར་ལོངས་སྤྲུལ་སོགས། །ཆུ་ཟླ་ལྟ་བུས་འགྲོ་ཀུན་བགྲོད་གྱུར་ཅིག །ཅེས་ནཱ་རོའི་ཆོས་དྲུག་གི་རྟོགས་པ་དོན་གྱི་བརྒྱུད་པའི་གསོལ་འདེབས་ཚིག་ཉུང་ངུར་བགྱིས་པ་འདིའང༌། ལྕམ་ཞབས་དྲུང་གི་བཀས་བསྐུལ་བའི་ངོར། ལམ་འདིས་འགྲོ་བ་ཀུན་སྒྲོལ་བ་འབྲི་གུང་པ་ཆོས་ཀྱི་གྲགས་པས། ཤིང་སྤྲེལ་རྒྱལ་གྱི་ཟླ་བའི་དཀར་པོའི་རྫོགས་པ་གཉིས་པ་མཁའ་འགྲོ་འདུ་བའི་དུས་དགེ་བར༑ འབྲི་ཆོས་ཕྱུག་ཀླུང་གི་ཕོ་བྲང་སྲ་བརྟན་རྡོ་རྗེའི་གནས་སུ་མགྱོགས་པར་སྦྱར་བའོ།། །།སིདྡྷི་རསྟུ།།
