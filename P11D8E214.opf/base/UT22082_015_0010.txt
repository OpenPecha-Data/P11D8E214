གང་ཞིག་རྡུལ་བྲལ་ཁྱོད་སྐུ་གོ་འབྱེད་མདོག །སུམ་ཅུ་རྩ་གཉིས་མཚན་གྱི་དཔལ་འཆང་ཞིང༌། །ལེགས་སྤྱད་བརྒྱད་ཅུའི་དབང་ཚོན་མདངས་བཀྲ་བ། །ཐུབ་ཆེན་སྨན་གྱི་སངས་རྒྱས་ཁྱོད་ཕྱག་འཚལ། །ལུང་དང་རིག་སྨྲ་དཔའ་བོའི་མཚན་ཅན་དང༌། །བསྟན་པའི་གསལ་མཛད་ཡོན་ཏན་མགོན་པོ་མཆོག །ཟུང་འཇུག་དོན་མཁས་མཉམ་ཉིད་བཛྲ་སོགས། །བརྒྱུད་པའི་བླ་མ་རྣམས་ལ་གུས་པས་འདུད། །སྔོན་ཆད་བོད་འདིར་ཧ་ཅང་མ་གྲགས་ཀྱང༌། །དྷརྨ་སྭ་མའི་ལེགས་བཤད་ལས་བྱུང་བ། །གྲང་སྦྱོར་རིན་ཆེན་རིལ་བུའི་གདམས་པ་ཞིག །འདིར་སྤྲོ་སྐལ་ལྡན་ཚོགས་རྣམས་དེས་པས་ཉོན། །ས་བདག་བཅུད་བསྡུས་ཚ་བས་གཡའ་ཕྱི་ཞིང༌། །ཀྵ་ར་རྣམ་ལྔས་ལེགས་བཀྲུ་གཙོ་བོའི་ཕྱིར། །ཟ་བྱེད་ཁམས་ལྔ་འཆིང་བྱེད་ལྕགས་བརྒྱད་ཀྱིས། །བཙོ་ཞིང་དགྲ་སྤྲད་སྲ་མཁྲེགས་ལྡན་པ་དང༌། །གསེར་དངུལ་ཟངས་ལྕགས་ཨ་བྷ་ཀ་བསྲེགས་ཐལ། །ལྕགས་མདོག་གཉིས་དང་བཛྲ་ཧི་ར་དང༌། །ཁྱུང་སྐྱུགས་མན་ཤེལ་པདྨ་རཱ་ག་དང༌། །ཨིནྡྲ་ནཱི་ལ་ནལ་དང་གཡུ་སྤོས་ཤེལ། །སྤུག་དང་མུ་ཏིག་དངུལ་དང་མུ་མེན་དང༌། །བྱུ་རུ་ཉ་ཕྱིས་སྦལ་རྒྱབ་གངས་ཀྱི་ཐིགས། །མཉྫི་ར་དང་དཀར་པོ་ཆིག་ཐུབ་དང༌། །ཁབ་ལེན་ཅོང་ཞི་ལྕགས་རྡོ་མདུང་རྩེ་དམར། །མཚལ་དཀར་ཟེ་ཚྭ་འགྲོན་ཐལ་བཟང་པོ་དྲུག །གླ་རྩི་དོམ་མཁྲིས་མ་གི་ཏ་དང་ནི། །སྔོ་ལེབ་སྤུ་དང་ཧ་རི་ཏ་ཀ་དང༌། །དུག་མེད་གསུམ་དང་དཔའ་བོ་དཀར་པོ་དང༌། །དཔའ་རྒོད་སུག་སྨེལ་གླང་ཆེན་ཆིག་ཐུབ་དང༌། །ལས་མཁན་གསེར་གྱི་རྒྱལ་མཚན་ཁཎྜ་དང༌། །གང་ཆུང་གཡུ་རྒྱན་མ་དང་བེ་ལྗང་དང༌། །ཙི་ཏྲ་མཱུ་ལ་གྷིདྡ་དྷཱུ་པེ་དང༌། །མུ་མེན་ག་ཤ་ཅན་དང་ཀོ་ཏྲ་བ། །སྤང་རྒྱན་དཀར་དང་དུང་གི་ཐག་པ་ཅན། །ཡུངས་ཀར་ཆོ་ལོ་པད་ཁང་ཁྱོར་བ་རེ། །སྨན་མཆོག་སྡོང་པོ་ཙནྡན་དཀར་དམར་དང༌། །ཨ་ག་རུ་ནག་ཤིང་གསུམ་བར་ཤུན་དང༌། །སྟབ་སེང་སྲད་དཀར་མདོངས་དང་བྱི་ཐུར་དང༌། །སྨ་ར་སྤྱི་མཆན་སྤུ་དང་ཀ་ལེ་ཛ། །ཉྭ་ཤ་ཤ་ཆེན་ཤ་ཚིལ་ཆུ་ཚན་གསུམ། །ས་བཅུད་རྣམས་ནི་གཞུང་སྨན་བརྒྱད་ཅུ་གསུམ། །ཁ་སྐོང་བྲག་ཁྱུང་པུ་པར་སྤྱི་སྤང་ཨུཏ། །སྟག་ཏེ་ཕུར་ལྷོག་གསེར་ཏིག་སྟོང་ཏང་དང༌། །པྲི་ཤུག་སྦྱར་ཏེ་ཞིབ་བཏགས་པི་ཞུས་གཡོ། །འདམ་བུར་བཏགས་ལ་ཞག་བདུན་བར་དུ་འཐག

།དུལ་ཚེ་སྲན་ཕོས་རྩི་དྲིལ་གྲིབ་མར་བསྐམ། །རབ་ཏུ་གནས་ལ་ཉི་གནམ་མ་མཐོང་བྱ། །བསྟེན་ཐབས་ཕན་ཡོན་ལོགས་སུ་བཀོད་དེར་ལྟོས། །འབྲིང་སྦྱོར་འབིགས་བྱེད་གསེར་དངུལ་ཟངས་དང་ལྕགས། །ཨ་བྷ་གཡུ་སྤོས་བྱི་མུ་མུ་སྲན་བླ། །བོང་བ་དཀར་དམར་སུག་སྨེལ་དོམ་གྱི་རྩི། །ཅུ་གུར་ལི་ཛཱ་སུག་ཀ་ཀོ་ཡུངས་མདོངས། །ལྕགས་རྩི་ཨ་རུ་དཔའ་བོ་སེར་པོ་དང༌། །མཆིན་པ་རྣམས་སྦྱར་དྲིལ་ཐབས་གོང་བཞིན་བྱ། །ཆུང་སྦྱོར་རྒྱལ་པོ་དང་ནི་བློན་པོ་ལྔ། །དམག་དཔོན་སེར་དང་འབངས་དྲུག་ནོར་ལེགས་པོ། །གཟི་མདངས་དམག་གིས་ཀསྟུ་རི་མཚོན་ཆ་ཐོགས། །ཨ་ཕུབ་ཁུར་ཏེ་ཡུངས་ཀྱི་རྟ་ལ་ཞོན། །འཐག་ཐབས་གྲིབ་མར་སྐམ་ཐབས་གོང་ལྟར་བྱ། །ཞིབ་པ་སློབ་དཔོན་མཁས་པའི་ངག་ལ་ཡོད། །དེ་མིན་འདི་ལ་བྱ་ཐབས་བཟོ་མ་མཆིས། །འོལ་ཚོད་བྱས་ན་ཅིས་ཀྱང་འདི་མི་འབྱུང༌། །ཕན་ཡོན་ལ་སོགས་གང་དགོས་ལོགས་སུ་ཡོད། །འདི་ནི་མ་ཚང་མེད་བསྡུས་ཞིབ་བཀོད་པ། །ཀྱེ་མ་དྷརྨ་སྭ་མི་གངས་ཀྱི་རི་བོ་ལས། །འོངས་པའི་ཡན་ལག་བརྒྱད་ལྡན་མན་ངག་ཆུ། །ཀོང་སྨན་དཀོན་རྒྱན་འབབ་སྟེགས་ཡིད་བཟང་པོར། །ལེགས་བསྐྲུན་དེ་ལས་ཡོངས་རྫོགས་རྒྱུན་བབ་སྟེང༌། །བདག་བློའི་བུམ་བཟང་ཡོངས་སུ་གང་བའི་མཐུས༑ ༑སྦྱོར་སྡེ་མཆོག་འདི་ཕན་སེམས་བློ་ཡིས་སྦྱར། །དགེ་བས་འགྲོ་ཀུན་ཐམས་ཅད་མཁྱེན་ཐོབ་ཤོག །ཅེས་པ་རིན་ཆེན་གྲང་སྦྱོར་རིལ་བུའི་སྦྱོར་ཐབས་འདིའང༌། གསོ་རིག་གི་གནས་ལ་ཅུང་ཟད་གོམས་པར་བགྱིས་པའི་དགེ་སློང་དྷརྨ་ཀཱིརྟིས་ཁྲོམ་སྟོད་ཀྱི་སྒར་དུ་སྦྱར་བའོ།། །།མངྒ་ལཾ་བྷ་ཝནྟུ།།
