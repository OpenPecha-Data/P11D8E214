༄༅། །ཡང་ཟློག་མེའི་སྤུ་གྲིའི་བསྙེན་སྒྲུབ་ལས་སྦྱོར་རྣམས་ཀྱི་ཚོད་འཛིན་ནང་རྟགས་དྲོད་

ཚད་རྒྱུད་ལས་བྱུང་བ་བཞུགས་སོ།། བླ་མ་དང་ལྷག་པའི་ལྷ་ལ་ཕྱག་འཚལ་ལོ། འདིར་དབང་ཐོབ་ཅིང་དམ་ཚིག་སྡོམ་པ་ལ་གནས་པའི་སྔགས་པས། འཇམ་དཔལ་གཤིན་རྗེའི་གཤེད་བསྙེན་ཅིང་སྒྲུབ་པའི་ཚེ་ན་ནང་རྟགས་འབྱུང་ཚུལ་རྣམས་བཀོལ་ཏེ་བྲི་བ་ནི། བསྙེན་པ་དང༌། སྒྲུབ་པ་དང༌། ལས་སྦྱོར་གསུམ་ལས། དང་པོ་ཟླ་གསང་གཞི་བསྙེན་ལ་རྟགས་ཐོན་པ་ཞིག་ཤིན་ཏུ་གལ་ཆེ་བས། གཞི་བསྙེན་གྱི་བསྙེན་རྟགས་ནི། གཤིན་རྗེའི་གཤེད་གསང་བ་དགུ་བཅུ་པའི་ལེའུ་གསུམ་པ་ལས། རྩ་སྔགས་རྒྱུན་དུ་བཟླས་བྱས་པས། །མཆོད་རྟེན་ལུས་སྐྱེས་ཁོང་ན་ཡོད། །བང་རིམ་ལ་འཛེགས་སྐུ་གཟུགས་འབྱུང༌། །ཆུ་ལ་མི་ནུབ་བོན་པོ་འབྱུང༌། །མཚན་མ་གཏན་ལ་དབབ་པ་ཡིན། །དུས་ནི་ལོ་ཟླ་ཡུན་དུ་བསྲིང༌། །གྲངས་ནི་བྱེ་བ་ཆིག་འབུམ་རྫོགས། །ཞེས་དང༌། རྒྱུད་དེའི་སྲོག་ལ་དྲངས་པའི་ལེའུ་ལས། སྲོག་སྙིང་ཉི་མ་ཤར་ལྟས་འབྱུང༌། །དགེ་བཤེས་བོན་པོ་མི་ནག་འབྱུང༌༑ ༑ཕྱོགས་བཅུ་གཤིན་རྗེའི་ཞིང་ཁམས་ནས། །བྱ་རྒོད་གཟན་ལ་བཅུམ་བཞིན་འབྱུང༌། །ཞེས་དང༌། ཡང་དེ་ཉིད་ཀྱི་སྔགས་ཀྱི་ཚད་མའི་ལེའུ་ལས༑ རྩ་བ་རྒྱན་གྱི་དྲག་སྔགས་ལ། །མི་ལུས་ཕྲ་མེན་མགོ་ཅན་རབ། །མཆོད་རྟེན་སྒྲ་ཅན་བྱུང་ན་འབྲིང༌། །མེ་འབར་ཆུ་ལ་མེ་བྱིང་ཐ། །ཞེས་རབ་འབྲིང་ཐ་མ་གསུམ་དུ་གསུངས་ཤིང༌། གཤིན་རྗེའི་གཤེད་ནག་པོ་སྔགས་རྒྱུད་ཀྱི་ལེའུ་བརྒྱད་པ་ལས། བདེ་གཤེགས་གཟུགས་བརྙན་མཁའ་ལ་ཉི་ཟླ་ཤར། །མགོན་གཉེར་བཤེས་སོགས་རྨི་ན་རྟགས༑ ༑ཞེས་སོགས་གསུངས་སོ། །ཉེ་བར་བསྙེན་པའི་རྟགས་ནི། གསང་བ་དགུ་བཅུ་པའི་རྒྱུད་ཀྱི་ལེའུ་དྲུག་པ་ལས། རྟགས་ནི་ཡང་དག་མངོན་དུ་སྟོན། །མགོ་ལ་ཉི་ཤར་ལྷོ་སྒོ་འབྱེད༑ ༑དགེ་བཤེས་མོན་པ་རྟགས་ཅན་འབྱུང༌། །ཞེས་སོ། །གཤིན་རྗེ་དམར་ནག་འབར་བའི་དམེ་རྒྱུད་ཀྱི་ལེའུ་གསུམ་པ་ལས། སྔགས་ཉིད་ལྷ་ཡི་སྐུར་གྱུར་པ། སྣང་བའི་དུས་ན་སྐུར་གསལ་བའི། །སྟོང་པའི་དུས་ན་རྫོགས་པ་ཉིད། །སྒྱུ་ལུས་ལྷ་ཡི་སྐུ་རུ་འཕོ། །ངག་ནི་གྲག་སྟོང་སྔགས་སུ་འཕོ། །སེམས་ཉིད་གསལ་སྟོང་ཆེན་པོར་འཕོ། །སྒྲིབ་པ་གསུམ་ལས་རིམ་པར་གྲོལ༑ ༑ནམ་མཁའ་བཞིན་དུ་མཉམ་རྫོགས་ཆེ། །འཁོར་ལོ་ཐམས་ཅད་རྫོགས་པར་འདོད། །དཀའ་ཐུབ་ཆེན་པོ་བྱེད་པ་ན། །བསད་པའི་ལས་ཀྱི་སྔགས་ཀྱང་གདངས།

སྐུ་གཟུགས་བཞད་དམ་འགུལ་བའམ། །ལུང་སྟོན་སྒྲའམ་གཟུགས་བརྙན་ནམ། །རྨི་ལམ་རྣམས་ནི་མང་དུ་འབྱུང༌། །བརྟན་པར་རིག་ནས་དངོས་གྲུབ་འབྱུང༌། །ཞེས་དང༌། གཤིན་རྗེ་དམར་ནག་འབར་བའི་དམེ་རྒྱུད་ཀྱི་ལེའུ་ཉེར་བརྒྱད་པ་ལས། སྙིང་པོ་ཡི་གེ་བཅུ་བདུན་བཟླས། །དེ་ཚེ་མཚན་རྟགས་མང་པོ་འབྱུང༌། །སྐུ་གཟུགས་བཞད་དང་འགུལ་བའམ། །ལུང་སྟོན་སྒྲའམ་གཟུགས་བརྙན་ནམ། །རྨི་ལམ་རྟགས་རྣམས་མང་པོ་འབྱུང༌། །དེ་ནས་བསྟོད་ཅིང་ཕྲིན་ལས་བརྩམ། །བསྙེན་པའི་མངོན་རྟགས་མ་བྱུང་ན། །ས་མ་ཡ་ལ་སེལ་ཡོད་པས། །ལུས་ངག་འབད་པས་བཤགས་པ་བྱ། །དེས་ཀྱང་མངོན་དུ་མ་གྱུར་ན། །ལས་འཕྲོ་མེད་དམ་བསོད་ནམས་ཞན། །དེ་ཡི་བྱ་བ་མ་གཏོགས་པ། །ཀུན་གྱི་བྱས་ན་རང་ལ་གནོད། །ཅེས་དང༌། གཤིན་རྗེ་ནག་པོ་སྔགས་རྒྱུད་ཀྱི་ལེའུ་ལྔ་པ་ལས། མཐོང་དང་ཐོས་དང་རྨི་ལམ་གསུམ། །བསྙེན་པའི་རྟགས་སུ་ཤེས་པར་བྱ། །ཞེས་གསུངས་པས། ལྷའི་གཟུགས་མཐོང་ན་རབ་དང༌། སྔགས་ཀྱི་སྒྲའི་དོན་ལུང་སྟོན་པའི་སྒྲ་ཐོས་པ་འབྲིང་དང༌། དེ་གཉིས་རྨི་ལམ་དུ་བྱུང་བ་ནི་བསྙེན་རྟགས་ཐ་མར་བཤད་ཅིང༌། འཇམ་དཔལ་ཁྲོ་བོ་གསང་བའི་རྒྱུད་ལེའུ་ལྔ་པ་ལས། དེ་ཚེ་མཚན་མའི་ཡུལ་ལ་འབད། །འོད་ཕུང་འཁྱིལ་དང་ལུང་སྟོན་སྒྲ། །རྨི་ལམ་རྟགས་རྣམས་མང་དུ་འབྱུང༌། །མཚན་མ་མང་པོ་བྱུང་བ་དང༌། །དངོས་གྲུབ་བླང་བའི་ཆོ་ག་བྱ། །རྟགས་རྣམས་མངོན་དུ་མ་བྱུང་ན། །དམ་ཚིག་བཤགས་ལ་བསྙེན་པ་སྤོག །དཀའ་ཐུབ་ལྡན་པའི་བརྟུལ་ཞུགས་སྤྱོད། །མི་སྨྲ་མི་སྤུར་མི་འཁྱམ་པར། །ལྷོ་ཡི་ཕྱོགས་སུ་ཁ་བལྟས་ཏེ། །ཀུ་ཤའི་སྟན་ལ་འདུག་ནས་ནི། །རི་མོ་དག་ལ་སེམས་བལྟ་ཞིང༌། །བཟླས་པའི་ཡི་གེའི་རིམ་པ་བྱ། །བྱིང་ནས་ཤེས་པ་རྨུགས་གྱུར་ན། །འཇམ་པའི་དབྱངས་ཀྱིས་རབ་བསྐུལ་བྱ། །རྒོད་ཅིང་གཡེང་བར་གྱུར་པ་ན། །མགོ་ནི་རྨ་བྱ་བཞིན་བཀུག་ལ། །མཚན་མ་ལྷ་ཡི་སྐུ་ལ་བརྟག །ཇི་སྲིད་དུ་ནི་ཆོས་སྐྱོང་བར། །རྩེ་གཅིག་སེམས་ཀྱིས་བལྟ་བར་བྱ། །བེ་ཤིང་སེང་ལྡེང་ཐོགས་ནས་ནི། །ཞིང་ལྤགས་བགོ་ཞིང་དོམ་ལྤགས་འདིང༌། །བགྲོད་དཀའི་དུར་ཁྲོད་གནས་སུ་འགྲིམ། །མྱུར་བ་མ་ཡིན་བུལ་བ་མིན། །དེ་བཞིན་འདྲེས་པའི་ཡི་གེ་མིན། །ཤིན་ཏུ་གསལ་ཞིང་དག་པར་སྨྲ། །ཞེས་གསུངས་པ་ལྟར།

བསྒྲིམས་ཏེ་རྩེ་གཅིག་བཟླ་དགོས་སོ། །སྒྲུབ་པ་ཆེན་པོའི་དུས་སུ་ནི་རྟགས་བྱུང་མ་བྱུང་ཅི་ཡིན་ཡང་སྔགས་འབུམ་ཕྲག་གཅིག་པོ་གཉིད་སོགས་སྤངས་ནས་ཐེམས་མ་ཐག་དངོས་གྲུབ་ཀྱི་རྫས་ལ་ལོངས་སྤྱོད་དོ། །འཆི་བདག་བདུད་སྒྲུབ་ཀྱི་ཚེ་རྟགས་ནི། འཇམ་དཔལ་ཁྲོ་བོ་གསང་བའི་རྒྱུད་ཀྱི་ལེའུ་ལྔ་པ་ལས། བཤན་པ་ལས་མཁན་བྱུང་བའི་ཚེ། །དམིགས་པ་དགུ་པའི་ལས་བྱ་སྟེ། །སྔགས་དང་ཏིང་འཛིན་ཕོ་ཉ་དགུག །ཅེས་དང༌། གཤིན་རྗེ་ནག་པོ་སྔགས་ཀྱི་རྒྱུད་ཀྱི་ལེའུ་བདུན་པ་ལས། གོ་གྱོན་མཚོན་ཐོགས་དམག་སྣ་འདྲེན་པ་དང༌། །དགྲ་ཡི་བླ་བཀུག་ཐུགས་རྗེས་བསྒྲལ་རྨིས་ན། །ཕྲིན་ལས་དྲག་པོའི་ལས་ལ་འཇུག་པར་བྱ། །ཞེས་སོགས་གསུངས་པས། མདོར་ན་བསྙེན་པ་དང༌། ཉེ་བར་བསྙེན་པ། སྒྲུབ་པ་ཆེན་པོ་རྣམས་ཚད་དུ་ཐེམས་ཤིང་དྲོད་རྟགས་ཐོབ་པ། འཆི་བདག་བདུད་སྒྲུབ་ཀྱི་ཚེ་འཆི་བདག་གཤིན་རྗེ་སོགས་སྡེ་བརྒྱད་འགྲུབ་ཅིང་བྲན་དུ་འཁོལ་བའི་རྟགས་ཐོབ་ནས་སྲུང་ཟློག་བསད་མནན་སོགས་ལས་སྦྱོར་ལ་འཇུག་པར་བྱ་ཡི། གང་རུང་བསྙེན་སྒྲུབ་དྲོད་རྟགས་མ་ཐོབ་པར་ལས་སྦྱོར་བྱེད་པ་མི་ཕུང་རང་བརླག་གི་རྒྱུ་ཡིན་པར་གསུངས་སོ། །བསྙེན་སྒྲུབ་ཐེམས་ནས་ལས་སྦྱོར་གྱིས་ཚེ་སྲུང་བ་དང༌། ཟློག་པ་དང༌། བསད་པ་གསུམ་རྟགས་དང་ལྡན་པར་འགྲུབ་པ་ཞིག་གལ་ཆེ་ཞིང༌། དེའང་སྲུང་བའི་ཚེ་མཁར་དུ་ཞུགས་པ་ལ་སོགས་པ་དང༌། ཟློག་པའི་ཚེ་ཤྭ་སྦུད་ཐོག་སེར་ཟློག་པ་སོགས་དང༌། བསད་པའི་ཚེ་ཐོག་མར་དགྲ་བོའི་བླ་ཁུག་པ་ཞིག་གལ་ཆེ་སྟེ། བླ་དྭགས་བརྒྱད་དམ་བཅུ་གཉིས་སོགས་ཁུག་པའི་མཚན་མ་མཐོང་བ། དེའང་སྲུང་མ་དང༌། བདག་དང༌། དགྲ་བོའི་བླ་མ་ནོར་བ་གལ་ཆེ་ཞིང༌། བླ་ཁུག་ནས་ལྷ་དང་ཕྲལ་བའི་རྟགས་དང༌། དེ་ནས་བསད་པའི་རྟགས་གློ་སྙིང་སྟེར་བ་སོགས་དྲོད་ཚད་ནང་རྟགས་ངེས་ངེས་ལན་འགའ་ཡར་གྱིས་གཏན་ལ་ཕབ་ནས། མཐའ་རྒྱན་མནན་སྲེག་འཕང་གསུམ་སོགས་བགྱིས་ནས་ཀྱང་ནང་རྟགས་་་་་་ལས་སྦྱོར་གྱི་སྒྲུབ་པའི་ཚེ་ནང་རྟགས་མ་བྱུང་ན་སྐབས་སུ་གང་བབ་ཀྱི་སྔགས་བསྡེབས་རྣམས་གནད་མ་འཆུག་པར་གསང་སྔགས་སྙིང་གཟེར་ནས་བཤད་པ་ལྟར་བྱའོ། །ལྡན་པས་དུས་བཏབ་པས་ཆོག་པ་ཞིག་གལ་ཆེ་བ་ནི།

ཡང་ཟློག་མེའི་སྤུ་གྲིའི་གཞུང་དང་ཞལ་གདམས་རྣམས་སུ་མདོར་བསྡུས་འགག་དོན་དྲིལ་ནས་ཡོད་པ་རྣམས་དང༌། གཞན་ཡང་རྡོ་རྗེ་གཙུག་ལག་གི་རྒྱུད་ལྕགས་སྡིག་གི་དྲོད་ཚད་གསེར་ཐུར་རྣམ་གསུམ་དང༌། རྫོགས་འཕྲང་སྲོག་གསུམ། གཏེར་ཁྱིམ། བླ་ཁྱིམ། སྲོག་ཁྱིམ་སོགས་དང༌། རྗེ་འབངས་གྲོགས་གསུམ་གྱི་མཐུའི་ཞུས་ལན་སྐོར་དང་ནོར་ར་རེའི་བརྟག་པ་སོགས་མངོན་སྤྱོད་སྤྱི་གཞུང་རྣམས་ནས་འབྱུང་བ་ལྟར་དཔྱད་ཅིང༌། ལར་ནི་ཕྲིན་ལས་བཞི་གང་སྒྲུབ་ཀྱང་ནང་རྟགས་དྲོད་ཚད་ལ་མཁས་པ་ཞིག་ཤིན་ཏུ་གལ་ཆེའོ། །དེ་དག་ཐམས་ཅད་ལ་སྤྱི་སྡོམ་དུ་གལ་ཆེ་བ་ནི། གཤིན་རྗེ་གསང་བ་དགུ་བཅུ་པའི་རྒྱུད་ལས། དེ་ནས་བཅོམ་ལྡན་འདས་ཀྱིས་བདག་གི་ཚད་མའི་རིམ་པ་འདི་གསུངས་སོ། །མཆོད་རྟེན་ཉི་ཤར་ཚུར་མེ་འགྱུར། །མཚོན་ཆ་བོན་པོ་སློབ་དཔོན་དང༌། །ཕ་མ་མོན་ནག་སྨྲ་བ་དང༌༑ ༑བྱིས་སྦ་བྱུང་བ་ལས་སྦྱོར་གཡོན། །ཉི་མ་བཞི་སྐོར་ནུབ་མོ་དྲིལ། །རྟགས་ནི་འོད་དང་གོ་ཁྲབ་དང༌། །མཚོན་ཆ་དང་ནི་ཙཽ་ར་ཐོབ། །ཁུག་ནས་བསད་པའི་ལས་ལ་འབྱུང༌། །ལྕེ་འོག་ཚ་དང་འབྲུམ་པས་ཁེངས། །མ་ནོར་ཚངས་ལས་འགྲུབ་རྟགས་ཡིན། །དམ་ཚིག་ཉམས་བསལ་མྱུར་དུ་འགྲུབ། །སྙིང་ཟ་སེང་སེང་པོར་བྱེད་ན། །མི་འགྲུབ་མ་ཚང་ཚད་མ་ཡིན། །བྱིང་ན་སྒྲིབ་ཡོད་བཤགས་པ་བྱ། །བདག་གི་ལུས་སྦྲིད་ཤ་འདར་ན། །ཐུགས་ཀྱི་བསྐངས་བས་དགྲ་རྣམས་སྒྲོལ། །གཤིན་རྗེའི་གཤེད་གསང་བ་དགུ་བཅུ་པའི་རྒྱུད་ཀྱི་རྒྱལ་པོ་ལས། བདག་གི་དམ་ཚིག་ཚད་མའི་ལེའུ་སྟེ་ཉི་ཤུ་གཅིག་པའོ། །དེ་ནས་བཅོམ་ལྡན་འདས་ཀྱིས་བསྒྲུབ་བྱའི་ཚད་མ་འདི་གསུངས་སོ། །གཡག་དང་ར་དང་བ་གླང་ཉེ། །རྟ་དང་བོང་བུ་མི་འགྲུབ་རྟགས། །ཏིང་འཛིན་མི་གསལ་ཁྲོ་འདོད་ན། །རྒྱབ་བརྟེན་ཡོད་པས་དབྱེ་བསལ་བྱ། །ལུག་བསད་རང་གི་འཁོར་གཡོག་ཆེ། །དགྲ་རྣམས་དངོས་སུ་བྱུང་ན་རབ། །ཏིང་འཛིན་གསལ་ཞིང་དངོས་བསད་ན༑ ༑ཞག་བདུན་ཟླ་བདུན་བར་གྱིས་འཆི། །ཁུང་ཆུང་ཁུག་རྣམས་ཆེན་པོར་བཏང༌། །དུག་གསུམ་བྱུང་ན་ཡ་ཙ་ར། །བཅུ་བདུན་མཐར་སྦྱར་མཁས་པས་བསྐོར། །ལུས་ཟད་ལ་སོགས་ཟད་པ་བརྒྱད༑ ༑རིགས་གསུམ་མགོན་པོས་བརླག་པར་འགྱུར། །གཤིན་རྗེའི་གཤེད་གསང་བ་དགུ་བཅུ་པའི་རྒྱུད་ཀྱི་རྒྱལ་པོ་ལས། །བསྒྲུབ་བྱ་དགྲའི་ཚད་མའི་ལེའུ་སྟེ་ཉི་ཤུ་རྩ་དགུ་པའོ། །དེ་ནས་བཅོམ་ལྡན་འདས་ཀྱིས་མཚན་མ་བརྟག་པའི་ཚད་མ་འདི་གསུངས་སོ། །རྟགས་ལ་རྣམ་པ་གཉིས་སུ་ཤེས། །དངོས་སུ་སྣང་བ་མཚན་མར་རོ། །དངོས་སུ་སྣང་བ་ཕྱི་ནང་གཉིས། ཁྲ་སྤྱང་མང་པོ་བྱུང་བ་དང༌། །མོན་པ་སྦྲུལ་ཞགས་ཁྱེར་བ་བྱུང༌། །གང་བྱུང་རྟགས་བྱུང་དྲག་སྔགས་གནད། །ཡུངས་ཀར་ལ་སོགས་བཟླས་ལ་གཏོར། །དེ་ཡིས་རང་གཤེད་འབེབས་པར་བྱེད། །དྲེགས་པ་དེ་དག་འཁོར་བཅས་ཀྱི། །དེད་དཔོན་སྡོང་ཟླ་གྲོགས་དང་ནི། །ཕོ་ཉ་ལས་མཁན་ཕྲ་མེན་དང༌། །བར་ཆད་བྱེད་པའི་ལོག་འདྲེན་རྣམས། །སྲོག་གི་ཡི་གེའི་གནད་དག་ནི། །ཤེས་པས་གདུག་པ་ཡང་དག་ཞི། །ཤིན་ཏུ་འཁྲུག་ན་འཁོར་ལོ་བསྐོར། །རྟགས་རྣམས་འཁྲུགས་ན་སྤྱི་སྲོག་བཅད། །རང་གཤེད་གང་མོས་གང་འགྲོ་བལྟ། །་་་་་སྲུང་ཟློག་བསད་མནན་གྱི་སྒྲུབ་པའི་སྐབས་ནང་རྟགས་རྨི་ལམ་དང་བསྟུན་པའི་རང་གཤེད་དང་དཀྲུག་སྦྱོར་སོགས་སྐབས་སུ་གང་བབ་གནད་དང་བསྟུན་ནས་བྱེད་པ་གལ་ཆེའོ། །སྲུང་དང་ཟློག་དང་བསད་དང་མནན༑ ༑མེ་ལ་སྲེག་དང་དམ་བདར་ལ། །དྲེགས་པ་ཞི་དང་དངོས་གྲུབ་འཐོབ། །ཅེས་གསུངས་སོ། །འཇམ་དཔལ་གཤིན་རྗེའི་གཤེད་དམར་ནག་འབར་བའི་དམེ་རྒྱུད་ལས། རྡོ་རྗེ་ནང་འགྲས་རྟགས་ཀྱི་ལེའུ་སྟེ་ཞེ་བརྒྱད་པའོ། །ཞེས་གསུངས་པ་རྣམས་ཉམས་སུ་བསྟར་ནས་སྒྲུབ་པའི་དུས་ནང་རྟགས་དྲོད་ཚད་རྨི་ལམ་གྱིས་འཛིན་དགོས་པས་གཙོ་ཆེར་སྲུང་ཟློག་བསད་མནན་གྱི་སྔགས་འབད་ཅིང༌། ཁ་བསྒྱུར་ནང་རྟགས་དང་བསྟུན་པའི་སྔགས་ཀྱི་ཤམ་འདོགས་ནི་སྡེབས་རྒྱུད་རྣམས་དང༌། ལྕགས་སྡིག་སོགས་གཞུང་ཆེན་རྣམས་ཀྱིས་ཁ་བསྐངས་པ་ཐུན་འོག་རྣམས་སུ་བཟླའམ༑ ཡང་ན་སྒྲུབ་པོ་མང་ན་ནང་རྟགས་དང་བསྟུན་པའི་སྔགས་རྣམས་དང༌། བླ་འགུགས་སྔགས་སོགས་ཤམ་འདོགས་རྣམས་རང་གིས་བཟླ། སྲུང་ཟློག་བསད་མནན་སོགས་སྒྲུབ་གཞུང་གི་སྔགས་རྣམས་ནི་སྒྲུབ་གྲོགས་རྣམས་ཀྱིས་ཉིན་མཚན་ཁོར་ཡུག་ཏུ་བཟླའོ། །གཞན་ཡང་གནད་རྣམས་ཞལ་ཤེས་སོ། །འདི་ལྟར་ཚོད་འཛིན་དྲོད་ཚད་རྣམས་ཀྱང་ཤིན་ཏུ་གལ་ཆེ་བའི་ཕྱིར་འཁྲུལ་རྨོངས་རྣམས་ཀྱི་ངག་སྒྲོས་ལ་མ་ལྟོས་པར་རང་ཉིད་འཇམ་དཔལ་གཤིན་རྗེའི་རྒྱུད་རྣམས་ནས་རང་ཉིད་ཀྱིས་བསྡུས་པ་སྟེ།

འདིའང་གཤིན་རྗེའི་གཤེད་ཀྱི་རྣལ་འབྱོར་པ་ཆོས་ཀྱི་གྲགས་པས༑ ཆུ་སྤྲེལ་ཆོ་འཕྲུལ་ཟླ་བའི་ཚེས་བཅུ་བདུན་ལ་མཚམས་མལ་གྱི་ཁོངས་ནས། ཕོ་བྲང་བཀྲ་ཤིས་ཚུགས་ཀྱི་རེའུ་མིག་པདྨ་འོད་དུ་བྲིས་པ་སྟེ་དགེ་བར་གྱུར་ཅིག །ཡང་ཟླ་གསང་འཇམ་དཔལ་ཐུགས་ཀྱི་རྩ་རྒྱུད་ཉམས་སུ་བླང་བའི་ལེའུ་བཅོ་ལྔ་པ་ལས། གང་དང་ལྡན་པར་འབྲེལ་་་་་་ཕྲིན་ལས་བཞི་གང་སྒྲུབ་ཀྱང༌། བའི་རྟགས། །ཤེས་པས་་་་་་རང་གི་སེམས། རྟགས་ཀྱི་ཚད་བཟུང་བྱ། །རིམ་་་་་་བསྙེན་སྒྲུབ་ལས་སྦྱོར་གོ་རིམ་མ་ལོག་པ། པ་ངེས་པའི་ཚད་མ་དང༌། །ཆོ་ག་ལས་་་་་་ལས་གང་སྒྲུབ་ཀྱི་རིམ་པ་མ་ནོར་བ། ཀྱི་ཚད་མ་དང༌། དྲག་པོ་སྔགས་་་་་་སྲུང་ཟློག་བསད་མནན་འབོད་རྦད་བསད་པ་སོགས་གང་ཡིན་གྱི་སྔགས་མ་ནོར་བ། ཀྱི་ཚད་མ་དང༌། །རྣལ་འབྱོར་བརྟན་་་་་་ལས་གང་སྒྲུབ་ཀྱི་དམིགས་པ་གསལ་ལ་མ་ནོར་བ། པའི་ཚད་མ་དང༌། །གཟས་པ་ཁུག་་་་་་ནང་རྟགས་གང་ཡིན་ལན་གསུམ་ལས་མི་ཉུང་བར་བྱུང་བ། པའི་ཚད་མ་དང༌། །སྦྱོར་བ་་་་་་ནང་རྟགས་དང་བསྟུན་པའི་ལས་སྦྱོར་བཞི་གང་ཡིན་ཡང་མ་ནོར་བ། ངེས་པའི་ཚད་མའོ། །ཞེས་ཚད་མ་དྲུག་གིས་གཏན་ལ་འབེབས་དགོས་གསུངས་འདུག་གོ། །ཡང་འཇམ་དཔལ་རྩ་རྒྱུད་ཀྱི་ལེའུ་བཅུ་དྲུག་པར། བཟླས་པའི་སྔགས་ནི་ཐམས་ཅད་པ། །རྨི་ལམ་དགེ་དང་མི་དགེ་དང༌། །སེམས་ཅན་མཆོག་དང་ཐ་མ་ཡིན། །སྤྱོད་པ་ཡོན་ཏན་རྒྱ་ཆེར་ལྡན༑ ༑ཞེས་བཟླས་པའི་དྲོད་ཚད་ནང་རྟགས་རྨི་ལམ་ལ་བརྟག་པར་གསུངས་ལ། ཡང་རྒྱུད་དེ་ཉིད་ཀྱི་ལེའུ་ཉེར་དྲུག་པར། གང་གི་དུས་དང་མཚན་མ་ལས། །དུས་མིན་འགྲུབ་པར་འདོད་པ་ནི༑ ༑མདོར་ན་བརྒྱ་བྱིན་ཡིན་ན་ཡང༌། །དེ་ཡིས་སྔགས་རྣམས་མི་འགྲུབ་བོ། །ཞེས་སྒྲུབ་པ་ལ་གྲངས་དུས་རྟགས་གསུམ་གའི་བསྙེན་པ་འབད་ནས། ནང་རྟགས་ཀྱི་ཚད་ངེས་པར་གཏན་ལ་འབེབས་པ་གལ་ཆེའོ། །འདི་དགེས་ཉེས་མུན་སྨག་རུམ་མཐར་བྱེད་ཅིང༌། །བུདྡྷའི་བསྟན་དར་དེ་འཛིན་ཞབས་བརྟན་ཅིང༌།

།དེར་འཚེའི་དགྲ་ཞི་དྷརྨའི་དགའ་སྟོན་གྱིས། །སཾ་གྷའི་འདུས་རྒྱས་འགྲོ་ཀུན་བདེར་འགོད་ཤོག༑ །།སརྦ་མངྒ་ལཾ།།
