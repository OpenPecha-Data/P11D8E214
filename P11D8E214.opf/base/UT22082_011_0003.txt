༄༅། །ཡང་ཟློག་མེའི་སྤུ་གྲིའི་ཁ་སྐོང་ཚོགས་ཀྱི་འཁོར་ལོ་བཞུགས་སོ།།

བླ་མ་དང་ནི་ལྷག་པའི་ལྷར། །གུས་བཏུད་ཡང་ཟློག་མེའི་སྤུ་གྲིའི། །ཁ་སྐོང་ཚོགས་ཀྱི་འཁོར་ལོ་བཤད། །ཉ་སྟོང་ཚེས་བརྒྱད་ཉེར་གཉིས་དང༌། །ཡང་ན་བསྙེན་སྒྲུབ་ལས་སྦྱོར་ཚེ། །ཞིང་ངམ་གཅན་གཟན་པགས་པའི་སྟེང༌། །ཐབས་རྫས་བཟའ་བ་གཡས་སུ་བཤམ། །ཤེས་རབ་མ་ད་ན་གཡོན་ལ། །དུག་ཁྲག་གིས་སྦགས་ཉ་བོ་ནི། །ཤ་ཁྲག་རུས་པ་བཅས་པ་དང༌། །ཐུན་ཕུར་གྲི་ལྕགས་བཅས་པ་བཤམ། །ཚོགས་རྫས་མེ་རླུང་ཆུས་སྦྱངས་ཏེ། རཾ་ཡཾ་ཁཾ། རཾ་རྡོ་རྗེ་ལྟ་བུའི་ཏིང་འཛིན་གྱིས། །ཆོས་ཀུན་སྒྱུ་མར་རིག་པ་ལས། །རིག་པའི་ཡེ་ཤེས་རཾ་ཞེས་བྱ། །རཾ་ལས་ཡེ་ཤེས་མེ་བྱུང་བས། །དངོས་དང་དངོས་མིན་བག་ཆགས་དང༌། །ཚོགས་ཀྱི་རྣམ་རྟོག་ངན་པ་བསྲེགས། །རཾ་བཛྲ་ཛྭ་ལ་རཾ། ཡཾ་རྡོ་རྗེ་ལྟ་བུའི་ཏིང་འཛིན་གྱིས། །ཆོས་ཀུན་སྒྱུ་མར་རིག་པ་ལས། །རིག་པའི་ཡེ་ཤེས་ཡཾ་ཞེས་བྱ། །ཡཾ་ལས་ཡེ་ཤེས་རླུང་བྱུང་བས། །དངོས་དང་དངོས་མིན་བག་ཆགས་དང༌། །ཚོགས་ཀྱི་རྣམ་རྟོག་ངན་པ་གཏོར། །ཡཾ་བཛྲ་སྥ་ར་ཎ་ཡཾ། ཁཾ་རྡོ་རྗེ་ལྟ་བུའི་ཏིང་འཛིན་གྱིས། །ཆོས་ཀུན་སྒྱུ་མར་རིག་པ་ལས། །རིག་པའི་ཡེ་ཤེས་ཁཾ་ཞེས་བྱ། །ཁཾ་ལས་ཡེ་ཤེས་ཆུ་བྱུང་བས། །དངོས་དང་དངོས་མིན་བག་ཆགས་དང༌། །ཚོགས་ཀྱི་རྣམ་རྟོག་ངན་པ་བཀྲུས། །ཁཾ་བཛྲ་བི་ཤུདྡྷེ་ཁཾ། ཚོགས་རྫས་ལ་བདུད་རྩི་བྲན། ཨོཾ་ཀཱ་ལ་རཱུ་པས་བསངས། ཨོཾ་སྭ་བྷཱ་ཝས་སྦྱངས༑ སྟོང་པའི་ངང་ལས་ཡཾ་ལས་རླུང༌། རཾ་ལས་མེ། ཀཾ་ལས་མི་མགོའི་སྒྱེད་པུ་གསུམ་གྱི་སྟེང་དུ། ཨ་ལས་ཚོགས་ཀྱི་སྣོད་བྷནྡྷའི་གཞོང་པ་ཡངས་ཤིང་རྒྱ་ཆེ་བའི་ནང་དུ། དབུས་སུ་ན་ལས་མིའི་ཤ། ཤར་དུ་གོ་ལས་བ་ལང་གི་ཤ། ལྷོར་ཀུ་ལས་ཁྱིའི་ཤ། ནུབ་ཏུ་ད་ལས་རྟའི་ཤ། བྱང་དུ་ཧ་ལས་གླང་པོ་ཆེའི་ཤ། དེ་རྣམས་ཀྱི་སྟེང་དུ་དབུས་སུ་ཧཱུྂ། ཕྱོགས་བཞིར་བྷྲཱུཾ་ཨཾ་ཛྲཱིཾ་ཁཾ༑ སླར་ཡང་དབུས་སུ་མཱུ་ལས་དྲི་ཆུ། ཤར་ལྷོར་བི་ལས་དྲི་ཆེན། ལྷོ་ནུབ་ཏུ་ར་ལས་རཀྟ། ནུབ་བྱང་དུ་ཤུ་ལས་བྱང་སེམས།

བྱང་ཤར་དུ་མ་ལས་གོ་རོ་ཙ་ན། དེ་རྣམས་ཀྱི་སྟེང་དུ་མཱཾ་ལཱཾ་བཱཾ་པཱཾ་ཏཱཾ་རྣམས་ཀྱིས་མཚན་པར་གྱུར། ཧཱུྂ་སྣ་རླུང་གིས་རླུང་གཡོས་མེ་སྦར་ཐོད་པ་དྲོས། རྫས་རྣམས་ཁོལ། མ་དག་པའི་ཉེས་སྐྱོན་ཐམས་ཅད་སྦྱངས། དག་པ་ཡེ་ཤེས་ཀྱི་བདུད་རྩིའི་རྒྱ་མཚོ་ཆེན་པོར་གྱུར། ལག་མཐིལ་གཡས་གཡོན་གཉིས་ཀར་བྱ་ཁྱུང་གི་ཕྱག་རྒྱ་རྣམས་བསྟན་ལ། ཨོཾ་ཨཱཿཧཱུྂ་ཧ་ཧོཿཧྲཱིཿ གསུམ་བརྗོད། ཚོགས་ཆ་གསུམ་དུ་སྤྲོས་པའི་དང་པོ་མཆོད་པ་ནི། གཏོར་མར་བསྟབ། སྨན་རཀ་བྲན་ལ། ཧཱུྂ་༌། ཆོས་དབྱིངས་དྲག་པོ་ཨེ་ཡི་ཞིང་ཁམས་སུ། །གཏི་མུག་མུན་ལྟར་འཐིབས་པའི་ཀློང་དཀྱིལ་ནས། །བཅོམ་ལྡན་ཕྱག་རྒྱ་ཟིལ་གནོན་འཁོར་དང་བཅས། །ཡེ་ཤེས་ཚོགས་ལ་སྤྱན་འདྲེན་གཤེགས་སུ་གསོལ། །འདོད་པའི་ཡོན་ཏན་ལྔ་ཡི་ཚོགས་མཆོད་འབུལ། །ཚོགས་ཀྱི་མཆོད་པ་དམ་པ་ཡར་བཞེས་ལ། །ཚེ་དང་དབང་གི་དངོས་གྲུབ་བསྩལ་དུ་གསོལ༑ ༑འདོད་ཆགས་ཆུ་ལྟར་འབྲུབ་པའི་ཀློང་དཀྱིལ་ནས། །ཡུམ་མཆོག་མ་མོ་ཁྲག་མིག་འཁོར་དང་བཅས། །ཡེ་ཤེས་ཚོགས་ལ་སྤྱན་འདྲེན་གཤེགས་སུ་གསོལ། །འདོད་པའི་ཡོན་ཏན་ལྔ་ཡི་ཚོགས་མཆོད་འབུལ། །ཚོགས་ཀྱི་མཆོད་པ་དམ་པ་ཡར་བཞེས་ལ། །ཚེ་དང་དབང་གི་དངོས་གྲུབ་བསྩལ་དུ་གསོལ། །ང་རྒྱལ་རི་ལྟར་དཔུངས་པའི་ཀློང་དཀྱིལ་ནས། །སྲས་མཆོག་ཡ་མ་རཱ་ཙ་འཁོར་དང་བཅས། །ཡེ་ཤེས་ཚོགས་ལ་སྤྱན་འདྲེན་གཤེགས་སུ་གསོལ། །འདོད་པའི་ཡོན་ཏན་ལྔ་ཡི་ཚོགས་མཆོད་འབུལ། །ཚོགས་ཀྱི་མཆོད་པ་དམ་པ་ཡར་བཞེས་ལ། །ཚེ་དང་དབང་གི་དངོས་གྲུབ་བསྩལ་དུ་གསོལ། །ཞེ་སྡང་མེ་ལྟར་འབར་བའི་ཀློང་དཀྱིལ་ནས། །སྐུ་སྤྲུལ་འཆི་བདག་ནག་པོ་འཁོར་དང་བཅས། །ཡེ་ཤེས་ཚོགས་ལ་སྤྱན་འདྲེན་གཤེགས་སུ་གསོལ༑ ༑འདོད་པའི་ཡོན་ཏན་ལྔ་ཡི་ཚོགས་མཆོད་འབུལ། །ཚོགས་ཀྱི་མཆོད་པ་དམ་པ་ཡར་བཞེས་ལ། །ཚེ་དང་དབང་གི་དངོས་གྲུབ་བསྩལ་དུ་གསོལ། །ཕྲག་དོག་རླུང་ལྟར་འཚུབས་པའི་དཀྱིལ་འཁོར་ནས། །གསུང་སྤྲུལ་སྲོག་བདག་ནག་པོ་འཁོར་དང་བཅས། །ཡེ་ཤེས་ཚོགས་ལ་སྤྱན་འདྲེན་གཤེགས་སུ་གསོལ། །འདོད་པའི་ཡོན་ཏན་ལྔ་ཡི་ཚོགས་མཆོད་འབུལ། །ཚོགས་ཀྱི་མཆོད་པ་དམ་པ་ཡར་བཞེས་ལ། །ཚེ་དང་དབང་གི་དངོས་གྲུབ་བསྩལ་དུ་གསོལ། །བཀའ་དང་དམ་ཚིག་གཉན་པོའི་གཞལ་ཡས་ནས། །འཁོར་གྱི་མ་གཤིན་དྲེགས་པའི་ཚོགས་རྣམས་ཀུན། །ཡེ་ཤེས་ཚོགས་ལ་སྤྱན་འདྲེན་གཤེགས་སུ་གསོལ། །འདོད་པའི་ཡོན་ཏན་ལྔ་ཡི་ཚོགས་མཆོད་འབུལ། །ཚོགས་ཀྱི་མཆོད་པ་དམ་པ་ཡར་བཞེས་ལ། །ཚེ་དང་དབང་གི་དངོས་གྲུབ་བསྩལ་དུ་གསོལ། །ཨོཾ་ཨེ་ཀ་ཡ་མ་དུ་རུ་ཙཀྲ་ག་ཎ་ཙཀྲ་པཱུ་ཛ་ཁཱ་ཧི།

ཧེ་ཀ་ཡ་མ་དུ་རུ་ཙཀྲ༴ གྷྲི་མ་ཡ་མ་དུ་རུ་ཙཀྲ༴ ཨོཾ་རཱ་ཏྲིའི་སྲོག་ལ་མཱ་ར་ཡ་རྦད༴ ཀླུ་བདུད་ནག་པོ་དགྲ་བོའི་སྲོག་ལ་མཱ་ར་ཡ་རྦད༴ ཡ་མཱ་རི་ས་པ་རི་ཝཱ་ར་ག་ཎ་ཙཀྲ་པཱུ་ཛ་ཁཱ་ཧི། སླར་མཆོད་པ་ནི། ཧཱུྂ་༌། དཔལ་ཆེན་ཕྱག་རྒྱ་ཟིལ་གནོན་འཁོར་བཅས་ལ། །ཕྱག་རྒྱ་གླུ་གར་ཕྱི་ནང་མཆོད་པ་དང༌། །སྦྱོར་སྒྲོལ་གཏོར་མ་རཀྟའི་མཆོད་པ་དང༌། །དམ་རྫས་བདུད་རྩི་རྫས་ཀྱི་མཆོད་པ་དང༌། །ཆོས་ཉིད་རོལ་པའི་མཆོད་པ་འབུལ་ལགས་ན། །གཉིས་མེད་བླང་དོར་མེད་པར་བཞེས་ནས་ཀྱང༌། །ཐུན་མཆོག་རིགས་ཀྱི་དངོས་གྲུབ་བསྩལ་དུ་གསོལ། །ཡུམ་མཆོག་མ་མོ་ཁྲག་མིག་འཁོར་བཅས་ལ། །ཕྱག་རྒྱ་གླུ་གར་ཕྱི་ནང་མཆོད་པ་དང༌། །སྦྱོར་སྒྲོལ་གཏོར་མ་རཀྟའི་མཆོད་པ་དང༌། །དམ་རྫས་བདུད་རྩི་རྫས་ཀྱི་མཆོད་པ་དང༌། །ཆོས་ཉིད་རོལ་པའི་མཆོད་པ་འབུལ་ལགས་ན། །གཉིས་མེད་བླང་དོར་མེད་པར་བཞེས་ནས་ཀྱང༌། །ཐུན་མཆོག་རིགས་ཀྱི་དངོས་གྲུབ་བསྩལ་དུ་གསོལ། །སྲས་མཆོག་ཡ་མ་རཱ་ཙ་འཁོར་བཅས་ལ། །ཕྱག་རྒྱ་གླུ་གར་ཕྱི་ནང་མཆོད་པ་དང༌། །སྦྱོར་སྒྲོལ་གཏོར་མ་རཀྟའི་མཆོད་པ་དང༌། །དམ་རྫས་བདུད་རྩི་རྫས་ཀྱི་མཆོད་པ་དང༌། །ཆོས་ཉིད་རོལ་པའི་མཆོད་པ་འབུལ་ལགས་ན། །གཉིས་མེད་བླང་དོར་མེད་པར་བཞེས་ནས་ཀྱང༌། །ཐུན་མཆོག་རིགས་ཀྱི་དངོས་གྲུབ་བསྩལ་དུ་གསོལ། །སྐུ་སྤྲུལ་འཆི་བདག་ནག་པོ་འཁོར་བཅས་ལ། །ཕྱག་རྒྱ་གླུ་གར་ཕྱི་ནང་མཆོད་པ་དང༌། །སྦྱོར་སྒྲོལ་གཏོར་མ་རཀྟའི་མཆོད་པ་དང༌། །དམ་རྫས་བདུད་རྩི་རྫས་ཀྱི་མཆོད་པ་དང༌། །ཆོས་ཉིད་རོལ་པའི་མཆོད་པ་འབུལ་ལགས་ན། །གཉིས་མེད་བླང་དོར་མེད་པར་བཞེས་ནས་ཀྱང༌། །ཐུན་མཆོག་རིགས་ཀྱི་དངོས་གྲུབ་བསྩལ་དུ་གསོལ། །གསུང་སྤྲུལ་སྲོག་བདག་ནག་པོ་་་་་་དམར་པོ། འཁོར་བཅས་ལ། །ཕྱག་རྒྱ་གླུ་གར་ཕྱི་ནང་མཆོད་པ་དང༌། །སྦྱོར་སྒྲོལ་གཏོར་མ་རཀྟའི་མཆོད་པ་དང༌། །དམ་རྫས་བདུད་རྩི་རྫས་ཀྱི་མཆོད་པ་དང༌། །ཆོས་ཉིད་རོལ་པའི་མཆོད་པ་འབུལ་ལགས་ན༑ ༑གཉིས་མེད་བླང་དོར་མེད་པར་བཞེས་ནས་ཀྱང༌། །ཐུན་མཆོག་རིགས་ཀྱི་དངོས་གྲུབ་བསྩལ་དུ་གསོལ། །འཁོར་གྱི་མ་གཤིན་དྲེགས་པ་འཁོར་བཅས་ལ། །ཕྱག་རྒྱ་གླུ་གར་ཕྱི་ནང་མཆོད་པ་དང༌། །སྦྱོར་སྒྲོལ་གཏོར་མ་རཀྟའི་མཆོད་པ་དང༌། །དམ་རྫས་བདུད་རྩི་རྫས་ཀྱི་མཆོད་པ་དང༌། །ཆོས་ཉིད་རོལ་པའི་མཆོད་པ་འབུལ་ལགས་ན། །གཉིས་མེད་བླང་དོར་མེད་པར་བཞེས་ནས་ཀྱང༌། །ཐུན་མཆོག་རིགས་ཀྱི་དངོས་གྲུབ་བསྩལ་དུ་གསོལ། །ཨོཾ་ཡ་མཱནྟ་ཀ་ས་པ་རི་ཝཱ་ར་མུ་དྲ་གཱི་ཏི་ནྲྀ་ཏྱཻ་པཱུ་ཛ་མ་ཧཱ་པཱུ་ཛ་ཡོ་ག་ཏ་རཱ་བ་ལིཾ་རུ་དྷཱི་ར་ཨ་མྲྀ་ཏ་དྷརྨ་དྷཱ་ཏུ་བ་ལིངྟ་སརྦ་པཱུ་ཛ་མེ་གྷ་ས་མུ་དྲ་སྥ་ར་ཎ་ས་མ་ཡ་ཤྲཱི་ཡེ་ཧཱུྂ།

ཞེས་དབུལ། བར་པ་བསྐང་ཞིང་བཤགས་པ་ནི། སྨན་རཀ་སྒྲོན་ལ། ཧཱུྂ་༌། བདག་ཅག་མ་རིག་ལུས་དང་ངག་སེམས་ཀྱི། །བག་མེད་སྤྱོད་པས་དམ་ལས་འགལ་གྱུར་ཏེ། །འདོད་ཁྲོ་གཏི་མུག་དབང་གིས་ནོངས་གྱུར་པས། །ཕྱག་རྒྱ་ཟིལ་གནོན་ཐུགས་དང་ཅི་འགལ་བ། །ཞལ་ཟས་ཚོགས་ཀྱི་མཆོད་པས་བསྐང་ཞིང་བཤགས། །ཐུགས་དམ་བསྐངས་ལ་དངོས་གྲུབ་བསྩལ་དུ་གསོལ། །བདག་ཅག་མ་རིག་ལུས་དང་ངག་སེམས་ཀྱི། །བག་མེད་སྤྱོད་པས་དམ་ལས་འགལ་གྱུར་ཏེ། །འདོད་ཁྲོ་གཏི་མུག་དབང་གིས་ནོངས་གྱུར་པས། །ཡུམ་མཆོག་མ་མོ་ཁྲག་མིག་ཐུགས་དང་ཅི་འགལ་བ། །ཞལ་ཟས་ཚོགས་ཀྱི་མཆོད་པས་བསྐང་ཞིང་བཤགས། །ཐུགས་དམ་བསྐངས་ལ་དངོས་གྲུབ་བསྩལ་དུ་གསོལ༑ ༑བདག་ཅག་མ་རིག་ལུས་དང་ངག་སེམས་ཀྱི། །བག་མེད་སྤྱོད་པས་དམ་ལས་འགལ་གྱུར་ཏེ། །འདོད་ཁྲོ་གཏི་མུག་དབང་གིས་ནོངས་གྱུར་པས། །སྲས་མཆོག་ཡ་མ་རཱ་ཙའི་ཐུགས་དང་ཅི་འགལ་བ། །ཞལ་ཟས་ཚོགས་ཀྱི་མཆོད་པས་བསྐང་ཞིང་བཤགས། །ཐུགས་དམ་བསྐངས་ལ་དངོས་གྲུབ་བསྩལ་དུ་གསོལ། །བདག་ཅག་མ་རིག་ལུས་དང་ངག་སེམས་ཀྱི། །བག་མེད་སྤྱོད་པས་དམ་ལས་འགལ་གྱུར་ཏེ། །འདོད་ཁྲོ་གཏི་མུག་དབང་གིས་ནོངས་གྱུར་པས། །སྐུ་སྤྲུལ་འཆི་བདག་ནག་པོའི་ཐུགས་དང་ཅི་འགལ་བ། །ཞལ་ཟས་ཚོགས་ཀྱི་མཆོད་པས་བསྐང་ཞིང་བཤགས། །ཐུགས་དམ་བསྐངས་ལ་དངོས་གྲུབ་བསྩལ་དུ་གསོལ། །བདག་ཅག་མ་རིག་ལུས་དང་ངག་སེམས་ཀྱི། །བག་མེད་སྤྱོད་པས་དམ་ལས་འགལ་གྱུར་ཏེ། །འདོད་ཁྲོ་གཏི་མུག་དབང་གིས་ནོངས་གྱུར་པས། །གསུང་སྤྲུལ་སྲོག་བདག་ནག་་་་་་དམར་པོ། པོའི་ཐུགས་དང་ཅི་འགལ་བ། །ཞལ་ཟས་ཚོགས་ཀྱི་མཆོད་པས་བསྐང་ཞིང་བཤགས། །ཐུགས་དམ་བསྐངས་ལ་དངོས་གྲུབ་བསྩལ་དུ་གསོལ། །བདག་ཅག་མ་རིག་ལུས་དང་ངག་སེམས་ཀྱི། །བག་མེད་སྤྱོད་པས་དམ་ལས་འགལ་གྱུར་ཏེ། །འདོད་ཁྲོ་གཏི་མུག་དབང་གིས་ནོངས་གྱུར་པས༑ ༑འཁོར་གྱི་མ་གཤིན་དྲེགས་པའི་ཚོགས་ཀྱི་ཐུགས་དང་ཅི་འགལ་བ། །ཞལ་ཟས་ཚོགས་ཀྱི་མཆོད་པས་བསྐང་ཞིང་བཤགས། །ཐུགས་དམ་བསྐངས་ལ་དངོས་གྲུབ་བསྩལ་དུ་གསོལ༑ ༑ཨོཾ་ཡ་མཱནྟ་ཀ་ཡ་མ་རཱ་ཛ་ས་མ་ཡ།

མ་ནུ་པཱ་ལ་ཡ། སོགས་ཡིག་བརྒྱ་བརྗོད། གསུམ་པ་བསྒྲལ་ཞིང་བསྟབ་པ་ནི། ཉ་བོ་ལ་དགུག་གཞུག་བྱས་ལ། ནྲྀ་ཤ་ཀུ་རུ། ནྲྀ་ཤ་ཀུ་རུ། ཁ་སོ་པ་ཏི། ཁ་སོ་པ་ཏི༑ ཡཾ་རུ་ཏི། ཡཾ་སེན་ཡཾ་སེན། མན་ཡཾ། སྙིང་ཚལ་པ་རྦད། བཾ་རིལ་རྦད། དམ་ཉམས་དགྲ་བོ་བྱད་མ་ཁྱོད་ཀྱི་བླ་རྟེན་ཚེ་རྟེན་སྲོག་རྟེན་ཅི་ཡོད་པ་ཐམས་ཅད་འདིར་ཁུག་མྱུར་དུ་ཤོག་ཅིག ༑ཨོཾ་ཨ་ཀྲོ་ཏེ་ཀ་ཡ་མཱནྟ་ཀ་ཀཱ་ལ་རཱུ་པ་དགྲ་བོ་ནྲྀ་ཨཾ་ཀུ་ཤ་ཛ་ཛཿ ཞེས་དགུག །ཡ་མ་རཱ་ཛ་རླུང་ལིང་སང་སོགས་དང༌། ཨ་ནྲྀ་ཏྲཀ་རཀྴ་མཱ་ར་ཡ་སོགས་ཀྱིས་ལྷ་དང་དབྱེ། ཨོཾ་ཨེ་ཀ་ཡ་མ་དུ་རུ་ཙཀྲས་སོད། སོགས་བསད་སྔགས་ཀྱིས་ཐུན་བྲབ། ཕུར་བུ་བཏབ། གྲི་ལྕགས་ཀྱིས་བསྒྲལ་བའི་ལིང་རོ་ཚོགས་ཕུད་ཆ་གསུམ་དང་ལྷན་ཅིག་བསྒྲོན་ལ། ཧཱུྂ་༌། བསྟབ་བོ་བསྟབ་བོ་ཞལ་དུ་བསྟབ། །ཁྲོ་རྒྱལ་གཤིན་རྗེའི་གཤེད་པོ་ཞལ་ཕྱེས་ཤིག །མཆོད་རྫས་ཉོན་མོངས་དགྲ་བགེགས་ཞལ་དུ་བསྟབ། །བཞེས་ཤིག་རོལ་ཅིག་བསྟབ་བོ་ཆེ། །བསྟབ་བོ་བསྟབ་བོ་ཞལ་དུ་བསྟབ། །ཡུམ་མཆོག་མ་མོ་ཁྲག་མིག་ཞལ་ཕྱེས་ཤིག །མཆོད་རྫས་ཉོན་མོངས་དགྲ་བགེགས་ཞལ་དུ་བསྟབ། །བཞེས་ཤིག་རོལ་ཅིག་བསྟབ་བོ་ཆེ། །བསྟབ་བོ་བསྟབ་བོ་ཞལ་དུ་བསྟབ། །སྲས་མཆོག་ཡ་མ་རཱ་ཙ་ཞལ་ཕྱེས་ཤིག །མཆོད་རྫས་ཉོན་མོངས་དགྲ་བགེགས་ཞལ་དུ་བསྟབ། །བཞེས་ཤིག་རོལ་ཅིག་བསྟབ་བོ་ཆེ། །བསྟབ་བོ་བསྟབ་བོ་ཞལ་དུ་བསྟབ། །སྐུ་སྤྲུལ་འཆི་བདག་ནག་པོ་ཞལ་ཕྱེས་ཤིག །མཆོད་རྫས་ཉོན་མོངས་དགྲ་བགེགས་ཞལ་དུ་བསྟབ། །བཞེས་ཤིག་རོལ་ཅིག་བསྟབ་བོ་ཆེ། །བསྟབ་བོ་བསྟབ་བོ་ཞལ་དུ་བསྟབ། །གསུང་སྤྲུལ་སྲོག་བདག་ནག་པོ་ཞལ་ཕྱེས་ཤིག ༑མཆོད་རྫས་ཉོན་མོངས་དགྲ་བགེགས་ཞལ་དུ་བསྟབ། །བཞེས་ཤིག་རོལ་ཅིག་བསྟབ་བོ་ཆེ། །བསྟབ་བོ་བསྟབ་བོ་ཞལ་དུ་བསྟབ། །འཁོར་གྱི་མ་གཤིན་དྲེགས་པ་ཞལ་ཕྱེས་ཤིག །མཆོད་རྫས་ཉོན་མོངས་དགྲ་བགེགས་ཞལ་དུ་བསྟབ། །བཞེས་ཤིག་རོལ་ཅིག་བསྟབ་བོ་ཆེ། །ཡབ་གཅིག་དཔལ་ཆེན་ཤ་ལ་དགྱེས། །ཤ་ཡི་རང་བཞིན་གཏི་མུག་ཆེ། །གཏི་མུག་མཆོད་པ་བླ་ན་མེད། །བླ་མེད་མཆོད་པ་འདི་བཞེས་ལ། །སྐུ་ཡི་དངོས་གྲུབ་བསྩལ་དུ་གསོལ། །ཡབ་གཅིག་དཔལ་ཆེན་ཁྲག་ལ་དགྱེས། །ཁྲག་གི་རང་བཞིན་འདོད་ཆགས་ཆེ། །འདོད་ཆགས་མཆོད་པ་བླ་ན་མེད། །བླ་མེད་མཆོད་པ་འདི་བཞེས་ལ། །གསུང་གི་དངོས་གྲུབ་བསྩལ་དུ་གསོལ། །ཡབ་གཅིག་དཔལ་ཆེན་རུས་ལ་དགྱེས། །རུས་པའི་རང་བཞིན་ཞེ་སྡང་ཆེ། །ཞེ་སྡང་མཆོད་པ་བླ་ན་མེད། །བླ་མེད་མཆོད་པ་འདི་བཞེས་ལ། །ཐུགས་ཀྱི་དངོས་གྲུབ་བསྩལ་དུ་གསོལ། །དགྲ་བགེགས་མཁྲིས་པ་སྨན་གྱི་རྒྱུ། །སྨན་གྱི་རང་བཞིན་ང་རྒྱལ་ཆེ། །ང་རྒྱལ་མཆོད་པ་བླ་ན་མེད༑ ༑བླ་མེད་མཆོད་པ་འདི་བཞེས་ལ། །ཡོན་ཏན་དངོས་གྲུབ་བསྩལ་དུ་གསོལ། །དགྲ་བགེགས་གློ་སྙིང་གསང་བའི་རྟེན། །གསང་བའི་རང་བཞིན་ཕྲག་དོག་ཆེ། །ཕྲག་དོག་མཆོད་པ་བླ་ན་མེད། །བླ་མེད་མཆོད་པ་འདི་བཞེས་ལ། །ཕྲིན་ལས་དངོས་གྲུབ་བསྩལ་དུ་གསོལ། །དུག་ལྔའི་མཆོད་པ་བླ་ན་མེད། །བླ་མེད་མཆོད་པ་འདི་བཞེས་ལ། །སྐུ་ལྔའི་དངོས་གྲུབ་བསྩལ་དུ་གསོལ། །དུག་གསུམ་མཆོད་པ་བླ་ན་མེད། །བླ་མེད་མཆོད་པ་འདི་བཞེས་ལ། །སྐུ་གསུམ་དངོས་གྲུབ་བསྩལ་དུ་གསོལ། །ཨོཾ་ཨེ་ཀ་ཡ་མ་དུ་རུ་ཙཀྲ་མ་ཧཱ་མཾ་ས་ལ་ཁཱ་ཧི།

མ་ཧཱ་རཀྟ༴ མ་ཧཱ་ཀེཾ་ནི་རི་ཏི༴ མ་ཧཱ་ཙིཏྟ༴ མ་ཧཱ་གོ་རོ་ཙ་ན༴ མ་ཧཱ་བ་སུ་ཏ༴ མ་ཧཱ་བོ་དྷི་ཙིཏྟ༴ མ་ཧཱ་སྲོག་ཨཱཛྙཱ༴ མ་ཧཱ་པཉྩ་ཨ་མྲྀ་ཏ༴ མ་ཧཱ་རཀྟ༴ མ་ཧཱ་བ་ལིངྟ་ལ་ཁཱ་ཧི། དེ་བཞིན། ཧེ་ཀ་ཡ་མ་དུ་རུ་ཙཀྲ༴ གྷྲི་མ་ཡ་མ་དུ་རུ་ཙཀྲ༴ ཨོཾ་རཱ་ཏྲི༴ ཀླུ་བདུད་ནག་པོ༴ ཡ་མཱ་རི་ས་པ་རི་ཝཱ་ར་མ་ཧཱ་མཾ་ས་ལ་ཁཱ་ཧི། མ་ཧཱ་རཀྟ༴ མ་ཧཱ་ཀེཾ་ནི་རི་ཏི༴ མ་ཧཱ་ཙིཏྟ༴ མ་ཧཱ་གོ་རོ་ཙ་ན༴ མ་ཧཱ་བ་སུ་ཏ༴ མ་ཧཱ་བོ་དྷི་ཙིཏྟ༴ མ་ཧཱ་སྲོག་ཨཱཛྙཱ༴ མ་ཧཱ་པཉྩ་ཨ་མྲྀ་ཏ༴ མ་ཧཱ་རཀྟ༴ མ་ཧཱ་བ་ལིངྟ་ལ་ཁཱ་ཧི། ཞེས་བསྟབས། བག་དང་ལྡན་པས་ལོངས་སྤྱད། དམ་ཚིག་གཅིག་པ་ཁོ་ནའི་ལྷག་མ་བསྡུས་པ་ཆ་གསུམ་དུ་བཤམས་ལ། དྲག་པོའི་གཏོར་མ་བྱིན་རླབས་ཀྱི་སྔགས་རྒྱ་ལྟར་ལ། ལྷག་མའི་གཏོར་མ་ཞེས་བསྒྱུར་བས་བྱིན་གྱིས་བརླབས། བྷྱོཿ དང་པོ་ཕུད་ཀྱིས་མི་མཆོད་རྒྱུ། །དཔལ་ཆེན་ཟིལ་གནོན་ཕུད་ཀྱི་བདག །བར་དུ་ཚོགས་ལ་མི་རོལ་རྒྱུ། །རྣལ་འབྱོར་ལོངས་སྤྱོད་ཆེན་པོའི་བདག །ཐ་མར་ལྷག་མས་མི་མཆོད་རྒྱུ། །མ་གཤིན་དྲེགས་པ་ལྷག་མའི་བདག །མཆོད་ནས་ལས་ལ་མི་བསྐུལ་རྒྱུ། །ཆོས་འཁོར་བསྟན་པ་བཤིག་པ་བྱུང༌། །མཉམ་གསུམ་གཅིག་དུ་མི་བསྲེ་རྒྱུ། །ཆོས་ཀུན་གདོད་ནས་རྒྱུ་འབྲས་གཅིག །ཅེས་ཐོག་མཐའ་གཉིས་དབུས་མར་དྲིལ། མེ་མར་བཙུགས། སྨན་རཀ་བྲན་ལ། བྷྱོཿ གཤིན་རྗེ་རོལ་པའི་དུར་ཁྲོད་ཆེན་པོ་ནས། །བཀའ་ཉན་རོལ་པའི་ལྷག་སྡུད་རྣམས། །སྔགས་ཚིག་བཤུག་པས་མཚོན་ནས་ཀྱང༌། །མ་ལུས་འཁོར་བཅས་གཤེགས་སུ་གསོལ༑ ༑ཞིང་ཆེན་གཡབ་མོས་མཚོན་ནས་ཀྱང༌། །མ་ལུས་འཁོར་བཅས་གཤེགས་སུ་གསོལ། །གླུ་གར་སྒྲ་ཡིས་མཚོན་ནས་ཀྱང༌། །མ་ལུས་འཁོར་བཅས་གཤེགས་སུ་གསོལ། །རཀྟ་དམར་གྱིས་མཚོན་ནས་ཀྱང༌། །མ་ལུས་འཁོར་བཅས་གཤེགས་སུ་གསོལ། །རྐང་གླིང་ཐོད་རྔས་མཚོན་ནས་ཀྱང༌། །མ་ལུས་འཁོར་བཅས་གཤེགས་སུ་གསོལ། །འདིར་བྱོན་དམ་རྫས་ལྷག་གཏོར་བཞེས། །དཔལ་གྱི་ཕ་འབབ་འདི་བཞེས་ལ། །བཅོལ་བའི་ཕྲིན་ལས་གྲུབ་པར་མཛོད། །ཨོཾ་ཡ་མཱ་རི་ཨུཙྪིཥྚ་བ་ལིངྟ་བྷཀྵ་བྷྱཿསྭཱ་ཧཱ།

ཞེས་གཙང་སར་དབུལ་ལོ། །ཡང་ཟློག་མེ་ཡི་སྤུ་གྲི་ཡི། །ལྷན་ཐབས་ཚོགས་ཀྱི་འཁོར་ལོ་འདི། །བདེ་གཤེགས་འདུས་པའི་རྒྱས་བཀབ་ནས། །ཁ་སྐོང་ཚུལ་དུ་བཀོད་པ་ཡིན། །བདེ་གཤེགས་འདུས་པའི་ལྷོ་ཕྱོགས་ཀྱི། །དཀྱིལ་འཁོར་གཙོ་མཆོག་གཤིན་རྗེའི་གཤེད། །ཡིན་ཕྱིར་དེ་ཡིས་རྒྱས་བཀབ་པ། །རྒྱུད་མཐུན་མཆོག་ཏུ་ཤིས་མིན་ནམ། །རིགས་མི་མཐུན་པའི་ལྷ་སྔགས་སོགས། །རྒྱས་འདེབས་རང་བཟོ་གཟུ་ལུམ་ཅན། །འགའ་ཞིག་མཐོང་ཞིང་འགའ་ཞིག་ཐོས། །གསང་སྔགས་བསྟན་པའི་རུ་ཏྲར་སྣང༌། །འདི་ནི་བསྙེན་སྒྲུབ་ལས་སྦྱོར་ལ༑ ༑མཁོ་ཞིང་ཁྱད་པར་བཟའ་བ་ཡི། །དམ་ཚིག་མཆོག་ཏུ་བསྔགས་པས་ན། །ཚོགས་ཀྱི་འཁོར་ལོའི་མན་ངག་འདི། །སྔགས་འཆང་ཆོས་ཀྱིས་གྲགས་པས་བྲིས། །དགེ་བས་རྫོགས་བྱང་མྱུར་ཐོབ་ཤོག། །།སརྦ་མངྒ་ལཾ།།
