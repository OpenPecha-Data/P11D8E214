ཡ་མཱ་རི་ན་མོ། གྲིབ་འཇོམས་རྡོ་རྗེ་ཕ་ལམ་གྱི་ལས་མཐའ་གྲིབ་མདོས་བསྡུས་པ་ནི། བྲ་སྲན་འབྲུ་ནག་དང་ཚིག་ཕྱེ་ལ་ཟན་ནག་བརྫིས་ཏེ་ལྕགས་སྣོད་ལ་སོགས་པའི་ནང་དུ་རི་རབ་གྲུ་བཞི་བང་རིམ་བཞི་པ་བྱས་པའི་བང་རིམ་དང་པོར་མི་ནག་པོ་མགོ་ལ་འཕྲུ་ནག་པོ་བཙུགས་པ། རྟ་ནག་ཞོན་ཅིང་ཀླད་ན་བྱ་ནག་ལྡིང་བ། རོལ་དུ་ཁྱི་ནག་ཁྲིད་པ། གོས་ནག་གྱོན་པ། ལག་ན་དར་ནག་འཕྱར་བ། རྒྱབ་ཏུ་བྱ་མོ་ནག་མོས་སྒོང་སྐོགས་ཁུར་བ་བྱ། བང་རིམ་གཉིས་པའི་ཕྱོགས་བཞིར་མི་གཟུགས་བྱ་སྤྲུལ་ཕག་སྦལ་པའི་མགོ་ཅན་བཞི་བཞག །བང་རིམ་གསུམ་པར་ཤར་སྟོད་སྟག་སོགས་རང་རང་འབྱུང་བའི་མདོག་གམ་ཐམས་ཅད་ནག་པོ་བྱས་ལ་བཀོད། བང་རིམ་བཞི་པར། མར་མེ། བཤོས་བུ། ཆང་བུ། རིལ་བུ། ཨོང་སྐྱུ་རྣམས་བསྒྲུབ་བྱའི་ལོ་གྲངས་དང་མཉམ་པ་ཐམས་ཅད་ཁ་དོག་ནག་པོ་བྱས་ལ་རང་གི་མདུན་དུ་གཞི་མ་ནག་པོའི་སྟེང་དུ་བཤམ། སྣོད་གཅིག་ཏུ་བྲའོ། རམ་བུ། སྦང་མ། གཡེར་མ། རུས་པ་སོགས་མི་ཁའི་རྫས་བསག །སྣོད་གཅིག་ཏུ་བསྒྲུབ་བྱའི་ལོ་གྲངས་དང་མཉམ་པའི་ངར་གླུད་ཆང་བུ་མཐེབ་སྐྱུ་རྣམས་བཤམ། གཞན་ཡང་ཡ་མཱ་རི་དང་སྨེ་བརྩེགས་ལ་དབུལ་གཏོར། མཆོད་པ། སྨན་རཀ། རོལ་མོ། གཡབ་དར་ནག་པོ་སོགས་འདུ་བྱ། རང་ཡ་མཱ་རིར་གསལ་བའི་ཐུགས་ཀར་སྨེ་བརྩེགས་བསྐྱེད། བུམ་པ་གཞུང་ལྟར་བཅས་ལ་སྨེ་བརྩེགས་བསྐྱེད་བཟླས་སོགས་གྲིབ་འཇོམས་རྡོ་རྗེ་ཕ་ལམ་ལྟར་བྱས་རྗེས་གཡབ་དར་ནག་པོ་བསྒྲུབ་བྱའི་མགོ་ལ་བསྐོར་ཞིང༌། ཡ་མཱ་རི་སོགས་ཀྱི་བདེན་སྟོབས་བརྗོད་ནས་གྲིབ་བདག་བཀུག་ནས་གཟུགས་ལ་བསྟིམ། དེ་ནས་དབྱངས་ཀྱིས་གྱེར་བ་ནི། ཧཱུྂ་༌། ང་ནི་འཇམ་དཔལ་གཤིན་རྗེའི་གཤེད། །ང་ནི་ཁྲོ་རྒྱལ་སྨེ་བ་བརྩེགས། །རྟོག་འཇོམས་རྒྱལ་པོ་ཧེ་རུ་ཀཿ །སྣང་སྲིད་དྲེགས་པ་ཐམས་ཅད་དང༌། །གྲིབ་བདག་ཚོགས་རྣམས་ང་ལ་ཉོན། །ཀྱཻ། ཉི་མ་ནུབ་ཕྱོགས་ཕ་གི་ན། །མུན་པ་ནག་པོའི་གཏིང་རུམ་ན། །གྲིབ་མཁར་ནག་པོའི་ནང་ཤེད་ན། །གྲིབ་བདག་རྒྱལ་པོ་ཆེན་པོ་སྟེ། །སྐུ་མདོག་ནག་པོ་མུན་པའི་མདོག

།འོག་ཏུ་གྲིབ་རྟ་ནག་པོ་ཞོན། །སྟེང་དུ་གྲིབ་བྱ་ནག་པོ་ལྡིང༌། །གྲིབ་གོས་ནག་པོ་ལུས་ལ་གྱོན། །གྲིབ་དར་ནག་པོ་ལག་ན་འཕྱར། །གྲིབ་འཕྲུ་ནག་པོ་མགོ་ལ་བཙུགས། །གྲིབ་སྣོད་སྒོང་ནག་རྒྱབ་ཏུ་ཁུར། །གྲིབ་ཁྱི་ནག་པོ་རོལ་ཏུ་ཁྲིད། །གནས་འདིར་འབོད་དོ་དམ་ལ་འདུ། །གྲིབ་བདག་ཆེན་པོ་གྲིབ་ལོངས་ཤིག །ཀྱཻ། དེ་ཡི་བཀའ་ཉན་ལས་བྱེད་བཞི། །འདོད་ཆགས་བྱ་ནག་མགོ་བོ་ཅན། །ཞེ་སྡང་སྦྲུལ་ནག་མགོ་བོ་ཅན། །གཏི་མུག་ཕག་ནག་མགོ་བོ་ཅན། །མ་རིག་སྦལ་ནག་མགོ་བོ་ཅན། །ཁྱེད་རྣམས་གྲིབ་ཀྱི་བདག་པོ་སྟེ། །གནས་འདིར་འབོད་དོ་དམ་ལ་འདུས། །གྲིབ་ཀྱི་བདག་པོ་གྲིབ་ལོངས་ཤིག །ཀྱཻ། གྲིབ་བདག་རྒྱལ་པོ་ཆེན་པོ་དང༌། །དེ་ཡི་བཀའ་ཉན་བློན་པོ་བཞི། །ཁྱེད་རྣམས་གྲིབ་ཀྱི་བདག་པོ་ཡིན། །ཁྱེད་རྣམས་གཞི་བྱེས་བདག་པོ་ཡིན། །བདག་དང་རྒྱུ་སྦྱོར་འཁོར་བཅས་ལ། །ཟས་གྲིབ་གོས་གྲིབ་ནོར་གྱི་གྲིབ། །གསོན་གྲིབ་ཤི་གྲིབ་རོ་ཡི་གྲིབ། །དམ་ཉམས་གྲིབ་དང་རྨེ་གྲིབ་དང༌། །མཛེ་གྲིབ་འཁོན་གྲིབ་ཡུག་གྲིབ་དང༌། །ལྷ་ཀླུ་གཉན་སོགས་སྡེ་བརྒྱད་གྲིབ། །རྣམ་རྟོག་འཁྲུལ་བའི་གྲིབ་ལ་སོགས། །དུག་གསུམ་དུག་ལྔ་ལས་ཀྱི་གྲིབ། །ཉོན་མོངས་སྔོན་ལས་འཕྲལ་རྐྱེན་གྲིབ། །དེ་འདྲའི་གྲིབ་དང་ཉེས་པའི་ཚོགས། །གྲིབ་བདག་རྣམས་ཀྱིས་ད་ལོངས་ལ༑ ༑བདག་ཅག་གྲིབ་རྣམས་སངས་གྱུར་ཅིག །ཀྱཻ། གྲིབ་བདག་ཆེན་པོའི་བཀའ་འཁོར་ནི། །ཤར་སྟོད་སྟག་གི་མགོ་བོ་ཅན། །ནུབ་སྟོད་སྤྲེའུའི་མགོ་བོ་ཅན། །ཁྱོད་གཉིས་བདུན་ཟུར་དགྲ་གཤེད་དེ། །འཁོར་དུ་བགེགས་ཕྲན་འབུམ་གྱིས་བསྐོར། །ཁྱེད་རྣམས་གྲིབ་ཀྱི་བདག་པོ་སྟེ། །བདག་ཅག་གྲིབ་རྣམས་ད་ལོངས་ལ། །བདུན་ཟུར་གནོད་པ་ཞི་བར་མཛོད༑ ༑ཤར་སྨད་ཡོས་ཀྱི་མགོ་བོ་ཅན། །ནུབ་སྨད་བྱ་ཡི་མགོ་བོ་ཅན། །ཁྱེད་གཉིས་བྱེ་བྲལ་དགྲ་གཤེད་དེ། །འཁོར་དུ་བགེགས་ཕྲན་འབུམ་གྱིས་བསྐོར། །ཁྱེད་རྣམས་གྲིབ་ཀྱི་བདག་པོ་སྟེ༑ ༑བདག་ཅག་གྲིབ་རྣམས་ད་ལོངས་ལ།

བྱེ་བྲལ་གནོད་པ་ཞི་བར་མཛོད། །ལྷོ་སྟོད་སྦྲུལ་གྱི་མགོ་བོ་ཅན། །བྱང་སྟོད་ཕག་གི་མགོ་བོ་ཅན། །ཁྱེད་གཉིས་འཁོན་པའི་དགྲ་གཤེད་དེ། །འཁོར་དུ་བགེགས་ཕྲན་འབུམ་གྱིས་བསྐོར། །ཁྱེད་རྣམས་གྲིབ་ཀྱི་བདག་པོ་སྟེ། །བདག་ཅག་གྲིབ་རྣམས་ད་ལོངས་ལ། །འཁོན་གྱི་གནོད་པ་ཞི་བར་མཛོད། །ལྷོ་སྨད་རྟ་ཡི་མགོ་བོ་ཅན། །བྱང་སྨད་བྱི་བའི་མགོ་བོ་ཅན། །ཁྱེད་གཉིས་རུ་གཅོད་དགྲ་གཤེད་དེ། །འཁོར་དུ་བགེགས་ཕྲན་འབུམ་གྱིས་བསྐོར། །ཁྱེད་རྣམས་གྲིབ་ཀྱི་བདག་པོ་སྟེ། །བདག་ཅག་གྲིབ་རྣམས་ད་ལོངས་ལ༑ ༑འཁོན་གྱི་གནོད་པ་ཞི་བར་མཛོད། །ཤར་ལྷོ་འབྲུག་གི་མགོ་བོ་ཅན། །ནུབ་བྱང་ཁྱིའི་མགོ་བོ་ཅན། །ཁྱེད་གཉིས་ཐོག་རྫིས་དགྲ་གཤེད་དེ། །འཁོར་དུ་བགེགས་ཕྲན་འབུམ་གྱིས་བསྐོར༑ ༑ཁྱེད་རྣམས་གྲིབ་ཀྱི་བདག་པོ་སྟེ། །བདག་ཅག་གྲིབ་རྣམས་ད་ལོངས་ལ། །ཐོག་རྫིས་གནོད་པ་ཞི་བར་མཛོད། །ལྷོ་ནུབ་ལུག་གི་མགོ་བོ་ཅན། །བྱང་ཤར་གླང་གི་མགོ་བོ་ཅན། །ཁྱེད་གཉིས་ཕུང་གྱོད་དགྲ་གཤེད་དེ། །འཁོར་དུ་བགེགས་ཕྲན་འབུམ་གྱིས་བསྐོར། །ཁྱེད་རྣམས་གྲིབ་ཀྱི་བདག་པོ་སྟེ། །བདག་ཅག་གྲིབ་རྣམས་ད་ལོངས་ལ། །ཕུང་གྱོད་གནོད་པ་ཞི་བར་མཛོད༑ ༑བདག་ཅག་རྒྱུ་སྦྱོར་འཁོར་བཅས་ཀྱི། །གྲིབ་དང་མི་གཙང་གནོད་པ་དང༌། །དགྲ་གཤེད་སྐག་ངན་སྦྱོར་བ་རྣམས། །གདོན་ཆེན་གནས་པ་བཅུ་གཉིས་པོ། །ད་ལྟ་ལོངས་ལ་ཞི་བར་མཛོད༑ ༑དེ་ལ་ཤིང་མེ་ས་ལྕགས་ཆུ། །ལྔ་ལྔར་ཕྱེ་བ་དྲུག་ཅུས་བསྐོར། །འཁོར་དུ་བགེགས་ཕྲན་འབུམ་དང་བཅས། །ལོ་ཟླ་ཞག་དུས་དགྲ་གཤེད་དང༌། །དུས་བཞི་ཁམས་ལྔའི་གྲིབ་ལ་སོགས༑ ༑མི་གཙང་ངན་དགུ་བྱུང་བ་རྣམས། །གདོན་ཆེན་རྣམས་ཀྱིས་ད་ལོངས་ལ། །བདག་ཅག་གྲིབ་རྣམས་སངས་གྱུར་ཅིག །ཅེས་བརྗོད། གླུད་རྣམས་སྔགས་དྲུག་ཕྱག་རྒྱ་དྲུག་གིས་བྱིན་གྱིས་བརླབས་ལ། ཀྱཻ༑ སངས་རྒྱས་ཆོས་ཚོགས་མཆོག་གསུམ་གྱི། །བཀའ་ལ་མི་ཉན་སུ་ཞིག་ཡོད། །སྐྱ་རིང་ལམ་ལ་མི་འགྲོ་མེད། །འབབ་པ་ཆུ་ལ་མི་འཐུང་མེད། །བརྟན་པའི་གཞི་ལ་མི་གནས་མེད། །འཚོ་བའི་སྲོག་ལ་མི་གཅེས་མེད། །དྲང་པོའི་ཚིག་ལ་མི་ཉན་མེད། །རྔན་སྦྱིན་ཡས་ལ་མི་སེམས་མེད། །བདག་ཅག་རྒྱུ་སྦྱོར་བཅས་པ་ཡི། །ལུས་ནི་ཤ་ཁྲག་མི་གཙང་རྫས། །ངག་ནི་བྲག་ཆ་སྟོང་པའི་སྒྲ། །ཡིད་ནི་རླུང་ལྟར་དངོས་པོ་བྲལ། །དེས་ནི་ཁྱོད་རྣམས་ཅི་བྱར་ཡོད། །མི་ལས་ལྷག་པའི་ངར་གླུད་འདི། །འབྲུ་སྣ་ལྔ་དང་རིན་ཆེན་སྨན། །དར་ཟབ་ཟས་སྣ་མཁོ་དགུའི་ནོར། །ཀུན་བསྡུས་དབང་པོ་སྐྱེ་མཆེད་ཚང༌། །ལུས་ནི་མཛེས་པའི་གར་སྟབས་བསྒྱུར། །ངག་ནི་སྙན་པའི་གླུ་དབྱངས་ལེན། །ཡིད་ནི་མཛའ་བའི་བློ་སེམས་ཅན། །མི་ལས་དགའ་བའི་གླུད་འདི་འབུལ། །བདག་ཅག་རྒྱུ་སྦྱོར་འཁོར་བཅས་ཀྱི། །ཤ་ཡི་གླུད་འདི་ས་ལ་གཏོང༌། །ས་ལ་གནས་པའི་ལྷ་འདྲེ་རྣམས། །མཐུན་པའི་གླུད་ཆེན་འདི་བཞེས་ལ། །རང་དོན་ཡིན་ན་བདག་ལ་ཞེས་དང༌།

གཞན་དོན་ཡིན་ན། འཁོར་འདིར་གནོད་པ་མ་བྱེད་ཅིག །ཁྲག་གི་གླུད་འདི་ཆུ་ལ་གཏོང༌། །ཆུ་ལ་གནས་པའི་ལྷ་འདྲེ་རྣམས། །མཐུན་པའི་གླུད་ཆེན་འདི་བཞེས་ལ། །འཁོར་འདིར་གནོད་པ་མ་བྱེད་ཅིག །རུས་པའི་གླུད་འདི་བྲག་ལ་གཏོང༌། །བྲག་ལ་གནས་པའི་ལྷ་འདྲེ་རྣམས། །མཐུན་པའི་གླུད་ཆེན་འདི་བཞེས་ལ། །འཁོར་འདིར་གནོད་པ་མ་བྱེད་ཅིག །དྲོད་ཀྱི་གླུད་འདི་མེ་ལ་གཏོང༌། །མེ་ལ་གནས་པའི་ལྷ་འདྲེ་རྣམས། །མཐུན་པའི་གླུད་ཆེན་འདི་བཞེས་ལ། །འཁོར་འདིར་གནོད་པ་མ་བྱེད་ཅིག །དབུགས་ཀྱི་གླུད་འདི་རླུང་ལ་གཏོང༌། །རླུང་ལ་གནས་པའི་ལྷ་འདྲེ་རྣམས། །མཐུན་པའི་གླུད་ཆེན་འདི་བཞེས་ལ། །འཁོར་འདིར་གནོད་པ་མ་བྱེད་ཅིག །སྐྲ་དང་བ་སྦུའི་གླུད་འདི་ཤིང་ལ་གཏོང༌། །ཤིང་ལ་གནས་པའི་ལྷ་འདྲེ་རྣམས༑ ༑མཐུན་པའི་གླུད་ཆེན་འདི་བཞེས་ལ། །འཁོར་འདིར་གནོད་པ་མ་བྱེད་ཅིག །ནང་ཁྲོལ་གླུད་འདི་འཕྲང་ལ་གཏོང༌། །འཕྲང་ལ་གནས་པའི་ལྷ་འདྲེ་རྣམས། །མཐུན་པའི་གླུད་ཆེན་འདི་བཞེས་ལ། །འཁོར་འདིར་གནོད་པ་མ་བྱེད་ཅིག །སེམས་ཀྱི་གླུད་འདི་ནམ་མཁར་གཏོང༌། །ནམ་མཁར་གནས་པའི་ལྷ་འདྲེ་རྣམས། །མཐུན་པའི་གླུད་ཆེན་འདི་བཞེས་ལ། །འཁོར་འདིར་གནོད་པ་མ་བྱེད་ཅིག །གླུད་བདག་རྣམས་ཀྱིས་གླུད་བཞེས་ཤིག །ཡས་བདག་རྣམས་ཀྱིས་ཡས་ཁྱེར་ཅིག །གྲིབ་བདག་རྣམས་ཀྱིས་གྲིབ་ལོངས་ཤིག །མཐུན་པའི་གླུད་ཆེན་འདི་བཞེས་ལ། །བདག་ཅག་རྒྱུ་སྦྱོར་འཁོར་དང་བཅས། །བཟུང་བ་ཐོང་ལ་བཅིངས་པ་ཁྲོལ། །བསྡམས་པ་ལྷོད་ལ་མནན་པ་ཁྱོག །ཆག་ཆེ་ཉམ་ང་ལ་སོགས་པའི། །མི་མཐུན་ཕྱོགས་རྣམས་ཞི་བར་མཛོད། །ཅེས་གྲིབ་བདག་གི་མདོས་སྣོད་ཀྱི་ནང་དུ་གླུད་རྣམས་འབུལ་ལོ། །དེ་ནས་རྫོངས་བཏབ་པ་ནི། ལྷ་སྔགས་ཀྱིས་བསྒྲུབས་པའི་བུམ་ཆུས་བསྒྲུབ་བྱ་བཀྲུས་པའི་ཁུ་བ་དང༌། བསྒྲུབ་བྱའི་ཁ་བཤལ་བའི་ཆུ་དང༌༑ སྐྲ་དང་སེན་མོ་སོགས་དྲི་མ་བསྲེས་པ་དང༌། བྲ་སྲན་འབྲུ་ནག་ཚིག་ཕྱེ་དང༌།

གཏོར་ཆུང་ནག་པོ། ངར་མི། ཆང་བུ་རྣམས་སྒོང་སྐོགས་སུ་བླུགས་ཤིང༌། ཨོཾ་ཡ་མཱནྟ་ཀ་ཧཱུྂ་ཨོཾ་བྷུར་ཀུཾ་མ་ཧཱ་པྲཱ་ཎཱ་ཡ་བྷུར་ཙི་བྷུརྐི་བི་མ་ལེ་ཨུ་ཙུཥྨ་ཀྲོ་དྷ་ཧཱུྂ་ཕཊ། བཛྲ་བྷུར་ཀུཾ་རི་ལི་ཤུ་ཐ་ར་ཤ་ཧུར་ཤེ་བི་མ་ལེ་ཨུ་ཙུཥྨ་ཀྲོ་དྷ་ཧཱུྂ་ཧཱུྂ་ཕཊ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་སཏྭ་ཧཱུྂ། གྲིབ་གནོན་ནག་པོ་ཞི་ཞི་མལ་མལ་སྭཱ་ཧཱ། ཨོཾ་ཨཱཿཧཱུྂ་ཨེ་ཨེ་ཧོཿ ཨ་དུཾ་ཨཱ་སྭཱ་ཧཱ། ཨ་དུཾ་ཨཱ་སྭཱ་ཧཱ། ཨོཾ་ཨེ་ཧོ་ཤུདྡྷེ། བཾ་ཧོ་ཤུདྡྷེ། རཾ་ཧོ་ཤུདྡྷེ། ལཾ་ཧོ་ཤུདྡྷེ། ཡཾ་ཧོ་ཤུདྡྷེ། ཨེ་བཾ་རཾ་ལཾ་ཡཾ་ཤུདྡྷེ་ཤུདྡྷེ་ཨ་ཨཿསྭཱ་ཧཱ། ཞེས་བཏབ་ཅིང༌། ཀྱཻ། གྲིབ་བདག་ཆེན་པོ་འཁོར་བཅས་ལ། །སྔགས་འཆང་བདག་དང་ཡོན་མཆོད་ཀྱིས། །དེ་རིང་གཤེགས་སྐྱེམས་རྫོངས་བཏབ་བོ། །བྲ་སྲན་འབྲུ་ནག་ཚིག་ཕྱེ་དང༌། །གྲིབ་གཏོར་ནག་པོ་དམ་རྫས་དང༌། །ངར་མི་ལམ་ཆས་ཆང་བུ་དང༌། །གྲིབ་རྫས་རྣམ་པ་སྣ་ཚོགས་ཀྱིས། །གྲིབ་བདག་ཁྱོད་ལ་རྫོངས་བཏབ་བོ། །གཞན་ཡང་ཁྱོད་ལ་རྫོངས་བཏབ་པ། །ནད་དང་གདོན་གྱི་རྫོངས་བཏབ་བོ། །དལ་དང་རིམས་ཀྱི་རྫོངས་བཏབ་བོ། །ཐན་དང་ལྟས་ངན་རྫོངས་བཏབ་བོ། །གྱོད་དང་ཁ་སྨྲས་རྫོངས་བཏབ་བོ། །མི་ཁ་ཁ་མཆུའི་རྫོངས་བཏབ་བོ། །རྨི་ལྟས་མཚན་ངན་རྫོངས་བཏབ་བོ། །མོ་པྲ་རྩིས་ངན་རྫོངས་བཏབ་བོ། །ལ་ཡོགས་གཤེད་ཀྱི་རྫོངས་བཏབ་བོ། །ལན་ཆགས་བུ་ལོན་རྫོངས་བཏབ་བོ། །ཤ་འཁོན་སྐྱིན་གྱི་རྫོངས་བཏབ་བོ། །མི་མཐུན་བར་ཆད་རྫོངས་བཏབ་བོ། །སྐག་ངན་གཤེད་ཀྱི་རྫོངས་བཏབ་བོ། །བྱད་ཕུར་རྦོད་གཏོང་རྫོངས་བཏབ་བོ། །བསམ་ངན་སྦྱོར་རྩུབ་རྫོངས་བཏབ་བོ། །དུག་མཚོན་དམོད་པས་རྫོངས་བཏབ་བོ། །ལ་ཡོགས་ཁ་ཉེས་རྫོངས་བཏབ་བོ༑ ༑གྲིབ་དང་མི་གཙང་རྫོངས་བཏབ་བོ། །མི་མཐུན་ཕྱོགས་ཀུན་རྫོངས་བཏབ་བོ། །རྫོངས་སུ་བཏབ་བོ་ད་ཁྱེར་ཅིག །གཤེགས་སྐྱེམས་འབུལ་གྱི་ད་གཤེགས་ཤིག །རིངས་པར་རིངས་པར་ད་གཤེགས་ཤིག །མ་འགོར་མྱུར་དུ་ད་གཤེགས་ཤིག

།ཅེས་བརྗོད། ལམ་བསྟན་པ་ནི། མདོས་ཁ་ཕྱིར་བསྒྱུར་དཔལ་འབར་སྦར་ལ་བཙུགས། ཀྱཻ། ད་ནི་གཤེགས་པའི་ལམ་བསྟན་གྱི། །གྲིབ་ཁ་འཁོར་བཅས་ལམ་དེར་གཤེགས། །ཉི་མ་ནུབ་ཕྱོགས་ཕ་གི་ན། །མུན་པ་ནག་པོའི་ཀློང་དཀྱིལ་ན། །གྲིབ་བདག་ཆེན་པོའི་གནས་དེ་བདེ། །གྲིབ་མཁར་ནག་པོ་ལྡེམས་སེ་ལྡེམ། །གྲིབ་མཁར་ནག་པོ་རྫོང་རེ་བཙན། །གྲིབ་མཁར་ནག་པོའི་ནང་ཤེད་ན། །གྲིབ་ཀྱི་ན་བུན་ཐིབས་སེ་ཐིབ། །མུན་པ་ཐིབས་པ་བརྗིད་རེ་ཆེ། །བརྗིད་ལྡན་མུན་པའི་མཁར་ནང་ན། །གྲིབ་བདག་ཆེན་པོ་ད་གཤེགས་ཤིག །གདོང་ཅན་བཞི་པོ་ད་གཤེགས་ཤིག །ཕྱོགས་མཚམས་ཤིང་མེ་ས་ལྕགས་ཆུ། །འབྱུང་བ་རྣམ་ལྔའི་ཕོ་བྲང་ན། །གནས་པ་བཅུ་གཉིས་ད་གཤེགས་ཤིག །རང་རང་བཞུགས་གནས་ཉམས་དགའ་བར། །གདོན་བགེགས་ཐམས་ཅད་ད་གཤེགས་ཤིག །སྔགས་འཆང་བདག་གི་གནས་འདི་ནི། །ཁྱེད་ཅག་བཞུགས་པ་དང་མི་སྤྲོ། །འདི་ན་ཁྱོད་ཀྱི་གནས་མ་མཆིས། །མུན་པའི་ཀློང་རུམ་འདི་ན་མེད། །གྲིབ་མཁར་ནག་པོ་འདི་ན་མེད། །གྲིབ་ཀྱི་ན་བུན་འདི་ན་མེད། །ཁྱོད་ལ་དགྱེས་པའི་ཡོ་བྱད་མེད། །དེ་བས་གྲིབ་བདག་འཁོར་དང་བཅས༑ ༑རྒྱ་གྲམ་རླུང་གིས་བཏེག་པ་ལྟར། །གྲིབ་བདག་སོ་སོའི་གནས་སུ་སོངས། །ལས་ཀྱི་རླུང་གིས་དེད་པ་ལྟར། །གྲིབ་བདག་སོ་སོའི་གནས་སུ་སོངས། །འཇིགས་པའི་མུན་པས་བསུས་པ་ལྟར། །གྲིབ་བདག་སོ་སོའི་གནས་སུ་སོངས། །ནམ་མཁར་སྤྲིན་ཚོགས་དེངས་པ་ལྟར། །གྲིབ་བདག་རང་རང་གནས་སུ་གཤེགས། །བདེའོ་སྐྱིད་དོ་རང་གི་གནས། །གྲིབ་བདག་གྲིབ་དང་བཅས་པ་རྣམས། །ཕོ་ལ་བསྒྱུར་རོ་གཡས་སུ་བྷྱོཿ མོ་ལ་བསྒྱུར་རོ་གཡོན་དུ་བྷྱོཿ ཆུང་ལ་བསྒྱུར་རོ་ཐད་དུ་བྷྱོཿ ཁྱིམ་མཚེས་བསམ་པ་ངན་ལ་བྷྱོཿ གང་དང་འཕྲད་པའི་བགེགས་ལ་བྷྱོཿ བྷྱོཿཟློག་བསྒྱུར་རོ་ས་མ་ཡ་ཧོཿ །ཞེས་བརྗོད་ཅིང་མདོས་ཕྱི་མིག་མི་ལྟ་བར་ཁྱེར་ལ་ཕྱོགས་ངན་ནམ་བདུད་གཅོད་དམ། འདྲེ་ལྔའི་ཕྱོགས་སམ། ཆུ་ཆེན་ནམ། ལམ་ཆེན་པོར་ཁ་ཕར་ལ་བསྟན་ལ་བསྐྱལ།

མདོས་སྐྱེལ་མི་དང་རང་འཁོར་གནས་ཁང་བཅས་པ་བུམ་ཆུས་ཁྲུས་བྱའོ། །དེ་ལྟར་གྲིབ་འཇོམས་རྡོ་རྗེ་ཕ་ལམ་གྱི་ལས་མཐའ་གྲིབ་མདོས་བསྡུས་པ་འདིའང༌། སྨེ་བརྩེགས་ལྗང་སྔོན་གྱི་ལས་མཐར་ཇོ་བོ་རྗེས་ཀྱང་གྲིབ་མདོས་ཆུང་ངུ་མཛད་གནང་བ་ལྟར་འདིའང་སྨེ་བརྩེགས་ཀྱི་ལས་མཐར་མི་རུང་བ་མ་ཡིན་པར་མཐོང་ནས། དབུར་སྟོད་འབྲི་གུང་སྐྱུ་རའི་རིགས་སུ་སོན་པ་དུས་མཐའི་དགེ་སློང་སྔགས་འཆང་རྡོ་རྗེ་འཛིན་པ་ཐུགས་ཀྱི་རྡོ་རྗེས། མེ་ཕག་མཆུ་ཟླའི་དམར་པོའི་རྒྱལ་བ་དང་པོར། རྩེ་བ་གཞིས་ཀ་སྲ་བརྟན་རྡོ་རྗེའི་གནས་མཆོག་ཏུ་སྦྱར་བའོ།། །།སརྦ་མངྒ་ལཾ།།
