༄༅། །ཁྱབ་བདག་རྡོ་རྗེ་འཆང་ཆེན་པོ་ལ་བསྟོད་པ་གཞོམ་མེད་རྡོ་རྗེའི་གླུ་དབྱངས་ཞེས་བྱ་བ། བླ་མ་དང་མགོན་པོ་འཇམ་པའི་དབྱངས་ལ་ཕྱག་འཚལ་ལོ། །འགྱུར་བྲལ་མཆོག་གི་བདེ་བ་མ་ཉམས་རྡོ་རྗེའི་རང་བཞིན་ཧེ་རུ་ཀཿ །རྣམ་ཀུན་མཆོག་ལྡན་ཤེས་རབ་བཙུན་མོས་རྣམ་པ་ཀུན་ཏུ་ལེགས་འཁྱུད་ཅིང༌། །ཟུང་འཇུག་བདེ་ཆེན་རོ་ལ་རྟག་རོལ་དགྱེས་པ་མཆོག་གི་གར་གྱིས་རྩེན། །ཁྱབ་བདག་ཆོས་ཉིད་སྐུ་ཅན་དེ་ལ་རྣམ་པ་ཀུན་ཏུ་བདག་འདུད་དོ། །གདོད་མའི་མགོན་པོ་མིག་སྟོང་ལྡན་པའི་གར། །གདུལ་བྱ་སྣ་ཚོགས་བཻ་ཌཱུར་འཛིན་མ་ལ། །གྲངས་མེད་ཞིང་དང་སྤྲུལ་སྐུའི་གར་ཅི་ཡང༌། །སྟོན་མཛད་དཔའ་བོ་ཁྱོད་ལ་གུས་ཕྱག་འཚལ། །ལུས་ནི་དྭངས་ཤིང་རྡུལ་ཕྲན་དག །མཉམ་པ་ནམ་མཁའ་དང་མཚུངས་ལ། །མཚན་དཔེའི་དྲ་བ་རབ་རྫོགས་པ། །རྨི་ལམ་བཞིན་དུ་སྣང་དེར་འདུད། །སྐད་ནི་ཀུན་ནས་ཆད་པ་མེད༑ ༑དུ་མའི་སྐད་གཞན་དག་གིས་ཀྱང༌། །གཞན་གྱི་སྙིང་ལ་འཇུག་པ་སྟེ། །སྒྲ་བྲལ་སྒྲ་དབྱངས་ལྡན་དེར་འདུད། །སེམས་ནི་དམ་པའི་བདེས་གང་ཞིང༌། །མི་གཡོ་ཐམས་ཅད་དུས་ཀུན་ཏུ། །ལྷན་ཅིག་སྐྱེས་པས་རབ་འཁྱུད་པ། །བཅོ་བརྒྱད་མ་འདྲེས་མངའ་དེར་འདུད། །ཕྲ་རབ་རྡུལ་གྱི་ཆོས་ཉིད་ལས་འདས་ཤིང༌། །པྲ་ཕབ་པ་ལྟར་སྟོང་པ་ཉིད་ཀྱི་གཟུགས། །ཤེས་བྱ་ཀུན་གྱི་རྣམ་པར་རབ་སྣང་བ། །ལོངས་སྤྱོད་རྫོགས་པའི་སྐུ་ཅན་དེ་ལ་འདུད། །རྣམ་ཀུན་མཆོག་ལྡན་དམིགས་བཅས་སྟོང་ཉིད་དང༌། །དམིགས་མེད་སྙིང་རྗེ་མཆོག་ཏུ་མི་འགྱུར་བའི། །བདེ་ཆེན་དབྱེར་མེད་ཟུང་འཇུག་ལ་རོལ་བ། །ཁ་སྦྱོར་རྡོ་རྗེའི་སྐུ་མངའ་དེ་ལ་འདུད། །གང་ཡང་རྟོག་པ་མེད་པ་ཉིད། །ཚོར་བ་སྡུག་བསྔལ་གྱི་རྩ་བ། །གཟུང་འཛིན་རྣམ་རྟོག་གཏན་སྤངས་པ། །བདེ་ཆེན་རྟོག་མེད་དེ་ལ་འདུད། །གཤིས་ཀྱི་གནས་ལུགས་སྤྲོས་བྲལ་བ། །རྫས་སམ་དངོས་པོ་དེར་འཛིན་གྱིས། །བདེན་པར་གྲུབ་པ་མེད་པའི་ཕྱིར། །རང་བཞིན་སྐྱེ་མེད་དེ་ལ་འདུད། །དགའ་དང་མཆོག་དགའ་དགའ་བྲལ་ལྷན་ཅིག་སྐྱེས། །དེ་དང་དེར་འཛིན་གཅིག་དང་དུ་མ་སོགས། །མི་མངའ་ཟག་མེད་འཕོ་བྲལ་རྟག་བརྟན་པའི། །འགྱུར་མེད་སྙིང་རྗེ་ངེས་གང་དེ་ལ་འདུད། །སྔར་ཉིད་ཇི་བཞིན་ཕྱིས་དེ་བཞིན༑ ༑དངོས་མེད་རྟག་པ་མ་ཡིན་པར། །ཀུན་གྱི་རྟག་པའི་སྐུ་ཅན་ཉིད། །རྒྱུན་མི་འཆད་པ་དེ་ལ་འདུད། །ཅང་མེད་ཅི་ཡང་མེད་ལ་སོགས་པ་ཡི། །སྟོང་པ་ཆད་པའི་མཐའ་ལས་རིང་འདས་ཤིང༌། །རྣམ་པ་ཀུན་ཏུ་ཆད་པ་མི་མངའ་བའི། །འགག་པ་མེད་པའི་ལུས་ཅན་དེ་ལ་འདུད། །རྒྱལ་བ་བསྐྱེད་ཡུམ་སྲིད་པའི་ཉེས་པས་མ་གོས་སྒོམ་པར་འགྱུར་བྲལ་ཞིང༌། །ཅེས་པ་རྒྱལ་སྲས་ཏིལླི་པ་ལ་བསྟོད་པ་གསོལ་འདེབས་དང་བཅས་པ་གྲུབ་པ་མཆོག་གི་ཟློས་གར་ཞེས་བྱ་བ་འདིའང༌། ར་ཁྱིམ་ནས་དགེ་སློང་ངག་དབང་ཕུན་ཚོགས་ཀྱིས་བསྐུལ་བའི་ངོར། འབྲི་གུང་པ་ཤཱཀྱའི་དགེ་སློང་ཆོས་ཀྱི་གྲགས་པས། ཕོ་བྲང་རྩེ་བ་གཞིས་ཀར་སྦྱར་བ་སིདྡྷི་རསྟུ།། །།མངྒ་ལཾ།། ༈ །པཎ་ཆེན་ནཱ་རོ་པ་ལ་བསྟོད་པ་སྙན་གྲགས་ལྷ་ཡི་རྒྱུད་མང་ཞེས་བྱ་བ། བླ་མ་དང་མགོན་པོ་འཇམ་དཔལ་གཞོན་ནུར་གྱུར་པ་ལ་ཕྱག་འཚལ་ལོ། །མཁས་དང་གྲུབ་པའི་གཙུག་རྒྱན་མེ་མུར་བའི། །སྙན་གྲགས་ལྷ་དབང་གཞོན་ནུའི་དུནྡ་བྷིཿ །ཐོས་པས་དད་པའི་རྒྱན་ལྡན་མཻ་ན་ཀཿ །རབ་དྭངས་དྲུག་སྐྱེས་རྒྱུད་མང་འདི་མཆོད་སྤྲིན། །ཀསྨཱི་ར་ཡུལ་གཉིས་སྐྱེས་རིགས་གཙང་མར། །ཞི་བའི་གོ་ཆ་དཔལ་གྱི་བློ་གྲོས་མའི། །སྲིད་པ་བཟུང་ཚེ་དགེ་མཚན་ལྟས་དང་བཅས། །ཀུན་ཏུ་བཟང་པོར་བཏགས་དེར་མགོས་ཕྱག་འཚལ། །དེར་ཡང་ཁྱོད་ཀྱིས་སྒྲ་ཚད་སོགས། །རྣམ་དག་རིག་ལམ་བགྲངས་ཡས་ཀྱི། །མཁས་པའི་གྲགས་པས་ཕྱོགས་ཀྱི་ནི། །འཁོར་ལོ་འགེངས་མཛད་དེ་ལ་འདུད། །བུདྡྷ་པཱ་ལས་རབ་བྱུང་ནས། །སྡེ་སྣོད་གསུམ་གྱི་ཕ་མཐར་སོན། །གཙུག་ལག་ཆེན་པོའི་སྒོ་སྲུང་མཛད། །སྙན་གྲགས་བཟང་པོར་གཙུག་གིས་འདུད། །རིག་ལམ་སྟོང་ཕྲག་དལ་འགྲོའི་རྒྱུན་ལྡན་སྡེ་སྣོད་དབུ་མའི་ལང་ཚོ་ངོམས། །ལེགས་བཤད་ཟེགས་མའི་ཕ་འབབ་བཟང་པོས་འདུས་ཚོགས་འབུམ་ཕྲག་ཉེར་འཛིན་ཀྱང༌། །རང་སེམས་བ་ཡི་ནུ་འཐེན་པ་ཡི་ཡུན་ཙམ་ཞིག་ཀྱང་གནས་མིན་པ། །ཀྱེ་མ་ཐོས་བསམ་སྒོམ་པ་ཁོ་ནས་ངེས་དོན་རྟོགས་མིན་ཞེས་ཁྱོད་སེམས༑ ༑ད་ནི་སེམས་ལ་དབང་འབྱོར་བའི། །དམ་པ་སུ་ཞེས་དཔལ་ཁྲག་འཐུང༌། །ཉིད་ལ་དྲིས་ཚེ་ཤར་ཕྱོགས་སུ། །ཏིལླིའི་རིགས་ཞེས་དེས་ལུང་བསྟན། །དེ་ནས་དེ་ཉིད་ཡིད་ལ་རབ་སེམས་པས༑ ༑ཕྱོགས་རྣམས་ཀུན་ཏུ་རྒྱུ་བའི་བརྟུལ་ཞུགས་ཀྱིས། །བཙལ་ཚེ་དེ་ཡི་ཆོ་འཕྲུལ་ཡ་མ་ཟུང༌། །མཐོང་ཡང་སོམ་ཉིའི་ཟོལ་གྱིས་དཀའ་བ་སྤྱད། །དམ་པ་མཆོག་དེའི་གསུང་གི་བཻ་ཌཱུར་ཆེ། །འཁོར་འདས་ཀུན་གྱི་འབྱོར་བའི་ཕུལ་ལ་དབང་བསྒྱུར་མ་ཆགས་གང་དེ་ཡི། །འདི་ཡི་སྣ་དང་རྣ་བའི་ཚོམ་བུ་ཁྲག་གིས་སྦགས་པའི་མཎྜལ་ལ། །དགྱེས་པ་ལྟར་མཛད་ངེས་དོན་ནོར་བུ་དེ་ལ་བསྩལ་ཞེས་འདི་ཡ་མཚན། །དེ་ནས་དེ་ལ་ཐོབ་པའི་ནོར། དབུར་སྟོད་འབྲི་གུང་དུ་སོན་པ་དགེ་སློང་རྡོ་རྗེ་འཛིན་པ་ཆོས་ཀྱི་གྲགས་པས། རྩེ་བ་སྲ་བརྟན་རྡོ་རྗེའི་གནས་མཆོག་ཏུ་སྦྱར་བ་དགེ་བར་གྱུར་ཅིག། །།སརྦ་མངྒ་ལཾ།། ༈ །མངའ་བདག་མར་པ་ལོ་ཙཱ་ལ་བསྟོད་པ་ནགས་ཀྱི་གར་མཁན་ཞེས་བྱ་བ། བླ་མ་དང་མགོན་པོ་འཇམ་པའི་དཔལ་ལ་ཕྱག་འཚལ་ལོ། །དཔལ་ལྡན་རྡོ་རྗེ་འཛིན་པ་འཁོར་ལོས་སྒྱུར། །ཡོན་ཏན་དབྱར་རྔ་ཐོས་པས་གཙུག་ཕུད་ཅན། །དད་པའི་འདབ་མ་གྱེན་དུ་བསྐྱོད་པ་ཡི། །བསྟོད་དབྱངས་སྙན་པའི་རོལ་པ་ཅི་ཡང་འཛིན། །ཧེ་རུ་ཀ་དཔལ་དགྱེས་མཛད་རྡོ་རྗེ་ནི། །སྐལ་བཟང་ཡིད་མཁར་སྣ་ཚོགས་གར་མཛད་པ། །ཞིང་འདིར་མར་པ་ཞེས་བྱ་ས་གསུམ་དུ། །སྙན་པའི་བ་དན་འཛིན་དེར་གུས་པས་འདུད། །མི་འགྱུར་བདེ་བ་ཆེན་པོའི་ཀུན་རང་བཞིན། །རྣམ་ཀུན་སྟོང་ཉིད་མཆོག་གི་ཡེ་ཤེས་དང༌། །གཉིས་སུ་མེད་པ་ཁ་སྦྱོར་རྡོ་རྗེ་སེམས། །ཁྱབ་བདག་ཧེ་རུ་ཀ་དཔལ་དེ་ལ་འདུད། །སྤྲོས་པའི་མཚན་མ་འོད་གསལ་མེ་ཆེན་གྱིས། །བསྲེགས་ནས་དྷཱུ་ཏིའི་ཁྱིམ་དུ་གཉིད་ལོག་པ། །སྔོ་བསངས་དགའ་མའི་གླུ་ཡིས་སད་པའི་ཚེ། །ཟུང་འཇུག་སྒྱུ་མར་ལྡང་བ་དེ་ལ་འདུད། །བཅུ་དྲུག་ལང་ཚོའི་ཆ་ཤས་མ་ཉམས་པའི། །དཔལ་ལྡན་རྡོ་རྗེ་བཙུན་མོའི་་་་མཁའ་དབྱིངས་སུ། །བདེ་སྟོང་གཉིས་སུ་མེད་པའི་རོ་མྱང་བ། །ལྷན་སྐྱེས་རྟག་པའི་འཁོར་ལོ་དེ་ལ་འདུད། །ཐབས་ཤེས་ཨ་ཧཾ་དྭངས་མའི་རྒྱུན་བཟང་གིས། །འབྱུང་ལྡོག་དགའ་བ་བཅུ་དྲུག་མཐར་སོན་ཏེ། །ཚེ་གཅིག་རྡོ་རྗེ་འཛིན་པའི་ས་བསྒྲུབས་པ། །ལོངས་སྤྱོད་རྫོགས་པའི་སྐུ་མཆོག་དེ་ལ་འདུད། ༑བདེ་ཆེན་ཧེ་རུ་ཀ་ལ་གུས་ཕྱག་འཚལ། །མཁའ་ལྟར་དངོས་མེད་ཨེ་ཡིག་སེངྒེའི་ཁྲིར། །དངོས་པོ་མཆོག་ལྡན་ཝཾ་ཡིག་རྡོ་རྗེའི་སྐུ། །མི་ཕྱེད་གཉུག་མའི་ངང་དུ་རོ་མཉམ་པ། །གདོད་ནས་རབ་ཞི་་ཁྱོད་ལ་གུས་ཕྱག་འཚལ། །དག་དང་མ་དག་གཉིས་སྣང་སྒྱུ་མ་ཡི། །ཀུན་རྟོག་འོད་གསལ་མེ་ལྕེས་རབ་སྦྱངས་ནས། །དེ་ལས་སླར་ཡང་ཟུང་འཇུག་སྒྱུ་མའི་སྐུར། །སྐད་ཅིག་ལྡང་བ་ཁྱོད་ལ་གུས་ཕྱག་འཚལ། །གང་གི་མཁྱེན་པའི་འོད་སྣང་ཆེར་འབར་བས། །ལྟ་ངན་འབྱུང་པོའི་འདབ་ཆགས་མིག་ལྡོང་ཞིང༌། །བསྐལ་བཟང་བློ་གྲོས་འདབ་སྟོང་རབ་གཡོ་བ། །ངེས་དོན་བསྟན་པའི་ཉི་མ་ཁྱོད་ཕྱག་འཚལ། །གུས་པའི་ཡིད་ནི་ཀེཾ་ཤུ་ཀ་ལྟར་གཡོས། །ལུས་ཀྱི་སྤུ་ལྡང་མཆི་མའི་ཆར་རྒྱུན་བབ། །གདུང་བའི་དབྱངས་ཀྱིས་རྩེ་གཅིག་གསོལ་འདེབས་ན༑ ༑བརྩེ་བས་གཟིགས་ཤིག་ཐུགས་རྗེའི་གཏེར་ཆེན་པོ། །བསྐལ་མང་རྙེད་དཀའི་གྲུ་ཆེན་འདི་ཐོབ་ནས། །དོན་མེད་ཚེ་འདིའི་བྱ་བ་འབའ་ཞིག་གིས། །ཡུན་གྱི་དོན་ཆེན་ཡལ་བར་འདོར་བ་ཡི༑ ༑སྙིང་མེད་པོ་ལ་ཐུགས་རྗེའི་སྤྱན་གྱིས་གཟིགས། །འདུས་བྱས་མི་རྟག་རླུང་ནང་མར་མེ་འདྲ། །རྟག་པར་འཛིན་པའི་བློ་ཡི་ཞེ་གདེང་གིས། །གཉིད་དང་སྙོམས་ལས་ལེ་ལོའི་དབང་གཏོང་བ། །སྙིང་མེད་པོ་ལ་ཐུགས་རྗེའི་སྤྱན་གྱིས་གཟིགས། །མི་གཙང་མེ་དོང་ལྟ་བུའི་འཁོར་བ་ལ། །བདེ་བར་འཛིན་པའི་བློ་གྲོས་ངན་པ་འདིས། །དེ་ལས་ཐར་པའི་གོ་སྐབས་མི་འབྱེད་པར། །སྙིང་མེད་པོ་ལ་ཐུགས་རྗེའི་སྤྱན་གྱིས་གཟིགས། །ལས་འབྲས་བསླུ་བ་མེད་པ་ཤེས་བཞིན་དུ། །དགེ་སྡིག་བླང་དོར་ཚུལ་བཞིན་མི་བྱེད་པའི། །སྡུག་བསྔལ་སྤོང་འདོད་དེ་རྒྱུ་སྒྲུབ་པ་ཡི། །སྙིང་མེད་པོ་ལ་ཐུགས་རྗེའི་སྤྱན་གྱིས་གཟིགས། །མཐའ་ཡས་སེམས་ཅན་ཁམས་རྣམས་བསྒྲལ་བའི་ཕྱིར། །མཆོག་གསུམ་སྐྱབས་ཀྱི་མཆོག་ཏུ་ལེགས་བཟུང་ཞིང༌། །ཚུལ་ཁྲིམས་མིག་འབྲས་བཞིན་དུ་སྲུང་བ་དང༌༑ ༑བྱམས་དང་སྙིང་རྗེ་བྱང་ཆུབ་སེམས་གཉིས་འབྱོང༌། །སྨིན་ལམ་དབང་གི་ཆུ་བོས་རྒྱུད་སྦྱངས་ནས། །རིམ་པ་གཉིས་ཀྱི་ལམ་མཆོག་མཐར་ཕྱིན་པས། །ཡན་ལག་བདུན་ལྡན་རྡོ་རྗེ་འཆང་ཆེན་པོའི། །གོ་འཕང་ཚེ་འདིར་ཐོབ་པར་བྱིན་གྱིས་རློབས། །ཞེས་དབང་ཕྱུག་བཞད་པ་རྡོ་རྗེ་ལ་བསྟོད་པ་གསོལ་འདེབས་དང་བཅས་པ་ཡིད་ཀྱི་ཤིང་རྟ་ཞེས་བྱ་བ་འདིའང༌།

སྡོམ་བརྩོན་དམ་པ་དགེ་སློང་ཀ་བཅུ་པ་ངག་དབང་ཕུན་ཚོགས་ཀྱིས་བསྐུལ་ངོར༑ འབྲི་གུང་པ་རྡོ་རྗེ་འཛིན་པ་ཆོས་ཀྱི་གྲགས་པས། འབྲི་ཆོས་ཕྱུག་གི་ཕོ་བྲང་དུ་མགྱོགས་པར་བྲིས་པ་ཤུ་བྷཾ།། །།སརྦ་མངྒ་ལཾ།། ༈ །མཉམ་མེད་སྒམ་པོ་ལ་བསྟོད་པ་དད་པའི་རོལ་མཚོ་ཞེས་བྱ་བ། བླ་མ་དང་མགོན་པོ་འཇམ་པའི་རྡོ་རྗེ་ལ་ཕྱག་འཚལ་ལོ། །རྒྱལ་བའི་འཁོར་ལོ་བར་ཐའི་དགོངས་དོན་རྣམས། །གངས་ཅན་ཁྲོད་འདིར་གྱི་ནར་སྟོན་པ་ཀུན། །དྲང་པོར་གཟུད་མཛད་འཚོ་མཛད་དེ་དེ་ཡི། །ཡོན་ཏན་དྲན་པས་བསྟོད་དབྱངས་མགྲིན་པར་འཛིན། །གངས་རིའི་དབང་པོ་དང་ལྡན་པའི། །ས་འདིར་སྟོང་ཉིད་སྨྲ་བོ་ལོ། །ལོག་སྨྲའི་ཝ་ཚོགས་ཇི་སྙེད་པ། །རླུང་ལྟར་འཕྱོ་བ་སྙིགས་མའི་དུས། །དཔལ་ལྡན་ཞི་བའི་སྤོ་ལ་རབ་གནས་ཤིང༌། །ངེས་དོན་གདམས་པའི་རལ་པ་འཐུག་པོ་འཛིན། །སྟོང་ཉིད་ཇི་བཞིན་སྨྲ་བའི་སྦར་ཁྱིམ་གྱིས། །ལྟ་ངན་ཝ་ཚོགས་ཀླད་པ་འགེམས་དེར་འདུད། །ཁྱོད་སྔོན་མེ་ཏོག་ཟླ་མཛེས་ཚེ། །དཔའ་བས་བྱིན་གྱིས་རྣམ་འཚེ་ཡང༌། །བྱང་ཆུབ་སེམས་ལ་གོམས་པའི་མཐུས། །ཁྲོ་བ་མི་མངའ་དེ་ལ་འདུད། །བསྐལ་བཟང་འདི་ལ་འདྲེན་མཆོག་ཤཱཀྱ་ཡི། །སྤྱན་སྔར་ཟླ་འོད་གཞོན་ནུར་ཞེས་གྲགས་པ། །མདོ་སྡེ་ཏིང་འཛིན་རྒྱལ་པོ་ལ་སོགས་པས། །ཡོངས་སུ་ཚིམས་པར་མཛད་དེར་མགོས་ཕྱག་འཚལ། །སྲིད་འདིར་བལྟམས་ནས་གསོ་རིག་ལ། །ལེགས་སྦྱངས་ཁྱིམ་ནི་དུག་སྦྲུལ་གྱིས། །བཅིངས་པ་ལྟར་གཟིགས་ངེས་པར་འབྱུང༌། །དྲང་སྲོང་ཆེན་པོ་དེ་ལ་འདུད། །མར་མེ་མཛད་དཔལ་བྱང་ཆུབ་སེམས་ཀྱི་རྒྱུན། །བྱ་དང་ལྕགས་རི་གཉུག་རུམ་ལ་སོགས་ལ། །གང་བབ་ཁྱེད་ཐུགས་གསེར་གྱི་ཀུམྦྷ་རུ། །ལེགས་པར་གདམས་པར་དེ་ལ་གུས་ཕྱག་འཚལ། །གང་གིས་སྒྲུབ་ལ་གཞོལ་བའི་ཚེ། །གསེར་འོད་དམ་པ་ནས་གསུངས་ལྟར། །ཇི་བཞིན་མཚན་མ་རབ་རྫོགས་པ། །ས་བཅུ་མངོན་མཛད་ཁྱོད་ཕྱག་འཚལ། །རིགས་དང་དཀྱིལ་འཁོར་མ་ལུས་དབང་བསྒྱུར་ཁྱབ་བདག་བཞད་པ་རྡོ་རྗེ་ཡི། །མཚན་ཐོས་ཙམ་གྱིས་མོས་པར་གྱུར་གང་ཀེཾ་ཤུའི་འདབ་མ་རླུང་བསྐྱོད་བཞིན། །དེ་ཡི་བྱིན་རླབས་སྐད་ཅིག་ལ་ཞུགས་འཁོར་འདས་ཆོས་ཀུན་སྟོང་ཉིད་རྟོགས། །སེམས་ཅན་ཀུན་རྒྱུད་ཤེས་པའི་ཉམས་སོགས་ཡོན་ཏན་མངའ་བདག་དེ་ལ་འདུད། །དེ་ནུབ་དེ་ན་གསོལ་བཏབ་པས། །ཡ་མཚན་མཚན་མ་ཀུན་མཐོང་ནས། །ཆས་པས་དམ་པ་དེ་ཞབས་འདུད། །ལས་ཅན་ཁྱོད་ལ་བདག་འདུད་དོ། །གྲུབ་ཆེན་རྒྱལ་པོ་དེ་ཞབས་མྱུར་ཕྱིན་ཚེ། །སྣོད་བཅུད་རྟེན་འབྲེལ་ར་ས་ཡཱ་ན་ཡི། རྩེ་བ་གཞིས་ཀར་སྦྱར་བ་སྭ་སྟི་པྲ་ཛེ་བྷྱཿ སརྦ་མངྒ་ལཾ།། །། ༈ །མགོན་པོ་ཕག་མོ་གྲུ་པ་ལ་བསྟོད་པ་རབ་དཀར་ལྷག་བསམ་གྱི་བ་དན་ཞེས་བྱ་བ། བླ་མ་དང་མགོན་པོ་ངག་གི་བདག་པོ་ལ་ཕྱག་འཚལ་ལོ། །མཁས་དང་གྲུབ་པའི་བདག་པོ་ས་གསུམ་ན། །ངུར་སྨྲིག་གར་གྱིས་རོལ་པ་ས་གསུམ་གྱི། །སྐྱེ་དགུའི་མགོན་པོ་དེ་ལ་མགོས་ཕྱག་འཚལ། །འཁོར་ལོ་བར་པའི་དགོངས་དོན་ཕ་རོལ་ཕྱིན། །ཟབ་མོ་སྟོང་ཉིད་རབ་སྟོན་ཚད་མའི་མདོ། །ཐེག་ཆེན་དབུ་མ་ལ་སོགས་ལེགས་སྦྱངས་པས། །ཐོས་པ་རྒྱ་མཚོའི་དེ་ཉིད་སྒོམ་པའི་ཚེ། །ལམ་རྟགས་དྲོད་ཚད་འགའ་ཞིག་མངོན་དུ་མཛད། །བརྩོན་བྱས་འཁོར་ལོ་གཉིས་ཀྱིས་དུས་འདའ་བ། །མཉམ་མེད་འགྲོ་མགོན་དེ་ལ་བདག་ཕྱག་འཚལ། །ཨཱ་ནནྡ་ལ་གདམས་པའི་བོགས་དབྱུང་ཚེ། །མཐོང་ལམ་ཡིན་ཞེས་མུ་ཅོར་ལ་བབ་པས། །མཉམ་མེད་འཚོ་བྱེད་ཞབས་རྡུལ་སྤྱིར་བླངས་ནས། །འཁྲུལ་མེད་ལམ་བཟང་བརྙེས་དེར་ཕྱག་འཚལ་ལོ། །རྒྱལ་བ་མཆོག་དེའི་བྱིན་རླབས་དོན་དམ་པའི། །འཕོ་བས་འཁོར་འདས་ཆོས་རྣམས་སྐད་ཅིག་རྟོགས། །བླ་མའི་ཐུགས་དང་རང་སེམས་གཅིག་ཏུ་འདྲེས། །མངོན་པར་བྱང་ཆུབ་ཁྱོད་ལ་བདག་ཕྱག་འཚལ། །ཞིང་ཁམས་ཀུན་གཙོ་ཀུན་བཟང་ནགས་ཁྲོད་དུ། །བསྟན་པའི་རྒྱལ་མཚན་སྒྲེང་བར་མཛད་པའི་ཚེ། །རྒྱལ་དང་དེ་སྲས་པཎ་གྲུབ་དུ་མ་ཡིས། །མངོན་སུམ་རྗེས་བཟུང་དེ་ལ་ཕྱག་འཚལ་ལོ། །གསང་སྔགས་རྡོ་རྗེའི་ཐེག་ལ་དབང་བསྒྱུར་བས། །རྩ་རླུང་ཐིག་ལེ་དྭངས་པས་ཌཱ་ཀི་ནཱི། །སྣ་ཚོགས་དབང་བསྡུས་འཕོ་མེད་བདེ་སྟོང་འབར། །འཁོར་ལོའི་དབང་ཕྱུག་ཁྱོད་ལ་བདག་འདུད་དོ། །ཐོག་མེད་དུས་ནས་མངོན་པར་སངས་རྒྱས་ཀྱང༌། །སྲིད་འདིར་སྣ་ཚོགས་རྣམ་འཕྲུལ་སྒྱུ་མའི་གར། །ཅིར་ཡང་བསྟན་པས་འགྲོ་ལ་དགེ་བའི་ལམ། །གསལ་མཛད་འགྲོ་མགོན་དེ་ལ་བདག་ཕྱག་འཚལ༑ ༑ཤར་ཕྱོགས་རྡུལ་དང་བྲལ་བའི་ཞིང་ཁམས་སུ། །རྒྱལ་བ་མི་བསྐྱོད་རྒྱལ་པོའི་སྲས་ཐུ་བོ། །བློ་གྲོས་དྲི་མ་མེད་པར་འགྱུར་ཞེས་གསུངས། །དུས་གསུམ་རྒྱལ་བ་ཁྱོད་ལ་ཕྱག་འཚལ་ལོ༑ ༑ཀུན་བཟང་ནགས་ཀྱིས་བཀྲ་བའི་ཞིང་ཁམས་དེར། །དུས་གསུམ་འགྲོ་བའི་མགོན་པོ་རྡོ་རྗེ་རྒྱལ། །རིས་མེད་འགྲོ་ལ་དམ་པའི་ལམ་སྟོན་པ། །མཉམ་མེད་བླ་མའི་ཞབས་ལ་གསོལ་བ་འདེབས༑ ༑བདག་སོགས་མཐའ་ཡས་སེམས་ཅན་མ་ལུས་པ། །ངན་སོང་གསུམ་ལས་རྣམ་པར་གྲོལ་བ་དང༌། །དལ་འབྱོར་མཆོག་ཐོབ་རྒྱལ་བའི་བསྟན་ལ་འཇུག

།དམ་པས་རྗེས་བཟུང་རྒྱུ་ཡི་ཐེག་པ་ཡིས། །བློ་སྦྱངས་སྨིན་བྱས་སྔགས་ཀྱི་སྒོར་འཇུག་ཅིང༌། །རིམ་གཉིས་ཟབ་མོའི་ལམ་ལ་རྟག་ཞུགས་ནས། །ཁྱབ་བདག་ཟུང་འཇུག་རྡོ་རྗེ་འཆང་དབང་པོའི། །གོ་འཕང་བདེ་བླག་འགྲུབ་པའི་རྒྱུར་གྱུར་ཅིག །ཅེས་མགོན་པོ་ཕག་མོ་གྲུ་པ་ལ་བསྟོད་པ་གསོལ་འདེབས་དང་བཅས་པ་རབ་དཀར་ལྷག་བསམ་གྱི་བ་དན་ཞེས་བྱ་བ་འདིའང༌། དགེ་སློང་ངག་དབང་ཕུན་ཚོགས་ཀྱིས་བསྐུལ་ངོར། འབྲི་གུང་པ་ཤཱཀྱའི་དགེ་སློང་ཆོས་ཀྱི་གྲགས་པས། ཕོ་བྲང་རྩེ་བ་གཞིས་ཀར་མགྱོགས་པར་བྲིས་པ་མངྒ་ལཾ།། །། ༈ །འཇིག་རྟེན་གསུམ་གྱི་མགོན་པོ་འབྲི་གུང་པ་ཆེན་པོ་ལ་ཁྱད་པར་དུ་འཕགས་པའི་རྣམ་པར་ཐར་པ་བརྒྱད་ཀྱི་སྒོ་ནས་བསྟོད་པ་ཚངས་པའི་དབྱངས་ཀྱི་སྒྲ་བརྒྱ་པ་ཞེས་བྱ་བ། བླ་མ་དང་མགོན་པོ་འཇམ་པའི་དབྱངས་ལ་ཕྱག་འཚལ་ལོ། །འཇིག་རྟེན་གསུམ་ན་སྐྱེ་དགུ་མཐའ་དག་གི། །མགོན་པོ་གཅིག་སྤྲུལ་རྒྱལ་བ་ཉིད་སྤྲུལ་པ། །འབྲི་གུང་པར་གྲགས་ཆེན་པོ་ཀུན་གཙོ་དེའི། །བསྟོད་དབྱངས་གསེར་མངལ་ལྡན་བའི་སྒྲ་འདི་ཀོ། །དམ་པ་མཆོག་དེའི་རྣམ་ཐར་མི་ཟད་སྒོ། །རྒྱལ་དང་དེ་སྲས་ཕྱོགས་བཅུར་གང་བཞུགས་པ། །ཀུན་གྱི་ལྕེ་བརྒྱ་བསྐྱོད་པས་བསྙད་ན་ཡང༌། །རྫོགས་པར་མི་ནུས་འོན་ཀྱང་དད་པའི་ཡིད། །རྣམ་པར་གཡོས་པས་ཆ་ཤས་རེ་རེ་ཡང༌། །བརྗོད་པར་རབ་འབྱམས་ཚོགས་ཀྱི་ཕུལ་ཆེན་པོར། །འགྱུར་ཞེས་རྣམ་འདྲེན་མཆོག་གིས་ཡང་དང་ཡང༌། །གང་གསུངས་ཡིད་ལ་བརྣགས་པ་དོན་ལྡན་ཉིད། །དེ་ཕྱིར་སྙན་ངག་མཁན་དབང་རྒན་པོ་ནི། །དབྱངས་ཅན་མགྲིན་པར་ཆགས་པས་ལེགས་བཤད་གཏམ། །མོས་པའི་སོར་པད་སྙིང་ཁར་ཁ་ཟུམ་པ། །རབ་བཀོད་བསྟོད་དབྱངས་སྒྲ་བརྒྱ་འདི་མཆོད་སྤྲིན། །ཁྱོད་སྔོན་རྒྱལ་བ་ཉིད་ལས་སླར་དེའི་སྤྲུལ། །ལི་ཙ་བཱི་དང་འཕགས་མཆོག་ཀླུ་སྒྲུབ་སོགས། །མཁས་དང་གྲུབ་པའི་རྣམ་རོལ་བགྲངས་ཡས་ཀྱིས། །འགྲོ་ཀུན་འདྲེན་པར་མཛད་ལ་བདག་ཕྱག་འཚལ། །སྲིད་འདིར་རྩར་སྟོད་ཙོང་ངུ་ཞེས་བྱ་བར། །ཡབ་ཡུམ་བཛྲ་བྷ་དནྟིའི་ཁབ་ཏུ། །པུ་རང་ལོ་ལ་སྤྲུལ་པའི་སྣང་བ་བསྟན། །བསམ་བཞིན་སྒྱུ་འཕྲུལ་དྲ་བ་སྟོན་ལ་འདུད། །ཉེར་ལྔའི་ལོ་ལ་དམ་པ་ཁྱོད། །ཀུན་བཟང་ནགས་ཀྱི་དྲ་བ་རུ། །མགོན་པོ་རྡོ་རྗེ་རྒྱལ་ཞབས་པད། །ཡན་ལག་མཆོག་རྩེར་བསྙེན་ལ་འདུད། །སོ་ལྔ་བཞེས་ཚེ་ཆོས་འདུལ་གསོལ་བཞི་ཡི། །ལས་ཀྱིས་བསྙེན་རྫོགས་སུམ་ཅུ་རྩ་དགུ་ནས། །དོན་ལྔའི་བར་དུ་བྱང་ཆུབ་གླིང་བཞུགས་ཏེ། །བསྟན་པའི་སྒྲོན་མེ་སྦར་དེར་བདག་ཕྱག་འཚལ། །ཐོག་མེད་དུས་ནས་སྐྱོན་ཀུན་དག་ཅིང་ཡོན་ཏན་མ་ལུས་རྫོགས་པའི་སངས་རྒྱས་རྣམ་པ་ཐམས་ཅད་མཁྱེན་བརྙེས་ཀྱང༌། །གཤིས་བབས་དགེ་སྡིག་རྒྱུ་འབྲས་རྟེན་འབྲེལ་སྐད་ཅིག་མ་ཡི་འཁོར་ལོར་རྒྱུ་ལམ་འབྲས་བུ་ཐམས་ཅད་དུ་འབྱུང་བས༑ ༑རྟོག་པ་ཕྲ་བའང་ཕྱག་རྒྱ་བཞིས་སྤོང་དགེ་བ་དག་ཀྱང་དེ་སྒྲུབ་མི་ཟད་རྒྱན་གྱི་འཁོར་ལོར་ནི། །མཐའ་ཡས་མཛད་ཅིང་དེ་ཉིད་གཞན་ལའང་སྟོན་པར་མཛད་འདི་ཁྱད་པར་འཕགས་པའི་རྣམ་པར་ཐར་པའི་དང་པོའོ། །ཚུལ་ཁྲིམས་ཀྱིས་ནི་ཐོག་མཐའ་བར། །ཀུན་གྱི་ལམ་བགྲོད་ཡིན་མཁྱེན་ནས། །ཤ་ཆང་དྲི་ཡང་ཁྱོད་ཀྱིས་སྤངས། །མྱ་ངན་འདའ་ཁར་སྔ་དྲོ་ཡང༌། །ཁྲིམས་དཔང་མེད་པར་དར་བའང་སྤངས། །མགུལ་གློར་ཕན་ཞེས་གློ་བའི་སྨན། །སྨན་པས་དྲངས་ཀྱང་མ་གནང་བར། །རྣམ་གསུམ་དག་པའི་ཤ་ཉིད་ཀྱང༌། །དེ་འདྲ་ན་ནི་ཉེས་པའི་ཚོགས། །སུམ་ཅུ་རྩ་ལྔ་ལྡན་པའི་ཆང༌། །ལྟ་ཅི་སྨོས་ཞེས་རྩད་ནས་སྤངས། །མདོར་ན་ཆབ་ཚ་རེ་ཙམ་ཡང༌། །མཐར་ཐུག་ངེས་པའི་ཚོགས་སུ་རོལ། །གཞན་ལའང་སྟོན་འདི་གཉིས་པ་ཡིན། །ཐུབ་པས་བསྔགས་པའི་ཆོས་གོས་གསུམ་དང་ལྷུང་བཟེད་ཙམ་ལས་རྫས་དང་ཡོ་བྱད་གཞན་ཇི་སྙེད། །འཛིན་ཞེན་ཆགས་དང་བདག་ཏུ་བཟུང་བ་སྐད་ཅིག་མི་མངའ་མདོར་ན་ཕྱག་དར་ཁྲོད་ཀྱི་རྡུལ།
