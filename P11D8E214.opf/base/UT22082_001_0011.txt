༄༅། །སྐྱེས་རབས་གསོལ་འདེབས་མོས་གུས་ཀྱི་བ་དན་གཡོ་བ་ཞེས་བྱ་བ་བཞུགས་སོ།།

འཕགས་ཡུལ་ལྗོངས་སུ་རྒྱལ་བ་ཤཱཀྱ་ཐུབ། །སླར་ཡང་ནཱ་གརྫུ་ནའི་ཚུལ་དག་ལས། །ཞིང་འདིར་འདྲེན་པའི་མཆོག་གྱུར་རཏྣ་ཤྲཱི༑ ༑མཉམ་མེད་བླ་མ་རྣམས་ལ་གསོལ་བ་འདེབས། །རྒྱལ་བའི་སྤྱན་སྔར་བྲམ་ཟེ་སྒྲ་གཅན་འཛིན། །པདྨས་རྗེས་བཟུང་འཇིགས་མེད་སེངྒེ་ཞེས། །རིག་གྲོལ་གཉིས་ལྡན་ཁྱེའུ་དགའ་བྱེད་ཞབས། །བསྟན་པའི་གསལ་བྱེད་མཆོག་དེར་གསོལ་བ་འདེབས། །འཆི་མེད་གྲུབ་བརྙེས་དྲང་སྲོང་ཉི་མ་མགོན། །ཟངས་གླིང་སྲིད་སྐྱོང་རྒྱལ་པོ་འོད་མཛེས་ཞབས། །ལྔ་རིག་པཎ་ཆེན་ལྷ་ཡི་ཟླ་བ་ཞེས། །གྲུབ་བརྙེས་མཁས་པ་གང་དེར་གསོལ་བ་འདེབས། །བརྟུལ་ཞུགས་མཆོག་བརྙེས་རཀྵ་ཙཎྚ་དང༌། །ཙ་ཏུརྨུ་ཁའི་གྲུབ་མཆོག་རྒྱལ་པོའི་ཞབས། །ངེས་པའི་དོན་རྟོགས་མེ་ལོང་རྡོ་རྗེ་ཞེས། །གྲུབ་པའི་དབང་ཕྱུག་མཆོག་དེར་གསོལ་བ་འདེབས། །བྷངྒ་ལ་རུ་ཆོས་རྒྱལ་ལེགས་བྱིན་མཆོག །རྒྱ་བོད་མཁས་མཆོག་པཎ་ཆེན་ཆོས་ཀྱི་ཡིད། །རྡོ་རྗེ་གདན་པ་ཆོས་ཀྱི་ཕྲེང་བའི་ཞབས། །མཁས་གྲུབ་མང་ཐོས་མཆོག་དེར་གསོལ་བ་འདེབས། །གནུབས་ཆུང་སློབ་མ་བནྡེ་ཚུལ་ཁྲིམས་དང༌། །ཇོ་བོ་སྤྱན་དྲངས་ཁུ་སྟོན་ཆོས་ཀྱི་རྗེ། །ཚུལ་ཁྲིམས་མཆོག་མངའ་གུ་ཎ་མ་ཏིའི་ཞབས། །བསླབ་གསུམ་རྒྱལ་མཚན་འཛིན་དེར་གསོལ་བ་འདེབས། །འཇམ་དཔལ་ཁྲོ་ཆུའི་རྣལ་འབྱོར་རྡོ་རྗེ་གྲགས། །འཁོར་ལོ་སྡོམ་པའི་གྲུབ་བརྙེས་ཧཱུྂ་ཀ་ར། །ཞལ་བཞི་བྲན་ཁོལ་གྲགས་པ་ཡེ་ཤེས་ཞབས། །རིམ་གཉིས་མཐར་སོན་མཆོག་དེར་གསོལ་བ་འདེབས། །གཙུག་ལག་རྩིས་ལ་མཁས་པ་ལེགས་གྲུབ་དང༌༑ ༑བླ་མའི་དགོངས་པ་ཀུན་སྒྲུབ་གྲགས་པའི་མཆོག །སྤྲོས་བྲལ་ཆོས་སྐུ་མངོན་གྱུར་རིན་ཆེན་ཆོས། །གཉུག་མའི་དོན་རྟོགས་མཆོག་དེར་གསོལ་བ་འདེབས། །སྐྱོབ་པའི་དམ་ཆོས་མཛོད་འཛིན་ཨ་ཀ་ར། །སྐད་གཉིས་སྨྲ་བ་དཔང་ལོ་བློ་གྲོས་ཞབས། །སྒྲ་རིག་མཁས་མཆོག་ཤོང་སྟོན་རྡོ་རྗེའི་མཚན། །ལེགས་སྦྱར་གཞུང་སྨྲ་མཆོག་ལ་གསོལ་བ་འདེབས། །ཤཱཀྱ་མཆོག་དྲུང་པཎ་ཆེན་དོན་ཡོད་ཞེས༑ ༑སྒྲུབ་བརྒྱུད་བསྟན་འཛིན་རྡོ་རྗེ་རིན་ཆེན་དང༌། །གདུང་བརྒྱུད་སྲིད་སྐྱོང་སྒོམ་པ་རཏྣའི་མཚན། །དྲང་སྲོང་ཆོས་འཛིན་མཆོག་དེར་གསོལ་བ་འདེབས། །རིག་པ་འཛིན་མཆོག་རྗེ་བཙུན་བི་ཤྭའི་ཞབས༑

༑འགྲུ་རྒྱལ་རིགས་འདིར་སླར་ཡང་སྲིད་པ་བཟུང༌། །དེ་ལས་མྱུར་ལྡོག་དྷརྨ་ཀཱིརྟི་ཞེས། །ཚུལ་ལྡན་མང་ཐོས་མཆོག་དེར་གསོལ་བ་འདེབས། །འདི་འོག་དོན་གྲུབ་གྲགས་པ་ཆོས་ཀྱི་འབངས༑ ༑སྔགས་འཆང་ཆེན་པོ་བདུད་འདུལ་གྲགས་པ་དང༌། །རྒྱ་གར་བྱང་གླིང་པཎ་ཆེན་ཉི་མའི་སྙིང༌། །གཏེར་འབྱིན་མཁས་པའི་མཆོག་དེར་གསོལ་བ་འདེབས། །ཀློང་ཡངས་དྲུང་དུ་བཞིན་བཟང་ཟླ་བས་བྱིན། །རབ་གསལ་སྤྱན་སྔར་སྤོབས་པའི་སྣང་བ་ཅན། །མ་འོངས་སངས་རྒྱས་དགུ་པ་མེ་ཏོག་ཞེས། །འབྱུང་འགྱུར་བསྟན་པའི་བདག་པོར་གསོལ་བ་འདེབས༑ ༑དེ་ལྟར་གསོལ་བ་བཏབ་པའི་བྱིན་རླབས་ཀྱིས། །བདག་སོགས་མོས་ལྡན་འགྲོ་བ་མ་ལུས་པ། །མཉམ་ཉིད་གཉུག་མ་ཆེན་པོའི་དོན་རྟོགས་ནས། །རྫོགས་པའི་སངས་རྒྱས་ས་ལ་འགོད་གྱུར་ཅིག །ཅེས་པ་འདི་ཡང་ཆོས་མཛད་དོ་པོ་སྒང་པས་སྐྱེ་རབས་ཀྱི་གསོལ་འདེབས་ཞིག་དགོས་ཞེས་བསྐུལ་བའི་ངོར་བགྱིས་ཏེ། ངན་པ་རང་བསྟོད་ཀྱི་ཚུལ་དུ་ཤཱཀྱའི་དགེ་སློང་དྷརྨ་ཀཱིརྟིས་སྦྱར་བའོ།། །།མངྒ་ལཾ།།
