རྒྱལ་བ་ཀུན་གྱི་རྒྱལ་བར་གྱུར་ན་ཡང༌། །རྒྱལ་བའི་སྲས་ཚུལ་རྒྱལ་བ་འཇིག་རྟེན་དབང༌། །རྒྱལ་བའི་གདུང་འཚོབ་རྒྱལ་བའི་འབྱུང་གནས་ཉིད། །རྒྱལ་བའི་བསྟན་ལ་རྒྱལ་བ་གཉིས་པ་ཁྱོད། །མེས་པོ་འཇོམས་བྱེད་ཚོགས་བདག་རི་ཡི་མགོན་པོ་སོགས་རྣམས་ཀྱིས། །ཁྱོད་ཀྱི་ཞབས་སེན་ཟླ་བའི་དཀྱིལ་འཁོར་འཇིགས་སྐྱོབ་ངོ་མཚར་བ། །མགོན་པོར་དང་བས་མཆོད་པ་དག་ལ་རྗེས་སུ་འདོད་སྦྱིན་པ། །འདྲེན་པ་ཞི་བ་ཡང་དག་རིག་པ་རྒྱལ་བ་དེའི་དྲུང་དུ། །གང་ཐུགས་མཐའ་བྲལ་རོལ་གནས་དབྱིངས་སུ་ཇི་སྙེད་ཤེས་བྱ་ནམ་མཁའི་གླིང༌། །མཐའ་ཡས་བཟླུམས་པས་རིག་པའི་སྤྲིན་སྐྱེས་སྙན་ངག་ཉམས་བརྒྱའི་བདེ་འཐུང་དབང༌། །བྲྀཏྟ་ཛཱ་ཏིའི་སྐད་གསང་དང་བཅས་ལེགས་བཤད་ཡོངས་ལྷུང་དགའ་སྟོན་ཆེ། །འདིར་གནས་འོད་སྲུང་བྱིན་གྱི་དོ་རར་སྐད་ཅིག་མ་ཉམས་སྨིན་མ་ཐག །དེ་ཚེ་དངོས་ཅག་མདོངས་ཐ་ཅན། །རབ་དགའ་ཉམས་བརྒྱའི་རོལ་གར་ནི། །ཅི་ཡང་བརྟས་པའི་མཻ་ན་ཀཿ ༑འཇོ་སྒེག་མཆོད་སྤྲིན་འདི་འབུལ་ལོ། །མཐའ་བྲལ་ཆོས་དབྱིངས་དག་པའི་རང་གཟུགས་ནི། །རླུང་གི་རྒྱལ་མཚན་ཇི་བཞིན་འགྲོ་འདི་ཀོ། །མ་དྲངས་འདུལ་སླད་གང་གང་དེར་སྣང་བ། །ཁྱོད་ལ་གཅམ་བུ་མིན་པར་ཉེར་སེམས་སོ། །འདི་ནའང་ཉེར་འགྲོའི་ཁུར་ཆེན་པོས། །ངལ་བའི་ན་ཚོད་གཏུགས་པ་ཡི༑ ༑ལྔ་པའི་དགའ་སྟོན་མ་མཐོང་ཙམ། །རབ་ཞིའི་གནས་ལ་བརྩོན་ལགས་སོ། །ཤཱཀྱའི་རིགས་ལས་ཤཱཀྱའི་དབང་པོ་བལྟམས་པ་ཇི་བཞིན་དུ། །སཱ་ལའི་ལྗོན་པ་ལ་སྡང་ཁམས་ཀྱི་རྒྱལ་པོ་ཟ་ཧོ་རའི༑ ༑རིགས་ལས་རྒྱལ་བ་མཆོག་ཁྱོད་ལེགས་པར་བལྟམས་ཚེ་མཛེས་པའི་སྐུ། །ཀཱ་ཤིའི་སླེབ་མོས་མི་ལེན་ཚངས་དང་བརྒྱ་བྱིན་གཉིད་དམ་ཤི། །བསྔགས་འོས་འབུམ་ཕྲག་དུ་མ་ལས། །ལེགས་འཁྲུངས་ཁྱོད་སྐུ་ཀུ་ལི་ཤའི། །རང་བཞིན་གཞལ་དུ་མེད་པ་དག །ཨེ་མ་འགྲོ་འདིའི་བསོད་ནམས་ཕུལ། །འཇིག་རྟེན་དབང་ཕྱུག་བརྩེ་བའི་བདག་ཉིད་ཡན་ལག་བདུན་དང་བཅོ་བརྒྱད་ཆོས་ལྡན་མཁའ་ལྟར་རབ་ཟབ་འདུས་མ་བྱས། །གདུལ་བྱ་དག་པའི་ཡིད་མཁར་འགོག་མེད་ངེས་པ་ལྔ་ལྡན་རྣམ་རོལ་གར་གྱིས་ཐེག་ཆེན་ཆོས་ཀྱི་རྔ་སྒྲ་སྒྲོག

།རྣམ་འགྱུར་སྟོང་གི་རོལ་སྒེག་གར་གྱིས་རྩེན། །འོན་ཀྱང་ཉི་འོག་དབུས་གནས་དག་པའི་ཞིང༌། །བསྟན་དང་སྐལ་བཟང་དུ་མའི་འཇུག་ངོགས་འདིར། །འདྲེན་པ་ཁྱོད་ཉིད་བལྟ་ཞིང་དམ་ཆོས་ཀྱི། །བདུད་རྩིའི་དགའ་སྟོན་རོ་ལ་སྤྱོད་འདོད་པས། །ཚ་གདུང་སྟོང་ལ་འགོང་བའི་གདུང་ཤུགས་ནི། །ཇི་ལྟར་རྒྱལ་བ་ཡུམ་གྱི་དོན་སླད་དུ། །སུམ་རྩེན་གནས་སུ་ཕྱར་ཚུལ་ལ་གཞོལ་ཚེ། །ཞིང་འདིའི་ལུས་ཅན་གདུང་བས་ཡིད་མྱོས་ཏེ། །རྫུ་འཕྲུལ་སྟོང་མངའ་མཽདྒལ་གྱི་བུ། །སྦྲན་ཏེ་མྱུར་དུ་གཤེགས་པར་གསོལ་བཏབ་བཞིན། །བདག་ལྟའི་ལུས་ཅན་ལོ་ཟླའི་ཁུར་མང་གིས། །དུབ་པའི་འབྱུང་བཞིའི་མགྲོན་ཁང་སྒྱུ་མའི་གྲོང༌། །སེམས་ཀྱི་མགྲོན་པོས་སྤོང་ལ་ཉེ་བ་རྣམས། །རྒྱལ་བ་ཁྱོད་ཞལ་ཡུད་ཙམ་མཐོང་བའི་མོད། །ལྔ་པར་གྱུར་ཀྱང་ཡིད་རབ་རྫོགས་སེམས་པས། །མྱུར་དུ་ཕྱོགས་འདིར་གཤེགས་ལ་གསོལ་འདེབས་ཞེས། །ལུས་ངག་ཡིད་གསུམ་རྩེ་གཅིག་སྨོན་ལགས་ན། །དེ་འབྲས་རེ་བ་སྐོང་བའི་ཐུགས་བསྐྱེད་ཀྱི། །ཞིང་འདིར་མྱུར་བར་མྱུར་བར་ཕེབས་པར་གསོལ། །དེ་སྐད་གསོལ་ཕྱིར་ཡིད་སྤྲུལ་མཆོད་པའི་སྤྲིན། །ཀུན་བཟང་རྣམ་རོལ་ལ་འཁུ་དང་འགྲོགས་ཏེ། །དངོས་འབྱོར་ལྷ་རྫས་དཀར་པོ་དྲི་མེད་དང༌། །རུང་བའི་ཚོས་ལྡན་དྲས་ཤིང་གྲུབ་པ་ཡི། །སྐུ་ཆོས་དག་དང་ལེབ་རྒན་མདངས་འཚོབ་པའི། །རྔུལ་གཟན་རྣམས་ལ་ཨེ་ཝཾ་ཟུང་འཇུག་གི༑ ༑རྟགས་ཀྱིས་བཀྲ་བ་རྣམ་པར་རྒྱལ་བའི་ལོའི། །འདུ་བྱེད་ཟླ་བའི་དམར་པོའི་གྲལ་ཚེས་ནི། །དགའ་བ་གཉིས་པར་འབྲི་གུང་ཕྱོགས་གནས་པའི། །སྐྱུ་རའི་རིགས་ལས་ཤཱཀྱའི་སྲས་དགེ་སློང༌།། རྒན་པོར་གྱུར་པས་རྩེ་བ་དག་ནས་ཕུལ། །ཞེས་པ་འདི་རྒྱལ་བ་ཐམས་ཅད་མཁྱེན་པ་ཆེན་པོར་རྒྱ་ནག་ཏུ་ཕུལ་བའོ།། །།མངྒ་ལཾ།།
