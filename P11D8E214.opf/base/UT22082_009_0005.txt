གཞི་རྣམ་དག་ལམ་དུས་ཀྱི་ལམ་དུས་ཀྱི་རྣམ་དག་འབྲས་བུ་རྣམ་དག 

ཨོཾ་་་་་་་་གཞིའི་རང་བཞིན་གྱི་འོད་གསལ་པ།ཨོཾ་་་་་དག་བྱ།ཀུན་གཞིའི་རྣམ་པར་ཤེས་པ། ཨོཾདག་བྱེད། སྐྱེ་བ་དང་འགག་པ་མེད་པ། ཨོཾཐོབ་བྱ། དཔལ་ཧེ་རུ་ཀ་ཆེན་པོ། ཧྲཱིཿ་་་་་་་གཞིའི་རང་བྱུང་གི་སྙིང་པོ་ཉིད། ཧྲཱིཿ་་་་དག་བྱ།ཉོན་མོངས་པ་ཅན་གྱི་ཡིད། ཧྲཱིཿདག་བྱེད། དེ་བཞིན་འོད་གསལ་བ། ཧྲཱིཿཐོབ་བྱ། རྡོ་རྗེ་རྣམ་འབྱོར་མ། ཧ་་་་་་་་གཞིའི་བྱ་བ་དང་བྱེད་པའི་ཚོགས་དང་བྲལ་བ།ཧ་་་་་་་དག་བྱ།ཡིད་ཀྱི་རྣམ་པར་ཤེས་པ། ཧདག་བྱེད། གཞན་མ་ཡིན་པའི་རྟོགས་པ། ཧཐོབ་བྱ། མཁའ་འགྲོ་མ། ཧ གཞིའི་ཡུལ་སྣང་སྣ་ཚོགས་ལས་རབ་ཏུ་རྒྱལ་བ།ཧ་་་་་་་དག་བྱ།སྒོ་ལྔའི་རྣམ་པར་ཤེས་པ། ཧདག་བྱེད། གདོད་མ་ནས་ཞི་བ། ཧཐོབ་བྱ། མཛེས་མ། ཧུཾ༴་་་་་་གཞིའི་རབ་ཏུ་གདུག་པ་དང་མཚོན་ཡུལ་འཇོམས་པ།ཧུཾ་་་་་་་དག་བྱ།ཚོར་བ་ལས་གྱུར་པའི་བདེ་བ། ཧུཾདག་བྱེད། གང་གིས་ཀྱང་འདུས་མ་བྱས་པ། ཧུཾཐོབ་བྱ། དུམ་སྐྱེས་མ། ཧཱུྂ གཞིའི་མི་སྡུག་པ་དང་ངན་པ་སེལ་བ།ཧཱུྂ་་་་་་་དག་བྱ།ཚོར་བ་ལས་གྱུར་པའི་སྡུག་བསྔལ། ཧཱུྂདག་བྱེད། ཡང་དག་པའི་མཐའ་ལས་འདས་པ། ཧཱུྂཐོབ་བྱ།གཟུགས་ཅན་མ། ཕཊ྄ ་་གཞིའི་ཀུན་ཏུ་སྦྱོར་བའི་བག་ཆགས་སྤོང་བ་རྣམས་སོ།། ་་་ཕཊ྄་་་་་་དག་བྱ།ཚོར་བ་ལས་གྱུར་བའི་བཏང་སྙོམས། ཕཊ྄དག་བྱེད༌།ཆོས་ཀྱི་དབྱིངས་རྣམ་ཀུན་མཆོག་ལྡན།་་་་ཕཊ྄ཐོབ་བྱ།དཔའ་བོ་དང་དཔའ་མོའི་ཚོགས་སོ།། ཕྱི་རོལ་རྣམ་དག་དང་སྦྱར་བ་བདུན་ཕྲག་དང་པོ། ཕྱི་རོལ་རྣམ་དག་དང་སྦྱར་བ་བདུན་ཕྲག་གཉིས་པ།་་ཕྱི་རོལ་རྣམ་དག་དང་སྦྱར་བ་བདུན་ཕྲག་གསུམ་པ་སྟེ་ཕྱིའི་དག་པ་ཉེར་གཅིག ནང་ལྷ་དང་སྦྱར་བའི་དག་པ་བདུན་ཕྲག་གཅིག ། དག་པའི་ཡོན་ཏན་བདུན། དག་པར་བྱེད་པའི་ཐབས་བདུན། སྔགས་ཀྱི་ངོ་བོ་བདུན། ཨོཾ་་་་ཡོན་ཏན། སྐུ་གསུམ་གྱི་བདག་ཉིད། ཨོཾ་་་་ཐབས། རང་གི་སྙིང་དབུས་སུ་སད་པའི་ཡེ་ཤེས་ངོ་སྤྲོད་པ། ཨོཾ ་ངོ་བོ། དྲན་པ་ཡང་དག་བྱང་ཆུབ་ཀྱི་ཡན་ལག་བདུན། མ་བྱོན་སངས་རྒྱས་བདུན་པ་རབ་གསལ་གྱི། །

ཧྲཱིཿ་་་ཡོན་ཏན། ཆགས་པའི་དངོས་པོ་དང་བྲལ་བ། ཧྲཱིཿ་་་ཐབས། ཆགས་པ་ཆེན་པོའི་ཡིད་ཀྱིས་ལུས་བསྟན་པ། ཧྲཱིཿ་་་་ངོ་བོ། ཆོས་རབ་དུ་རྣམ་པར་འབྱེད་པ་ཡང་དག་བྱང་ཆུབ་ཀྱི་ཡན་ལག ་་་་་རྣམ་འཁྲུལ་པཎ་ཆེན་ཤཱཀྱ་ཤྲཱི་ལ། ། ཧ་་་་་ཡོན་ཏན། འཁོར་འདས་ཀྱི་ཆོས་ཀུན་ལ་ཁྱབ་པ། ཧ་་་་་ཐབས། བདེ་བའི་ངོ་བོ་ནམ་མཁའི་རྣལ་འབྱོར་དུ་ངོ་སྤྲོད་པ། ཧ ་ངོ་བོ། བརྩོན་འགྲུས་ཡང་དག་བྱང་ཆུབ་ཀྱི་ཡན་ལག མཁའ་འགྲོས་ནམ་མཁར་སིལ་སྙན་དཀྲོལ་བའི་སྒྲར།། ཧ ་ཡོན་ཏན།་་་དངོས་པོ་དང་བྲལ་བའི་བདེ་བ་ཆེན་པོའི་བདག་ཉིད། ་་་་ཧ་་་་་ཐབས། བདེ་སྟོང་ལྷན་སྐྱེས་ཀྱི་དགའ་བ་རབ་ཏུ་རྟོགས་པ། ཧ ་ངོ་བོ། དགའ་བ་ཡང་དག་བྱང་ཆུབ་ཀྱི་ཡན་ལག ་་་མཁའ་འགྲོ་དྲ་བ་སྡོམ་པའི་ཡི་གེ་བདུན། ། ཧུཾ༴་་་ཡོན་ཏན། ཡེ་ཤེས་ཆེན་པོ་འབར་ཞིང་རབ་ཏུའབར་བ། ཧུཾ་་་་་ཐབས། དགའ་བྲལ་གྱི་དགའ་བ་མངོན་དུ་བྱས་པ།ཧུཾ་་་་་ངོ་བ། ཤིན་ཏུ་སྦྱངས་པ་ཡང་དག་བྱང་ཆུབ་ཀྱི་ཡན་ལག གཞི་ལམ་འབྲས་བུའི་དག་པ་ཡི་གེ་བདུན། །ཧཱུྂ ་ཡོན་ཏན། བདེ་བ་དང་སྟོང་པ་མཆོག་གི་རྒྱུད་རླན་པ། ཧཱུྂ་་་་ཐབས། མཆོག་ཏུ་དགའ་བའི་དངོས་པོ་མཐོང་བ།ཧཱུྂ ་ངོ་བོ། ཏིང་ངེ་འཛིན་ཡང་དག་བྱང་ཆུབ་ཀྱི་ཡན་ལག ཕྱི་ནང་གཞན་གསུམ་ངེས་པའི་གསང་དོན་ནི། །ཕཊ྄་་་ཡོན་ཏན། དེ་ཁོ་ན་ཉིད་སེམས་སུ་ཐག་གཅད་པ། ཕཊ྄་་ཐབས། རབ་ཏུ་དགའ་བ་ཏིང་ངེ་འཛིན་དུ་རྟོགས་པ།ཕཊ྄་་་ངོ་བོ། བཏང་སྙོམས་ཡང་དག་བྱང་ཆུབ་ཀྱི་ཡན་ལག ཆོས་ཀྱི་གྲགས་པས་བསྡུས་ཏེ་གོ་བདེར་བྲིས།་ །གཞན་ཡང་གསང་བ་ཡེ་ཤེས་དང་སྦྱར་བའི་དག་པ་བདུན་ཕྲག་གཅིག། དེ་ཁོ་ན་ཉིད་སྦྱར་བའི་དག་པ་བདུན་ཕྲག་གཅིག ཕྱི་ནང་གཞན་དེ་ཁོ་ན་ཉིད་ཀུན་ལ་ཐུན་མོང་དུ་སྦྱར་བའི་དག་པ་བདུན་ཕྲག་གཅིག་གོ།།་་་་་་་་་་་སརྦ་མངྒ་ལཾ།། །། ༄༅། །དྲ་བ་སྡོམ་པའི་ཡི་གེ་བདུན་གྱི་རྣམ་པར་བཤད་པ་བཞུགས་སོ།།
