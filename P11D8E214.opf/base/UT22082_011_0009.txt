༄༅། །ཡང་ཟློག་མེའི་སྤུ་གྲིའི་གཏོར་མས་ཟློག་པའི་ལས་སྒྲུབ་པའི་ལག་ལེན་བཞུགས་སོ།།

ཡ་མཱ་རི་ན་མོ། ཡང་ཟློག་མེའི་སྤུ་གྲིའི་སྲུང་ཟློག་བསད་མནན་གྱི་སྒྲུབ་པའམ་སྲུང་ཟློག་རྐྱང་པའི་སྒྲུབ་པ་སོགས་ངག་མཚམས་མི་བྱེད་པར་གཏོར་ཟློག་ཙམ་ལ་བརྟེན་པར་བྱེད་ན། ལྕགས་སྣོད་དུ་འབྲུ་ནག་རྗེན་བཏགས་ལ་སྔར་གྱི་གཏོར་ཟན་རྙིང་པའི་ཕབས་རྒྱུན་དང་བཙན་དུག་ཆང་ཁྲག་གིས་སྦྲུས་པའི་གཏོར་མ་བྲ་གཟུགས་ལ་ལོགས་གསུམ་དབལ་གཉིས་དང་དབུས་སུ་ཁྲམ་རིས་ཅན། བྲང་ནག་པོ། དབལ་ཁྲག་གིས་བྱུགས་པར་དཀར་དམར་གྱི་རྒྱན་དང་རྩེ་རྒྱན་ཐོག་མདེའུ་ལྟ་བུ་ཕྱག་མཚན་གྱིས་སྤྲས་པ། འཁོར་བཤོས་བུ་གྲུ་གསུམ་བཞི་དམར་གྱིས་བརྒྱན་པ་མདུན་རྒྱབ་གཡས་གཡོན་བཞིར་བཞག །ཕྱི་སྐོར་དུ་མདའ་དར་རམ་རྒྱལ་མཚན་ཕྱོགས་མདོག །དྲག་པོའི་ཤིང་གི་རྩང་སྟོད་སྨད་རོ་སོལ་ཁྲག་དང་སྦྱར་བས་ནག་པོ། སྐེད་པ་གྲི་ཁྲག་དང་བཙག་གིས་དམར་པོར་བྱུགས་པ་བསྣོལ་མར་བཙུགས་པའི་རྩེ་མོར་ཤ་དུམ་དང་ལིངྒ་མགོ་ཐུར་བསྟན་རེ་བསྐྱོན། གཙོ་འཁོར་ལ་རྒྱན་རྩང་དང་མཚེ་ཡུངས་གཟུགས། གལ་ཆེ་ན་བསྲུང་བའི་འཁོར་ལོ་གཞུང་ལྟར་བྲིས་ལ་བལྟབས་པ་ཚོན་སྐུད་གྱིས་བཅིངས་པ་གཏོར་གདུགས་ནག་པོའི་དབུས་སུ་བཞུགས། ཟློག་པ་རྐྱང་པའི་འཁོར་ལོ་ཤམ་འདོགས་ཅན་གཏོར་མ་གཙོ་བོའི་སུམ་ཆ་ཡར་བཅད་སུམ་ཆ་གཉིས་མར་བཅད་པའི་མཚམས་སུ་བཅུག །འཁོར་བཞི་ལའང་རང་རང་གི་སྔགས་ཟློག་ཤམ་བཏགས་པ་དྲིལ་ལ་བཅུག །འབྲུ་ནག་གི་ཆོས་འབྱུང་རིམ་གསུམ་གྱི་དབུས་སུ་ཚོམ་བུ་ལྔ་བཀོད་པའི་སྟེང་དུ་ཁྲིའི་ཁར་བཤམ། དེ་དག་གི་མཐའ་སྐོར་དུ། ཁྲག་ཆེན་གྱི་ཨརྒྷཾ། དུག་ཁྲག་གི་ཞབས་བསིལ། དབང་པོ་ལྔའི་མེ་ཏོག །ཤ་ཆེན་དང་གུ་གུལ་སྦྱར་བའི་སྤོས། མཁྲིས་ཆེན་ཆང་དང་སྦྱར་བའི་དྲི་ཆབ། ལྷ་བཤོས་དམར་པོ་ཟུར་གསུམ་ཤ་ཁྲག་རུས་པས་སྦགས་པ། རྫ་རྔ་རྐང་གླིང་གི་རོལ་མོ་རྣམས་གཡོན་སྐོར་དུ་བཤམས། གཏོར་ཟོར་ནས་འབྲས་གྲོ་སྲན་སོ་བ་ལྔ་རྗེན་ཆེའི་ཟུར་གསུམ་དུག་ཁྲག་བྱུགས་པ་མདའ་མདུང་མཚེ་ཡུངས་རྩང་གིས་བརྒྱན་པ་ཐོད་པའི་ལྕགས་སྣོད་དུ་བཤམས།

ཡང་ཐོད་པའི་ལྕགས་སྣོད་དུ་སྲན་མ་བྲ་བོ་སོགས་འབྲུ་ནག་གི་ཕྱེ་མར་ལིངྒ་མཁྱིད་གང་པ་དམིགས་བྱ་ཡོད་ན་དེའི་མིང་རུས་དུག་ཁྲག་གིས་བྲིས་པ་སྙིང་གར་བཅུག་ཅིང་དྲི་མས་སྦགས་པ་དང༌། བགེགས་ཡིན་ན་དེ་གནས་པའི་ས་ཆུ་ཤིང་སོགས་དང་མིང་བྲིས་པ་དུག་ཁྲག་བྱུགས་པ་ཁོང་པར་བཅུག་ཅིང་ཕྱི་ལའང་དུག་ཁྲག་བྱུགས་པ་ཕུང་རེའམ་དར་རས་ནག་པོ་བཀབ་པ་གཏོར་མའི་མདུན་དུ་བཤམས། གཞན་ཡང་སྨན་རཀ། ཐུན། ཕུར་པ། གྲི་ལྕགས་ཀྱི་རལ་གྲི༑ གཏུན་བུ། གཏུན། གཡབ་དར་ལ་སོགས་པ་མཁོ་བའི་རྫས་ཐམས་ཅད་མ་འཛོམས་པ་མེད་པ་བཤམས། རང་ཉིད་སྟན་བདེ་བར་འདུག་ལ། སྔོན་འགྲོ་སྐྱབས་སེམས་དང་ཆ་གསུམ་གཏོར་མ་བཏང༌། ངག་འདོན་གཞུང་བསྲངས་ལ་སྲུང་སྔགས་བརྒྱ་སྟོང་སོགས་བཟླས་ལ་ཉལ། ཕྱི་ནངས་ནས་ཐོ་རངས་སྐྱབས་སེམས་དང་ངག་འདོན་བསྲུང་བའི་ཁ་བསྒྱུར་བྱས་པ་གཞུང་བསྲངས་པས་བདག་དང་གཏོར་མ་ལྷན་ཅིག་ཏུ་ལྷར་བསྐྱེད། ཉི་ཤར་བར་སྲུང་བསྔགས་བཟླ། ཉིན་ཤར་ཐུན་གཏོར་ལ་དགུག་གཞུག་ཐབས་མཆོག་བདེ་ཆེན་མ་སྦས་ལྷ་གཉིས་སྦྱར་བས་གཏོར་ཆེན་ལ་བསྟབ་སྨན་རཀ་བསྒྲོན་ལ། སླར་ཐུན་གཏོར་གཅིག་བཤམ་སྨན་རཀ་བསྒྲོན་ལ་བཞག །སླར་ངག་འདོན་གྱི་དམིགས་པ་ཐུགས་ཀའི་འཇམ་དཔལ་སྐུ་མདོག་དམར་ནག་ཅེས་པ་ནས་གཡོ་འགུལ་མེད་པར་གྱུར་བར་བརྗོད་ཅིང་ཟློག་པའི་དམིགས་པ་བསྒྱུར་ལ་ཟློག་སྔགས་ལ་འབད། ཉིན་གུང་གི་ཐུན་ལ་ཚོགས་མཆོད་དང་བསྐང་བཤགས་བྱ། དགོང་ཉི་མ་གྲིབ་སོ་གུག་པ་དང༌། གཏོར་ཆུང་ཤ་ཁྲག་རུས་པ་དང་སྦྲགས་པ་ལྔ། དྲག་པོའི་གཏོར་མ་བྱིན་རླབས་ལྟར་བྱས་ལ་བསྟོད་རིང་གི་བ་ལིང་ཤ་ཁྲག་གཏོར་མ་འདི་བཞེས་ལ་ཟེར་དུས་རེ་རེ་ནས་གཏོར་ཆེན་ལ་བསྟབ་སྨན་རཀ་བསྒྲོན་ལ། དེ་རྗེས་ཉ་བོ་དང་ཕུར་ཐུན་མཚོན་ཆ་གཏུན་བུ་གཡབ་དར་སོགས་རང་གི་མདུན་དུ་བཤམས་ལ། གཏོར་ཆེན་ལ་ཤ་ཟན་དང་སྨན་རཀ་བྲན་ལ་ཟློག་པའི་ཕྲིན་ལས་གཞུང་བསྲང་ཞིང་གྱེར། དེ་རྗེས་གསེར་སྐྱེམས་བཤམས་ལ་མཁྱེན་དགོངས་དཔང་འཛུགས་དམོད་བཅོལ་བྱ།

བྲུབ་ཁུང་བསྐྱེད་ཆོག་བྱས་ལ་དགུག་གཞུག་དྲག་པོ་ཅི་མང་གི་མཐར་གཟུ་དཔང་བླ་དགུག་བླ་དབྱེ་བྱ། བྱད་མ་རང་གཤེད་དུ་བསྐུལ་བ། ཉེས་བྱས་དྲག་བསྐུལ་དང༌། མོན་པ་བཞི་བསྐུལ་དང༌། སྡེ་བརྒྱད་འབྱུང་ཚུལ་ལྟར་བསྐུལ་བ་དང༌། སོ་སྡེབས་དྲེགས་པའི་བསྐུལ། སླར་ཀིཾ་ཀང་གི་བླ་འགུགས་དང་ལྷ་དབྱེ་བྱ། ཐུན་བསྐུལ་བྱས་ལ་ཐུན་བྲབ། བྱང་ཆུབ་ལམ་བསྟན་ཡང་བྱས་ལ། སླར་ཕུར་གདབ་དང་བསྒྲལ་ཞིང་གཏུན་བརྡུང་རྒྱས་པར་བྱ། བསྒྲལ་བའི་ལིང་རོ་གཏོར་མར་བསྟབ་ཅིང་སྨན་རཀ་བསྒྲོན་ལ། བསྟབ་པོ་དང༌། གསང་བ་གསོལ་གདབ་མངའ་གསོལ་བྱ། ཟློག་བྱང་ཆུང་བ་ཡང་གཏང༌། རྟགས་བསྟོད་སོགས་ཅི་འགྲུབ་བྱ། སྲོད་ཐུན་ལ་ན་མོ་བླ་མའི་ཐལ་མོ་བརྡབས་རྗེས། ཐུན་གཏོར་དགུག་གཞུག་གསུམ་དང་དྲག་པོའི་གཏོར་མ་བྱིན་རླབས་གཏོར་བསྔོས་བྱས་ལ། དྲག་བསྐུལ་ལམ་བསྟོད་རིང་གྲུ་གསུམ་མཐིང་ནག་མ་གང་རུང་གིས་བསྐུལ་ལ། ལྕགས་སམ་བྷནྡྷ་དང༌། ཟློག་སྔགས་དང༌། འབྱུང་བ་ཁྲམ་ཟློག་དང༌། རྦོད་གཏོང་དྲག་པོའི་ཟོར་འཕེན་པའི་སྔགས་སྡེབས་དང་བཅས་པ་བཟླ་ཞིང་ཕྱིར་འཕང༌། ཐུན་གཏོར་གཅིག་བཤམས་ལ་སྨན་རཀ་བྲན་ལ་བཞག །ཚུལ་དེ་ལྟ་བུ་ཞག་གསུམ། ལྔ། བདུན། དགུ་སོགས་གལ་ཆེ་ཆུང་དང་ལོང་སྐབས་དང་བསྟུན་པ་བྱས་རྗེས༑ ནུབ་ཐ་མ་མངའ་གསོལ་གྱི་རྗེས་སུ་ཐོག་སྟེང་དུ། གསེར་སྐྱེམས་འབུལ། གཏོར་ཟོར་ལྕགས་སམ་བྷནྡྷས་ཟློག་སྔགས་རྦོད་གཏོང་དྲག་པོའི་ཟོར་འཕེན་པའི་སྔགས་སྡེབས་དང་འབྱུང་བ་ཁྲམ་ཟློག་དང་བཅས་གང་དམིགས་པའི་ཕྱོགས་སུ་འཕང༌། ནང་དུ་ལོག་ལ་གཏོར་ཆེན་སྒུལ་ཤ་ཟན་སྨན་རཀ་བྲན་ལ་དགུག་གཞུག་དང་གཏོར་བསྔོས་བྱས་རྗེས། གྲུ་གསུམ་མཐིང་ནག་མ་བྱས་ལ་གཏོར་མའི་ལྷ་ཡེ་ཤེས་པ་གཤེགས་དམ་ཚིག་པ་རང་ལ་བསྡུ། སླར་གཏོར་མ་གཤིན་རྗེའི་དམག་དཔུང་དུ་བསྐྱེད་ལ་བསྟོད་རིང་ ་་་་་ཨེ་རིང་ བསྐུལ། སྒོ་ཕྱིར་སྤགས་ལ་ཟློག་བྱང་ཆུང་བ་གྱེར། ཤ་ཟན་དང་སྨན་རཀ་བྲན། དབྱེ་དགུག་གཏོར་བསྔོས་བྱས་ལ་དྲག་བསྐུལ་

་་་་་ཨེ་ཐུང་ དང༌། ལྕགས་སམ་བྷནྡྷ། ཟློག་སྔགས། རྦོད་གཏོང་ཟོར་འཕེན་པའི་སྔགས་སྡེབས། འབྱུང་བ་ཁྲམ་ཟློག་དང་བཅས་པས་དགྲ་འདྲེ་གནོད་པའི་ཕྱོགས་ཐེབས་ཐོག་ཏུ་འཕང༌། གཏོར་མནན་སོགས་ཟློག་པ་དཀྱུས་ལྟར་བྱ་བ་ཡིན་ནོ། །ཞེས་པའང་གཤིན་རྗེ་ཚེ་བདག་ཡང་ཟློག་མེའི་སྤུ་གྲིའི་ཉེར་མཁོ་གཏོར་མ་ཁོ་ནས་ཟློག་པའི་ལས་སྒྲུབ་པའི་ལག་ལེན་དུ། འབྲི་གུང་པ་སྔགས་འཆང་ཐུགས་ཀྱི་རྡོ་རྗེས། འབྲི་གུང་བྱང་ཆུབ་ཀྱི་གླིང་གི་ཐོག་ཁ་གསེར་ཁང་དུ་བྲིས་པའོ།། །།སརྦ་མངྒ་ལཾ།།
