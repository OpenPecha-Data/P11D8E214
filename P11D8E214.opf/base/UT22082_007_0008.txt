ཕྱག་རྒྱ་བཞིའི་ཟུར་འདེབས་ཀྱི་ཐོག་མར་དམ་རྒྱ་ནི། ལྷ་རྣམས་ཅད་ཀྱི་ཐུགས་ཀར་ཟླ་བའི་དབུས་སུ་ཐོག་མཐའ་མེད་པའི་ཆོས་ཉིད་ཀྱི་ངོ་བོ་དང་པོའི་རྡོ་རྗེ་རྩེ་ལྔ་པའི་མཚན་པར་གྱུར། ས་མ་ཡ་སྟྭཾ། བཛྲ་དྲྀཥྚི་མ་ཊ། ཨོཾ་སརྦ་བིད་དྲྀ་ཤྱ། ཛཿཧཱུྂ་བཾ་ཧོཿས་མ་ཡ་སྟྭཾ། ས་མ་ཡ་སྟྭཾ། ཨ་ཧཾ། སརྦ་བིད། ཞེས་སྙིང་པོ་ལན་གསུམ་དང༌། ཨོཾ་སརྦ་བིད་སརྦ་ཨཱ་ཝ་ར་ཎ་བི་ཤོ་དྷ་ཡ་ཧ་ན་ཧཱུྂ་ཕཊ། ཅེས་གཙོ་བོའི་སྔགས་ལན་གསུམ་བརྗོད། དེ་བཞིན་དུ། ས་མ་ཡ་སྟྭཾ། བཛྲ་དྲྀཥྚི་མ་ཊ། ཨོཾ་སརྦ་བིད་དྲྀཤྱ། ཛཿཧཱུྂ་བཾ་ཧོཿ ས་མ་ཡ་སྟྭཾ། ས་མ་ཡ་སྟྭཾ། ཨ་ཧཾ། སརྦ་བིད། ཅེས་ལན་གསུམ། ཨོཾ་སརྦ་བིད་ཧཱུྂ། ཞེས་ལན་གསུམ་བརྗོད༑ ས་མ་ཡ་སྟྭཾ། བཛྲ་དྲྀཥྚི་མ་ཊ། ཨོཾ་སརྦ་བིད་དྲྀཤྱ། ཛཿཧཱུྂ་བཾ་ཧོཿ ས་མ་ཡ་སྟྭཾ། ས་མ་ཡ་སྟྭཾ། ཨ་ཧཾ། སརྦ་བིད། ཅེས་ལན་གསུམ། ཨོཾ་སརྦ་བིད་ཕཊ། ཅེས་ལན་གསུམ། ས་མ་ཡ་སྟྭཾ། བཛྲ་དྲྀཥྚི་མ་ཊ། ཨོཾ་སརྦ་བིད་དྲྀཤྱ། ཛཿཧཱུྂ་བཾ་ཧོཿ ས་མ་ཡ་སྟྭཾ། ས་མ་ཡ་སྟྭཾ། ཨ་ཧཾ། སརྦ་བིད། ཅེས་ལན་གསུམ། ཨོཾ་སརྦ་བིད་ཨཱཿ ཞེས་ལན་གསུམ། ས་མ་ཡ་སྟྭཾ། བཛྲ་དྲྀཥྚི་མ་ཊ། ཨོཾ་སརྦ་བིད་དྲྀཤྱ། ཛཿཧཱུྂ་བཾ་ཧོཿ ས་མ་ཡ་སྟྭཾ། ས་མ་ཡ་སྟྭཾ། ཨ་ཧཾ། སརྦ་བིད། ཅེས་ལན་གསུམ། ཨོཾ་སརྦ་བིད་ཏྲཊ། ཅེས་ལན་གསུམ། དེ་བཞིན་དུ། ༈ སུ་ར་ཏེ་སྟྭཾ། བཛྲ་དྲྀཥྚི་མ་ཊ། བུདྡྷ་ལོ་ཙ་ནི་དྲྀཤྱ། ཛཿཧཱུྂ་བཾ་ཧོཿ ས་མ་ཡ་སྟྭཾ། ས་མ་ཡ་སྟྭཾ། ཨ་ཧཾ། བུདྡྷ་ལོ་ཙ་ནི། ཞེས་ལན་གསུམ། ཨོཾ་བུདྡྷ་ལོ་ཙ་ནི་ཧཱུྂ། ཞེས་ལན་གསུམ། སུ་ར་ཏེ་སྟྭཾ། བཛྲ་དྲྀཥྚི་མ་ཊ། བཛྲ་མཱ་མ་ཀཱི་དྲྀཤྱ། ཛཿཧཱུྂ་བཾ་ཧོཿ ས་མ་ཡ་སྟྭཾ། ས་མ་ཡ་སྟྭཾ། ཨ་ཧཾ། བཛྲ་མཱ་མ་ཀཱི། ཞེས་ལན་གསུམ། ཨོཾ་བཛྲ་མཱ་མ་ཀཱི་ཏྲཱཾ། ཞེས་ལན་གསུམ། དེ་བཞིན་དུ་ཀུན་ལ་སྙིང་པོའི་སྔགས་སོ་སོ་ལ་སྦྱར་ནས་རིག་འགྲེའོ། །བཛྲ་པཱཎྜ་རི། ཨོཾ་བཛྲ་པཱཎྜ་རི་ཧྲཱིཿ བཛྲ་ཏཱ་རེ། ཨོཾ་བཛྲ་ཏཱ་རེ་ཨཱཿ དེ་བཞིན་དུ། ཨོཾ་བཛྲ་སཏྭ་ཧཱུྂ། ཨོཾ་བཛྲ་རཱ་ཛ་ཛཿ ཨོཾ་བཛྲ་རཱ་ག་ཧོཿ

ཨོཾ་བཛྲ་སཱ་དྷུ་དྷུ། ཨོཾ་བཛྲ་རཏྣ་ཏྲཱཾ། ཨོཾ་བཛྲ་ཏེ་ཛ་རྻ། ཨོཾ་བཛྲ་ཀེ་ཏུ་བྷྲི། ཨོཾ་བཛྲ་ཧཱ་ས་ཧཱ། ཨོཾ་བཛྲ་དྷརྨ་ཧྲཱིཿ ཨོཾ་བཛྲ་ཏཱིཀྵ་ཎ་དྷཱིཿ ཨོཾ་བཛྲ་ཧེ་ཏུ་ཀྲུཾ། ཨོཾ་བཛྲ་བྷཱ་ཥ་བྷཱ། ཨོཾ་བཛྲ་ཀརྨ་ཀཾ། ཨོཾ་བཛྲ་རཀྵ་རཾ། ཨོཾ་བཛྲ་ཡཀྵ་ཀྵཾ། ཨོཾ་བཛྲ་སནྡྷི་ཥྚཾ། ཨོཾ་བཛྲ་ལཱ་སྱེ་ཧཱུྂ། ཨོཾ་བཛྲ་མཱ་ལེ་ཏྲཱཾ། ཨོཾ་བཛྲ་གཱི་ཏི་ཧྲཱིཿ ཨོཾ་བཛྲ་ནྲྀཏྱེ་ཨཱཿ ཨོཾ་བཛྲ་དྷཱུ་པེ་ཧཱུྂ། ཨོཾ་བཛྲ་པུཥྤེ་ཏྲཱཾ། ཨོཾ་བཛྲ་ཨཱ་ལོ་ཀེ་ཧྲཱིཿ ཨོཾ་བཛྲ་གནྡྷེ་ཨཱཿ ཨོཾ་བཛྲ་ཨཾ་ཀུ་ཤ་ཛཿ ཨོཾ་བཛྲ་པཱ་ཤ་ཧཱུྂ། ཨོཾ་བཛྲ་སྥོ་ཊ་བཾ། ཨོཾ་བཛྲ་གྷཎྚ་ཧོཿ ༈ ཀལྤ་བྷ་དྲ། ཨོཾ་སརྦ་སཾ་སྐཱ་ར་པ་རི་ཤོ་དྷ་ན་དྷརྨ་ཏེ་ག་ག་ན་ས་མུདྒ་ཏེ་མ་ཧཱ་ན་ཡ་པ་རི་ཝཱ་རེ་སྭཱ་ཧཱ། ༈ །ཤྲ་ཝ་ཀ །རྩ་རིག་བརྗོད། པྲ་ཏྱེཀ །རྩ་རིག་བརྗོད། སུ་ར་ཏ་སྟྭཾ། བཛྲ་དྲྀཥྚི་མ་ཊ། བཛྲ་ཀྲོ་དྷ་དྲྀཤྱ། ཛཿཧཱུྂ་བཾ་ཧོཿ ས་མ་ཡ་སྟྭཾ། ས་མ་ཡ་སྟྭཾ། ཨ་ཧཾ། བཛྲ་ཀྲོ་དྷ། ཞེས་ལན་གསུམ་བརྗོད། རྩ་རིག་ལན་གསུམ་བརྗོད། གཉིས་པ་ཆོས་རྒྱ་ནི། ལྷ་ཐམས་ཅད་ཀྱི་ལྐོག་མར་ཧྲཱིཿལས་པདྨ་དམར་པོ་འདབ་མ་བརྒྱད་པ་མདུན་གྱི་འདབ་མ་གཅིག་ལྕེར་གྱུར་པའི་སྟེང་དུ་ཨ་ལས་ཟླ་བའི་དབུས་སུ་ཧཱུྂ་ལས་རྡོ་རྗེ་དཀར་པོ་རྩེ་ལྔ་པ་ཆུར་ཉལ་བ། རྡོ་རྗེའི་ལྟེ་བར་ལྷ་རང་རང་གི་སྔགས་ཀྱིས་མཚན་པར་གྱུར་པར་དམིགས་ཏེ། ཨོཾ་བཛྲ་ཛི་ཧྭཱ། ཨོཾ་སརྦ་བིད་སརྦ་ཨཱ་ཝ་ར་ཎ་བི་ཤོ་དྷ་ཡ་ཧ་ན་ཧཱུྂ་ཕཊ། ཨོཾ་སརྦ་བིད་ཨོཾ། ཨོཾ་སརྦ་བིད་ཧཱུྂ། ཨོཾ་སརྦ་བིད་ཕཊ། ཨོཾ་སརྦ་བིད་ཨཱཿ ཨོཾ་སརྦ་བིད་ཏྲཊ། ཨོཾ་བུདྡྷ་ལོ་ཙ་ནི་ཧཱུྂ། ཞེས་པ་ནས། ཨོཾ་བཛྲ་གྷཎྚ་ཧོཿ ཨོཾ་སརྦ་སཾ་སྐཱ་ར་སོགས། ཨོཾ་ན་མོ་བྷ་ག་ཝ་ཏེ་སོགས་རྩ་རིག་ལན་གསུམ་བརྗོད། གསུམ་པ་ལས་རྒྱ་ནི། ལྷ་ཐམས་ཅད་ཀྱི་ཐུགས་ཀར་ཟླ་བའི་སྟེང་དུ་སྣ་ཚོགས་རྡོ་རྗེས་མཚན་པར་གྱུར་པར་བསམས་ལ། གར་དང་གླུ་དང་རོ་དང་ཟས། །སྤྱོད་ལམ་ལ་སོགས་བདེ་བ་རྣམས། །སངས་རྒྱས་རྣམས་ལ་ཕུལ་ནས་ནི། །ལས་ཀྱི་ཕྱག་རྒྱས་དབང་དུ་འགྱུར། །སརྦ་བིད།

སརྦ་བིད། སརྦ་བིད། སརྦ་བིད། སརྦ་བིད། བུདྡྷ་ལོ་ཙ་ནི། ཞེས་པ་ནས། བཛྲ་ཀྲོ་དྷ། ཞེས་པའི་བར་དུ་བརྗོད། བཞི་པ་ཕྱག་ཆེན་ནི། ལྷ་ཐམས་ཅད་ཀྱི་ཐུགས་ཀར་ཟླ་བའི་སྟེང་དུ་ཐོག་མའི་རྡོ་རྗེ་དཀར་པོ་རྩེ་ལྔ་པས་མཚན་པར་གྱུར། སརྦ་བིད། སོགས་ནས། བཛྲ་ཀྲོ་དྷའི་བར་གོང་ལྟར་བརྗོད་ནས། ཨོཾ་བུདྡྷ་ཨ་བྷི་ཥིཉྩ་ཨོཾ། ཨོཾ་བཛྲ་ཨ་བྷི་ཥིཉྩ་ཧཱུྂ། ཨོཾ་རཏྣ་ཨ་བྷི་ཥིཉྩ་ཏྲཱཾ། ཨོཾ་པདྨ་ཨ་བྷི་ཥིཉྩ་ཧྲཱིཿ ཨོཾ་ཀརྨ་ཨ་བྷི་ཥིཉྩ་ཨཱཿ དེ་ལྟར་དབང་བསྐུར་བས་རིགས་ལྔའི་བདག་ཉིད་ཅན་དུ་གྱུར། དེ་ནས་ཕྲིན་ལས་བཅོལ་བ་ནི། ཨོཾ་ན་མོ་བྷ་ག་ཝ་ཏེ་སརྦ་དུརྒ་ཏི་པ་རི་ཤོ་དྷ་ནི་རཱ་ཛཱ་ཡ། ཏ་ཐཱ་ག་ཏཱ་ཡ། ཨརྷ་ཏེ་སམྱཀྶཾ་བུདྡྷཱ་ཡ། ཏདྱ་ཐཱ། ཨོཾ་ཤོ་དྷ་ནི་ཤོ་དྷ་ནི། སརྦ་པཱ་པཾ་བི་ཤོ་དྷ་ནི། ཤུདྡྷེ་བི་ཤུདྡྷེ། སརྦ་ཀརྨ་ཨཱ་ཝ་ར་ཎ་བི་ཤོ་དྷ་ནེ་སྭཱ་ཧཱ། ཨོཾ་སརྦ་བིད་སརྦ་ཨཱ་ཝ་ར་ཎ་བི་ཤོ་དྷ་ཡ་ཧ་ན་ཧཱུྂ་ཕཊ། བཅོམ་ལྡན་འདས་ངན་སོང་ཐམས་ཅད་ཡོངས་སུ་སྦྱོང་བ་གཟི་བརྗིད་ཀྱི་རྒྱལ་པོ་ཀུན་རིག་རྣམ་པར་སྣང་མཛད་ཀྱིས། བདག་སོགས་སེམས་ཅན་ཐམས་ཅད་ཟབ་གསལ་གཉིས་སུ་མེད་པ་ལ་སོགས་པ་ལམ་གྱི་རྟོགས་པ་མཐའ་དག་རྒྱུད་ལ་སྐྱེ་བར་མཛད་དུ་གསོལ། ཅེས་དང༌། ཨོཾ་ན་མོ་བྷ་ག་ཝ་ཏེ་སོགས་རྩ་རིག་དང༌། ཨོཾ་སརྦ་བིད་ཧཱུྂ། སྦྱོང་བའི་རྒྱལ་པོ་དང༌། ཨོཾ་སརྦ་བིད་ཕཊ། རྒྱལ་མཆོག་རིན་ཆེན་དང༌། ཨོཾ་སརྦ་བིད་ཨཱཿ ཤཱཀྱ་རི་དབང་དང༌། ཨོཾ་སརྦ་བིད་ཏྲཊ། མེ་ཏོག་ཆེར་རྒྱས་ཏེ། དེ་བཞིན་གཤེགས་པ་བཞིས་ཀྱང་བདག་སོགས་སེམས་ཅན་ཐམས་ཅད་ཟབ་གསལ་གཉིས་སུ་མེད་པ་ལ་སོགས་པ་ལམ་གྱི་རྟོགས་པ་མཐའ་དག་རྒྱུད་ལ་སྐྱེ་བར་མཛད་དུ་གསོལ། ཨོཾ་ན་མོ་བྷ་ག་ཝ་ཏེ་རྩ་རིག་ཀུན་ལ་འགྲེའོ། ཨོཾ་བུདྡྷ་ལོ་ཙ་ནི་ཧཱུྂ། སངས་རྒྱས་སྤྱན་མ་དང༌། ཨོཾ་བཛྲ་མཱ་མ་ཀཱི་ཏྲཱཾ། རྡོ་རྗེ་མཱ་མ་ཀཱི་དང༌། ཨོཾ་བཛྲ་པཱཎྜ་རི་ཧྲཱིཿ གོས་དཀར་མོ་དང༌། ཨོཾ་བཛྲ་ཏཱ་རེ་ཨཱཿ སྒྲོལ་མ་སྟེ། ཡུམ་བཞིས་ཀྱང་བདག་སོགས་སེམས་ཅན་ཐམས་ཅད་ཟབ་གསལ་གཉིས་སུ་མེད་པ་ལ་སོགས་པ་ལམ་གྱི་རྟོགས་པ་མཐའ་དག་རྒྱུད་ལ་སྐྱེ་བར་མཛད་དུ་གསོལ།

ཨོཾ་ན་མོ་བྷ་ག་ཝ་ཏེ་སོགས་རྩ་རིག་དང༌། ཨོཾ་བཛྲ་ས་ཏྭ་ཧཱུྂ། རྡོ་རྗེ་སེམས་དཔའ་དང༌། ཨོཾ་བཛྲ་རཱ་ཛ་ཛཿ རྡོ་རྗེ་རྒྱལ་པོ་དང༌། ཨོཾ་བཛྲ་རཱ་ག་ཧོཿ རྡོ་རྗེ་ཆགས་པ་དང༌། ཨོཾ་བཛྲ་སཱ་དྷུ་དྷུ། རྡོ་རྗེ་ལེགས་པ་སྟེ། ཤར་ཕྱོགས་ཀྱི་སེམས་དཔའ་བཞིས་ཀྱང་བདག་སོགས་སེམས་ཅན་ཐམས་ཅད་ཟབ་གསལ་གཉིས་སུ་མེད་པ་ལ་སོགས་པ་ལམ་གྱི་རྟོགས་པ་མཐའ་དག་རྒྱུད་ལ་སྐྱེ་བར་མཛད་དུ་གསོལ། ཨོཾ་ན་མོ་བྷ་ག་ཝ་ཏེ་སོགས་རྩ་རིག་དང༌། ཨོཾ་བཛྲ་རཏྣ་ཏྲཱཾ། རྡོ་རྗེ་རིན་ཆེན་དང༌། ཨོཾ་བཛྲ་ཏེ་ཛ་རྻ། རྡོ་རྗེ་གཟི་བརྗིད་དང༌། ཨོཾ་བཛྲ་ཀེ་ཏུ་བྷྲི། རྡོ་རྗེ་རྒྱལ་མཚན་དང༌། ཨོཾ་བཛྲ་ཧཱ་ས་ཧཱ། རྡོ་རྗེ་བཞད་པ་སྟེ། ལྷོ་ཕྱོགས་ཀྱི་སེམས་དཔའ་བཞིས་ཀྱང་བདག་སོགས་སེམས་ཅན་ཐམས་ཅད་ཟབ་གསལ་གཉིས་སུ་མེད་པ་ལ་སོགས་པ་ལམ་གྱི་རྟོགས་པ་མཐའ་དག་རྒྱུད་ལ་སྐྱེ་བར་མཛད་དུ་གསོལ། ཨོཾ་ན་མོ་བྷ་ག་ཝ་ཏེ་སོགས་རྩ་རིག་དང༌། ཨོཾ་བཛྲ་དྷརྨ་ཧྲཱིཿ རྡོ་རྗེ་ཆོས་དང༌། ཨོཾ་བཛྲ་ཏཱིཀྵ་ཎ་དྷཱིཿ རྡོ་རྗེ་རྣོན་པོ་དང༌། ཨོཾ་བཛྲ་ཧེ་ཏུ་ཀྲུཾ། རྡོ་རྗེ་རྒྱུ་དང༌། ཨོཾ་བཛྲ་བྷཱ་ཥ་བྷཱ། རྡོ་རྗེ་སྨྲ་བ་སྟེ། ནུབ་ཕྱོགས་ཀྱི་སེམས་དཔའ་བཞིས་ཀྱང་བདག་སོགས་སེམས་ཅན་ཐམས་ཅད་ཟབ་གསལ་གཉིས་སུ་མེད་པ་ལ་སོགས་པ་ལམ་གྱི་རྟོགས་པ་མཐའ་དག་རྒྱུད་ལ་སྐྱེ་བར་མཛད་དུ་གསོལ། ཨོཾ་ན་མོ་བྷ་ག་ཝ་ཏེ་སརྦ་དུརྒ་ཏི་པ་རི་ཤོ་དྷ་ནི་རཱ་ཛཱ་ཡ། ཏ་ཐཱ་ག་ཏཱ་ཡ། ཨརྷ་ཏེ་སམྱཀྶཾ་བུདྡྷཱ་ཡ༑ ཏདྱ་ཐཱ། ཨོཾ་ཤོ་དྷ་ནི་ཤོ་དྷ་ནི། སརྦ་པཱ་པཾ་བི་ཤོ་དྷ་ནི། ཤུདྡྷེ་བི་ཤུདྡྷེ། སརྦ་ཀརྨ་ཨཱ་ཝ་ར་ཎ་བི་ཤོ་དྷ་ནེ་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་ཀརྨ་ཀཾ། རྡོ་རྗེ་ལས་དང༌། ཨོཾ་བཛྲ་རཀྵ་རཾ། རྡོ་རྗེ་སྲུང་བ་དང༌༑ ཨོཾ་བཛྲ་ཡཀྵ་ཀྵཾ། རྡོ་རྗེ་གནོད་སྦྱིན་དང༌། ཨོཾ་བཛྲ་སནྡྷིཥྚཾ། རྡོ་རྗེ་ཁུ་ཚུར་ཏེ། བྱང་ཕྱོགས་ཀྱི་སེམས་དཔའ་བཞིས་ཀྱང་བདག་སོགས་སེམས་ཅན་ཐམས་ཅད་ཟབ་གསལ་གཉིས་སུ་མེད་པ་ལ་སོགས་པ་ལམ་གྱི་རྟོགས་པ་མཐའ་དག་རྒྱུད་ལ་སྐྱེ་བར་མཛད་དུ་གསོལ།

ཨོཾ་ན་མོ་བྷ་ག་ཝ་ཏེ་སོགས་རྩ་རིག་དང༌། ཨོཾ་བཛྲ་ལཱ་སྱེ་ཧཱུྂ། སྒེག་མོ་མ་དང༌། ཨོཾ་བཛྲ་མཱ་ལེ་ཏྲཱཾ། རྡོ་རྗེ་ཕྲེང་བ་མ་དང༌། ཨོཾ་བཛྲ་གཱི་ཏི་ཧྲཱིཿ གླུ་མ་དང༌། ཨོཾ་བཛྲ་ནྲྀཏྱེ་ཨཱཿ གར་མ་སྟེ། ནང་གི་སྒེག་མོ་བཞིས་ཀྱང་བདག་སོགས་སེམས་ཅན་ཐམས་ཅད་ཟབ་གསལ་གཉིས་སུ་མེད་པ་ལ་སོགས་པ་ལམ་གྱི་རྟོགས་པ་མཐའ་དག་རྒྱུད་ལ་སྐྱེ་བར་མཛད་དུ་གསོལ། ཨོཾ་ན་མོ་བྷ་ག་ཝ་ཏེ་སོགས་རྩ་རིག་དང༌། ཨོཾ་བཛྲ་དྷཱུ་པེ་ཧཱུྂ། བདུག་སྤོས་མ་དང༌། ཨོཾ་བཛྲ་པུཥྤེ་ཏྲཱཾ། མེ་ཏོག་མ་དང༌། ཨོཾ་བཛྲ་ཨཱ་ལོ་ཀེ་ཧྲཱིཿ སྣང་གསལ་མ་དང༌། ཨོཾ་བཛྲ་གནྡྷེ་ཨཱཿ དྲི་ཆབ་མ་སྟེ། ཕྱི་ཡི་སྒེག་མོ་བཞིས་ཀྱང་བདག་སོགས་སེམས་ཅན་ཐམས་ཅད་ཟབ་གསལ་གཉིས་སུ་མེད་པ་ལ་སོགས་པ་ལམ་གྱི་རྟོགས་པ་མཐའ་དག་རྒྱུད་ལ་སྐྱེ་བར་མཛད་དུ་གསོལ། ཨོཾ་ན་མོ་བྷ་ག་ཝ་ཏེ་སརྦ་དུརྒ་ཏི་པ་རི་ཤོ་དྷ་ནི་རཱ་ཛཱ་ཡ། ཏ་ཐཱ་ག་ཏཱ་ཡ། ཨརྷ་ཏེ་སམྱཀྶཾ་བུདྡྷཱ་ཡ། ཏདྱ་ཐཱ། ཨོཾ་ཤོ་དྷ་ནི་ཤོ་དྷ་ནི། སརྦ་པཱ་པཾ་བི་ཤོ་དྷ་ནི། ཤུདྡྷེ་བི་ཤུདྡྷེ། སརྦ་ཀརྨ་ཨཱ་ཝ་ར་ཎ་བི་ཤོ་དྷ་ནེ་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་ཨངྐུ་ཤ་ཛཿ རྡོ་རྗེ་ལྕགས་ཀྱུ་དང༌། ཨོཾ་བཛྲ་པཱ་ཤ་ཧཱུྂ། རྡོ་རྗེ་ཞགས་པ་དང༌། ཨོཾ་བཛྲ་སྥོ་ཊ་བཾ། རྡོ་རྗེ་ལྕགས་སྒྲོག་དང༌། ཨོཾ་བཛྲ་གྷཎྚ་ཧོཿ རྡོ་རྗེ་འབེབས་པ་སྟེ། སྒོ་མ་བཞིས་ཀྱང་བདག་སོགས་སེམས་ཅན་ཐམས་ཅད་ཟབ་གསལ་གཉིས་སུ་མེད་པ་ལ་སོགས་པ་ལམ་གྱི་རྟོགས་པ་མཐའ་དག་རྒྱུད་ལ་སྐྱེ་བར་མཛད་དུ་གསོལ། ཨོཾ་ན་མོ་བྷ་ག་ཝ་ཏེ་སོགས་རྩ་རིག་དང༌། ཨོཾ་སརྦ་སཾ་སྐཱ་ར་པ་རི་ཤུདྡྷ་དྷརྨ་ཏེ་ག་ག་ན་ས་མུདྒ་ཏེ་མ་ཧཱ་ན་ཡ་པ་རི་ཝཱ་རེ་སྭཱ་ཧཱ། བྱམས་པ། མཐོང་བ་དོན་ཡོད། ངན་སོང་ཀུན་འདྲེན། མྱ་ངན་མུན་འཇོམས། སྤོས་ཀྱི་གླང་པོ། དཔའ་བར་འགྲོ་བ། ནམ་མཁའ་མཛོད། ཡེ་ཤེས་ཏོག །འོད་དཔག་མེད། ཟླ་འོད། བཟང་སྐྱོང༌། དྲ་བ་ཅན། རྡོ་རྗེ་སྙིང་པོ། བློ་གྲོས་མི་ཟད་པ། སྤོབས་པ་བརྩེགས་པ། ཀུན་ཏུ་བཟང་པོ་སྟེ། བསྐལ་བཟང་གི་སེམས་དཔའ་བཅུ་དྲུག་གིས་ཀྱང་བདག་སོགས་སེམས་ཅན་ཐམས་ཅད་ཟབ་གསལ་གཉིས་སུ་མེད་པ་ལ་སོགས་པ་ལམ་གྱི་རྟོགས་པ་མཐའ་དག་རྒྱུད་ལ་སྐྱེ་བར་མཛད་དུ་གསོལ།

ཨོཾ་ན་མོ་བྷ་ག་ཝ་ཏེ་སརྦ་དུརྒ་ཏི་པ་རི་ཤོ་དྷ་ནི་རཱ་ཛཱ་ཡ། ཏ་ཐཱ་ག་ཏཱ་ཡ། ཨརྷ་ཏེ་སམྱཀྶཾ་བུདྡྷཱ་ཡ། ཏདྱ་ཐཱ། ཨོཾ་ཤོ་དྷ་ནི་ཤོ་དྷ་ནི། སརྦ་པཱ་པཾ་བི་ཤོ་དྷ་ནི། ཤུདྡྷེ་བི་ཤུདྡྷེ། སརྦ་ཀརྨ་ཨཱ་ཝ་ར་ཎ་བི་ཤོ་དྷ་ནེ་སྭཱ་ཧཱ། དགའ་བོ། སྒྲ་གཅན་འཛིན། ཤཱ་རིའི་བུ། མཽདྒལ་གྱི་བུ། རབ་འབྱོར། འཆར་ཀ །མ་འགགས་པ། ཀཽཎྜི་ནྱ། ཉེ་བར་འཁོར། རྟ་ཐུལ། བ་ལང་བདག །ཀུན་དགའ་བོ། མིང་ཆེན། ནོར་འཛིན། གང་པོ། འོད་སྲུང་སྟེ། ཉན་ཐོས་ཆེན་པོ་བཅུ་དྲུག་གིས་ཀྱང་བདག་སོགས་སེམས་ཅན་ཐམས་ཅད་ཟབ་གསལ་གཉིས་སུ་མེད་པ་ལ་སོགས་པ་ལམ་གྱི་རྟོགས་པ་མཐའ་དག་རྒྱུད་ལ་སྐྱེ་བར་མཛད་དུ་གསོལ༑ ཨོཾ་ན་མོ་བྷ་ག་ཝ་ཏེ་སོགས་རྩ་རིག་དང༌། གདུང་བ་སེལ་བ། སྣང་བ་བརྩེགས་པ། དཔའ་བོ་འདུལ་བ། ཤེས་རབ་མགྱོགས་ཆེན། སྨྲ་བའི་སྒྲ་གཅན། བསེ་རུ་ལྟ་བུ། རྫུ་འཕྲུལ་ཆེར་སྟོན༑ སེང་གེ་སྒྲ། འཇིགས་པ་མེད་པ། རི་བོ་རྒྱལ། དུག་ཆེན་འཇོམས་པ། ཡིད་མྱུར་བ་སྟེ། རང་རྒྱལ་ཆེན་པོ་བཅུ་གཉིས་ཀྱིས་ཀྱང་བདག་སོགས་སེམས་ཅན་ཐམས་ཅད་ཟབ་གསལ་གཉིས་སུ་མེད་པ་ལ་སོགས་པ་ལམ་གྱི་རྟོགས་པ་མཐའ་དག་རྒྱུད་ལ་སྐྱེ་བར་མཛད་དུ་གསོལ། ཨོཾ་ན་མོ་བྷ་ག་ཝ་ཏེ་སརྦ་དུརྒ་ཏི་པ་རི་ཤོ་དྷ་ནི་རཱ་ཛཱ་ཡ། ཏ་ཐཱ་ག་ཏཱ་ཡ། ཨརྷ་ཏེ་སམྱཀྶཾ་བུདྡྷཱ་ཡ༑ ཏདྱ་ཐཱ། ཨོཾ་ཤོ་དྷ་ནི་ཤོ་དྷ་ནི། སརྦ་པཱ་པཾ་བི་ཤོ་དྷ་ནི། ཤུདྡྷེ་བི་ཤུདྡྷེ། སརྦ་ཀརྨ་ཨཱ་ཝ་ར་ཎ་བི་ཤོ་དྷ་ནེ་སྭཱ་ཧཱ། ཞེས་དང༌། ཁྲོ་བོ་བཞི། བྲན་བཞི། ཕོ་ཉ་བཞི། མངགས་པ་བཞི༑ ཁྲོ་ཚོགས་ཆེན་པོ་བཅུ་དྲུག་གིས་ཀྱང་བདག་སོགས་སེམས་ཅན་ཐམས་ཅད་ཟབ་གསལ་གཉིས་སུ་མེད་པ་ལ་སོགས་པ་ལམ་གྱི་རྟོགས་པ་མཐའ་དག་རྒྱུད་ལ་སྐྱེ་བར་མཛད་དུ་གསོལ། ཞེས་སོ།། །།གཙུག་ཏོར་དྲི་མེད་ཀྱི་གཟུངས་ཐུང་ནི། ཨོཾ་ན་མ་སྟྲཻ་ཡ་དྷིའ་ཀཱ་ནཱཾ། སརྦ་ཏ་ཐཱ་ག་ཏ་ཧྲྀ་ད་ཡ་གརྦཧེ་ཛྭ་ལ་ཛྭ་ལ། དྷརྨ་དྷཱ་ཏུ་གརྦཧེ། སཾ་ཧ་ར་མ་མ་ཨཱ་ཡུ་སཾ་ཤོ་དྷ་ཡ་སརྦ་པཱ་པཾ། སརྦ་ཏ་ཐཱ་ག་ཏ་ས་མནྟོཥྞཱི་ཥ་བི་མ་ལེ་བི་ཤུདྡྷེ་ཧཱུྂ་ཧཱུྂ་ཧཱུྂ་ཧཱུྂ།

ཨཾ་བཾ་སཾ་ཛཾ་སྭཱ་ཧཱ། ཞེས་དང༌། གཙུག་ཏོར་རྣམ་རྒྱལ་གྱི་གཟུངས་ཐུང་ནི། ཨོཾ་བྷྲཱུཾ་སྭཱ་ཧཱ། ཨོཾ་མ་མྲྀ་ཏ་ཨཱ་ཡུརྡ་དེ་སྭཱ་ཧཱ། ཨོཾ་ཨཱཿཧཱུྂ་ཧྲཱིཿཨཾ་ཨཿརཀྵ་རཀྵ་མཱཾ་སརྦ་སཏྭཱ་ཤྩ་སྭཱ་ཧཱ། ཞེས་བྱང་ཆོག་སྐབས་སུ་སྦྱར་བར་བྱའོ།། །།
